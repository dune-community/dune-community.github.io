48 template <class GridViewImp, class SourceImp, class RangeImp>
49 class L2GlobalProjectionLocalizableOperator : public LocalizableOperatorBase<GridViewImp, SourceImp, RangeImp>
50 {
51  static_assert(Stuff::is_localizable_function<SourceImp>::value, "");
52  static_assert(is_discrete_function<RangeImp>::value, "");

54 
55 public:
56  using typename BaseType::GridViewType;
57  using typename BaseType::SourceType;
58  using typename BaseType::RangeType;

60  typedef typename Stuff::LA::Container<typename RangeType::RangeFieldType, VectorType::sparse_matrix_type>::MatrixType
61  MatrixType;

63 private:
64  typedef typename RangeType::SpaceType SpaceType;
65  typedef L2MatrixOperator<SpaceType, MatrixType, GridViewType> LhsOperatorType;
66  typedef L2VolumeVectorFunctional<SourceType, SpaceType, VectorType, GridViewType> RhsFunctionalType;
67 
68 public:
69  L2GlobalProjectionLocalizableOperator(const size_t over_integrate, GridViewType grd_vw, const SourceType& src,
70  RangeType& rng)
71  : BaseType(grd_vw, src, rng)
72  , lhs_operator_(over_integrate, range_.space(), grid_view_)
73  , rhs_functional_(over_integrate, source_, range_.space(), grid_view_)
74  , solved_(false)
75  {
76  this->add(lhs_operator_);
77  this->add(rhs_functional_);
78  issue_warning(this->range().space());
79  }
80 
81  L2GlobalProjectionLocalizableOperator(GridViewType grd_vw, const SourceType& src, RangeType& rng)
82  : L2GlobalProjectionLocalizableOperator(0, grd_vw, src, rng)
83  {
84  }
85 
86  void apply()
87  {
88  if (solved_)
89  return;
90  BaseType::apply();
91  try {
92  Stuff::LA::Solver<MatrixType>(lhs_operator_.matrix()).apply(rhs_functional_.vector(), range_.vector());
93  } catch (Stuff::Exceptions::linear_solver_failed& ee) {
94  DUNE_THROW(projection_error,
95  "L2 projection failed because a global matrix could not be inverted!\n\n"
96  << "This was the original error: "
97  << ee.what());
98  }
