58  template <class E, class D, size_t d, class R, size_t r, size_t rC, class RS, class V>
59  void apply(const Stuff::LocalizableFunctionInterface<E, D, d, R, r, rC>& source,
60  LocalDiscreteFunction<RS, V>& local_range) const
61  {
62  const auto& entity = local_range.entity();
63  if (!entity.hasBoundaryIntersections())
64  return;
65  const auto local_dirichlet_DoFs = local_range.space().local_dirichlet_DoFs(entity, boundary_info_);
66  if (local_dirichlet_DoFs.size() == 0)
67  return;
68  const auto lagrange_points = local_range.space().lagrange_points(entity);
69  const auto local_source = source.local_function(entity);
70  auto& local_range_DoF_vector = local_range.vector();
71  assert(lagrange_points.size() == local_range_DoF_vector.size());
72  for (const size_t& local_DoF_id : local_dirichlet_DoFs)
73  local_range_DoF_vector.set(local_DoF_id, local_source->evaluate(lagrange_points[local_DoF_id]));
74  } // ... apply(...)
