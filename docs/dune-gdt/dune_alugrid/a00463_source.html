293  template <class IntersectionType, class R>
294  void evaluate(const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionEntity,
295  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionNeighbor,
296  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseEntity,
297  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseEntity,
298  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseNeighbor,
299  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseNeighbor,
300  const IntersectionType& intersection, const Dune::FieldVector<DomainFieldType, 1>& localPoint,
301  Dune::DynamicMatrix<R>& entityEntityRet, Dune::DynamicMatrix<R>& neighborNeighborRet,
302  Dune::DynamicMatrix<R>& entityNeighborRet, Dune::DynamicMatrix<R>& neighborEntityRet) const
303  {
304  // clear ret
305  entityEntityRet *= 0.0;
306  neighborNeighborRet *= 0.0;
307  entityNeighborRet *= 0.0;
308  neighborEntityRet *= 0.0;
309  // convert local point (which is in intersection coordinates) to entity/neighbor coordinates
310  const auto localPointEn = intersection.geometryInInside().global(localPoint);
311  const auto localPointNe = intersection.geometryInOutside().global(localPoint);
312  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
313  // evaluate local function
314  const auto functionValueEn = localFunctionEntity.evaluate(localPointEn);
315  const auto functionValueNe = localFunctionNeighbor.evaluate(localPointNe);
316  // compute penalty (see Epshteyn, Riviere, 2007)
317  const auto max_polorder =
318  std::max(testBaseEntity.order(),
319  std::max(ansatzBaseEntity.order(), std::max(testBaseNeighbor.order(), ansatzBaseNeighbor.order())));
320  const R sigma = internal::inner_sigma(max_polorder);
321  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
322  // evaluate bases
323  // * entity
324  // * test
325  const auto rowsEn = testBaseEntity.size();
326  const auto testValuesEn = testBaseEntity.evaluate(localPointEn);
327  const auto testGradientsEn = testBaseEntity.jacobian(localPointEn);
328  // * ansatz
329  const auto colsEn = ansatzBaseEntity.size();
330  const auto ansatzValuesEn = ansatzBaseEntity.evaluate(localPointEn);
331  const auto ansatzGradientsEn = ansatzBaseEntity.jacobian(localPointEn);
332  // * neighbor
333  // * test
334  const auto rowsNe = testBaseNeighbor.size();
335  const auto testValuesNe = testBaseNeighbor.evaluate(localPointNe);
336  const auto testGradientsNe = testBaseNeighbor.jacobian(localPointNe);
337  // * ansatz
338  const auto colsNe = ansatzBaseNeighbor.size();
339  const auto ansatzValuesNe = ansatzBaseNeighbor.evaluate(localPointNe);
340  const auto ansatzGradientsNe = ansatzBaseNeighbor.jacobian(localPointNe);
341  // compute the evaluations
342  assert(entityEntityRet.rows() >= rowsEn);
343  assert(entityEntityRet.cols() >= colsEn);
344  assert(entityNeighborRet.rows() >= rowsEn);
345  assert(entityNeighborRet.cols() >= colsNe);
346  assert(neighborEntityRet.rows() >= rowsNe);
347  assert(neighborEntityRet.cols() >= colsEn);
348  assert(neighborNeighborRet.rows() >= rowsNe);
349  assert(neighborNeighborRet.cols() >= colsNe);
350  // loop over all entity test basis functions
351  for (size_t ii = 0; ii < rowsEn; ++ii) {
352  auto& entityEntityRetRow = entityEntityRet[ii];
353  auto& entityNeighborRetRow = entityNeighborRet[ii];
354  // loop over all entity ansatz basis functions
355  for (size_t jj = 0; jj < colsEn; ++jj) {
356  // consistency term
357  entityEntityRetRow[jj] +=
358  -0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesEn[ii];
359  // symmetry term
360  entityEntityRetRow[jj] +=
361  -0.5 * ansatzValuesEn[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
362  // penalty term
363  entityEntityRetRow[jj] += penalty * ansatzValuesEn[jj] * testValuesEn[ii];
364  } // loop over all entity ansatz basis functions
365  // loop over all neighbor ansatz basis functions
366  for (size_t jj = 0; jj < colsNe; ++jj) {
367  // consistency term
368  entityNeighborRetRow[jj] +=
369  -0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesEn[ii];
370  // symmetry term
371  entityNeighborRetRow[jj] +=
372  0.5 * ansatzValuesNe[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
373  // penalty term
374  entityNeighborRetRow[jj] += -1.0 * penalty * ansatzValuesNe[jj] * testValuesEn[ii];
375  } // loop over all neighbor ansatz basis functions
376  } // loop over all entity test basis functions
377  // loop over all neighbor test basis functions
378  for (size_t ii = 0; ii < rowsNe; ++ii) {
379  auto& neighborEntityRetRow = neighborEntityRet[ii];
380  auto& neighborNeighborRetRow = neighborNeighborRet[ii];
381  // loop over all entity ansatz basis functions
382  for (size_t jj = 0; jj < colsEn; ++jj) {
383  // consistency term
384  neighborEntityRetRow[jj] +=
385  0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesNe[ii];
386  // symmetry term
387  neighborEntityRetRow[jj] +=
388  -0.5 * ansatzValuesEn[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
389  // penalty term
390  neighborEntityRetRow[jj] += -1.0 * penalty * ansatzValuesEn[jj] * testValuesNe[ii];
391  } // loop over all entity ansatz basis functions
392  // loop over all neighbor ansatz basis functions
393  for (size_t jj = 0; jj < colsNe; ++jj) {
394  // consistency term
395  neighborNeighborRetRow[jj] +=
396  0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesNe[ii];
397  // symmetry term
398  neighborNeighborRetRow[jj] +=
399  0.5 * ansatzValuesNe[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
400  // penalty term
401  neighborNeighborRetRow[jj] += penalty * ansatzValuesNe[jj] * testValuesNe[ii];
402  } // loop over all neighbor ansatz basis functions
403  } // loop over all neighbor test basis functions
404  } // ... evaluate(...)

487  template <class IntersectionType, class R>
488  void evaluate(const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunction,
489  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBase,
490  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBase,
491  const IntersectionType& intersection, const Dune::FieldVector<DomainFieldType, 1>& localPoint,
492  Dune::DynamicMatrix<R>& ret) const
493  {
494  // clear ret
495  ret *= 0.0;
496  // get local point (which is in intersection coordinates) in entity coordinates
497  const auto localPointEntity = intersection.geometryInInside().global(localPoint);
498  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
499  // evaluate local function
500  const auto functionValue = localFunction.evaluate(localPointEntity);
501  // compute penalty (see Epshteyn, Riviere, 2007)
502  const auto max_polorder = std::max(testBase.order(), ansatzBase.order());
503  const R sigma = internal::boundary_sigma(max_polorder);
504  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
505  // evaluate bases
506  // * test
507  const auto rows = testBase.size();
508  const auto testValues = testBase.evaluate(localPointEntity);
509  const auto testGradients = testBase.jacobian(localPointEntity);
510  // * ansatz
511  const auto cols = ansatzBase.size();
512  const auto ansatzValues = ansatzBase.evaluate(localPointEntity);
513  const auto ansatzGradients = ansatzBase.jacobian(localPointEntity);
514  // compute products
515  assert(ret.rows() >= rows);
516  assert(ret.cols() >= cols);
517  // loop over all test basis functions
518  for (size_t ii = 0; ii < rows; ++ii) {
519  auto& retRow = ret[ii];
520  // loop over all ansatz basis functions
521  for (size_t jj = 0; jj < cols; ++jj) {
522  // consistency term
523  retRow[jj] += -1.0 * functionValue * (ansatzGradients[jj][0] * unitOuterNormal) * testValues[ii];
524  // symmetry term
525  retRow[jj] += -1.0 * ansatzValues[jj] * functionValue * (testGradients[ii][0] * unitOuterNormal);
526  // penalty term
527  retRow[jj] += penalty * ansatzValues[jj] * testValues[ii];
528  } // loop over all ansatz basis functions
529  } // loop over all test basis functions
530  } // ... evaluate(...)

606  size_t
607  order(const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLF, rCLF>& localDiffusion,
608  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLR, rCLR>& localDirichlet,
609  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, rT, rCT>& testBase) const
610  {
611  const size_t testOrder = testBase.order();
612  const size_t testGradientOrder = boost::numeric_cast<size_t>(std::max(ssize_t(testOrder) - 1, ssize_t(0)));
613  const size_t diffusionOrder = localDiffusion.order();
614  const size_t dirichletOrder = localDirichlet.order();
615  return std::max(testOrder + dirichletOrder, diffusionOrder + testGradientOrder + dirichletOrder);
616  } // ... order(...)

622  template <class IntersectionType, class R>
623  void evaluate(const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDiffusion,
624  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDirichlet,
625  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& testBase,
626  const IntersectionType& intersection,
627  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& localPoint, Dune::DynamicVector<R>& ret) const
628  {
629  // clear ret
630  ret *= 0.0;
631  // get local point (which is in intersection coordinates) in entity coordinates
632  const auto localPointEntity = intersection.geometryInInside().global(localPoint);
633  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
634  // evaluate local functions
635  const auto diffusionValue = localDiffusion.evaluate(localPointEntity);
636  const auto dirichletValue = localDirichlet.evaluate(localPointEntity);
637  // compute penalty (see Epshteyn, Riviere, 2007)
638  const auto polorder = testBase.order();
639  const R sigma = internal::boundary_sigma(polorder);
640  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
641  // evaluate basis
642  const auto size = testBase.size();
643  const auto testValues = testBase.evaluate(localPointEntity);
644  const auto testGradients = testBase.jacobian(localPointEntity);
645  // compute
646  assert(ret.size() >= size);
647  // loop over all test basis functions
648  for (size_t ii = 0; ii < size; ++ii) {
649  // symmetry term
650  ret[ii] += -1.0 * dirichletValue * diffusionValue * (testGradients[ii][0] * unitOuterNormal);
651  // penalty term
652  ret[ii] += penalty * dirichletValue * testValues[ii];
653  } // loop over all test basis functions
654  } // ... evaluate(...)
