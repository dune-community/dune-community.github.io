45  // The following tag and the two ctors are unfortunately necessary. There should have been two ctors,
46  // L2LocalizableProduct(Args&& ...args)
47  // and
48  // L2LocalizableProduct(const size_t over_integrate, Args&& ...args),
49  // both using perfect forwarding on ...args (see the documentation of the ctor below for application szenarios).
50  // However, the compiler was not able to differentiate between the two. The work-around below was inspired by:
51  // http://stackoverflow.com/questions/32957830/c-variadic-template-constructor-and-common-constructors
52  template <bool anything>
53  struct tag
54  {
55  explicit tag(int)
56  {
57  }
58  }; // we do not want tag to be implicitely constructable, just to be safe
