175  template <class R>
176  void evaluate(
177  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>& local_diffusion,
178  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain>& local_diffusive_flux,
179  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& test_base,
180  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& ansatz_base,
181  const Dune::FieldVector<DomainFieldType, dimDomain>& local_point, Dune::DynamicMatrix<R>& ret) const
182  {
183  // evaluate local functions
184  const auto diffusion_value = local_diffusion.evaluate(local_point);
185  const R one_over_diffusion_value = 1.0 / diffusion_value;
186  const auto diffusive_flux_value = local_diffusive_flux.evaluate(local_point);
187  // evaluate test gradient
188  const size_t rows = test_base.size();
189  const auto test_gradients = test_base.jacobian(local_point);
190  auto left_sum = test_gradients[0];
191  assert(test_gradients.size() == rows);
192  // evaluate ansatz gradient
193  const size_t cols = ansatz_base.size();
194  const auto ansatz_gradients = ansatz_base.jacobian(local_point);
195  auto right_sum = ansatz_gradients[0];
196  assert(ansatz_gradients.size() == rows);
197  // compute
198  assert(ret.rows() >= rows);
199  assert(ret.cols() >= cols);
200  for (size_t ii = 0; ii < rows; ++ii) {
201  left_sum = test_gradients[ii];
202  left_sum[0] *= diffusion_value;
203  left_sum[0] += diffusive_flux_value;
204  auto& retRow = ret[ii];
205  for (size_t jj = 0; jj < cols; ++jj) {
206  right_sum = ansatz_gradients[jj];
207  right_sum[0] *= diffusion_value;
208  right_sum[0] += diffusive_flux_value;
209  retRow[jj] = one_over_diffusion_value * (left_sum[0] * right_sum[0]);
210  }
211  }
212  } // ... evaluate(...)

322  template <class R>
323  void evaluate(
324  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>& local_diffusion_factor,
325  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain, dimDomain>&
326  local_diffusion_tensor,
327  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain>& local_diffusive_flux,
328  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& test_base,
329  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& ansatz_base,
330  const Dune::FieldVector<DomainFieldType, dimDomain>& local_point, Dune::DynamicMatrix<R>& ret) const
331  {
332  typedef FieldVector<R, dimDomain> DomainType;
333  DomainType left_sum(0);
334  DomainType right_sum(0);
335  // evaluate local functions
336  const auto diffusion_factor_value = local_diffusion_factor.evaluate(local_point);
337  typedef Stuff::Common::FieldMatrix<R, dimDomain, dimDomain> TensorType;
338  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(local_point);
339  const TensorType diffusion_value = diffusion_tensor_value * diffusion_factor_value;
340  // TODO: there is no documented way to assert that the inversion was successfull, so find one or check the matrix
341  // beforehand
342  TensorType one_over_diffusion_value = diffusion_tensor_value;
343  one_over_diffusion_value.invert();
344  one_over_diffusion_value /= diffusion_factor_value;
345  const auto diffusive_flux_value = local_diffusive_flux.evaluate(local_point);
346  // evaluate test gradient
347  const size_t rows = test_base.size();
348  const auto test_gradients = test_base.jacobian(local_point);
349  assert(test_gradients.size() == rows);
350  // evaluate ansatz gradient
351  const size_t cols = ansatz_base.size();
352  const auto ansatz_gradients = ansatz_base.jacobian(local_point);
353  assert(ansatz_gradients.size() == rows);
354  // compute
355  assert(ret.rows() >= rows);
356  assert(ret.cols() >= cols);
357  for (size_t ii = 0; ii < rows; ++ii) {
358  left_sum = diffusion_value * test_gradients[ii];
359  left_sum += diffusive_flux_value;
360  auto& retRow = ret[ii];
361  for (size_t jj = 0; jj < cols; ++jj) {
362  right_sum = diffusion_value * ansatz_gradients[jj];
363  right_sum += diffusive_flux_value;
364  retRow[jj] = (one_over_diffusion_value * left_sum) * right_sum;
365  }
366  }
367  } // ... evaluate(...)
