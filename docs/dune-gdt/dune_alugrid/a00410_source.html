65  template <class GV, class V>
66  void apply(const Stuff::LocalizableFunctionInterface<EntityType, DomainFieldType, dimDomain, FieldType, 1>& source,
67  DiscreteFunction<DunePdelabRtSpaceWrapper<GV, 0, FieldType, dimDomain>, V>& range) const
68  {
69  const auto& rtn0_space = range.space();
70  auto& range_vector = range.vector();
71  const FieldType infinity = std::numeric_limits<FieldType>::infinity();
72  for (size_t ii = 0; ii < range_vector.size(); ++ii)
73  range_vector[ii] = infinity;
74  const LocalEllipticIpdgIntegrands::Inner<DiffusionFactorType, DiffusionTensorType> inner_evaluation(
75  diffusion_factor_, diffusion_tensor_);
76  const LocalEllipticIpdgIntegrands::BoundaryLHS<DiffusionFactorType, DiffusionTensorType> boundary_evaluation(
77  diffusion_factor_, diffusion_tensor_);
78  const Stuff::Functions::Constant<EntityType, DomainFieldType, dimDomain, FieldType, 1> constant_one(1);
79  DomainType normal(0);
80  DomainType xx_entity(0);
81  DynamicMatrix<FieldType> tmp_matrix(1, 1, 0);
82  DynamicMatrix<FieldType> tmp_matrix_en_en(1, 1, 0);
83  DynamicMatrix<FieldType> tmp_matrix_en_ne(1, 1, 0);
84  std::vector<typename DunePdelabRtSpaceWrapper<GV, 0, FieldType, dimDomain>::BaseFunctionSetType::RangeType>
85  basis_values(rtn0_space.mapper().maxNumDofs(),
86  typename DunePdelabRtSpaceWrapper<GV, 0, FieldType, dimDomain>::BaseFunctionSetType::RangeType(0));
87  // walk the grid
88  const auto entity_it_end = grid_view_.template end<0>();
89  for (auto entity_it = grid_view_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
90  const auto& entity = *entity_it;
91  const auto local_DoF_indices = rtn0_space.local_DoF_indices(entity);
92  const auto global_DoF_indices = rtn0_space.mapper().globalIndices(entity);
93  assert(global_DoF_indices.size() == local_DoF_indices.size());
94  const auto local_diffusion_factor = diffusion_factor_.local_function(entity);
95  const auto local_diffusion_tensor = diffusion_tensor_.local_function(entity);
96  const auto local_source = source.local_function(entity);
97  const auto local_basis = rtn0_space.base_function_set(entity);
98  const auto local_constant_one = constant_one.local_function(entity);
99  // walk the intersections
100  const auto intersection_it_end = grid_view_.iend(entity);
101  for (auto intersection_it = grid_view_.ibegin(entity); intersection_it != intersection_it_end;
102  ++intersection_it) {
103  const auto& intersection = *intersection_it;
104  if (intersection.neighbor() && !intersection.boundary()) {
105  const auto neighbor_ptr = intersection.outside();
106  const auto& neighbor = *neighbor_ptr;
107  if (grid_view_.indexSet().index(entity) < grid_view_.indexSet().index(neighbor)) {
108  const auto local_diffusion_factor_neighbor = diffusion_factor_.local_function(neighbor);
109  const auto local_diffusion_tensor_neighbor = diffusion_tensor_.local_function(neighbor);
110  const auto local_source_neighbor = source.local_function(neighbor);
111  const auto local_constant_one_neighbor = constant_one.local_function(neighbor);
112  const size_t local_intersection_index = intersection.indexInInside();
113  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
114  // do a face quadrature
115  FieldType lhs = 0;
116  FieldType rhs = 0;
117  const size_t integrand_order = inner_evaluation.order(*local_diffusion_factor,
118  *local_diffusion_tensor,
119  *local_diffusion_factor_neighbor,
120  *local_diffusion_tensor_neighbor,
121  *local_constant_one,
122  *local_source,
123  *local_constant_one_neighbor,
124  *local_source_neighbor);
125  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
126  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
127  const auto quadrature_it_end = quadrature.end();
128  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
129  const auto& xx_intersection = quadrature_it->position();
130  xx_entity = intersection.geometryInInside().global(xx_intersection);
131  normal = intersection.unitOuterNormal(xx_intersection);
132  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
133  const FieldType weigth = quadrature_it->weight();
134  // evalaute
135  local_basis.evaluate(xx_entity, basis_values);
136  const auto& basis_value = basis_values[local_DoF_index];
137  tmp_matrix_en_en *= 0.0;
138  tmp_matrix_en_ne *= 0.0;
139  inner_evaluation.evaluate(*local_diffusion_factor,
140  *local_diffusion_tensor,
141  *local_diffusion_factor_neighbor,
142  *local_diffusion_tensor_neighbor,
143  *local_constant_one,
144  *local_source,
145  *local_constant_one_neighbor,
146  *local_source_neighbor,
147  intersection,
148  xx_intersection,
149  tmp_matrix_en_en, // <- we are interested in this one
150  tmp_matrix,
151  tmp_matrix_en_ne, // <- and this one
152  tmp_matrix);
153  // compute integrals
154  assert(tmp_matrix_en_en.rows() >= 1);
155  assert(tmp_matrix_en_en.cols() >= 1);
156  assert(tmp_matrix_en_ne.rows() >= 1);
157  assert(tmp_matrix_en_ne.cols() >= 1);
158  lhs += integration_factor * weigth * (basis_value * normal);
159  rhs += integration_factor * weigth * (tmp_matrix_en_en[0][0] + tmp_matrix_en_ne[0][0]);
160  } // do a face quadrature
161  // set DoF
162  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
163  // and make sure we are the first to do so
164  assert(!(range_vector[global_DoF_index] < infinity));
165  range_vector[global_DoF_index] = rhs / lhs;
166  }
167  } else if (intersection.boundary() && !intersection.neighbor()) {
168  const size_t local_intersection_index = intersection.indexInInside();
169  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
170  // do a face quadrature
171  FieldType lhs = 0;
172  FieldType rhs = 0;
173  const size_t integrand_order = boundary_evaluation.order(
174  *local_diffusion_factor, *local_diffusion_tensor, *local_source, *local_constant_one);
175  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
176  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
177  const auto quadrature_it_end = quadrature.end();
178  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
179  const auto xx_intersection = quadrature_it->position();
180  normal = intersection.unitOuterNormal(xx_intersection);
181  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
182  const FieldType weigth = quadrature_it->weight();
183  xx_entity = intersection.geometryInInside().global(xx_intersection);
184  // evalaute
185  local_basis.evaluate(xx_entity, basis_values);
186  const auto& basis_value = basis_values[local_DoF_index];
187  tmp_matrix *= 0.0;
188  boundary_evaluation.evaluate(*local_diffusion_factor,
189  *local_diffusion_tensor,
190  *local_constant_one,
191  *local_source,
192  intersection,
193  xx_intersection,
194  tmp_matrix);
195  // compute integrals
196  assert(tmp_matrix.rows() >= 1);
197  assert(tmp_matrix.cols() >= 1);
198  lhs += integration_factor * weigth * (basis_value * normal);
199  rhs += integration_factor * weigth * tmp_matrix[0][0];
200  } // do a face quadrature
201  // set DoF
202  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
203  // and make sure we are the first to do so
204  assert(!(range_vector[global_DoF_index] < infinity));
205  range_vector[global_DoF_index] = rhs / lhs;
206  } else
207  DUNE_THROW(Stuff::Exceptions::internal_error, "Unknown intersection type!");
208  } // walk the intersections
209  } // walk the grid
210  } // ... apply(...)
