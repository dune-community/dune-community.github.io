440  template <class R, class IntersectionType>
441  void evaluate(
442  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& local_diffusion_factor_en,
443  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain, dimDomain>&
444  local_diffusion_tensor_en,
445  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& local_diffusion_factor_ne,
446  const Stuff::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain, dimDomain>&
447  local_diffusion_tensor_ne,
448  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& test_base_en,
449  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& ansatz_base_en,
450  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& test_base_ne,
451  const Stuff::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& ansatz_base_ne,
452  const IntersectionType& intersection, const Dune::FieldVector<DomainFieldType, dimDomain - 1>& local_point,
453  Dune::DynamicMatrix<R>& ret_en_en, Dune::DynamicMatrix<R>& ret_ne_ne, Dune::DynamicMatrix<R>& ret_en_ne,
454  Dune::DynamicMatrix<R>& ret_ne_en) const
455  {
456  typedef Stuff::Common::FieldMatrix<R, dimDomain, dimDomain> TensorType;
457  // clear ret
458  ret_en_en *= 0.0;
459  ret_ne_ne *= 0.0;
460  ret_en_ne *= 0.0;
461  ret_ne_en *= 0.0;
462  // convert local point (which is in intersection coordinates) to entity/neighbor coordinates
463  const auto local_point_en = intersection.geometryInInside().global(local_point);
464  const auto local_point_ne = intersection.geometryInOutside().global(local_point);
465  const auto normal = intersection.unitOuterNormal(local_point);
466  // evaluate local function
467  const auto local_diffusion_factor_value_en = local_diffusion_factor_en.evaluate(local_point_en);
468  const TensorType local_diffusion_tensor_value_en = local_diffusion_tensor_en.evaluate(local_point_en);
469  const auto local_diffusion_factor_value_ne = local_diffusion_factor_ne.evaluate(local_point_ne);
470  const TensorType local_diffusion_tensor_value_ne = local_diffusion_tensor_ne.evaluate(local_point_ne);
471  const auto diffusion_value_en = local_diffusion_tensor_value_en * local_diffusion_factor_value_en;
472  const auto diffusion_value_ne = local_diffusion_tensor_value_ne * local_diffusion_factor_value_ne;
473  // // this evaluation has to be linear wrt the diffusion factor, so no other averaging method is allowed here!
474  // const auto local_diffusion_factor = (local_diffusion_factor_en + local_diffusion_factor_ne) * 0.5;
475  // compute penalty factor (see Epshteyn, Riviere, 2007)
476  const size_t max_polorder = std::max(
477  test_base_en.order(), std::max(ansatz_base_en.order(), std::max(test_base_ne.order(), ansatz_base_ne.order())));
478  const R sigma = LocalSipdgIntegrands::internal::inner_sigma(max_polorder);
479  // compute weighting (see Ern, Stephansen, Zunino 2007)
480  const R delta_plus = normal * (/*local_diffusion_tensor_ne*/ diffusion_value_ne * normal);
481  const R delta_minus = normal * (/*local_diffusion_tensor_en*/ diffusion_value_en * normal);
482  const R gamma = (delta_plus * delta_minus) / (delta_plus + delta_minus);
483  const R penalty = (/*local_diffusion_factor **/ sigma * gamma) / std::pow(intersection.geometry().volume(), beta_);
484  const R weight_plus = delta_minus / (delta_plus + delta_minus);
485  const R weight_minus = delta_plus / (delta_plus + delta_minus);
486  // evaluate bases
487  // * entity
488  // * test
489  const size_t rows_en = test_base_en.size();
490  const auto test_values_en = test_base_en.evaluate(local_point_en);
491  const auto test_gradients_en = test_base_en.jacobian(local_point_en);
492  // * ansatz
493  const size_t cols_en = ansatz_base_en.size();
494  const auto ansatz_values_en = ansatz_base_en.evaluate(local_point_en);
495  const auto ansatz_gradients_en = ansatz_base_en.jacobian(local_point_en);
496  // * neighbor
497  // * test
498  const size_t rows_ne = test_base_ne.size();
499  const auto test_values_ne = test_base_ne.evaluate(local_point_ne);
500  const auto test_gradients_ne = test_base_ne.jacobian(local_point_ne);
501  // * ansatz
502  const size_t cols_ne = ansatz_base_ne.size();
503  const auto ansatz_values_ne = ansatz_base_ne.evaluate(local_point_ne);
504  const auto ansatz_gradients_ne = ansatz_base_ne.jacobian(local_point_ne);
505  // compute the evaluations
506  assert(ret_en_en.rows() >= rows_en && ret_en_en.cols() >= cols_en);
507  assert(ret_en_ne.rows() >= rows_en && ret_en_ne.cols() >= cols_ne);
508  assert(ret_ne_en.rows() >= rows_ne && ret_ne_en.cols() >= cols_en);
509  assert(ret_ne_ne.rows() >= rows_ne && ret_ne_ne.cols() >= cols_ne);
510  // loop over all entity test basis functions
511  for (size_t ii = 0; ii < rows_en; ++ii) {
512  auto& ret_en_en_row = ret_en_en[ii];
513  auto& ret_en_ne_row = ret_en_ne[ii];
514  // loop over all entity ansatz basis functions
515  for (size_t jj = 0; jj < cols_en; ++jj) {
516  // consistency term
517  ret_en_en_row[jj] +=
518  -weight_minus * ((diffusion_value_en * ansatz_gradients_en[jj][0]) * normal) * test_values_en[ii];
519  // symmetry term
520  ret_en_en_row[jj] +=
521  -weight_minus * ansatz_values_en[jj] * ((diffusion_value_en * test_gradients_en[ii][0]) * normal);
522  // penalty term
523  ret_en_en_row[jj] += penalty * ansatz_values_en[jj] * test_values_en[ii];
524  } // loop over all entity ansatz basis functions
525  // loop over all neighbor ansatz basis functions
526  for (size_t jj = 0; jj < cols_ne; ++jj) {
527  // consistency term
528  ret_en_ne_row[jj] +=
529  -weight_plus * ((diffusion_value_ne * ansatz_gradients_ne[jj][0]) * normal) * test_values_en[ii];
530  // symmetry term
531  ret_en_ne_row[jj] +=
532  weight_minus * ansatz_values_ne[jj] * ((diffusion_value_en * test_gradients_en[ii][0]) * normal);
533  // penalty term
534  ret_en_ne_row[jj] += -1.0 * penalty * ansatz_values_ne[jj] * test_values_en[ii];
535  } // loop over all neighbor ansatz basis functions
536  } // loop over all entity test basis functions
537  // loop over all neighbor test basis functions
538  for (size_t ii = 0; ii < rows_ne; ++ii) {
539  auto& ret_ne_en_row = ret_ne_en[ii];
540  auto& ret_ne_ne_row = ret_ne_ne[ii];
541  // loop over all entity ansatz basis functions
542  for (size_t jj = 0; jj < cols_en; ++jj) {
543  // consistency term

545  weight_minus * ((diffusion_value_en * ansatz_gradients_en[jj][0]) * normal) * test_values_ne[ii];
546  // symmetry term
547  ret_ne_en_row[jj] +=
548  -weight_plus * ansatz_values_en[jj] * ((diffusion_value_ne * test_gradients_ne[ii][0]) * normal);
549  // penalty term
550  ret_ne_en_row[jj] += -1.0 * penalty * ansatz_values_en[jj] * test_values_ne[ii];
551  } // loop over all entity ansatz basis functions
552  // loop over all neighbor ansatz basis functions
553  for (size_t jj = 0; jj < cols_ne; ++jj) {
554  // consistency term
555  ret_ne_ne_row[jj] +=
556  weight_plus * ((diffusion_value_ne * ansatz_gradients_ne[jj][0]) * normal) * test_values_ne[ii];

962  const auto local_point_entity = intersection.geometryInInside().global(local_point);
963  const auto normal = intersection.unitOuterNormal(local_point);
964  // evaluate local functions
965  const auto dirichlet_value = local_dirichlet.evaluate(local_point_entity);
966  const auto diffusion_factor_value = local_diffusion_factor.evaluate(local_point_entity);
967  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(local_point_entity);
968  // compute penalty (see Epshteyn, Riviere, 2007)
969  const size_t polorder = test_base.order();
970  const R sigma = LocalSipdgIntegrands::internal::boundary_sigma(polorder);
971  // compute weighting (see Ern, Stephansen, Zunino 2007)
972  const R gamma = normal * (diffusion_tensor_value * normal);
973  const R penalty = (diffusion_factor_value * sigma * gamma) / std::pow(intersection.geometry().volume(), beta_);

975  const auto diffusion_value = diffusion_tensor_value * diffusion_factor_value;
976  // evaluate basis
977  const size_t size = test_base.size();
978  const auto test_values = test_base.evaluate(local_point_entity);
979  const auto test_gradients = test_base.jacobian(local_point_entity);
980  // compute
981  assert(ret.size() >= size);
982  // loop over all test basis functions
983  for (size_t ii = 0; ii < size; ++ii) {
984  // symmetry term
985  ret[ii] += -1.0 * dirichlet_value * ((diffusion_value * test_gradients[ii][0]) * normal);
986  // penalty term
987  ret[ii] += penalty * dirichlet_value * test_values[ii];
988  } // loop over all test basis functions
989  } // ... evaluate(...)
