246  const TimeFieldType scale_factor_min = 0.2, const TimeFieldType scale_factor_max = 5,
247  const MatrixType& A = ButcherArrayProviderType::A(),
248  const VectorType& b_1 = ButcherArrayProviderType::b_1(),
249  const VectorType& b_2 = ButcherArrayProviderType::b_2(),
250  const TimeVectorType& c = ButcherArrayProviderType::c())
251  : BaseType(t_0, initial_values)
252  , op_(op)
253  , r_(r)
254  , tol_(tol)
255  , scale_factor_min_(scale_factor_min)
256  , scale_factor_max_(scale_factor_max)
257  , u_tmp_(BaseType::current_solution())
258  , A_(A)

263  , num_stages_(A_.rows())
264  {
265  assert(DSC::FloatCmp::gt(tol_, 0.0));
266  assert(DSC::FloatCmp::le(scale_factor_min_, 1.0));
267  assert(DSC::FloatCmp::ge(scale_factor_max_, 1.0));
268  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
269  assert(b_1_.size() == A_.rows());
270  assert(b_2_.size() == A_.rows());
271  assert(c_.size() == A_.rows());
272 #ifndef NDEBUG
273  for (size_t ii = 0; ii < A_.rows(); ++ii) {
274  for (size_t jj = ii; jj < A_.cols(); ++jj) {
275  assert(DSC::FloatCmp::eq(A_[ii][jj], 0.0)
276  && "A has to be a lower triangular matrix with 0 on the main diagonal");
277  }
278  }
279 #endif // NDEBUG
280  // store as many discrete functions as needed for intermediate stages
281  for (size_t ii = 0; ii < num_stages_; ++ii) {
282  u_intermediate_stages_.emplace_back(current_solution());
283  }
284  } // constructor AdaptiveRungeKuttaTimeStepper

289  TimeFieldType step(const TimeFieldType dt, const TimeFieldType max_dt)
290  {
291  TimeFieldType actual_dt = std::min(dt, max_dt);
292  RangeFieldType mixed_error = std::numeric_limits<RangeFieldType>::max();
293  TimeFieldType time_step_scale_factor = 1.0;
294 
295  auto& t = current_time();
296  auto& u_n = current_solution();
297 
298  while (DSC::FloatCmp::gt(mixed_error, tol_)) {
299  actual_dt *= time_step_scale_factor;
300  size_t first_stage_to_compute = 0;
301  if (last_stage_of_previous_step_) {
302  u_intermediate_stages_[0].vector() = last_stage_of_previous_step_->vector();
303  first_stage_to_compute = 1;
304  }
305 
306  for (size_t ii = first_stage_to_compute; ii < num_stages_; ++ii) {
307  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
308  u_tmp_.vector() = u_n.vector();
309  for (size_t jj = 0; jj < ii; ++jj)
310  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
311  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
312  }
313 
314  // compute error vector
315  u_tmp_.vector() = u_intermediate_stages_[0].vector() * b_diff_[0];
316  for (size_t ii = 1; ii < num_stages_; ++ii)
317  u_tmp_.vector() += u_intermediate_stages_[ii].vector() * b_diff_[ii];
318  u_tmp_.vector() *= actual_dt * r_;
319 
320  // calculate u at timestep n+1
321  for (size_t ii = 0; ii < num_stages_; ++ii)
322  u_n.vector() += u_intermediate_stages_[ii].vector() * (actual_dt * r_ * b_1_[ii]);
323 
324  // scale error, use absolute error if norm is less than 0.01 and relative error else
325  auto& diff_vector = u_tmp_.vector();
326  for (size_t ii = 0; ii < diff_vector.size(); ++ii) {
327  if (std::abs(u_n.vector()[ii]) > 0.01)
328  diff_vector[ii] /= std::abs(u_n.vector()[ii]);
329  }
330  mixed_error = diff_vector.sup_norm();
331  // scale dt to get the estimated optimal time step length, TODO: adapt formula
332  time_step_scale_factor =
333  std::min(std::max(0.9 * std::pow(tol_ / mixed_error, 1.0 / 5.0), scale_factor_min_), scale_factor_max_);
334 
335  if (mixed_error > tol_) { // go back from u at timestep n+1 to timestep n
336  for (size_t ii = 0; ii < num_stages_; ++ii)
337  u_n.vector() += u_intermediate_stages_[ii].vector() * (-1.0 * r_ * actual_dt * b_1_[ii]);
338  }
339  } // while (mixed_error > tol_)
340 
341  if (!last_stage_of_previous_step_)
342  last_stage_of_previous_step_ = DSC::make_unique<DiscreteFunctionType>(u_n);
343  last_stage_of_previous_step_->vector() = u_intermediate_stages_[num_stages_ - 1].vector();
344 
345  t += actual_dt;
346 
347  return actual_dt * time_step_scale_factor;
348  } // ... step(...)
