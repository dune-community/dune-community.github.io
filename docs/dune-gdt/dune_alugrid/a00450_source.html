86 private:
87  template <class SourceType, class RangeType>
88  void apply_dg_fem(const SourceType& source, RangeType& range) const
89  {
90  // data structures we need
91  // * a map from a global vertex index to global DoF indices
92  // given a vertex, one obtains a set of all global DoF ids, which are associated with this vertex
93  std::map<size_t, std::set<size_t>> global_vertex_id_to_global_DoF_id_map;
94  // * a map from a global DoF index to the global index of its associated vertex
95  std::vector<size_t> global_DoF_id_to_global_vertex_id_map(source.space().mapper().size());
96  // * a set to hold the global id of all boundary vertices
97  std::set<size_t> boundary_vertices;
98 
99  const auto entity_it_end = grid_view_.template end<0>();
100  // walk the grid to create the maps explained above and to find the boundary vertices
101  for (auto entity_it = grid_view_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
102  const auto& entity = *entity_it;
103  const size_t num_vertices = boost::numeric_cast<size_t>(entity.template count<dimDomain>());
104  const auto basis = source.space().base_function_set(entity);
105  if (basis.size() != num_vertices)
106  DUNE_THROW(Dune::Stuff::Exceptions::internal_error, "basis.size() = " << basis.size());
107 
108  // loop over all vertices of the entitity, to find their associated global DoF indices
109  for (size_t local_vertex_id = 0; local_vertex_id < num_vertices; ++local_vertex_id) {
110  const auto vertex_ptr = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(local_vertex_id));
111  const auto global_vertex_id = grid_view_.indexSet().index(*vertex_ptr);
112  const auto vertex = vertex_ptr->geometry().center();
113  // find the local basis function which corresponds to this vertex
114  const auto basis_values = basis.evaluate(entity.geometry().local(vertex));
115  if (basis_values.size() != num_vertices)
116  DUNE_THROW(Dune::Stuff::Exceptions::internal_error, "basis_values.size() = " << basis_values.size());
117  size_t ones = 0;
118  size_t zeros = 0;
119  size_t failures = 0;
120  size_t local_DoF_index = 0;
121  for (size_t ii = 0; ii < basis.size(); ++ii) {
122  if (std::abs(basis_values[ii][0] - 1.0) < 1e-14) {
123  local_DoF_index = ii;
124  ++ones;
125  } else if (std::abs(basis_values[ii][0] - 0.0) < 1e-14)
126  ++zeros;
127  else
128  ++failures;
129  }
130  if (ones != 1 || zeros != (basis.size() - 1) || failures > 0) {
131  std::stringstream ss;
132  ss << "ones = " << ones << ", zeros = " << zeros << ", failures = " << failures
133  << ", num_vertices = " << num_vertices << ", entity " << grid_view_.indexSet().index(entity) << ", vertex "
134  << local_vertex_id << ": [ " << vertex << "], ";
135  Stuff::Common::print(basis_values, "basis_values", ss);
136  DUNE_THROW(Dune::Stuff::Exceptions::internal_error, ss.str());
137  }
138  // now we know that the local DoF index of this vertex is ii
139  const size_t global_DoF_index = source.space().mapper().mapToGlobal(entity, local_DoF_index);
140  global_DoF_id_to_global_vertex_id_map[global_DoF_index] = global_vertex_id;
141  global_vertex_id_to_global_DoF_id_map[global_vertex_id].insert(global_DoF_index);
142  } // loop over all vertices
143 
144  if (zero_boundary_) {
145  // in order to determine the boundary vertices, we need to
146  // loop over all intersections
147  const auto intersectionEndIt = grid_view_.iend(entity);
148  for (auto intersectionIt = grid_view_.ibegin(entity); intersectionIt != intersectionEndIt; ++intersectionIt) {
149  const auto& intersection = *intersectionIt;
150  if (intersection.boundary() && !intersection.neighbor()) {
151  const auto& intersection_geometry = intersection.geometry();
152  for (auto local_intersection_corner_id : DSC::valueRange(intersection_geometry.corners())) {
153  const auto global_intersection_corner = intersection_geometry.corner(local_intersection_corner_id);
154  // now, we need to find the entity's vertex this intersection's corner point equals to, so we
155  // loop over all vertices of the entity
156  for (size_t local_vertex_id = 0; local_vertex_id < num_vertices; ++local_vertex_id) {
157  const auto vertex_ptr = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(local_vertex_id));
158  const auto global_vertex_id = grid_view_.indexSet().index(*vertex_ptr);
159  const auto vertex = vertex_ptr->geometry().center();
160  if (Stuff::Common::FloatCmp::eq(global_intersection_corner, vertex))
161  boundary_vertices.insert(global_vertex_id);
162  } // loop over all vertices of the entity
163  } // loop over all intersection corners
164  } // if (intersection.boundary() && !intersection.neighbor())
165  } // loop over all intersections
166  } // if(zero_boundary)
167  } // walk the grid for the first time
168 
169  // walk the grid for the second time
170  for (auto entity_it = grid_view_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
171  const auto& entity = *entity_it;
172  const auto num_vertices = boost::numeric_cast<size_t>(entity.template count<dimDomain>());
173  // get the local functions
174  const auto local_source = source.local_discrete_function(entity);
175  const auto& local_source_DoF_vector = local_source->vector();
176 
177  // * loop over all local DoFs
178  for (size_t local_DoF_id = 0; local_DoF_id < num_vertices; ++local_DoF_id) {
179  const size_t global_DoF_index = source.space().mapper().mapToGlobal(entity, local_DoF_id);
180  const size_t global_vertex_id = global_DoF_id_to_global_vertex_id_map[global_DoF_index];
181  // if we are on the domain boundary
182  if (zero_boundary_ && boundary_vertices.count(global_vertex_id)) {
183  // set the dof to zero (we have dirichlet zero)
184  range.vector().set_entry(global_DoF_index, FieldType(0));
185  } else {
186  // do the oswald projection
187  const size_t num_DoFS_per_vertex = global_vertex_id_to_global_DoF_id_map[global_vertex_id].size();
188  // * get the source DoF
189  const FieldType source_DoF_value = local_source_DoF_vector.get(local_DoF_id);
190  // * and add it to all target DoFs
191  for (size_t target_global_DoF_id : global_vertex_id_to_global_DoF_id_map[global_vertex_id])
192  range.vector().add_to_entry(target_global_DoF_id, source_DoF_value / num_DoFS_per_vertex);
193  } // if (boundary_vertices.find(global_vertex_id))
194  } // loop over all local DoFs
195  } // walk the grid for the second time
196  } // ... apply_dg_fem(...)
