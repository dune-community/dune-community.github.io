79 template <class VectorType>
80 struct ChooseLimiter<SlopeLimiters::minmod, VectorType>
81 {
82  static VectorType limit(const VectorType& slope_left, const VectorType& slope_right,
83  const VectorType& /*centered_slope*/)
84  {
85  VectorType ret;
86  for (size_t ii = 0; ii < slope_left.size(); ++ii) {
87  const auto slope_left_abs = std::abs(slope_left[ii]);
88  const auto slope_right_abs = std::abs(slope_right[ii]);
89  if (slope_left_abs < slope_right_abs && slope_left[ii] * slope_right[ii] > 0)
90  ret[ii] = slope_left[ii];
91  else if (DSC::FloatCmp::ge(slope_left_abs, slope_right_abs) && slope_left[ii] * slope_right[ii] > 0)
92  ret[ii] = slope_right[ii];
93  else
94  ret[ii] = 0.0;
95  }
96  return ret;
97  }
98 };

163  template <class SourceType, class IntersectionType, class SpaceType, class VectorType>
164  void apply(const SourceType& source, const IntersectionType& intersection,
165  LocalDiscreteFunction<SpaceType, VectorType>& local_range_entity,
166  LocalDiscreteFunction<SpaceType, VectorType>& local_range_neighbor) const
167  {
168  const auto entity_ptr = intersection.inside();
169  const auto neighbor_ptr = intersection.outside();
170  const auto local_source_entity = source.local_function(*entity_ptr);
171  const auto local_source_neighbor = source.local_function(*neighbor_ptr);
172  const auto geometry_intersection = intersection.geometry();
173  const auto local_functions_tuple_entity = numerical_flux_.local_functions(*entity_ptr);
174  const auto local_functions_tuple_neighbor = numerical_flux_.local_functions(*neighbor_ptr);
175  const DSC::FieldVector<typename LocalDiscreteFunction<SpaceType, VectorType>::DomainFieldType,
176  LocalDiscreteFunction<SpaceType, VectorType>::dimRange> result =
177  numerical_flux_.evaluate(local_functions_tuple_entity,
178  local_functions_tuple_neighbor,
179  *local_source_entity,
180  *local_source_neighbor,
181  intersection,
182  geometry_intersection.local(geometry_intersection.center()));
183  local_range_entity.vector().add(result * (1.0 / entity_ptr->geometry().volume()));
184  local_range_neighbor.vector().add(result * (-1.0 / neighbor_ptr->geometry().volume()));
185  }
186 
187 private:
188  const NumericalFluxType numerical_flux_;
189 };

251 template <class MatrixImp, class BoundaryValueFunctionImp, SlopeLimiters slope_limiter>
252 class LocalReconstructionFvOperator
253  : public LocalOperatorInterface<internal::LocalReconstructionFvOperatorTraits<MatrixImp, BoundaryValueFunctionImp,
254  slope_limiter>>
255 {
256  typedef internal::LocalReconstructionFvOperatorTraits<MatrixImp, BoundaryValueFunctionImp, slope_limiter> Traits;
257  typedef typename Traits::RangeFieldType RangeFieldType;
258  static const size_t dimRange = Traits::dimRange;
259  typedef typename DSC::FieldVector<RangeFieldType, dimRange> StuffFieldVectorType;

273  template <class SourceType, class RangeSpaceType, class VectorType>
274  void apply(const SourceType& source, LocalDiscreteFunction<RangeSpaceType, VectorType>& local_range) const
275  {
276  const auto& entity = local_range.entity();
277  const auto& grid_view = local_range.space().grid_view();
278  // get reconstruction vector and mapper
279  auto& reconstruction_vector = local_range.vector();
280  const auto& reconstruction_mapper = local_range.space().mapper();
281  const auto entity_center = entity.geometry().center();
282  // walk over intersections to get values of discrete function on left and right neighbor entity
283  typename SourceType::RangeType u_left, u_right;
284  typename SourceType::RangeType u_entity =
285  source.local_discrete_function(entity)->evaluate(entity.geometry().local(entity_center));
286  const auto i_it_end = grid_view.iend(entity);
287  for (auto i_it = grid_view.ibegin(entity); i_it != i_it_end; ++i_it) {
288  const auto& intersection = *i_it;
289  if (intersection.neighbor()) {
290  const auto neighbor_ptr = intersection.outside();
291  const auto neighbor_center = neighbor_ptr->geometry().center();
292  const bool boundary = intersection.boundary();
293  if ((neighbor_center[0] < entity_center[0] && !boundary) || (neighbor_center[0] > entity_center[0] && boundary))
294  u_left =
295  source.local_discrete_function(*neighbor_ptr)->evaluate(neighbor_ptr->geometry().local(neighbor_center));
296  else
297  u_right =
298  source.local_discrete_function(*neighbor_ptr)->evaluate(neighbor_ptr->geometry().local(neighbor_center));
299  } else {
300  if (intersection.geometry().center()[0] < entity_center[0])
301  u_left = boundary_values_.local_function(entity)->evaluate(intersection.geometryInInside().center());
302  else
303  u_right = boundary_values_.local_function(entity)->evaluate(intersection.geometryInInside().center());
304  }
305  }
306 
307  // diagonalize the system of equations from u_t + A*u_x = 0 to w_t + D*w_x = 0 where D = R^(-1)*A*R, w = R^(-1)*u
308  // and R matrix of eigenvectors of A
309  const StuffFieldVectorType w_left(eigenvectors_inverse_ * u_left);
310  const StuffFieldVectorType w_right(eigenvectors_inverse_ * u_right);
311  const StuffFieldVectorType w_entity(eigenvectors_inverse_ * u_entity);
312 
313  const StuffFieldVectorType w_slope_left = w_entity - w_left;
314  const StuffFieldVectorType w_slope_right = w_right - w_entity;
315  const StuffFieldVectorType w_centered_slope = w_right * RangeFieldType(0.5) - w_left * RangeFieldType(0.5);
316  const StuffFieldVectorType w_slope = internal::ChooseLimiter<slope_limiter, StuffFieldVectorType>::limit(
317  w_slope_left, w_slope_right, w_centered_slope);
318  const StuffFieldVectorType half_w_slope = w_slope * RangeFieldType(0.5);
319  const StuffFieldVectorType w_reconstructed_left = w_entity - half_w_slope;
320  const StuffFieldVectorType w_reconstructed_right = w_entity + half_w_slope;
321 
322  // convert back to u variable
323  const StuffFieldVectorType u_reconstructed_left(eigenvectors_ * w_reconstructed_left);
324  const StuffFieldVectorType u_reconstructed_right(eigenvectors_ * w_reconstructed_right);
325 
326  for (size_t factor_index = 0; factor_index < dimRange; ++factor_index) {
327  // set values on dofs, dof with local index 0 for each factor space corresponds to basis function 1 - x, local
328  // index 1 to x
329  reconstruction_vector.set(reconstruction_mapper.mapToLocal(factor_index, entity, 0),
330  u_reconstructed_left[factor_index]);
331  reconstruction_vector.set(reconstruction_mapper.mapToLocal(factor_index, entity, 1),
332  u_reconstructed_right[factor_index]);
333  }
334  } // void apply(...)
335 
336 private:
337  const MatrixType& eigenvectors_;
338  const MatrixType& eigenvectors_inverse_;
339  const BoundaryValueFunctionImp& boundary_values_;
340 };
