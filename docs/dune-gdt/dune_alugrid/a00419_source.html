117  template <class E, class D, size_t d, class R, size_t rT, size_t rCT, size_t rA, size_t rCA>
118  void apply2(const Stuff::LocalfunctionSetInterface<E, D, d, R, rT, rCT>& test_base,
119  const Stuff::LocalfunctionSetInterface<E, D, d, R, rA, rCA>& ansatz_base,
120  Dune::DynamicMatrix<R>& ret) const
121  {
122  const auto& entity = ansatz_base.entity();
123  const auto local_functions = integrand_.localFunctions(entity);
124  // create quadrature
125  const size_t integrand_order = integrand_.order(local_functions, ansatz_base, test_base) + over_integrate_;
126  const auto& quadrature = QuadratureRules<D, d>::rule(entity.type(), boost::numeric_cast<int>(integrand_order));
127  // prepare storage
128  const size_t rows = test_base.size();
129  const size_t cols = ansatz_base.size();
130  ret *= 0.0;
131  assert(ret.rows() >= rows);
132  assert(ret.cols() >= cols);
133  Dune::DynamicMatrix<R> evaluation_result(rows, cols, 0.); // \todo: make mutable member, after SMP refactor
134  // loop over all quadrature points
135  for (const auto& quadrature_point : quadrature) {
136  const auto xx = quadrature_point.position();
137  // integration factors
138  const auto integration_factor = entity.geometry().integrationElement(xx);
139  const auto quadrature_weight = quadrature_point.weight();
140  // evaluate the integrand
141  integrand_.evaluate(local_functions, test_base, ansatz_base, xx, evaluation_result);
142  // compute integral
143  for (size_t ii = 0; ii < rows; ++ii) {
144  const auto& evaluation_result_row = evaluation_result[ii];
145  auto& ret_row = ret[ii];
146  for (size_t jj = 0; jj < cols; ++jj)
147  ret_row[jj] += evaluation_result_row[jj] * integration_factor * quadrature_weight;
148  } // compute integral
149  } // loop over all quadrature points
150  } // ... apply2(...)
151 
152 private:
153  const BinaryEvaluationType integrand_;
154  const size_t over_integrate_;
155 }; // class LocalVolumeIntegralOperator

189  void apply2(const Stuff::LocalfunctionSetInterface<E, D, d, R, rT, rCT>& test_base_en,
190  const Stuff::LocalfunctionSetInterface<E, D, d, R, rA, rCA>& ansatz_base_en,
191  const Stuff::LocalfunctionSetInterface<N, D, d, R, rT, rCT>& test_base_ne,
192  const Stuff::LocalfunctionSetInterface<N, D, d, R, rA, rCA>& ansatz_base_ne,
193  const IntersectionType& intersection, Dune::DynamicMatrix<R>& ret_en_en,
194  Dune::DynamicMatrix<R>& ret_ne_ne, Dune::DynamicMatrix<R>& ret_en_ne,
195  Dune::DynamicMatrix<R>& ret_ne_en) const
196  {
197  // local inducing function
198  const auto& entity = test_base_en.entity();
199  const auto& neighbor = test_base_ne.entity();
200  const auto local_functions_en = integrand_.localFunctions(entity);
201  const auto local_functions_ne = integrand_.localFunctions(neighbor);
202  // quadrature
203  const size_t integrand_order =
204  integrand_.order(
205  local_functions_en, local_functions_ne, test_base_en, ansatz_base_en, test_base_ne, ansatz_base_ne)
206  + over_integrate_;
207  const auto& quadrature =
208  QuadratureRules<D, d - 1>::rule(intersection.type(), boost::numeric_cast<int>(integrand_order));
209  // check matrices
210  ret_en_en *= 0.0;
211  ret_ne_ne *= 0.0;
212  ret_en_ne *= 0.0;
213  ret_ne_en *= 0.0;
214  const size_t rows_en = test_base_en.size();
215  const size_t cols_en = ansatz_base_en.size();
216  const size_t rows_ne = test_base_ne.size();
217  const size_t cols_ne = ansatz_base_ne.size();
218  assert(ret_en_en.rows() >= rows_en);
219  assert(ret_en_en.cols() >= cols_en);
220  assert(ret_ne_ne.rows() >= rows_ne);
221  assert(ret_ne_ne.cols() >= cols_ne);
222  assert(ret_en_ne.rows() >= rows_en);
223  assert(ret_en_ne.cols() >= cols_ne);
224  assert(ret_ne_en.rows() >= rows_en);
225  assert(ret_ne_en.cols() >= cols_en);
226  Dune::DynamicMatrix<R> evaluation_result_en_en(
227  rows_en, cols_en, 0.); // \todo: make mutable member, after SMP refactor
228  Dune::DynamicMatrix<R> evaluation_result_ne_ne(
229  rows_ne, cols_ne, 0.); // \todo: make mutable member, after SMP refactor
230  Dune::DynamicMatrix<R> evaluation_result_en_ne(
231  rows_en, cols_ne, 0.); // \todo: make mutable member, after SMP refactor
232  Dune::DynamicMatrix<R> evaluation_result_ne_en(
233  rows_ne, cols_en, 0.); // \todo: make mutable member, after SMP refactor
234  // loop over all quadrature points
235  for (const auto& quadrature_point : quadrature) {
236  const auto xx = quadrature_point.position();
237  const auto integration_factor = intersection.geometry().integrationElement(xx);
238  const auto quadrature_weight = quadrature_point.weight();
239  // evaluate local
240  integrand_.evaluate(local_functions_en,
241  local_functions_ne,
242  test_base_en,
243  ansatz_base_en,
244  test_base_ne,
245  ansatz_base_ne,
246  intersection,
247  xx,
248  evaluation_result_en_en,
249  evaluation_result_ne_ne,
250  evaluation_result_en_ne,
251  evaluation_result_ne_en);
252  // compute integrals
253  // loop over all entity test basis functions
254  for (size_t ii = 0; ii < rows_en; ++ii) {
255  auto& ret_en_en_row = ret_en_en[ii];
256  auto& ret_en_ne_row = ret_en_ne[ii];
257  const auto& evaluation_result_en_en_row = evaluation_result_en_en[ii];
258  const auto& evaluation_result_en_ne_row = evaluation_result_en_ne[ii];
259  // loop over all entity ansatz basis functions
260  for (size_t jj = 0; jj < cols_en; ++jj) {
261  ret_en_en_row[jj] += evaluation_result_en_en_row[jj] * integration_factor * quadrature_weight;
262  }
263  // loop over all neighbor ansatz basis functions
264  for (size_t jj = 0; jj < cols_ne; ++jj) {
265  ret_en_ne_row[jj] += evaluation_result_en_ne_row[jj] * integration_factor * quadrature_weight;
266  }
267  }
268  // loop over all neighbor test basis functions
269  for (size_t ii = 0; ii < rows_ne; ++ii) {
270  auto& ret_ne_ne_row = ret_ne_ne[ii];
271  auto& ret_ne_en_row = ret_ne_en[ii];
272  const auto& evaluation_result_ne_ne_row = evaluation_result_ne_ne[ii];
273  const auto& evaluation_result_ne_en_row = evaluation_result_ne_en[ii];
274  // loop over all neighbor ansatz basis functions
275  for (size_t jj = 0; jj < cols_ne; ++jj) {
276  ret_ne_ne_row[jj] += evaluation_result_ne_ne_row[jj] * integration_factor * quadrature_weight;
277  }
278  // loop over all entity ansatz basis functions
279  for (size_t jj = 0; jj < cols_en; ++jj) {
280  ret_ne_en_row[jj] += evaluation_result_ne_en_row[jj] * integration_factor * quadrature_weight;
281  }
282  }
283  } // loop over all quadrature points
