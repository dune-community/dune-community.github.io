291  template <class G, class S, size_t d, size_t r, size_t rC>
292  PatternType compute_face_and_volume_pattern(const GridView<G>& local_grid_view,
293  const SpaceInterface<S, d, r, rC>& ansatz_space) const
294  {
295  // prepare
296  PatternType pattern(mapper().size());
297  Dune::DynamicVector<size_t> global_rows(mapper().maxNumDofs(), 0);
298  Dune::DynamicVector<size_t> global_cols(ansatz_space.mapper().maxNumDofs(), 0);
299  for (const auto& entity : DSC::entityRange(local_grid_view)) {
300  const auto test_base_entity = base_function_set(entity);
301  const auto ansatz_base_entity = ansatz_space.base_function_set(entity);
302  mapper().globalIndices(entity, global_rows);
303  ansatz_space.mapper().globalIndices(entity, global_cols);
304  // compute entity/entity
305  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
306  for (size_t jj = 0; jj < ansatz_base_entity.size(); ++jj) {
307  pattern.insert(global_rows[ii], global_cols[jj]);
308  }
309  }
310  // walk the intersections
311  const auto intersection_it_end = local_grid_view.iend(entity);
312  for (auto intersection_it = local_grid_view.ibegin(entity); intersection_it != intersection_it_end;
313  ++intersection_it) {
314  const auto& intersection = *intersection_it;
315  // get the neighbour
316  if (intersection.neighbor() && !intersection.boundary()) {
317  const auto neighbour_ptr = intersection.outside();
318  const auto& neighbour = *neighbour_ptr;
319  // get the basis
320  const auto ansatz_base_neighbour = ansatz_space.base_function_set(neighbour);
321  ansatz_space.mapper().globalIndices(neighbour, global_cols);
322  // compute entity/neighbour
323  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
324  for (size_t jj = 0; jj < ansatz_base_neighbour.size(); ++jj) {
325  pattern.insert(global_rows[ii], global_cols[jj]);
326  }
327  }
328  } // get the neighbour
329  } // walk the intersections
330  } // walk the grid view
331  pattern.sort();
332  return pattern;
