45 template <class TestSpaceType, class AnsatzSpaceType, class GridViewType>
46 class ConstraintsWrapper<TestSpaceType, AnsatzSpaceType, GridViewType,
47  DirichletConstraints<typename GridViewType::Intersection>>
48  : public Stuff::Grid::internal::Codim0Object<GridViewType>
49 {
50  static_assert(is_space<TestSpaceType>::value, "TestSpaceType has to be derived from SpaceInterface!");
51  static_assert(is_space<AnsatzSpaceType>::value, "AnsatzSpaceType has to be derived from SpaceInterface!");
52  typedef Stuff::Grid::internal::Codim0Object<GridViewType> BaseType;
53  typedef DirichletConstraints<typename GridViewType::Intersection> ConstraintsType;
54 
55 public:
56  using typename BaseType::EntityType;
57 
58  ConstraintsWrapper(const DS::PerThreadValue<const TestSpaceType>& test_space,
59  const DS::PerThreadValue<const AnsatzSpaceType>& ansatz_space,
60  const Stuff::Grid::ApplyOn::WhichEntity<GridViewType>* where, ConstraintsType& constraints)
61  : test_space_(test_space)
62  , ansatz_space_(ansatz_space)
63  , where_(where)
64  , constraints_(constraints)
65  , thread_local_constraints_(constraints_.boundary_info(), constraints_.size())
66  {
67  }
68 
69  virtual ~ConstraintsWrapper() = default;
70 
71  virtual bool apply_on(const GridViewType& gv, const EntityType& entity) const override final
72  {
73  return where_->apply_on(gv, entity);
74  }
75 
76  virtual void apply_local(const EntityType& entity) override final
77  {
78  test_space_->local_constraints(*ansatz_space_, entity, *thread_local_constraints_);
79  }
80 
81  virtual void finalize() override final
82  {
83  std::lock_guard<std::mutex> DUNE_UNUSED(mutex_guard)(constraints_.mutex_);
84  constraints_.dirichlet_DoFs_.insert(thread_local_constraints_->dirichlet_DoFs_.begin(),
85  thread_local_constraints_->dirichlet_DoFs_.end());
86  }
87 
88 private:
89  const DS::PerThreadValue<const TestSpaceType>& test_space_;
90  const DS::PerThreadValue<const AnsatzSpaceType>& ansatz_space_;
91  const std::unique_ptr<const Stuff::Grid::ApplyOn::WhichEntity<GridViewType>> where_;
92  ConstraintsType& constraints_;
93  DS::PerThreadValue<ConstraintsType> thread_local_constraints_;
94 }; // class ConstraintsWrapper
