57  std::vector<size_t> local_DoF_indices_2dsimplex_order0(const EntityType& entity) const
58  {
59  static_assert(dimDomain == 2, "Not implemented!");
60  static_assert(polOrder == 0, "Not implemented!");
61  // prepare
62  const auto num_intersections = boost::numeric_cast<size_t>(entity.template count<1>());
63  std::vector<size_t> local_DoF_index_of_vertex(num_intersections, std::numeric_limits<size_t>::infinity());
64  std::vector<size_t> local_DoF_index_of_intersection(num_intersections, std::numeric_limits<size_t>::infinity());
65  typedef typename BaseFunctionSetType::DomainType DomainType;
66  std::vector<DomainType> vertices(num_intersections, DomainType(0));
67  std::vector<bool> lies_on_intersection(num_intersections, false);
68  DomainType corner(0);
69  typedef typename BaseFunctionSetType::RangeType RangeType;
70  const RangeType one(1);
71  std::vector<RangeType> basis_values(num_intersections, one);
72  const auto basis = this->base_function_set(entity);
73  assert(basis.size() == num_intersections);
74  const auto geometry = entity.geometry();
75  const auto& reference_element = ReferenceElements<DomainFieldType, dimDomain>::general(geometry.type());
76  // find the basis function index that corresponds to each vertex of the entity
77  // (find the basis function that evaluates to zero at the vertex, and nonzero at the other ones)
78  // therefore we walk the vertices
79  assert(num_intersections == boost::numeric_cast<size_t>(entity.template count<dimDomain>()));
80  for (size_t vv = 0; vv < num_intersections; ++vv) {
81  const auto vertex_ptr = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(vv));
82  const auto& vertex = *vertex_ptr;
83  // get the vertex coordinates
84  vertices[vv] = vertex.geometry().center();
85  const auto& vertex_entity = reference_element.position(boost::numeric_cast<int>(vv), dimDomain);
86  // evalaute the basis
87  basis.evaluate(vertex_entity, basis_values);
88  // and find the basis that evaluates zero here
89  size_t zeros = 0;
90  size_t nonzeros = 0;
91  for (size_t ii = 0; ii < num_intersections; ++ii) {
92  // we would like to check against 0, but there is a bug in dune-commons FloatCmp
93  if (Stuff::Common::FloatCmp::eq(basis_values[ii] + one, one)) {
94  // this is a candidate for the basis function we are looking for
95  local_DoF_index_of_vertex[vv] = ii;
96  ++zeros;
97  } else
98  ++nonzeros;
99  }
100  // make sure there was only one candidate
101  if (zeros != 1 || nonzeros != (num_intersections - 1))
102  DUNE_THROW(Stuff::Exceptions::internal_error,
103  "This must not happen for RTN0 in 2d!\n"
104  << " zeros = "
105  << zeros
106  << "\n"
107  << " nonzeros = "
108  << nonzeros);
109  } // walk the vertices
110  // so from here on we have the local DoF index that corresponds to each vertex vv in local_DoF_index_of_vertex[vv]
111  // now we need to find the intersection opposite to this vertex
112  // therefore we walk the intersections
113  size_t intersection_counter = 0;
114  const auto intersection_it_end = this->grid_view().iend(entity);
115  for (auto intersection_it = this->grid_view().ibegin(entity); intersection_it != intersection_it_end;
116  ++intersection_it) {
117  const auto& intersection = *intersection_it;
118  const auto intersection_geometry = intersection.geometry();
119  const size_t local_intersection_index = intersection.indexInInside();
120  // make sure this index has not been already taken by another intersection
121  assert(local_DoF_index_of_intersection[local_intersection_index] == std::numeric_limits<size_t>::infinity());
122  // walk the corners of the intersection
123  for (size_t cc = 0; cc < intersection_geometry.corners(); ++cc) {
124  corner = intersection_geometry.corner(boost::numeric_cast<int>(cc));
125  // check which vertices lie on the intersection
126  for (size_t vv = 0; vv < num_intersections; ++vv)
127  if (Stuff::Common::FloatCmp::eq(vertices[vv], corner))
128  lies_on_intersection[vv] = true;
129  } // walk the corners of the intersection
130  // now see if we find a vertex that does not lie on the intersection
131  size_t found = 0;
132  size_t missed = 0;
133  for (size_t vv = 0; vv < num_intersections; ++vv) {
134  if (!(lies_on_intersection[vv])) {
135  // this is a good candidate
136  // so the local DoF id that corresponds to this vertex is the one that corresponds to the intersection
137  local_DoF_index_of_intersection[local_intersection_index] = local_DoF_index_of_vertex[vv];
138  ++found;
139  } else
140  ++missed;
141  // and clear for the next intersection
142  lies_on_intersection[vv] = false;
143  } // walk the vertices of this entity
144  // make sure there was only one candidate
145  if (found != 1 || missed != (num_intersections - 1))
146  DUNE_THROW(Stuff::Exceptions::internal_error,
147  "This must not happen for RTN0 in 2d!\n"
148  << " found = "
149  << found
150  << "\n"
151  << " missed = "
152  << missed);
153  ++intersection_counter;
154  } // walk the intersection
155  assert(intersection_counter == num_intersections);
156  return local_DoF_index_of_intersection;
157  } // ... local_DoF_indices_2dsimplex_order0(...)
