202  const MatrixType& A = ButcherArrayProviderType::A(),

216  assert(c_.size() == A_.rows());

223  }
224 #endif // NDEBUG

240  virtual TimeFieldType step(const TimeFieldType dt, const TimeFieldType max_dt) override final
241  {
242  const TimeFieldType actual_dt = std::min(dt, max_dt);
243  auto& t = current_time();
244  auto& u_n = current_solution();
245  // calculate stages
246  for (size_t ii = 0; ii < num_stages_; ++ii) {
247  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
248  u_tmp_.vector() = u_n.vector();
249  for (size_t jj = 0; jj < ii; ++jj)
250  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
251  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
252  }
253 
254  // calculate value of u at next time step
255  for (size_t ii = 0; ii < num_stages_; ++ii)
256  u_n.vector() += u_intermediate_stages_[ii].vector() * (r_ * actual_dt * b_[ii]);
257 
258  // augment time
259  t += actual_dt;
260 
261  return dt;
262  } // ... step(...)

264  const std::pair<bool, TimeFieldType>
265  find_suitable_dt(const TimeFieldType initial_dt, const TimeFieldType dt_refinement_factor = 2,
266  const RangeFieldType treshold = 0.9 * std::numeric_limits<RangeFieldType>::max(),
267  const size_t max_steps_per_dt = 20, const size_t max_refinements = 20)
268  {
269  auto& t = current_time();
270  auto& u_n = current_solution();
271  assert(treshold > 0);
272  // save current state
273  DiscreteFunctionType initial_u_n = u_n;
274  TimeFieldType initial_t = t;
275  // start with initial dt
276  TimeFieldType current_dt = initial_dt;
277  size_t num_refinements = 0;
278  while (num_refinements < max_refinements) {
279  std::cout << "Trying time step length dt = " << current_dt << "... " << std::flush;
280  bool unlikely_value_occured = false;
281  size_t num_steps = 0;
282  // do max_steps_per_dt time steps...
283  while (!unlikely_value_occured) {
284  step(current_dt);
285  ++num_steps;
286  // ... unless there is a value above threshold
287  for (size_t kk = 0; kk < u_n.vector().size(); ++kk) {
288  if (std::abs(u_n.vector()[kk]) > treshold) {
289  unlikely_value_occured = true;
290  std::cout << "failed" << std::endl;
291  break;
292  }
293  }
294  // if we are able to do max_steps_per_dt time steps with this dt, we accept this dt
295  if (num_steps == max_steps_per_dt) {
296  std::cout << "looks fine" << std::endl;
297  u_n.vector() = initial_u_n.vector();
298  t = initial_t;
299  return std::make_pair(bool(true), current_dt);
300  }
301  }
302  // if there was a value above threshold start over with smaller dt
303  u_n.vector() = initial_u_n.vector();
304  t = initial_t;
305  current_dt /= dt_refinement_factor;
306  ++num_refinements;
307  }
308  return std::make_pair(bool(false), current_dt);
309  }
