88  static void globalIndices(const MsGridType& ms_grid, const std::vector<std::shared_ptr<const L>>& local_spaces,
89  const std::vector<size_t>& global_start_indices, const Comdim0EntityType& entity,
90  Dune::DynamicVector<size_t>& ret)
91  {
92  const size_t block = find_block_of_(ms_grid, entity);
93  local_spaces[block]->mapper().globalIndices(entity, ret);
94  const size_t num_dofs = local_spaces[block]->mapper().numDofs(entity);
95  assert(ret.size() >= num_dofs);
96  for (size_t ii = 0; ii < num_dofs; ++ii)
97  ret[ii] += global_start_indices[block];
98  }

100  static size_t mapToGlobal(const MsGridType& ms_grid, const std::vector<std::shared_ptr<const L>>& local_spaces,
101  const std::vector<size_t>& global_start_indices, const Comdim0EntityType& entity,
102  const size_t& localIndex)
103  {
104  const size_t block = find_block_of_(ms_grid, entity);
105  const size_t block_local_index = local_spaces[block]->mapper().mapToGlobal(entity, localIndex);
106  return global_start_indices[block] + block_local_index;
107  }

109  private:
110  static size_t find_block_of_(const MsGridType& ms_grid, const Comdim0EntityType& entity)
111  {
112  const auto global_entity_index = ms_grid.globalGridView().indexSet().index(entity);
113  const auto result = ms_grid.entityToSubdomainMap()->find(global_entity_index);
114 #ifndef NDEBUG
115  if (result == ms_grid.entityToSubdomainMap()->end())
116  DUNE_THROW(Stuff::Exceptions::internal_error,
117  "Entity " << global_entity_index
118  << " of the global grid view was not found in the multiscale grid!");
119 #endif // NDEBUG
120  const size_t subdomain = result->second;
121 #ifndef NDEBUG
122  if (subdomain >= ms_grid.size())
123  DUNE_THROW(Stuff::Exceptions::internal_error,
124  "The multiscale grid is corrupted!\nIt reports Entity " << global_entity_index
125  << " to be in subdomain "
126  << subdomain
127  << " while only having "
128  << ms_grid.size()
129  << " subdomains!");
130 #endif // NDEBUG
131  return subdomain;
132  } // ... find_block_of_(...)

135 public:
136  BlockMapper(const std::shared_ptr<const MsGridType> ms_grid,

145  DUNE_THROW(Stuff::Exceptions::shapes_do_not_match,
146  "You have to provide a local space for each subdomain of the multiscale grid!\n"
147  << " Size of the given multiscale grid: "
148  << ms_grid_->size()
149  << "\n"
150  << " Number of local spaces given: "
151  << local_spaces_.size());
152  for (size_t bb = 0; bb < num_blocks_; ++bb) {
153  max_num_dofs_ = std::max(max_num_dofs_, local_spaces_[bb]->mapper().maxNumDofs());
