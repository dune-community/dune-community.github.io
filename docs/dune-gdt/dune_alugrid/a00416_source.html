68 template <class GridViewImp, size_t rangeDim>
69 class FvMapper<GridViewImp, rangeDim, 1> : public MapperInterface<internal::FvMapperTraits<GridViewImp, rangeDim, 1>>
70 {
71  typedef MapperInterface<internal::FvMapperTraits<GridViewImp, rangeDim, 1>> InterfaceType;
72  static const size_t dimRange = rangeDim;
73 
74 public:
75  typedef internal::FvMapperTraits<GridViewImp, rangeDim, 1> Traits;
76  typedef typename Traits::GridViewType GridViewType;
77  typedef typename Traits::BackendType BackendType;
78  typedef typename Traits::EntityType EntityType;
79 
80  FvMapper(const GridViewType& grid_view)
81  : backend_(grid_view.indexSet())
82  {
83  }
84 
85  const BackendType& backend() const
86  {
87  return backend_;
88  }
89 
90  size_t size() const
91  {
92  return dimRange * backend_.size(0);
93  }
94 
95  size_t numDofs(const EntityType& /*entity*/) const
96  {
97  return dimRange;
98  }
99 
100  size_t maxNumDofs() const
101  {
102  return dimRange;
103  }
104 
105  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
106  {
107  if (ret.size() < dimRange)
108  ret.resize(dimRange);
109  const size_t base = dimRange * backend_.index(entity);
110  for (size_t ii = 0; ii < dimRange; ++ii)
111  ret[ii] = base + ii;
112  } // ... globalIndices(...)
113 
114  using InterfaceType::globalIndices;
115 
116  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
117  {
118  assert(localIndex < dimRange);
119  return (dimRange * backend_.index(entity)) + localIndex;
120  }
121 
122 private:
123  const BackendType& backend_;
124 }; // class FvMapper< ..., rangeDim, 1 >
