60 template <size_t ii>
61 struct visualize_helper<ii, true>
62 {
63  template <class DiscreteFunctionType>
64  static void visualize_factor(const std::string filename_prefix, const std::string filename_suffix,
65  const bool subsampling, const VTK::OutputType vtk_output_type,
66  const DiscreteFunctionType& discrete_function)
67  {
68  static_assert(ii < DiscreteFunctionType::SpaceType::num_factors, "This factor does not exist.");
69  const auto& space = discrete_function.space();
70  const auto& factor_space = space.template factor<ii>();
71  typename DiscreteFunctionType::VectorType factor_vector(factor_space.mapper().size());
72  const auto it_end = space.grid_view().template end<0>();
73  for (auto it = space.grid_view().template begin<0>(); it != it_end; ++it) {
74  const auto& entity = *it;
75  for (size_t jj = 0; jj < factor_space.mapper().numDofs(entity); ++jj)
76  factor_vector[factor_space.mapper().mapToGlobal(entity, jj)] =
77  discrete_function.vector()[space.mapper().mapToGlobal(ii, entity, jj)];
78  }
79  ConstDiscreteFunction<
80  typename std::tuple_element<ii, typename DiscreteFunctionType::SpaceType::SpaceTupleType>::type,
81  typename DiscreteFunctionType::VectorType> factor_discrete_function(factor_space, factor_vector);
82  factor_discrete_function.visualize(
83  filename_prefix + "_factor_" + DSC::to_string(ii) + "_" + filename_suffix, subsampling, vtk_output_type);
84  }
85 };
