51 template <class PdelabSpaceImp, size_t rangeDim>
52 class ProductDgMapper<PdelabSpaceImp, rangeDim, 1>
53  : public ProductMapperInterface<internal::ProductDgMapperTraits<PdelabSpaceImp, rangeDim, 1>>
54 {

90  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
91  {
92  if (ret.size() < numDofs(entity))
93  ret.resize(numDofs(entity));
94  const auto factor_num_dofs = factor_mapper_.numDofs(entity);
95  const auto factor_mapper_size = factor_mapper_.size();
96  const auto factor_global_indices = factor_mapper_.globalIndices(entity);
97  for (size_t ii = 0; ii < dimRange; ++ii) {
98  const auto factor_num_dofs_times_ii = factor_num_dofs * ii;
99  const auto factor_mapper_size_times_ii = factor_mapper_size * ii;
100  for (size_t jj = 0; jj < factor_num_dofs; ++jj) {
101  ret[factor_num_dofs_times_ii + jj] = factor_global_indices[jj] + factor_mapper_size_times_ii;
102  }
103  }
104  } // ... globalIndices(...)
105 
106  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
107  {
108  assert(localIndex < numDofs(entity));
109  size_t factor_index = 0;
110  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
111  while (localIndex >= factor_mapper_num_dofs) {
112  localIndex -= factor_mapper_num_dofs;
113  ++factor_index;
114  }
115  return factor_mapper_.mapToGlobal(entity, localIndex) + factor_index * factor_mapper_.size();
116  }
117 
118  using InterfaceType::globalIndices;

130  void globalIndices(const size_t factor_index, const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
131  {
132  assert(factor_index < dimRange);
133  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
134  if (ret.size() < factor_mapper_num_dofs)
135  ret.resize(factor_mapper_num_dofs);
136  const auto factor_mapper_size_times_factor_index = factor_mapper_.size() * factor_index;
137  const auto factor_mapper_global_indices = factor_mapper_.globalIndices(entity);
138  for (size_t jj = 0; jj < factor_mapper_num_dofs; ++jj)
139  ret[jj] = factor_mapper_global_indices[jj] + factor_mapper_size_times_factor_index;
140  } // ... globalIndices(...)

142  size_t mapToGlobal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
143  {
144  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
145  assert(factor_index < dimRange);
146  return factor_mapper_.mapToGlobal(entity, local_index_in_factor) + factor_index * factor_mapper_.size();
147  }
148 
149  size_t mapToLocal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
150  {
151  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
152  assert(factor_index < dimRange);
153  return factor_mapper_.numDofs(entity) * factor_index + local_index_in_factor;
154  }
