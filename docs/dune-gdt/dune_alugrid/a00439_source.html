69  typename std::enable_if<StaticCheck<E, D, d, R, r, rC, RangeSpaceType, VectorType>::value

71  void>::type
72  apply(const Stuff::LocalizableFunctionInterface<E, D, d, R, r, rC>& source,
73  LocalDiscreteFunction<RangeSpaceType, VectorType>& local_range) const
74  {
75  // create local L2 operator
76  typedef Stuff::Functions::Constant<E, D, d, R, 1> OneType;
77  const OneType one(1.); // <- is not actually used, just needed for the product evaluation
78  const LocalVolumeIntegralOperator<LocalProductIntegrand<OneType>> local_l2_operator(over_integrate_, one);
79  // and functional
80  typedef Stuff::LocalizableFunctionInterface<E, D, d, R, r, rC> SourceType;
81  const LocalVolumeIntegralFunctional<LocalProductIntegrand<SourceType>> local_l2_functional(over_integrate_, source);
82  // create local lhs and rhs
83  const auto& local_basis = local_range.basis();
84  const size_t size = local_basis.size();
85  Stuff::LA::CommonDenseMatrix<R> local_matrix(size, size);
86  Stuff::LA::CommonDenseVector<R> local_vector(size);
87  // assemble
88  local_l2_operator.apply2(local_basis, local_basis, local_matrix.backend());
89  local_l2_functional.apply(local_basis, local_vector.backend());
90  // solve
91  Stuff::LA::CommonDenseVector<R> local_solution(size);
92  try {
93  Stuff::LA::Solver<Stuff::LA::CommonDenseMatrix<R>>(local_matrix).apply(local_vector, local_solution);
94  } catch (Stuff::Exceptions::linear_solver_failed& ee) {
95  DUNE_THROW(projection_error,
96  "L2 projection failed because a local matrix could not be inverted!\n\n"
97  << "This was the original error: "
98  << ee.what());
99  }
100  // set local DoFs
101  auto& local_range_vector = local_range.vector();
102  assert(local_range_vector.size() == local_solution.size());
103  for (size_t ii = 0; ii < local_range_vector.size(); ++ii)
104  local_range_vector.set(ii, local_solution[ii]);
105  } // ... apply(...)

118  Stuff::LA::CommonDenseVector<R> local_vector(local_range.basis().size());

120  local_l2_functional.apply(local_range.basis(), local_vector.backend());
121  local_vector /= entity.geometry().volume();
122  auto& local_range_vector = local_range.vector();
123  for (size_t ii = 0; ii < local_range_vector.size(); ++ii)
124  local_range_vector.set(ii, local_vector[ii]);
125  } // ... apply(...) for FV spaces
