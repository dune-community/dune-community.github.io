238  template <class R, size_t r>
239  void evaluate(const Stuff::LocalfunctionInterface<E, D, d, R, 1, 1>& local_diffusion_factor,
240  const Stuff::LocalfunctionInterface<E, D, d, R, d, d>& local_diffusion_tensor,
241  const Stuff::LocalfunctionSetInterface<E, D, d, R, r, 1>& test_base,
242  const Stuff::LocalfunctionSetInterface<E, D, d, R, r, 1>& ansatz_base,
243  const Dune::FieldVector<D, d>& localPoint, Dune::DynamicMatrix<R>& ret) const
244  {
245  typedef Stuff::Common::FieldMatrix<R, d, d> TensorType;
246  ret *= 0.0;
247  // evaluate local functions
248  const auto diffusion_factor_value = local_diffusion_factor.evaluate(localPoint);
249  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(localPoint);
250  const auto diffusion_value = diffusion_tensor_value * diffusion_factor_value;
251  // evaluate bases
252  const auto testGradients = test_base.jacobian(localPoint);
253  const auto ansatzGradients = ansatz_base.jacobian(localPoint);
254  // compute elliptic evaluation
255  const size_t rows = test_base.size();
256  const size_t cols = ansatz_base.size();
257  assert(ret.rows() >= rows);
258  assert(ret.cols() >= cols);
259  for (size_t ii = 0; ii < rows; ++ii) {
260  auto& retRow = ret[ii];
261  for (size_t jj = 0; jj < cols; ++jj) {
262  for (size_t rr = 0; rr < r; ++rr) {
263  retRow[jj] += (diffusion_value * ansatzGradients[jj][rr]) * testGradients[ii][rr];
264  }
265  }
266  }
267  } // ... evaluate(...)
