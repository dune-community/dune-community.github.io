124  template <class T, class S, class V>
125  void redirect_apply(
126  const CgSpaceInterface<T, dimDomain, dimDomain, 1>& /*space*/,
127  const Stuff::LocalizableFunctionInterface<EntityType, DomainFieldType, dimDomain, FieldType, 1, 1>& source,
128  DiscreteFunction<S, V>& range) const
129  {
130  typedef typename Stuff::LA::Container<FieldType, V::sparse_matrix_type>::MatrixType MatrixType;
131  MatrixType lhs(
132  range.space().mapper().size(), range.space().mapper().size(), range.space().compute_volume_pattern());
133  V rhs(range.space().mapper().size());
134 
135  // walk the grid
136  const auto entity_it_end = grid_view_.template end<0>();
137  for (auto entity_it = grid_view_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
138  const auto& entity = *entity_it;
139  const auto local_function = function_.local_function(entity);
140  const auto local_source = source.local_function(entity);
141  const auto basis = range.space().base_function_set(entity);
142  // do a volume quadrature
143  const size_t integrand_order =
144  std::max(local_function->order() + ssize_t(local_source->order()) - 1, basis.order()) + basis.order();
145  const auto& quadrature =
146  QuadratureRules<DomainFieldType, dimDomain>::rule(entity.type(), boost::numeric_cast<int>(integrand_order));
147  const auto quadrature_it_end = quadrature.end();
148  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
149  const auto xx = quadrature_it->position();
150  const auto quadrature_weight = quadrature_it->weight();
151  const auto integration_element = entity.geometry().integrationElement(xx);
152  const ValueType function_value = local_function->evaluate(xx);
153  const auto source_gradient = local_source->jacobian(xx);
154  const auto basis_value = basis.evaluate(xx);
155  for (size_t ii = 0; ii < basis.size(); ++ii) {
156  const size_t global_ii = range.space().mapper().mapToGlobal(entity, ii);
157  rhs.add_to_entry(global_ii,
158  integration_element * quadrature_weight * -1.0
159  * ((function_value * source_gradient[0]) * basis_value[ii]));
160  for (size_t jj = 0; jj < basis.size(); ++jj) {
161  const size_t global_jj = range.space().mapper().mapToGlobal(entity, jj);
162  lhs.add_to_entry(
163  global_ii, global_jj, integration_element * quadrature_weight * (basis_value[ii] * basis_value[jj]));
164  }
165  }
166  } // do a volume quadrature
167  } // walk the grid
168 
169  // solve
170  try {
171  Stuff::LA::Solver<MatrixType>(lhs).apply(rhs, range.vector());
172  } catch (Stuff::Exceptions::linear_solver_failed& ee) {
173  DUNE_THROW(operator_error,
174  "Application of the Darcy operator failed because a matrix could not be inverted!\n\n"
175  << "This was the original error: "
176  << ee.what());
177  }
178  } // ... redirect_apply(...)

180  template <class T, class S, class V>
181  void redirect_apply(
182  const RtSpaceInterface<T, dimDomain, dimDomain, 1>& /*space*/,
183  const Stuff::LocalizableFunctionInterface<EntityType, DomainFieldType, dimDomain, FieldType, 1>& source,
184  DiscreteFunction<S, V>& range) const
185  {
186  static_assert(RtSpaceInterface<T, dimDomain, 1>::polOrder == 0, "Untested!");
187  const auto& rtn0_space = range.space();
188  auto& range_vector = range.vector();
189  const auto infinity = std::numeric_limits<FieldType>::infinity();
190  for (size_t ii = 0; ii < range_vector.size(); ++ii)
191  range_vector[ii] = infinity;
192  // walk the grid
193  const auto entity_it_end = grid_view_.template end<0>();
194  for (auto entity_it = grid_view_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
195  const auto& entity = *entity_it;
196  const auto local_DoF_indices = rtn0_space.local_DoF_indices(entity);
197  const auto global_DoF_indices = rtn0_space.mapper().globalIndices(entity);
198  assert(global_DoF_indices.size() == local_DoF_indices.size());
199  const auto local_function = function_.local_function(entity);
200  const auto local_source = source.local_function(entity);
201  const auto local_basis = rtn0_space.base_function_set(entity);
202  // walk the intersections
203  const auto intersection_it_end = grid_view_.iend(entity);
204  for (auto intersection_it = grid_view_.ibegin(entity); intersection_it != intersection_it_end;
205  ++intersection_it) {
206  const auto& intersection = *intersection_it;
207  if (intersection.neighbor() && !intersection.boundary()) {
208  const auto neighbor_ptr = intersection.outside();
209  const auto& neighbor = *neighbor_ptr;
210  if (grid_view_.indexSet().index(entity) < grid_view_.indexSet().index(neighbor)) {
211  const auto local_function_neighbor = function_.local_function(neighbor);
212  const auto local_source_neighbor = source.local_function(neighbor);
213  const size_t local_intersection_index = intersection.indexInInside();
214  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
215  // do a face quadrature
216  FieldType lhs = 0;
217  FieldType rhs = 0;
218  const size_t integrand_order = local_function->order();
219  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
220  intersection.type(), boost::numeric_cast<int>(integrand_order));
221  const auto quadrature_it_end = quadrature.end();
222  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
223  const auto xx_intersection = quadrature_it->position();
224  const auto normal = intersection.unitOuterNormal(xx_intersection);
225  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
226  const FieldType weight = quadrature_it->weight();
227  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
228  const auto xx_neighbor = intersection.geometryInOutside().global(xx_intersection);
229  // evaluate
230  ValueType function_value = local_function->evaluate(xx_entity);
231  function_value *= 0.5;
232  ValueType function_value_neighbor = local_function_neighbor->evaluate(xx_neighbor);
233  function_value_neighbor *= 0.5;
234  function_value += function_value_neighbor;
235  auto source_gradient = local_source->jacobian(xx_entity)[0];
236  source_gradient *= 0.5;
237  auto source_gradient_neighbor = local_source_neighbor->jacobian(xx_neighbor)[0];
238  source_gradient_neighbor *= 0.5;
239  source_gradient += source_gradient_neighbor;
240  const auto basis_values = local_basis.evaluate(xx_entity);
241  const auto basis_value = basis_values[local_DoF_index];
242  // compute integrals
243  lhs += integration_factor * weight * (basis_value * normal);
244  rhs += integration_factor * weight * -1.0 * compute_value(function_value, source_gradient, normal);
245  } // do a face quadrature
246  // set DoF
247  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
248  assert(!(range_vector[global_DoF_index] < infinity));
249  range_vector[global_DoF_index] = rhs / lhs;
250  }
251  } else if (intersection.boundary() && !intersection.neighbor()) {
252  const size_t local_intersection_index = intersection.indexInInside();
253  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
254  // do a face quadrature
255  FieldType lhs = 0;
256  FieldType rhs = 0;
257  const size_t integrand_order = local_function->order();
258  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
259  intersection.type(), boost::numeric_cast<int>(integrand_order));
260  const auto quadrature_it_end = quadrature.end();
261  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
262  const auto xx_intersection = quadrature_it->position();
263  const auto normal = intersection.unitOuterNormal(xx_intersection);
264  const auto integration_factor = intersection.geometry().integrationElement(xx_intersection);
265  const auto weight = quadrature_it->weight();
266  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
267  // evalaute
268  const ValueType function_value = local_function->evaluate(xx_entity);
269  const auto source_gradient = local_source->jacobian(xx_entity)[0];
270  const auto basis_values = local_basis.evaluate(xx_entity);
271  const auto basis_value = basis_values[local_DoF_index];
272  // compute integrals
273  lhs += integration_factor * weight * (basis_value * normal);
274  rhs += integration_factor * weight * -1.0 * compute_value(function_value, source_gradient, normal);
275  } // do a face quadrature
276  // set DoF
277  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
278  assert(!(range_vector[global_DoF_index] < infinity));
279  range_vector[global_DoF_index] = rhs / lhs;
280  } else
281  DUNE_THROW(Stuff::Exceptions::internal_error, "Unknown intersection type!");
282  } // walk the intersections
283  } // walk the grid
284  } // ... redirect_apply(...)
