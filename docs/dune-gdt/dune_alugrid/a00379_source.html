362  void solve(DiscreteSolutionType& solution) const
363  {
364 #if HAVE_EIGEN
365  try {
366  // get analytical flux, initial and boundary values
367  const std::shared_ptr<const AnalyticalFluxType> analytical_flux = problem().flux();
368  const std::shared_ptr<const InitialValueType> initial_values = problem().initial_values();
369  const std::shared_ptr<const BoundaryValueType> boundary_values = problem().boundary_values();
370  const std::shared_ptr<const RHSType> rhs = problem().rhs();
371 
372  // create a discrete function for the solution
373  DiscreteFunctionType u(*fv_space_, "solution");
374 
375  // project initial values
376  project(*initial_values, u);
377 
378  RangeFieldType t_end = test_case_.t_end();
379  const RangeFieldType CFL = problem().CFL();
380 
381  // calculate dx and choose initial dt
382  Dune::Stuff::Grid::Dimensions<typename SpaceType::GridViewType> dimensions(fv_space_->grid_view());
383  RangeFieldType dx = dimensions.entity_width.max();
384  if (dimDomain == 2)
385  dx /= std::sqrt(2);
386  RangeFieldType dt = CFL * dx;
387 
388  // create operators
389  const ConstantFunctionType dx_function(dx);
390  AdvectionOperatorType advection_operator =
391  internal::AdvectionOperatorCreator<AdvectionOperatorType, numerical_flux>::create(
392  *analytical_flux, *boundary_values, dx_function, dt, linear);
393  RHSOperatorType rhs_operator(*rhs);
394 
395  // create timestepper
396  OperatorTimeStepperType timestepper_op(advection_operator, u, -1.0);
397 
398  // do the time steps
399  const size_t num_save_steps = 100;
400  solution.clear();
401  if (problem().has_non_zero_rhs()) {
402  // use fractional step method
403  RHSOperatorTimeStepperType timestepper_rhs(rhs_operator, u);
404  TimeStepperType timestepper(timestepper_op, timestepper_rhs);
405  timestepper.solve(t_end, dt, num_save_steps, solution);
406  } else {
407  timestepper_op.solve(t_end, dt, num_save_steps, solution);
408  }
409 
410  } catch (Dune::Exception& e) {
411  std::cerr << "Dune reported: " << e.what() << std::endl;
412  std::abort();
413  }
414 #else // HAVE_EIGEN
415  static_assert(AlwaysFalse<DiscreteSolutionType>::value, "You are missing eigen!");
416 #endif // HAVE_EIGEN
417  }
