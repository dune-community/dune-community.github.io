44 template <class DomainFieldType>
45 struct product_quadrature_helper<DomainFieldType, 3>
46 {
47  static QuadratureRule<DomainFieldType, 2> get(const QuadratureRule<DomainFieldType, 1>& quadrature_1d)
48  {
49  QuadratureRule<DomainFieldType, 2> ret;
50  for (size_t ii = 0; ii < quadrature_1d.size(); ++ii)
51  for (size_t jj = 0; jj < quadrature_1d.size(); ++jj)
52  ret.push_back(
53  QuadraturePoint<DomainFieldType, 2>({quadrature_1d[ii].position()[0], quadrature_1d[jj].position()[0]},
54  quadrature_1d[ii].weight() * quadrature_1d[jj].weight()));
55  return ret;
56  }
