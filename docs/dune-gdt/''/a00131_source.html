313 template <class AnalyticalFluxType, size_t block_size = (AnalyticalFluxType::dimDomain == 1) ? 2 : 4>
314 class BlockedJacobianWrapper
315  : public JacobianWrapperBase<AnalyticalFluxType,
316  XT::Common::BlockedFieldMatrix<typename AnalyticalFluxType::RangeFieldType,
317  AnalyticalFluxType::dimRange / block_size,
318  block_size>,
319  XT::Common::BlockedFieldVector<typename AnalyticalFluxType::RangeFieldType,
320  AnalyticalFluxType::dimRange / block_size,
321  block_size>>
322 {
323  using BaseType = JacobianWrapperBase<AnalyticalFluxType,
324  XT::Common::BlockedFieldMatrix<typename AnalyticalFluxType::RangeFieldType,
325  AnalyticalFluxType::dimRange / block_size,
326  block_size>,
327  XT::Common::BlockedFieldVector<typename AnalyticalFluxType::RangeFieldType,
328  AnalyticalFluxType::dimRange / block_size,
329  block_size>>;
330 
331 public:
332  using BaseType::dimDomain;
333  using BaseType::dimRange;
334  using typename BaseType::RangeFieldType;
335  using typename BaseType::MatrixType;
336  using typename BaseType::VectorType;
337  using typename BaseType::DomainType;
338  using typename BaseType::EntityType;
339  using typename BaseType::JacobianType;
340  using typename BaseType::StateRangeType;
341  static constexpr size_t num_blocks = VectorType::num_blocks;
342  static_assert(dimRange % block_size == 0, "dimRange has to be a multiple of block_size");
343  using BlockedIntVectorType = XT::Common::BlockedFieldVector<int, num_blocks, block_size>;
344  using LocalVectorType = typename VectorType::BlockType;
345  using LocalMatrixType = typename MatrixType::BlockType;
346  using EigenSolverType = typename XT::LA::EigenSolver<LocalMatrixType>;
347  using LocalM = typename XT::Common::MatrixAbstraction<LocalMatrixType>;
348 
349  BlockedJacobianWrapper()
350  : work_(1)
351  {
352  jacobian() = std::make_unique<JacobianType>();
353  nonblocked_jacobians_ = std::make_unique<FieldVector<FieldMatrix<RangeFieldType, dimRange, dimRange>, dimDomain>>();
354 #if HAVE_MKL || HAVE_LAPACKE
355  // get optimal working size in work[0] (requested by lwork = -1)
356  int info = XT::Common::Lapacke::dgeev_work(XT::Common::Lapacke::row_major(),
357  /*do_not_compute_left_eigenvectors: */ 'N',
358  /*compute_right_eigenvectors: */ 'V',
359  static_cast<int>(block_size),
360  LocalM::data(jacobian(0).block(0)),
361  static_cast<int>(block_size),
362  eigenvalues_[0].data(),
363  &(dummy_complex_eigenvalues_[0]),
364  nullptr,
365  static_cast<int>(block_size),
366  LocalM::data(eigenvectors_[0].block(0)),
367  static_cast<int>(block_size),
368  work_.data(),
369  -1);
370  if (info != 0)
371  DUNE_THROW(Dune::XT::LA::Exceptions::eigen_solver_failed, "The lapack backend reported '" << info << "'!");
372  assert(work_[0] >= 0.);
373  work_.resize(static_cast<size_t>(work_[0] + 0.5));
374 #endif
375  }

377  using BaseType::jacobian;
378  using BaseType::compute;

380  virtual void compute(const size_t dd) override final
381  {
382  for (size_t jj = 0; jj < num_blocks; ++jj) {
383  if (block_size == 2) {
384  const auto& jac = jacobian(dd).block(jj);
385  const auto trace = jac[0][0] + jac[1][1];
386  const auto det = jac[0][0] * jac[1][1] - jac[0][1] * jac[1][0];
387  const auto sqrt_val = std::sqrt(0.25 * trace * trace - det);
388  const auto eigval1 = 0.5 * trace + sqrt_val;
389  const auto eigval2 = 0.5 * trace - sqrt_val;
390  if (std::abs(jac[1][0]) > std::abs(jac[0][1])) {
391  if (XT::Common::FloatCmp::ne(jac[1][0], 0.)) {
392  eigenvectors_[dd].block(jj)[0][0] = eigval1 - jac[1][1];
393  eigenvectors_[dd].block(jj)[0][1] = eigval2 - jac[1][1];
394  eigenvectors_[dd].block(jj)[1][0] = eigenvectors_[dd].block(jj)[1][1] = jac[1][0];
395  } else {
396  eigenvectors_[dd].block(jj)[0][0] = eigenvectors_[dd].block(jj)[1][1] = 1.;
397  eigenvectors_[dd].block(jj)[0][1] = eigenvectors_[dd].block(jj)[1][0] = 0.;
398  }
399  } else {
400  if (XT::Common::FloatCmp::ne(jac[0][1], 0.)) {
401  eigenvectors_[dd].block(jj)[1][0] = eigval1 - jac[0][0];
402  eigenvectors_[dd].block(jj)[1][1] = eigval2 - jac[0][0];
403  eigenvectors_[dd].block(jj)[0][0] = eigenvectors_[dd].block(jj)[0][1] = jac[0][1];
404  } else {
405  eigenvectors_[dd].block(jj)[0][0] = eigenvectors_[dd].block(jj)[1][1] = 1.;
406  eigenvectors_[dd].block(jj)[0][1] = eigenvectors_[dd].block(jj)[1][0] = 0.;
407  }
408  }
409  } else {
410 #if HAVE_MKL || HAVE_LAPACKE
411  int info = XT::Common::Lapacke::dgeev_work(XT::Common::Lapacke::row_major(),
412  /*do_not_compute_left_eigenvectors: */ 'N',
413  /*compute_right_eigenvectors: */ 'V',
414  static_cast<int>(block_size),
415  LocalM::data(jacobian(dd).block(jj)),
416  static_cast<int>(block_size),
417  eigenvalues_[jj].data(),
418  &(dummy_complex_eigenvalues_[0]),
419  nullptr,
420  static_cast<int>(block_size),
421  LocalM::data(eigenvectors_[dd].block(jj)),
422  static_cast<int>(block_size),
423  work_.data(),
424  static_cast<int>(work_.size()));
425  if (info != 0)
426  DUNE_THROW(Dune::XT::LA::Exceptions::eigen_solver_failed, "The lapack backend reported '" << info << "'!");
427 #else // HAVE_MKL || HAVE_LAPACKE
428  static XT::Common::Configuration eigensolver_options = hyperbolic_default_eigensolver_options<MatrixType>();
429  const auto eigensolver = EigenSolverType(jacobian(dd).block(jj), &eigensolver_options);
430  eigenvectors_[dd].block(jj) = eigensolver.real_eigenvectors();
431 #endif // HAVE_MKL || HAVE_LAPACKE
432  } // else (block_size == 2)
433  QR_[dd].block(jj) = eigenvectors_[dd].block(jj);
434  XT::LA::qr(QR_[dd].block(jj), tau_[dd].block(jj), permutations_[dd].block(jj));
435  } // jj
436  computed_[dd] = true;
437  }

497 {
498 public:
499  static constexpr size_t dim = MultiArrayType::dimensionality;
500  static constexpr size_t array_dim = dim == 0 ? 1 : dim;
501  using IndicesType = std::array<size_t, array_dim>;
502 
503  MultiIndexIterator(const MultiArrayType& multi_array, const IndicesType& indices)
504  : multi_array_(multi_array)
505  , indices_(indices)
506  {
507  }
508 
509  IndicesType& operator*()
510  {
511  return indices_;
512  }
513 
514  const IndicesType& operator*() const
515  {
516  return indices_;
517  }
518 
519  MultiIndexIterator& operator++()
520  {
521  size_t ii = 0;
522  for (; ii < array_dim; ++ii) {
523  indices_[ii]++;
524  if (indices_[ii] < (dim == 0 ? 1 : multi_array_.shape()[ii]))
525  break;
526  indices_[ii] = 0;
527  }
528  if (ii == array_dim) {
529  if (dim == 0)
530  indices_[0] = 1;
531  else
532  std::copy_n(multi_array_.shape(), array_dim, indices_.begin());
533  }
534  return *this;
535  } // ... operator++()
536 
537  MultiIndexIterator operator++(int)
538  {
539  MultiIndexIterator ret = *this;
540  this->operator++();
541  return ret;
542  } // ... operator++(int)
543 
544  bool operator==(const MultiIndexIterator& other) const
545  {
546  return indices_ == other.indices_;
547  }
548 
549  bool operator!=(const MultiIndexIterator& other) const
550  {
551  return indices_ != other.indices_;
552  }

554 private:
