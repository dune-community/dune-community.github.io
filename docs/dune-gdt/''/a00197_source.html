175 template <typename GatherScatter>
176 class DataGatherScatter
177 {
178 
179 public:
180  typedef std::size_t size_t;
181 
182  template <typename MessageBuffer, typename Entity, typename LocalView>
183  bool gather(MessageBuffer& buff, const Entity& /*e*/, const LocalView& local_view) const
184  {
185  for (std::size_t i = 0; i < local_view.size(); ++i)
186  _gather_scatter.gather(buff, local_view[i]);
187  return false;
188  }
189 
190  // default scatter - requires function space structure to be identical on sender and receiver side
191  template <typename MessageBuffer, typename Entity, typename LocalView>
192  bool scatter(MessageBuffer& buff, size_t n, const Entity& e, LocalView& local_view) const
193  {
194  if (partitions_.contains(e.partitionType())) {
195  if (local_view.size() != n)
196  DUNE_THROW(Exception,
197  "size mismatch in GridFunctionSpace data handle, have " << local_view.size() << "DOFs, but received "
198  << n);
199 
200  for (std::size_t i = 0; i < local_view.size(); ++i)
201  _gather_scatter.scatter(buff, local_view[i]);
202  return true;
203  } else {
204  if (local_view.size() != 0)
205  DUNE_THROW(Exception,
206  "expected no DOFs in partition '" << e.partitionType() << "', but have " << local_view.size());
207 
208  for (std::size_t i = 0; i < local_view.size(); ++i) {
209  typename LocalView::value_type dummy;
210  buff.read(dummy);
211  }
212  return false;
213  }
214  }
215 
216  DataGatherScatter(GatherScatter gather_scatter = GatherScatter())
217  : _gather_scatter(gather_scatter)
218  {
219  }
220 
221 private:
222  GatherScatter _gather_scatter;
223  // this is currently the default since we have no non-overlapping parallel code
224  static constexpr const Partitions::All partitions_{};
225 };

323 class GhostGatherScatter
324 {
325 public:
326  template <typename MessageBuffer, typename Entity, typename LocalView>
327  bool gather(MessageBuffer& buff, const Entity& e, LocalView& /*local_view*/) const
328  {
329  // Figure out where we are...
330  const bool ghost = e.partitionType() != Dune::InteriorEntity && e.partitionType() != Dune::BorderEntity;
331 
332  // ... and send something (doesn't really matter what, we'll throw it away on the receiving side).
333  buff.write(ghost);
334 
335  return false;
336  }
337 
338  template <typename MessageBuffer, typename Entity, typename LocalView>
339  bool scatter(MessageBuffer& buff, std::size_t /*n*/, const Entity& e, LocalView& local_view) const
340  {
341  // Figure out where we are - we have to do this again on the receiving side due to the asymmetric
342  // communication interface!
343  const bool ghost = e.partitionType() != Dune::InteriorEntity && e.partitionType() != Dune::BorderEntity;
344 
345  // drain buffer
346  bool dummy;
347  buff.read(dummy);
348 
349  for (std::size_t i = 0; i < local_view.size(); ++i)
350  local_view[i] = ghost;
351 
352  return true;
353  }

402 template <typename RankIndex>
403 class DisjointPartitioningGatherScatter
404 {
405 
406 public:
407  template <typename MessageBuffer, typename Entity, typename LocalView>
408  bool gather(MessageBuffer& buff, const Entity& /*e*/, LocalView& /*local_view*/) const
409  {
410  // We only gather from interior and border entities, so we can throw in our ownership
411  // claim without any further checks.
412  buff.write(rank_);
413 
414  return true;
415  }
416 
417  template <typename MessageBuffer, typename Entity, typename LocalView>
418  bool scatter(MessageBuffer& buff, std::size_t /*n*/, const Entity& e, LocalView& local_view) const
419  {
420  // Value used for DOFs with currently unknown rank.
421  const RankIndex unknown_rank = std::numeric_limits<RankIndex>::max();
422 
423  // We can only own this DOF if it is either on the interior or border partition.
424  const bool is_interior_or_border =
425  (e.partitionType() == Dune::InteriorEntity || e.partitionType() == Dune::BorderEntity);
426 
427  // Receive data.
428  RankIndex received_rank;
429  buff.read(received_rank);
430 
431  for (std::size_t i = 0; i < local_view.size(); ++i) {
432  // Get the currently stored owner rank for this DOF.
433  RankIndex current_rank = local_view[i];
434 
435  // We only gather from interior and border entities, so we need to make sure
436  // we relinquish any ownership claims on overlap and ghost entities on the
437  // receiving side. We also need to make sure not to overwrite any data already
438  // received, so we only blank the rank value if the currently stored value is
439  // equal to our own rank.
440  if (!is_interior_or_border && current_rank == rank_)
441  current_rank = unknown_rank;
442 
443  // Assign DOFs to minimum rank value.
444  local_view[i] = std::min(current_rank, received_rank);
445  }
446  return true;
447  }
448 
450 
453  DisjointPartitioningGatherScatter(RankIndex rank)
454  : rank_(rank)
455  {
456  }
457 
458 private:
459  const RankIndex rank_;
460 };
