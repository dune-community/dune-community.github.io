88 template <class GL, int p, class R>
89 class ContinuousLagrangeSpace
90  : public CgSpaceInterface<internal::ContinuousLagrangeSpaceTraits<GL, p, R>, GL::dimension, 1>
91 {
92 public:
93  using Traits = internal::ContinuousLagrangeSpaceTraits<GL, p, R>;
94 
95 private:
96  using BaseType = CgSpaceInterface<Traits, GL::dimension, 1>;
97  using ThisType = ContinuousLagrangeSpace<GL, p, R>;
98  using D = typename GL::ctype;
99  static const constexpr size_t d = BaseType::dimDomain;
100  using FiniteElementType = LagrangeLocalFiniteElement<EquidistantPointSet, d, D, R>;
101  using DofCommunicationChooserType = typename Traits::DofCommunicationChooserType;
102 
103 public:
104  using typename BaseType::GridLayerType;
105  using typename BaseType::EntityType;
106  using typename BaseType::MapperType;
107  using typename BaseType::BaseFunctionSetType;
108  using DomainType = typename BaseFunctionSetType::DomainType;
109  using DofCommunicatorType = typename Traits::DofCommunicatorType;
110 
111  ContinuousLagrangeSpace(GridLayerType grd_lr)
112  : grid_layer_(grd_lr)
113  , communicator_(DofCommunicationChooserType::create(grid_layer_))
114  , backend_(0)
115  , finite_elements_(new std::map<GeometryType, std::shared_ptr<FiniteElementType>>())
116  , lagrange_points_(new std::map<GeometryType, std::vector<DomainType>>())
117  , mapper_(nullptr)
118  {
119  // create finite elements and lagrange points
120  for (auto&& geometry_type : grid_layer_.indexSet().types(0)) {
121  if (geometry_type == GeometryType(GeometryType::pyramid, 3))
122  DUNE_THROW(space_error, "Continuous Lagrange space does not seem to have working jacobians on pyramid grids!");
123  auto fe = std::make_shared<FiniteElementType>(geometry_type, p);
124  const auto& lp = fe->localInterpolation().lagrangePoints();
125  std::vector<DomainType> lagrange_points(lp.size());
126  for (unsigned int ii = 0; ii < lp.size(); ++ii)
127  lagrange_points[ii] = lp[ii].point();
128  lagrange_points_->insert(std::make_pair(geometry_type, std::move(lagrange_points)));
129  finite_elements_->insert(std::make_pair(geometry_type, std::move(fe)));
130  }
131  // check
132  if (d == 3 && finite_elements_->size() != 1)
133  DUNE_THROW(space_error, "Continuous Lagrange space with multiple finite elements in 3d not supported (yet)!");
134  // create mapper
135  mapper_ = std::make_shared<MapperType>(grid_layer_, finite_elements_);
136  }

139  ContinuousLagrangeSpace(ThisType&&) = default;

142  ThisType& operator=(ThisType&&) = delete;

157  }

159  BaseFunctionSetType base_function_set(const EntityType& entity) const
160  {
161  const auto finite_element_search_result = finite_elements_->find(entity.geometry().type());
162  if (finite_element_search_result == finite_elements_->end())
163  DUNE_THROW(XT::Common::Exceptions::internal_error,
164  "This must not happen, the grid layer did not report all geometry types!"
165  << "\n entity.geometry().type() = "
166  << entity.geometry().type());
167  const auto& finite_element = *finite_element_search_result->second;
168  return BaseFunctionSetType(entity, finite_element);
169  }
170 
171  DofCommunicatorType& dof_communicator() const
172  {
173  DofCommunicationChooserType::prepare(*this, *communicator_);
174  return *communicator_;
175  }
176 
177  std::vector<DomainType> lagrange_points(const EntityType& entity) const
178  {
179  const auto lagrange_points_search_result = lagrange_points_->find(entity.geometry().type());
180  if (lagrange_points_search_result == lagrange_points_->end())
181  DUNE_THROW(XT::Common::Exceptions::internal_error,
182  "This must not happen, the grid layer did not report all geometry types!"
183  << "\n entity.geometry().type() = "
184  << entity.geometry().type());
185  return lagrange_points_search_result->second;
186  }
187 
188 private:
189  const GridLayerType grid_layer_;
190  mutable std::shared_ptr<DofCommunicatorType> communicator_;
191  const double backend_;
192  std::shared_ptr<std::map<GeometryType, std::shared_ptr<FiniteElementType>>> finite_elements_;
193  std::shared_ptr<std::map<GeometryType, std::vector<DomainType>>> lagrange_points_;
194  std::shared_ptr<MapperType> mapper_;
195 }; // class ContinuousLagrangeSpace
