52 template <class UnrestrictedSpace, class RestrictionGridLayer>
53 class RestrictedMapper
54  : public MapperInterface<internal::RestrictedMapperTraits<UnrestrictedSpace, RestrictionGridLayer>>
55 {
56  typedef MapperInterface<internal::RestrictedMapperTraits<UnrestrictedSpace, RestrictionGridLayer>> BaseType;
57  typedef RestrictedMapper<UnrestrictedSpace, RestrictionGridLayer> ThisType;
58 
59 public:
60  using typename BaseType::BackendType;
61  using typename BaseType::EntityType;
62 
63  RestrictedMapper(const UnrestrictedSpace& unrestricted_space, RestrictionGridLayer restriction_grid_layer)
64  : unrestricted_space_(unrestricted_space)
65  , grid_layer_(restriction_grid_layer)
66  , max_num_dofs_(0)
67  , map_to_restricted_()
68  , map_to_unrestricted_()
69  {
70  std::set<size_t> restricted_indices;
71  DynamicVector<size_t> unrestricted_indices(backend().maxNumDofs(), 0);
72  for (auto&& entity : elements(grid_layer_)) {
73  max_num_dofs_ = std::max(max_num_dofs_, backend().numDofs(entity));
74  backend().globalIndices(entity, unrestricted_indices);
75  for (size_t ii = 0; ii < backend().numDofs(entity); ++ii)
76  restricted_indices.insert(unrestricted_indices[ii]);
77  }
78  map_to_unrestricted_ = std::vector<size_t>(restricted_indices.size());
79  size_t counter = 0;
80  for (const auto& index : restricted_indices) {
81  map_to_restricted_[index] = counter;
82  map_to_unrestricted_[counter] = index;
83  ++counter;
84  }
85  DUNE_THROW_IF(
86  map_to_restricted_.size() != map_to_unrestricted_.size(), InvalidStateException, "This should not happen!");
87  }
88 
89  RestrictedMapper(const ThisType& other) = default;
90  RestrictedMapper(ThisType&& source) = default;
91 
92  ThisType& operator=(const ThisType& other) = delete;
93  ThisType& operator=(ThisType&& source) = delete;

95  template <class U, class V>
96  void restrict(const XT::LA::VectorInterface<U>& unrestricted_vector,
97  XT::LA::VectorInterface<V>& restricted_vector) const
98  {
99  DUNE_THROW_IF(unrestricted_vector.size() != backend().size(),
100  XT::Common::Exceptions::shapes_do_not_match,
101  "unrestricted_vector.size() = " << unrestricted_vector.size()
102  << "\n unrestricted_space.mapper().size() = "
103  << backend().size());
104  DUNE_THROW_IF(restricted_vector.size() != size(),
105  XT::Common::Exceptions::shapes_do_not_match,
106  "restricted_vector.size() = " << restricted_vector.size() << "\n size() = " << size());
107  // the actual work
108  for (size_t restricted_DoF = 0; restricted_DoF < map_to_unrestricted_.size(); ++restricted_DoF)
109  restricted_vector[restricted_DoF] = unrestricted_vector[map_to_unrestricted_[restricted_DoF]];
110  } // ... restrict(...)
