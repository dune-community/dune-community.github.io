187  template <class R>
188  void
189  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>& local_diffusion,
190  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain>&
191  local_diffusive_flux,
192  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& test_base,
193  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& ansatz_base,
194  const Dune::FieldVector<DomainFieldType, dimDomain>& local_point,
195  Dune::DynamicMatrix<R>& ret) const
196  {
197  // evaluate local functions
198  const auto diffusion_value = local_diffusion.evaluate(local_point);
199  const R one_over_diffusion_value = 1.0 / diffusion_value;
200  const auto diffusive_flux_value = local_diffusive_flux.evaluate(local_point);
201  // evaluate test gradient
202  const size_t rows = test_base.size();
203  const auto test_gradients = test_base.jacobian(local_point);
204  auto left_sum = test_gradients[0];
205  assert(test_gradients.size() == rows);
206  // evaluate ansatz gradient
207  const size_t cols = ansatz_base.size();
208  const auto ansatz_gradients = ansatz_base.jacobian(local_point);
209  auto right_sum = ansatz_gradients[0];
210  assert(ansatz_gradients.size() == rows);
211  // compute
212  assert(ret.rows() >= rows);
213  assert(ret.cols() >= cols);
214  for (size_t ii = 0; ii < rows; ++ii) {
215  left_sum = test_gradients[ii];
216  left_sum[0] *= diffusion_value;
217  left_sum[0] += diffusive_flux_value;
218  auto& retRow = ret[ii];
219  for (size_t jj = 0; jj < cols; ++jj) {
220  right_sum = ansatz_gradients[jj];
221  right_sum[0] *= diffusion_value;
222  right_sum[0] += diffusive_flux_value;
223  retRow[jj] = one_over_diffusion_value * (left_sum[0] * right_sum[0]);
224  }
225  }
226  } // ... evaluate(...)
227

230 private:
231  const DiffusionType& diffusion_;
232  const DiffusiveFluxType& diffusive_flux_;
233 }; // class LocalDiffusiveFluxEstimateESV2007Integrand< ..., void >

354  template <class R>
355  void evaluate(
356  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>& local_diffusion_factor,
357  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain, dimDomain>&
358  local_diffusion_tensor,
359  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain>&
360  local_diffusive_flux,
361  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& test_base,
362  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& ansatz_base,
363  const Dune::FieldVector<DomainFieldType, dimDomain>& local_point,
364  Dune::DynamicMatrix<R>& ret) const
365  {
366  typedef FieldVector<R, dimDomain> DomainType;
367  DomainType left_sum(0);
368  DomainType right_sum(0);
369  // evaluate local functions
370  const auto diffusion_factor_value = local_diffusion_factor.evaluate(local_point);
371  typedef XT::Common::FieldMatrix<R, dimDomain, dimDomain> TensorType;
372  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(local_point);
373  const TensorType diffusion_value = diffusion_tensor_value * diffusion_factor_value;
374  // TODO: there is no documented way to assert that the inversion was successfull, so find one or check the matrix
375  // beforehand
376  TensorType one_over_diffusion_value = diffusion_tensor_value;
377  one_over_diffusion_value.invert();
378  one_over_diffusion_value /= diffusion_factor_value;
379  const auto diffusive_flux_value = local_diffusive_flux.evaluate(local_point);
380  // evaluate test gradient
381  const size_t rows = test_base.size();
382  const auto test_gradients = test_base.jacobian(local_point);
383  assert(test_gradients.size() == rows);
384  // evaluate ansatz gradient
385  const size_t cols = ansatz_base.size();
386  const auto ansatz_gradients = ansatz_base.jacobian(local_point);
387  assert(ansatz_gradients.size() == rows);
388  // compute
389  assert(ret.rows() >= rows);
390  assert(ret.cols() >= cols);
391  for (size_t ii = 0; ii < rows; ++ii) {
392  left_sum = diffusion_value * test_gradients[ii][0];
393  left_sum += diffusive_flux_value;
394  auto& retRow = ret[ii];
395  for (size_t jj = 0; jj < cols; ++jj) {
396  right_sum = diffusion_value * ansatz_gradients[jj][0];
397  right_sum += diffusive_flux_value;
398  retRow[jj] = (one_over_diffusion_value * left_sum) * right_sum;
399  }
400  }
401  } // ... evaluate(...)
