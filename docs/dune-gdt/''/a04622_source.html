156  std::vector<DomainType> lagrange_points_order_1(const EntityType& entity) const
157  {
158  // check
159  static_assert(polOrder == 1, "Not tested for higher polynomial orders!");
160  if (dimRange != 1)
161  DUNE_THROW(NotImplemented, "Does not work for higher dimensions");
162  assert(this->grid_layer().indexSet().contains(entity));
163  // get the basis and reference element
164  const auto basis = this->base_function_set(entity);
165  typedef typename BaseType::BaseFunctionSetType::RangeType RangeType;
166  std::vector<RangeType> tmp_basis_values(basis.size(), RangeType(0));
167  const auto& reference_element = ReferenceElements<DomainFieldType, dimDomain>::general(entity.type());
168  const auto num_vertices = reference_element.size(dimDomain);
169  assert(num_vertices >= 0);
170  assert(boost::numeric_cast<size_t>(num_vertices) == basis.size() && "This should not happen with polOrder 1!");
171  // prepare return vector
172  std::vector<DomainType> local_vertices(num_vertices, DomainType(0));
173  // loop over all vertices
174  for (auto ii : Dune::XT::Common::value_range(num_vertices)) {
175  // get the local coordinate of the iith vertex
176  const auto local_vertex = reference_element.position(ii, dimDomain);
177  // evaluate the basefunctionset
178  basis.evaluate(local_vertex, tmp_basis_values);
179  // find the basis function that evaluates to one here (has to be only one!)
180  size_t ones = 0;
181  size_t zeros = 0;
182  size_t failures = 0;
183  for (size_t jj = 0; jj < basis.size(); ++jj) {
184  if (std::abs((tmp_basis_values)[jj][0] - RangeFieldType(1)) < compare_tolerance_) {
185  local_vertices[jj] = local_vertex;
186  ++ones;
187  } else if (std::abs((tmp_basis_values)[jj][0]) < compare_tolerance_)
188  ++zeros;
189  else
190  ++failures;
191  }
192  assert(ones == 1 && zeros == (basis.size() - 1) && failures == 0 && "This must not happen for polOrder 1!");
193  }
194  return local_vertices;
195  } // ... lagrange_points_order_1(...)
