314  template <class IntersectionType, class R>
315  void
316  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionEntity,
317  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionNeighbor,
318  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseEntity,
319  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseEntity,
320  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseNeighbor,
321  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseNeighbor,
322  const IntersectionType& intersection,
323  const Dune::FieldVector<DomainFieldType, 1>& localPoint,
324  Dune::DynamicMatrix<R>& entityEntityRet,
325  Dune::DynamicMatrix<R>& neighborNeighborRet,
326  Dune::DynamicMatrix<R>& entityNeighborRet,
327  Dune::DynamicMatrix<R>& neighborEntityRet) const
328  {
329  // clear ret
330  entityEntityRet *= 0.0;
331  neighborNeighborRet *= 0.0;
332  entityNeighborRet *= 0.0;
333  neighborEntityRet *= 0.0;
334  // convert local point (which is in intersection coordinates) to entity/neighbor coordinates
335  const auto localPointEn = intersection.geometryInInside().global(localPoint);
336  const auto localPointNe = intersection.geometryInOutside().global(localPoint);
337  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
338  // evaluate local function
339  const auto functionValueEn = localFunctionEntity.evaluate(localPointEn);
340  const auto functionValueNe = localFunctionNeighbor.evaluate(localPointNe);
341  // compute penalty (see Epshteyn, Riviere, 2007)
342  const auto max_polorder =
343  std::max(testBaseEntity.order(),
344  std::max(ansatzBaseEntity.order(), std::max(testBaseNeighbor.order(), ansatzBaseNeighbor.order())));
345  const R sigma = internal::inner_sigma(max_polorder);
346  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
347  // evaluate bases
348  // * entity
349  // * test
350  const auto rowsEn = testBaseEntity.size();
351  const auto testValuesEn = testBaseEntity.evaluate(localPointEn);
352  const auto testGradientsEn = testBaseEntity.jacobian(localPointEn);
353  // * ansatz
354  const auto colsEn = ansatzBaseEntity.size();
355  const auto ansatzValuesEn = ansatzBaseEntity.evaluate(localPointEn);
356  const auto ansatzGradientsEn = ansatzBaseEntity.jacobian(localPointEn);
357  // * neighbor
358  // * test
359  const auto rowsNe = testBaseNeighbor.size();
360  const auto testValuesNe = testBaseNeighbor.evaluate(localPointNe);
361  const auto testGradientsNe = testBaseNeighbor.jacobian(localPointNe);
362  // * ansatz
363  const auto colsNe = ansatzBaseNeighbor.size();
364  const auto ansatzValuesNe = ansatzBaseNeighbor.evaluate(localPointNe);
365  const auto ansatzGradientsNe = ansatzBaseNeighbor.jacobian(localPointNe);
366  // compute the evaluations
367  assert(entityEntityRet.rows() >= rowsEn);
368  assert(entityEntityRet.cols() >= colsEn);
369  assert(entityNeighborRet.rows() >= rowsEn);
370  assert(entityNeighborRet.cols() >= colsNe);
371  assert(neighborEntityRet.rows() >= rowsNe);
372  assert(neighborEntityRet.cols() >= colsEn);
373  assert(neighborNeighborRet.rows() >= rowsNe);
374  assert(neighborNeighborRet.cols() >= colsNe);
375  // loop over all entity test basis functions
376  for (size_t ii = 0; ii < rowsEn; ++ii) {
377  auto& entityEntityRetRow = entityEntityRet[ii];
378  auto& entityNeighborRetRow = entityNeighborRet[ii];
379  // loop over all entity ansatz basis functions
380  for (size_t jj = 0; jj < colsEn; ++jj) {
381  // consistency term
382  entityEntityRetRow[jj] +=
383  -0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesEn[ii];
384  // symmetry term
385  entityEntityRetRow[jj] +=
386  -0.5 * ansatzValuesEn[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
387  // penalty term
388  entityEntityRetRow[jj] += penalty * ansatzValuesEn[jj] * testValuesEn[ii];
389  } // loop over all entity ansatz basis functions
390  // loop over all neighbor ansatz basis functions
391  for (size_t jj = 0; jj < colsNe; ++jj) {
392  // consistency term
393  entityNeighborRetRow[jj] +=
394  -0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesEn[ii];
395  // symmetry term
396  entityNeighborRetRow[jj] +=
397  0.5 * ansatzValuesNe[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
398  // penalty term
399  entityNeighborRetRow[jj] += -1.0 * penalty * ansatzValuesNe[jj] * testValuesEn[ii];
400  } // loop over all neighbor ansatz basis functions
401  } // loop over all entity test basis functions
402  // loop over all neighbor test basis functions
403  for (size_t ii = 0; ii < rowsNe; ++ii) {
404  auto& neighborEntityRetRow = neighborEntityRet[ii];
405  auto& neighborNeighborRetRow = neighborNeighborRet[ii];
406  // loop over all entity ansatz basis functions
407  for (size_t jj = 0; jj < colsEn; ++jj) {
408  // consistency term
409  neighborEntityRetRow[jj] +=
410  0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesNe[ii];
411  // symmetry term
412  neighborEntityRetRow[jj] +=
413  -0.5 * ansatzValuesEn[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
414  // penalty term
415  neighborEntityRetRow[jj] += -1.0 * penalty * ansatzValuesEn[jj] * testValuesNe[ii];
416  } // loop over all entity ansatz basis functions
417  // loop over all neighbor ansatz basis functions
418  for (size_t jj = 0; jj < colsNe; ++jj) {
419  // consistency term
420  neighborNeighborRetRow[jj] +=
421  0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesNe[ii];
422  // symmetry term
423  neighborNeighborRetRow[jj] +=
424  0.5 * ansatzValuesNe[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
425  // penalty term
426  neighborNeighborRetRow[jj] += penalty * ansatzValuesNe[jj] * testValuesNe[ii];
427  } // loop over all neighbor ansatz basis functions
428  } // loop over all neighbor test basis functions
429  } // ... evaluate(...)

644  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLF, rCLF>& localDiffusion,
645  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLR, rCLR>& localDirichlet,
646  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, rT, rCT>& testBase)
647  const
648  {
649  const size_t testOrder = testBase.order();
650  const size_t testGradientOrder = boost::numeric_cast<size_t>(std::max(ssize_t(testOrder) - 1, ssize_t(0)));
651  const size_t diffusionOrder = localDiffusion.order();

660  template <class IntersectionType, class R>
661  void
662  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDiffusion,
663  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDirichlet,
664  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& testBase,
665  const IntersectionType& intersection,
666  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& localPoint,
667  Dune::DynamicVector<R>& ret) const
668  {
669  // clear ret
670  ret *= 0.0;
671  // get local point (which is in intersection coordinates) in entity coordinates
672  const auto localPointEntity = intersection.geometryInInside().global(localPoint);
673  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
674  // evaluate local functions
675  const auto diffusionValue = localDiffusion.evaluate(localPointEntity);
676  const auto dirichletValue = localDirichlet.evaluate(localPointEntity);
677  // compute penalty (see Epshteyn, Riviere, 2007)
678  const auto polorder = testBase.order();
679  const R sigma = internal::boundary_sigma(polorder);
680  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
681  // evaluate basis
682  const auto size = testBase.size();
683  const auto testValues = testBase.evaluate(localPointEntity);
684  const auto testGradients = testBase.jacobian(localPointEntity);
685  // compute
686  assert(ret.size() >= size);
687  // loop over all test basis functions
688  for (size_t ii = 0; ii < size; ++ii) {
689  // symmetry term
690  ret[ii] += -1.0 * dirichletValue * diffusionValue * (testGradients[ii][0] * unitOuterNormal);
691  // penalty term
692  ret[ii] += penalty * dirichletValue * testValues[ii];
693  } // loop over all test basis functions
694  } // ... evaluate(...)
