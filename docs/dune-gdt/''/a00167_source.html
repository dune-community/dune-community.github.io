176 template <class OperatorImp,
177  class DiscreteFunctionImp,
178  class SolverImp,
179  RosenbrockTimeStepperMethods method = RosenbrockTimeStepperMethods::GRK4T>
180 class RosenbrockTimeStepper : public TimeStepperInterface<DiscreteFunctionImp>
181 {
182  using BaseType = TimeStepperInterface<DiscreteFunctionImp>;
183  using ButcherArrayProviderType = internal::RosenbrockButcherArrayProvider<typename BaseType::RangeFieldType, method>;
184 
185 public:
186  using typename BaseType::DiscreteFunctionType;
187  using typename BaseType::DomainFieldType;
188  using typename BaseType::RangeFieldType;
189 
190  using OperatorType = OperatorImp;
191  using SolverType = SolverImp;
192  using MatrixType = typename Dune::DynamicMatrix<RangeFieldType>;
193  using VectorType = typename Dune::DynamicVector<RangeFieldType>;
194 
195  using BaseType::current_solution;
196  using BaseType::current_time;
197 
213  RosenbrockTimeStepper(const OperatorType& op,
214  const DiscreteFunctionType& initial_values,
215  const RangeFieldType r = 1.0,
216  const double t_0 = 0.0,
217  const RangeFieldType tol = 1e-4,
218  const RangeFieldType scale_factor_min = 0.2,
219  const RangeFieldType scale_factor_max = 5,
220  const MatrixType& A = ButcherArrayProviderType::A(),
221  const VectorType& b_1 = ButcherArrayProviderType::b_1(),
222  const VectorType& b_2 = ButcherArrayProviderType::b_2(),
223  const VectorType& c = ButcherArrayProviderType::c(),
224  const MatrixType& Gamma = ButcherArrayProviderType::Gamma())
225  : BaseType(t_0, initial_values)
226  , op_(op)
227  , r_(r)
228  , tol_(tol)
229  , scale_factor_min_(scale_factor_min)
230  , scale_factor_max_(scale_factor_max)
231  , u_tmp_(BaseType::current_solution())
232  , A_(A)
233  , m_1_(b_1)
234  , m_2_(b_2)
235  , c_(c)
236  , d_(Gamma.rows())
237  , Gamma_(Gamma)
238  , m_diff_(b_1)
239  , num_stages_(A_.rows())
240  , gamma_ii_equal_for_all_i_(true)
241  {
242  assert(Dune::XT::Common::FloatCmp::gt(tol_, 0.0));
243  assert(Dune::XT::Common::FloatCmp::le(scale_factor_min_, 1.0));
244  assert(Dune::XT::Common::FloatCmp::ge(scale_factor_max_, 1.0));
245  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
246  assert(Gamma_.rows() == Gamma_.cols() && "Gamma has to be a square matrix");
247  assert(Gamma_.rows() == A_.rows() && "Sizes of A and Gamma have to match!");
248  assert(b_1.size() == A_.rows());
249  assert(b_2.size() == A_.rows());
250  assert(c_.size() == A_.rows());
251 #ifndef NDEBUG
252  for (size_t ii = 0; ii < A_.rows(); ++ii) {
253  RangeFieldType c_calculated = 0;
254  for (size_t jj = 0; jj < ii; ++jj)
255  c_calculated += A_[ii][jj];
256  assert(Dune::XT::Common::FloatCmp::eq(c_calculated, c_[ii]));
257  for (size_t jj = ii; jj < A_.cols(); ++jj) {
258  assert(Dune::XT::Common::FloatCmp::eq(A_[ii][jj], 0.0)
259  && "A has to be a lower triangular matrix with 0 on the main diagonal!");
260  if (jj == ii)
261  assert(Dune::XT::Common::FloatCmp::ne(Gamma_[ii][jj], 0.0)
262  && "The diagonal entries of Gamma must not vanish!");
263  else
264  assert(Dune::XT::Common::FloatCmp::eq(Gamma_[ii][jj], 0.0) && "Gamma has to be a lower triangular matrix!");
265  }
266  }
267 #endif // NDEBUG
268  // store as many discrete functions as needed for intermediate stages
269  for (size_t ii = 0; ii < num_stages_; ++ii) {
270  u_intermediate_stages_.emplace_back(current_solution());
271  }
272  // transform variables for faster calculations
273  // C_ = diag(1/gamma_11, ... , 1/gamma_nn) - Gamma^(-1), A_ = A*Gamma^(-1), m = (b_1,...,b_n)*Gamma^(-1), see e.g.
274  // Hairer, Wanner (1996), Solving ordinary differential equations II: Stiff and differential-algebraic problems,
275  // pp 119ff
276  auto Gamma_inv = Gamma_;
277  Gamma_inv.invert();
278  C_ = Gamma_inv;
279  C_ *= -1.0;
280  for (size_t ii = 0; ii < C_.rows(); ++ii)
281  C_[ii][ii] += 1.0 / (Gamma_[ii][ii]);
282  A_.rightmultiply(Gamma_inv);
283  Gamma_inv.mtv(b_1, m_1_);
284  Gamma_inv.mtv(b_2, m_2_);
285 
286  auto gamma = Gamma[0][0];
287  for (size_t ii = 0; ii < Gamma.rows(); ++ii) {
288  if (Dune::XT::Common::FloatCmp::ne(gamma, Gamma_[ii]))
289  gamma_ii_equal_for_all_i_ = false;
290  d_[ii] = 0.0;
291  for (size_t jj = 0; jj <= ii; ++jj)
292  d_[ii] += Gamma[ii][jj];
293  }
294  } // constructor RosenbrockTimeStepper
295 
296  RangeFieldType step(const RangeFieldType dt, const RangeFieldType max_dt)
297  {
298  RangeFieldType actual_dt = std::min(dt, max_dt);
299  RangeFieldType mixed_error = std::numeric_limits<RangeFieldType>::max();
300  RangeFieldType time_step_scale_factor = 1.0;
301 
302  auto& t = current_time();
303  auto& u_n = current_solution();
304 
305  while (Dune::XT::Common::FloatCmp::gt(mixed_error, tol_)) {
306  actual_dt *= time_step_scale_factor;
307 
308  for (size_t ii = 0; ii < num_stages_; ++ii) {
309  // GRK4A and GRK4T use the same coefficients for the third and fourth stage
310  if ((method == RosenbrockTimeStepperMethods::GRK4A || method == RosenbrockTimeStepperMethods::GRK4T) && ii == 3)
311  u_intermediate_stages_[ii].vector() = u_intermediate_stages_[ii - 1].vector();
312  else {
313  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
314  u_tmp_.vector() = u_n.vector();
315  for (size_t jj = 0; jj < ii; ++jj)
316  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * A_[ii][jj];
317  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
318  u_intermediate_stages_[ii].vector() *= r_;
319  }
320 
321  // TODO: If ii == 0, calculate (negative of) jacobian of L and partielle_Ableitung_von_L_nach_t
322  //
323  //
324  //
325 
326  // if gamma is the same for all i, we only need to calculate the matrix in the first step
327  if ((gamma_ii_equal_for_all_i_ && ii == 0) || !gamma_ii_equal_for_all_i_) {
328  // create solver
329  system_matrix_ = jacobian_ * -1.0 * r_;
330  for (size_t row = 0; row < system_matrix_.rows(); ++row)
331  system_matrix_.add_to_entry(row, row, 1.0 / (Gamma_[ii][ii] * actual_dt));
332  solver_ = Dune::XT::Common::make_unique<SolverType>(system_matrix_);
333  }
334  for (size_t jj = 0; jj < ii; ++jj)
335  u_intermediate_stages_[ii].vector().axpy(C_[ii][jj] / actual_dt, u_intermediate_stages_[jj].vector());
336  u_intermediate_stages_[ii].vector() += partielle_Ableitung_von_L_nach_t_ * (r_ * actual_dt * d_[ii]);
337  // solve, TODO: maybe need to do a copy before applying solver?
338  solver_->apply(u_intermediate_stages_[ii].vector(), u_intermediate_stages_[ii].vector());
339  }
340 
341  // compute error vector
342  u_tmp_.vector() = u_intermediate_stages_[0].vector() * m_diff_[0];
343  for (size_t ii = 1; ii < num_stages_; ++ii)
344  u_tmp_.vector() += u_intermediate_stages_[ii].vector() * m_diff_[ii];
345 
346  // calculate u at timestep n+1
347  for (size_t ii = 0; ii < num_stages_; ++ii)
348  u_n.vector() += u_intermediate_stages_[ii].vector() * (m_1_[ii]);
349 
350  // scale error, use absolute error if norm is less than 0.01 and relative error else
351  auto& diff_vector = u_tmp_.vector();
352  for (size_t ii = 0; ii < diff_vector.size(); ++ii) {
353  if (std::abs(u_n.vector()[ii]) > 0.01)
354  diff_vector[ii] /= std::abs(u_n.vector()[ii]);
355  }
356  mixed_error = diff_vector.sup_norm();
357  // scale dt to get the estimated optimal time step length, TODO: adapt formula
358  time_step_scale_factor =
359  std::min(std::max(0.9 * std::pow(tol_ / mixed_error, 1.0 / 4.0), scale_factor_min_), scale_factor_max_);
360 
361  if (mixed_error > tol_) { // go back from u at timestep n+1 to timestep n
362  for (size_t ii = 0; ii < num_stages_; ++ii)
363  u_n.vector() += u_intermediate_stages_[ii].vector() * (-1.0 * m_1_[ii]);
364  }
365  } // while (mixed_error > tol_)
366 
367  t += actual_dt;
368 
369  return actual_dt * time_step_scale_factor;
370  } // ... step(...)
371 
372 private:
373  const OperatorType& op_;
374  const RangeFieldType r_;
375  const RangeFieldType tol_;
376  const RangeFieldType scale_factor_min_;
377  const RangeFieldType scale_factor_max_;
378  DiscreteFunctionType u_tmp_;
379  const MatrixType A_;
380  const VectorType m_1_;
381  const VectorType m_2_;
382  const VectorType c_;
383  const VectorType m_diff_;
384  VectorType d_;
385  MatrixType C_;
386  std::vector<DiscreteFunctionType> u_intermediate_stages_;
387  typename SolverType::MatrixType system_matrix_, jacobian_;
388  typename DiscreteFunctionType::VectorType partielle_Ableitung_von_L_nach_t_;
389  const size_t num_stages_;
390  std::unique_ptr<SolverType> solver_;
391  const MatrixType Gamma_;
392  bool gamma_ii_equal_for_all_i_;
393 };
