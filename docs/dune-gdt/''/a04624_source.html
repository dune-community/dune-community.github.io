83 template <class GridLayerImp, size_t rangeDim>
84 class FvMapper<GridLayerImp, rangeDim, 1> : public MapperInterface<internal::FvMapperTraits<GridLayerImp, rangeDim, 1>>

95  FvMapper(const GridLayerType& grd_layr)
96  : mapper_(new BackendType(grd_layr))

98  }
99 
100  const BackendType& backend() const
101  {
102  return *mapper_;
103  }
104 
105  size_t size() const
106  {
107  return dimRange * mapper_->size();
108  }
109 
110  template <int cd, class GridImp, template <int, int, class> class EntityImp>
111  size_t numDofs(const Entity<cd, EntityType::dimension, GridImp, EntityImp>& /*entity*/) const
112  {
113  return 0;
114  }
115 
116  size_t numDofs(const EntityType& /*entity*/) const
117  {
118  return dimRange;
119  }
120 
121  size_t maxNumDofs() const
122  {
123  return dimRange;
124  }
125 
126  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
127  {
128  if (ret.size() < dimRange)
129  ret.resize(dimRange);
130  const size_t base = dimRange * mapper_->subIndex(entity, 0, 0);
131  for (size_t ii = 0; ii < dimRange; ++ii)
132  ret[ii] = base + ii;
133  } // ... globalIndices(...)
134 
135  using InterfaceType::globalIndices;
136 
137  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
138  {
139  assert(localIndex < dimRange);
140  return (dimRange * mapper_->subIndex(entity, 0, 0)) + localIndex;
141  }
142 
143 private:
144  const std::shared_ptr<BackendType> mapper_;
145 }; // class FvMapper< ..., rangeDim, 1 >
