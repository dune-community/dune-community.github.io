45 template <class AnalyticalFluxType, class MatrixType, class VectorType>
46 class GodunovJacobianWrapper : public JacobianWrapper<AnalyticalFluxType, MatrixType, VectorType>
47 {
48  using BaseType = JacobianWrapper<AnalyticalFluxType, MatrixType, VectorType>;
49  using typename BaseType::V;
50 
51 public:
52  using BaseType::dimDomain;
53  using BaseType::dimRange;
54 
55  GodunovJacobianWrapper()
56  : eigvals_neg_(V::create(dimRange))
57  , eigvals_pos_(V::create(dimRange))
58  {
59  }
60 
61  virtual void compute(const size_t dd) override
62  {
63  BaseType::compute(dd);
64  std::fill(eigvals_neg_[dd].begin(), eigvals_neg_[dd].end(), 0.);
65  std::fill(eigvals_pos_[dd].begin(), eigvals_pos_[dd].end(), 0.);
66  for (size_t ii = 0; ii < eigenvalues_[dd].size(); ++ii)
67  (eigenvalues_[dd][ii] < 0 ? eigvals_neg_[dd][ii] : eigvals_pos_[dd][ii]) = eigenvalues_[dd][ii];
68  }
69 
70  const VectorType& negative_eigenvalues(const size_t direction) const
71  {
72  return eigvals_neg_[direction];
73  }
74 
75  const VectorType& positive_eigenvalues(const size_t direction) const
76  {
77  return eigvals_pos_[direction];
78  }
79 
80 protected:
81  using BaseType::eigenvalues_;
82  using BaseType::computed_;
83  FieldVector<VectorType, dimDomain> eigvals_neg_;
84  FieldVector<VectorType, dimDomain> eigvals_pos_;
85 };

116 template <class Traits>
117 class GodunovFluxImplementation
118 {
119 public:
120  typedef typename Traits::LocalfunctionTupleType LocalfunctionTupleType;
121  typedef typename Traits::EntityType EntityType;
122  typedef typename Traits::DomainFieldType DomainFieldType;
123  typedef typename Traits::RangeFieldType RangeFieldType;
124  typedef typename Traits::AnalyticalFluxType AnalyticalFluxType;
125  typedef typename Traits::DomainType DomainType;
126  typedef typename Traits::RangeType RangeType;
127  static constexpr size_t dimDomain = Traits::dimDomain;
128  static const size_t dimRange = Traits::dimRange;
129  using MatrixType = FieldMatrix<DomainFieldType, dimRange, dimRange>;
130  using VectorType = FieldVector<RangeFieldType, dimRange>;
131  using JacobianWrapperType = GodunovJacobianWrapper<AnalyticalFluxType, MatrixType, VectorType>;
132 
133  explicit GodunovFluxImplementation(const AnalyticalFluxType& analytical_flux,
134  XT::Common::Parameter param,
135  XT::Common::PerThreadValue<JacobianWrapperType>& jacobian_wrapper,
136  const bool boundary = false)
137  : analytical_flux_(analytical_flux)
138  , param_inside_(param)
139  , param_outside_(param)
140  , jacobian_wrapper_(jacobian_wrapper)
141  {
142  param_inside_.set("boundary", {0.}, true);
143  param_outside_.set("boundary", {double(boundary)}, true);
144  }
145 
146  template <class IntersectionType>
147  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple,
148  const IntersectionType& intersection,
149  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
150  const DomainType& x_in_inside_coords,
151  const DomainType& /*x_in_outside_coords*/,
152  const RangeType& u_i,
153  const RangeType& u_j) const
154  {
155  size_t direction = intersection.indexInInside() / 2;
156  // get unit outer normal
157  const auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
158  assert(XT::Common::FloatCmp::eq(std::abs(n_ij[direction]), 1.));
159 
160  // intialize jacobians
161  auto& jac = *jacobian_wrapper_;
162  initialize_jacobians(jac, direction, local_functions_tuple, x_in_inside_coords, u_i, u_j);
163 
164  // get jump at the intersection
165  const RangeType delta_u = u_i - u_j;
166 
167  // calculate waves
168 
169  // apply A D_{+,-} A^{-1}, where A is the matrix of eigenvectors and D_{+/-} is the diagonal matrix containing only
170  // positive/negative eigenvalues of A
171  RangeType waves, tmp_vec;
172  jac.apply_inverse_eigenvectors(direction, delta_u, tmp_vec);
173  const auto& eigenvalues =
174  n_ij[direction] > 0 ? jac.negative_eigenvalues(direction) : jac.positive_eigenvalues(direction);
175  for (size_t ii = 0; ii < dimRange; ++ii)
176  tmp_vec[ii] *= eigenvalues[ii];
177  jac.apply_eigenvectors(direction, tmp_vec, waves);
178 
179  // calculate flux
180  const auto& local_flux = std::get<0>(local_functions_tuple);
181  RangeType ret = local_flux->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
182  ret -= waves;
183  ret *= n_ij[direction];
184  return ret;
185  } // RangeType evaluate(...) const
186 
187  const AnalyticalFluxType& analytical_flux() const
188  {
189  return analytical_flux_;
190  }
191 
192 private:
193  // use simple linearized Riemann solver, LeVeque p.316
194  void initialize_jacobians(JacobianWrapperType& jac,
195  const size_t direction,
196  const LocalfunctionTupleType& local_functions_tuple,
197  const DomainType& x_local,
198  const RangeType& u_i,
199  const RangeType& u_j) const
200  {
201  // get jacobian
202  if (!jac.computed(direction) || !analytical_flux_.is_affine()) {
203  // calculate jacobian as jacobian(0.5*(u_i+u_j)
204  auto u_mean = (u_i + u_j) * 0.5;
205  const auto& local_flux = std::get<0>(local_functions_tuple);
206  local_flux->partial_u_col(direction, x_local, u_mean, jac.jacobian(direction), param_inside_);
207  jac.compute(direction);
208  }
209  } // void initialize_jacobians(...)
210 
211  const AnalyticalFluxType& analytical_flux_;
212  XT::Common::Parameter param_inside_;
213  XT::Common::Parameter param_outside_;
214  XT::Common::PerThreadValue<JacobianWrapperType>& jacobian_wrapper_;
215 }; // class GodunovFluxImplementation
