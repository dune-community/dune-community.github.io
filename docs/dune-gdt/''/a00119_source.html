88  // scalar version
89  void apply_limiter(const EntityType& entity,
90  const RangeFieldType theta_entity,
91  LocalReconstructedValuesType& local_reconstructed_values,
92  const RangeType& u_bar,
93  bool add_epsilon = true)
94  {
95  // apply slope limiter
96  assert(dynamic_cast<const EntropyFluxType*>(&analytical_flux_) != nullptr
97  && "analytical_flux_ has to be derived from EntropyBasedLocalFlux");
98  auto theta = add_epsilon ? theta_entity + epsilon_ : theta_entity;
99  if (theta > 0.) {
100  // std::cout << "limited with theta: " << theta << " and epsilon " << epsilon_ << std::endl;
101  if (theta > 1.)
102  theta = 1.;
103  for (auto& pair : local_reconstructed_values) {
104  auto& u = pair.second;
105  u = convex_combination(u, u_bar, theta);
106  }
107  }
108  check_solvability(entity, local_reconstructed_values, u_bar);
109  } // void apply_limiter(...)

111  // version for component-wise limiting
112  void apply_limiter(const EntityType& entity,
113  const RangeType& theta_entity,
114  LocalReconstructedValuesType& local_reconstructed_values,
115  const RangeType& u_bar,
116  bool add_epsilon = true)
117  {
118  // apply slope limiter
119  assert(dynamic_cast<const EntropyFluxType*>(&analytical_flux_) != nullptr
120  && "analytical_flux_ has to be derived from EntropyBasedLocalFlux");
121  for (size_t ii = 0; ii < dimRange; ++ii) {
122  auto theta_ii = add_epsilon ? theta_entity[ii] + epsilon_ : theta_entity[ii];
123  if (theta_ii > 0.) {
124  // std::cout << "limited with theta: " << theta_ii << " and epsilon " << epsilon_ << std::endl;
125  if (theta_ii > 1.)
126  theta_ii = 1.;
127  for (auto& pair : local_reconstructed_values) {
128  auto& u_ii = pair.second[ii];
129  u_ii = convex_combination(u_ii, u_bar[ii], theta_ii);
130  }
131  } // if (theta_ii > 0)
132  } // ii
133  check_solvability(entity, local_reconstructed_values, u_bar);
134  }
135 
136 protected:
137  void check_solvability(const EntityType& entity,
138  LocalReconstructedValuesType& local_reconstructed_values,
139  const RangeType& u_bar)
140 
141  {
142  // Try to solve optimization problem for all reconstructed values. If it fails for one value,
143  // we cannot guarantee realizability preservation, so disable reconstruction in that case.
144  const auto local_func = dynamic_cast<const EntropyFluxType*>(&analytical_flux_)->derived_local_function(entity);
145  try {
146  for (const auto& pair : local_reconstructed_values) {
147  const auto x_in_inside_coords = entity.geometry().local(pair.first);
148  const auto& u = pair.second;
149  local_func->get_alpha(x_in_inside_coords, u, param_, false, false);
150  } // local_reconstructed_values
151  } catch (const Dune::MathError&) {
152  // solving failed for reconstructed value, so check that it works with u_bar ...
153  local_func->get_alpha(entity.geometry().local(entity.geometry().center()), u_bar, param_, false, false);
154  // ... and set all reconstructed values to u_bar
155  for (auto& pair : local_reconstructed_values)
156  pair.second = u_bar;
157  std::cout << "Reconstruction disabled at time " << XT::Common::to_string(param_.get("t")[0], 15)
158  << " and entity with center " << XT::Common::to_string(entity.geometry().center(), 15) << std::endl;
159  }
160  }
161 
162  RangeType convex_combination(const RangeType& u, const RangeType& u_bar, const RangeFieldType& theta)
163  {
164  RangeType u_scaled = u;
165  u_scaled *= 1 - theta;
166  RangeType u_bar_scaled = u_bar;
167  u_bar_scaled *= theta;
168  return u_scaled + u_bar_scaled;
169  }
170 
171  RangeFieldType convex_combination(const RangeFieldType& u, const RangeFieldType& u_bar, const RangeFieldType& theta)
172  {
173  return u_bar * theta + u * (1. - theta);
174  }

182  const std::vector<RangeType>& basis_values_;
183  XT::Common::Parameter param_;

241  const RangeType u_bar =
242  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
243 
244  // vector to store thetas for each local reconstructed value
245  RangeType thetas(0.);
246 
247  for (const auto& pair : local_reconstructed_values) {

256  } // ii
257  } // ll
258 
259  const auto theta_max = *std::max_element(thetas.begin(), thetas.end());
260  BaseType::apply_limiter(entity, theta_max, local_reconstructed_values, u_bar, false);
261  } // void apply_local(...)

266  using BaseType::epsilon_;

270 class DgLocalRealizabilityLimiter
271  : public LocalRealizabilityLimiterBase<AnalyticalFluxImp, DiscreteFunctionImp, BasisfunctionImp>
272 {
273  using BaseType = LocalRealizabilityLimiterBase<AnalyticalFluxImp, DiscreteFunctionImp, BasisfunctionImp>;
274 
275 public:
276  using typename BaseType::AnalyticalFluxType;
277  using typename BaseType::DiscreteFunctionType;
278  using typename BaseType::EntityType;
279  using typename BaseType::GridLayerType;
280  using typename BaseType::QuadratureType;
281  using typename BaseType::RangeType;
282  using typename BaseType::RangeFieldType;
283  using typename BaseType::ReconstructedFunctionType;
284  using typename BaseType::BasisfunctionType;
285  static const size_t dimDomain = BaseType::dimDomain;
286  static const size_t dimRange = BaseType::dimRange;
287 
288  template <class... Args>
289  DgLocalRealizabilityLimiter(Args&&... args)
290  : BaseType(std::forward<Args>(args)...)
291  , triangulation_(basis_functions_.triangulation())
292  {
293  }
294 
295  void apply_local(const EntityType& entity)
296  {
297  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
298 
299  // get cell average
300  const RangeType u_bar =
301  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
302 
303  // vector to store thetas for each local reconstructed value
304  RangeType thetas(0.);
305 
306  for (const auto& pair : local_reconstructed_values) {
307  const auto& u = pair.second;
308  for (size_t ii = 0; ii < dimRange / 2; ++ii) {
309  const auto& u0 = u[2 * ii];
310  const auto& u1 = u[2 * ii + 1];
311  const auto& ubar0 = u_bar[2 * ii];
312  const auto& ubar1 = u_bar[2 * ii + 1];
313  const auto& vj = triangulation_[ii];
314  const auto& vjplus1 = triangulation_[ii + 1];
315  FieldVector<RangeFieldType, 3> thetas_ii;
316  if (!is_epsilon_realizable(ubar0, ubar1, vj, vjplus1, epsilon_)) {
317  thetas[2 * ii] = 1.;
318  } else {
319  thetas_ii[0] = (epsilon_ - u0) / (ubar0 - u0);
320  thetas_ii[1] =
321  (u0 * vj - u1 + epsilon_ * std::sqrt(std::pow(vj, 2) + 1)) / ((ubar1 - u1) - (ubar0 - u0) * vj);
322  thetas_ii[2] = (u0 * vjplus1 - u1 - epsilon_ * std::sqrt(std::pow(vjplus1, 2) + 1))
323  / ((ubar1 - u1) - (ubar0 - u0) * vjplus1);
324  for (size_t kk = 0; kk < 3; ++kk)
325  if (thetas_ii[kk] >= 0. && thetas_ii[kk] <= 1.)
326  thetas[2 * ii] = std::max(thetas[2 * ii], thetas_ii[kk]);
327  } // else (!realizable)
328  thetas[2 * ii + 1] = thetas[2 * ii];
329  } // ii
330  } // local_reconstructed_values
331  BaseType::apply_limiter(entity, thetas, local_reconstructed_values, u_bar, false);
332  } // void apply_local(...)
333 
334 private:
335  bool is_epsilon_realizable(const RangeFieldType ubar0,
336  const RangeFieldType ubar1,
337  const RangeFieldType v0,
338  const RangeFieldType v1,
339  const RangeFieldType eps) const
340  {
341  bool ret = (ubar0 >= eps) && (ubar1 <= v1 * ubar0 - eps * std::sqrt(std::pow(v1, 2) + 1))
342  && (v0 * ubar0 + eps * std::sqrt(std::pow(v0, 2) + 1) <= ubar1);
343  return ret;
344  }

347  using BaseType::epsilon_;
348  using BaseType::reconstructed_function_;
349  using BaseType::source_;
350  typename BasisfunctionImp::TriangulationType triangulation_;
351 }; // class DgLocalRealizabilityLimiter

393  void apply_local(const EntityType& entity)
394  {
395  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
396 
397  // get cell average
398  const RangeType u_bar =
399  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
400 
401  // vector to store thetas for each local reconstructed value
402  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
403 
404  size_t ll = -1;
405  for (const auto& pair : local_reconstructed_values) {
406  ++ll;
407  // rescale u_l, u_bar
408  auto u_l = pair.second;
409  auto u_bar_minus_u_l = u_bar - u_l;
410  const auto factor = std::max(basis_functions_.density(u_l), basis_functions_.density(u_bar)) / 2.;
411  u_l /= factor;
412  u_bar_minus_u_l /= factor;
413 
414  for (const auto& coeffs : *plane_coefficients_) {
415  const RangeType& a = coeffs.first;
416  const RangeFieldType& b = coeffs.second;
417  RangeFieldType theta_li = (b - a * u_l) / (a * u_bar_minus_u_l);
418  if (XT::Common::FloatCmp::le(theta_li, 1.))
419  thetas[ll] = std::max(thetas[ll], theta_li);
420  } // coeffs
421  } // ll
422  for (auto& theta : thetas)
423  theta = std::min(epsilon_ + theta, 1.);
424 
425  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
426  if (theta_entity > 0.) {
427  for (auto& pair : local_reconstructed_values) {
428  auto& u = pair.second;
429  auto u_scaled = u;
430  u_scaled *= (1 - theta_entity);
431  auto u_bar_scaled = u_bar;
432  u_bar_scaled *= theta_entity;
433  u = u_scaled + u_bar_scaled;
434  }
435  }
436  } // void apply_local(...)

438 private:
439  // calculate half space representation of realizable set
440  void calculate_plane_coefficients()
441  {
442  using orgQhull::Qhull;
443  Qhull qhull;
444  std::vector<FieldVector<RangeFieldType, dimRange>> points(quadrature_.size() + 1);
445  points[0] = FieldVector<RangeFieldType, dimRange>(0);
446  size_t ii = 1;
447  for (const auto& quad_point : quadrature_)
448  points[ii++] = basis_functions_.evaluate(quad_point.position());
449 
450  std::cout << "Starting qhull..." << std::endl;
451  qhull.runQhull("Realizable set", int(dimRange), int(points.size()), &(points[0][0]), "Qt T1");
452  std::cout << "qhull done" << std::endl;
453  // qhull.outputQhull("n");
454  const auto facet_end = qhull.endFacet();
455  plane_coefficients_ = std::make_shared<PlaneCoefficientsType>(qhull.facetList().count());
456  ii = 0;
457  for (auto facet = qhull.beginFacet(); facet != facet_end; facet = facet.next(), ++ii) {
458  for (size_t jj = 0; jj < dimRange; ++jj)
459  (*plane_coefficients_)[ii].first[jj] = *(facet.hyperplane().coordinates() + jj);
460  (*plane_coefficients_)[ii].second = -facet.hyperplane().offset();
461  }
462  }

526  void apply_local(const EntityType& entity)
527  {
528  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
529 
530  // get cell average
531  const RangeType u_bar =
532  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
533 
534  // vector to store thetas for each local reconstructed value
535  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
536 
537  size_t ll = -1;
538  for (const auto& pair : local_reconstructed_values) {
539  ++ll;
540  // rescale u_l, u_bar
541  auto u_l = pair.second;
542  if (XT::Common::FloatCmp::eq(u_l, u_bar))
543  continue;
544  const auto factor = std::max(basis_functions_.density(u_l), basis_functions_.density(u_bar)) / 2.;
545  u_l /= factor;
546  auto u_bar_scaled = u_bar;
547  u_bar_scaled /= factor;
548  // check positivity of first moment on each spherical triangle
549  for (size_t jj = 0; jj < num_blocks; ++jj) {
550  const size_t offset = jj * block_size;
551  if (u_l[offset] >= u_bar_scaled[offset])
552  continue;
553  thetas[ll] = std::max(thetas[ll], u_l[offset] / (u_l[offset] - u_bar_scaled[offset]));
554  }
555  // check convex hull property of other moments
556  for (size_t jj = 0; jj < num_blocks; ++jj) {
557  const size_t offset = jj * block_size;
558  BlockRangeType q_k, q_bar_k;
559  for (size_t mm = 0; mm < block_size - 1; ++mm) {
560  q_k[mm] = u_l[offset + mm + 1];
561  q_bar_k[mm] = u_bar_scaled[offset + mm + 1];
562  }
563  for (const auto& coeffs : (*plane_coefficients_)[jj]) {
564  const BlockRangeType& a = coeffs.first;
565  const RangeFieldType& b = coeffs.second;
566  const auto u0 = u_l[offset];
567  const auto ubar0 = u_bar_scaled[offset];
568  RangeFieldType theta_li = (b * u0 - a * q_k) / (a * (q_bar_k - q_k) - b * (ubar0 - u0));
569  if (!(theta_li > 1.))
570  thetas[ll] = std::max(thetas[ll], theta_li);
571  } // coeffs
572  } // jj
573  } // ll
574 
575  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
576  BaseType::apply_limiter(entity, theta_entity, local_reconstructed_values, u_bar);
577  } // void apply_local(...)

580  // calculate half space representation of realizable set
581  void calculate_plane_coefficients()
582  {
583  plane_coefficients_ = std::make_shared<PlaneCoefficientsType>();
584  FieldVector<std::vector<FieldVector<RangeFieldType, block_size - 1>>, num_blocks> points;
585  FieldVector<QuadratureType, num_blocks> blocked_quadrature;
586  for (const auto& quad_point : quadrature_) {
587  const auto face_indices = basis_functions_.get_face_indices(quad_point.position());
588  const size_t num_adjacent_faces = face_indices.size();
589  for (const auto& kk : face_indices)
590  blocked_quadrature[kk].emplace_back(quad_point.position(), quad_point.weight() / num_adjacent_faces);
591  } // ii
592  size_t num_faces;
593  for (size_t jj = 0; jj < num_blocks; ++jj) {
594  points[jj].resize(blocked_quadrature[jj].size() + 1);
595  for (size_t ii = 0; ii < blocked_quadrature[jj].size(); ++ii) {
596  const auto val = basis_functions_.evaluate(blocked_quadrature[jj][ii].position(), false, num_faces);
597  for (size_t ll = 0; ll < block_size - 1; ++ll)
598  points[jj][ii][ll] = val[block_size * jj + 1 + ll];
599  } // ii
600  points[jj][blocked_quadrature[jj].size()] = FieldVector<RangeFieldType, block_size - 1>(0.);
601  }
602  std::vector<std::thread> threads(num_blocks);
603  // Launch a group of threads
604  for (size_t jj = 0; jj < num_blocks; ++jj)
605  threads[jj] = std::thread(&ThisType::calculate_plane_coefficient_block, this, std::ref(points[jj]), jj);
606  // Join the threads with the main thread
607  for (size_t jj = 0; jj < num_blocks; ++jj)
608  threads[jj].join();
609  }

690  void apply_local(const EntityType& entity)
691  {
692  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
693  assert(local_reconstructed_values.size() == 2 * dimDomain);
694 
695  // get cell average
696  const RangeType u_bar =
697  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
698 
699  RangeFieldType theta_entity(-epsilon_);
700 
701  for (const auto& pair : local_reconstructed_values) {
702  const auto& u = pair.second;
703  if (XT::Common::FloatCmp::eq(u_bar, u))
704  continue;
705 
706  // solve LP:
707  // min \theta s.t.
708  // (\sum x_i v_i) + \theta (u - \bar{u}) = u
709  // x_i, \theta >= 0
710  theta_entity = std::max(theta_entity, solve_linear_program(u_bar, u));
711  } // ll
712  BaseType::apply_limiter(entity, theta_entity, local_reconstructed_values, u_bar, true);
713  } // void apply_local(...)

715 private:
716  void setup_linear_program()
717  {
718  if (!*lp_) {
719  // We start with creating a model with dimRange rows and num_quad_points+1 columns */
720  constexpr int num_rows = static_cast<int>(dimRange);
721  assert(quadrature_.size() < std::numeric_limits<int>::max());
722  int num_cols = static_cast<int>(quadrature_.size() + 1); /* variables are x_1, ..., x_{num_quad_points}, theta */
723  *lp_ = std::make_unique<ClpSimplex>(false);
724  auto& lp = **lp_;
725  // set number of rows
726  lp.resize(num_rows, 0);
727 
728  // Clp wants the row indices that are non-zero in each column. We have a dense matrix, so provide all indices
729  // 0..num_rows
730  std::array<int, num_rows> row_indices;
731  for (int ii = 0; ii < num_rows; ++ii)
732  row_indices[ii] = ii;
733 
734  // set columns for quadrature points
735  assert(int(basis_values_.size()) == num_cols - 1);
736  for (int ii = 0; ii < num_cols - 1; ++ii) {
737  const auto& v_i = basis_values_[ii];
738  // First argument: number of elements in column
739  // Second/Third argument: indices/values of column entries
740  // Fourth/Fifth argument: lower/upper column bound, i.e. lower/upper bound for x_i. As all x_i should be
741  // positive, set to 0/inf, which is the default.
742  // Sixth argument: Prefactor in objective for x_i, this is 0 for all x_i, which is also the default;
743  lp.addColumn(num_rows, row_indices.data(), &(v_i[0]));
744  }
745 
746  // add theta column (set to random values, will be set correctly in solve_linear_program)
747  // The bounds for theta should be [0,1], but we allow allow theta to be slightly
748  // negative so we can check if u_l is on the boundary and if so, move it a little
749  // away from the boundary
750  // Also sets the prefactor in the objective to 1 for theta.
751  lp.addColumn(num_rows, row_indices.data(), &(basis_values_[0][0]), -0.1, 1., 1.);
752  lp.setLogLevel(0);
753  } // if (!lp_)
754  }

756  // RangeFieldType solve_linear_program(const RangeType& u_bar, const RangeType& u_l, const size_t index)
757  RangeFieldType solve_linear_program(const RangeType& u_bar, const RangeType& u_l)
758  {
759  setup_linear_program();
760  auto& lp = **lp_;
761  constexpr int num_rows = static_cast<int>(dimRange);
762  int num_cols = static_cast<int>(quadrature_.size() + 1); /* variables are x_1, ..., x_{num_quad_points}, theta */
763  RangeFieldType theta;
764  const auto u_l_minus_u_bar = u_l - u_bar;
765 
766  // set rhs (equality constraints, so set both bounds equal
767  for (int ii = 0; ii < num_rows; ++ii) {
768  lp.setRowLower(ii, u_l[ii]);
769  lp.setRowUpper(ii, u_l[ii]);
770  }
771 
772  // Clp wants the row indices that are non-zero in each column. We have a dense matrix, so provide all indices
773  // 0..num_rows
774  std::array<int, num_rows> row_indices;
775  for (int ii = 0; ii < num_rows; ++ii)
776  row_indices[ii] = ii;
777 
778  // delete and reset theta column
779  const int last_col = num_cols - 1;
780  lp.deleteColumns(1, &last_col);
781  lp.addColumn(num_rows, row_indices.data(), &(u_l_minus_u_bar[0]), -0.1, 1., 1.);
782 
783  // Now solve
784  lp.primal();
785  theta = lp.objectiveValue();
786  if (!lp.isProvenOptimal())
787  theta = 1.;
788 
789  return theta;
790  }
