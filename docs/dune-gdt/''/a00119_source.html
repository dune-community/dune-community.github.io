160 template <class IntersectionImp, class LocalizableFunctionType>
161 class LocalMomentModelBoundaryValue : public LocalBoundaryValueInterface<typename LocalizableFunctionType::EntityType,
162  IntersectionImp,
163  typename LocalizableFunctionType::RangeType>
164 {
165  using BaseType = LocalBoundaryValueInterface<typename LocalizableFunctionType::EntityType,
166  IntersectionImp,
167  typename LocalizableFunctionType::RangeType>;
168  using RangeFieldType = typename LocalizableFunctionType::RangeFieldType;
169 
170 public:
171  using MatrixType = typename Dune::XT::LA::CommonSparseOrDenseMatrixCsr<RangeFieldType>;
172  using typename BaseType::BoundaryInfoType;
173  using typename BaseType::DomainType;
174  using typename BaseType::IntersectionType;
175  using typename BaseType::RangeType;
176  using BaseType::dimDomain;
177 
178  LocalMomentModelBoundaryValue(
179  const BoundaryInfoType& boundary_info,
180  const XT::Common::FieldVector<MatrixType, 2 * dimDomain>& reflection_matrices,
181  std::unique_ptr<const typename LocalizableFunctionType::LocalfunctionType>&& local_dirichlet_boundary_values)
182  : BaseType(boundary_info)
183  , reflection_matrices_(reflection_matrices)
184  , local_dirichlet_boundary_values_(std::move(local_dirichlet_boundary_values))
185  {
186  }
187 
188  virtual RangeType evaluate(const IntersectionType& intersection,
189  const DomainType& x,
190  const RangeType& u,
191  const XT::Common::Parameter& /*param*/ = {}) const override
192  {
193  RangeType ret(0);
194  const auto& boundary_type = boundary_info_.type(intersection);
195  if (boundary_type == XT::Grid::absorbing_boundary)
196  ret = u;
197  else if (boundary_type == XT::Grid::dirichlet_boundary)
198  ret = local_dirichlet_boundary_values_->evaluate(x);
199  else if (boundary_type == XT::Grid::reflecting_boundary) {
200  assert(intersection.indexInInside() >= 0);
201  const auto direction = static_cast<size_t>(intersection.indexInInside());
202  reflection_matrices_[direction].mv(u, ret);
203  } else
204  DUNE_THROW(Dune::NotImplemented,
205  "Boundary types other than Absorbing, Dirichlet or Reflecting are not implemented!");
206  return ret;
207  }
208 
209 private:
210  using BaseType::boundary_info_;
211  const XT::Common::FieldVector<MatrixType, 2 * dimDomain>& reflection_matrices_;
212  const std::unique_ptr<const typename LocalizableFunctionType::LocalfunctionType> local_dirichlet_boundary_values_;
213 };

217 class MomentModelBoundaryValue : public LocalizableBoundaryValueInterface<typename LocalizableFunctionType::EntityType,
218  IntersectionType,
219  typename BasisfunctionType::RangeType>
220 {
221  static_assert(std::is_same<typename BasisfunctionType::RangeType, typename LocalizableFunctionType::RangeType>::value,
222  "RangeTypes have to match!");
223  using BaseType = LocalizableBoundaryValueInterface<typename LocalizableFunctionType::EntityType,
224  IntersectionType,
225  typename BasisfunctionType::RangeType>;
226 
227 public:
228  using typename BaseType::BoundaryInfoType;
229  using typename BaseType::DomainType;
230  using typename BaseType::EntityType;
231  using typename BaseType::LocalBoundaryValueInterfaceType;
232  using typename BaseType::RangeType;
233  using LocalBoundaryValueType = LocalMomentModelBoundaryValue<IntersectionType, LocalizableFunctionType>;
234  using MatrixType = typename LocalBoundaryValueType::MatrixType;
235  static const size_t dimDomain = BasisfunctionType::dimDomain;
236  static const size_t dimRange = BasisfunctionType::dimRange;
237 
238  MomentModelBoundaryValue(std::unique_ptr<const BoundaryInfoType>&& boundary_info,
239  const BasisfunctionType& basis_functions,
240  std::unique_ptr<const LocalizableFunctionType>&& dirichlet_boundary_values)
241  : BaseType(std::move(boundary_info))
242  , dirichlet_boundary_values_(std::move(dirichlet_boundary_values))
243  {
244  for (size_t ii = 0; ii < dimDomain; ++ii) {
245  DomainType n(0);
246  n[ii] = -1.;
247  reflection_matrices_[2 * ii] = MatrixType(basis_functions.reflection_matrix(n));
248  n[ii] = 1.;
249  reflection_matrices_[2 * ii + 1] = MatrixType(basis_functions.reflection_matrix(n));
250  }
251  }
252 
253  virtual std::unique_ptr<LocalBoundaryValueInterfaceType> local_function(const EntityType& entity) const override
254  {
255  return std::make_unique<LocalBoundaryValueType>(
256  *boundary_info_, reflection_matrices_, dirichlet_boundary_values_->local_function(entity));
257  }
258 
259 private:
260  using BaseType::boundary_info_;
261  Dune::XT::Common::FieldVector<MatrixType, 2 * dimDomain> reflection_matrices_;
262  const std::unique_ptr<const LocalizableFunctionType> dirichlet_boundary_values_;
263 };
