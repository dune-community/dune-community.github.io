88  virtual void apply_local(const EntityType& entity) override final
89  {
90  static const CoordsType stencil_sizes = []() {
91  CoordsType ret;
92  const auto ax_size = axis_size; // avoid linker error
93  ret.fill(ax_size);
94  return ret;
95  }();
96  thread_local StencilType stencil(stencil_sizes);
97  bool valid = fill_stencil(stencil, entity);
98  // In a MPI parallel run, if entity is on boundary of overlap, we do not have to reconstruct
99  if (!valid)
100  return;
101  // get intersections
102  FieldVector<typename GridLayerType::Intersection, 2 * dimDomain> intersections;
103  for (const auto& intersection : Dune::intersections(grid_layer_, entity)) {
104  const size_t index = static_cast<size_t>(intersection.indexInInside());
105  intersections[index] = intersection;
106  }
107  const auto entity_index = grid_layer_.indexSet().index(entity);
108  auto& reconstructed_values_map = reconstructed_function_.values()[entity_index];
109 
110  // get jacobian
111  auto& jac = *jacobian_wrapper_;
112  if (!jac.computed() || !analytical_flux_.is_affine()) {
113  const auto& u_entity = source_values_[entity_index];
114  const DomainType x_in_inside_coords = entity.geometry().local(entity.geometry().center());
115  jac.get_jacobian(entity, analytical_flux_, x_in_inside_coords, u_entity, param_);
116  jac.compute();
117  if (analytical_flux_.is_affine())
118  jac.jacobian() = nullptr;
119  }
120 
121  for (size_t dd = 0; dd < dimDomain; ++dd) {
122  if (quadrature_.size() == 1) {
123  // no need to reconstruct in all directions, as we are only regarding the center of the face, which will always
124  // have the same value assigned, independent of the slope in the other directions
125  std::array<size_t, dimDomain> indices;
126  indices.fill(1);
127  FieldVector<VectorType, axis_size> stencil_1d, stencil_1d_char;
128  FieldVector<VectorType, 2> reconstructed_values;
129  for (size_t ii = 0; ii < axis_size; ++ii) { // transform to characteristic variables
130  indices[dd] = ii;
131  stencil_1d[ii] = *stencil(indices);
132  jac.apply_inverse_eigenvectors(dd, stencil_1d[ii], stencil_1d_char[ii]);
133  }
134  // perform the actual reconstruction
135  linear_reconstruction_1d(stencil_1d, stencil_1d_char, reconstructed_values, jac.eigenvectors(dd));
136 
137  // convert back to non-characteristic variables
138  auto tmp_value = reconstructed_values[0];
139  jac.apply_eigenvectors(dd, tmp_value, reconstructed_values[0]);
140  tmp_value = reconstructed_values[1];
141  jac.apply_eigenvectors(dd, tmp_value, reconstructed_values[1]);
142 
143  // store reconstructed values
144  reconstructed_values_map.emplace(intersections[2 * dd].geometryInInside().center(), reconstructed_values[0]);
145  reconstructed_values_map.emplace(intersections[2 * dd + 1].geometryInInside().center(),
146  reconstructed_values[1]);
147  } else {
148  thread_local MultiArrayType reconstructed_values(stencil_sizes);
149  thread_local auto tmp_multiarray = reconstructed_values;
150  tmp_multiarray.resize(stencil_sizes);
151 
152  // Transform values on stencil to characteristic variables of the current coordinate direction dd
153  for (size_t ii = 0; ii < stencil.num_elements(); ++ii)
154  jac.apply_inverse_eigenvectors(dd, *stencil.data()[ii], tmp_multiarray.data()[ii]);
155 
156  size_t curr_dir = dd;
157  size_t last_dir = curr_dir;
158  VectorType tmp_value;
159  CoordsType current_sizes;
160  for (size_t dir = 0; dir < dimDomain; ++dir) {
161  curr_dir = (dd + dir) % dimDomain;
162  // Transform to characteristic variables of the current reconstruction direction.
163  if (dir > 0) {
164  std::copy_n(reconstructed_values.shape(), dimDomain, current_sizes.begin());
165  tmp_multiarray.resize(current_sizes);
166  tmp_multiarray = reconstructed_values;
167  std::for_each(
168  tmp_multiarray.data(), tmp_multiarray.data() + tmp_multiarray.num_elements(), [&](VectorType& value) {
169  jac.apply_eigenvectors(last_dir, value, tmp_value);
170  jac.apply_inverse_eigenvectors(curr_dir, tmp_value, value);
171  });
172  } // if (dir > 0)
173  // perform the actual reconstruction
174  const auto& curr_quadrature = dir > 0 ? quadrature_ : get_left_right_quadrature();
175  linear_reconstruction(
176  curr_dir, curr_quadrature, tmp_multiarray, reconstructed_values, jac.eigenvectors(curr_dir));
177  last_dir = curr_dir;
178  } // dir
179  // convert back to non-characteristic variables
180  std::for_each(reconstructed_values.data(),
181  reconstructed_values.data() + reconstructed_values.num_elements(),
182  [&](VectorType& value) {
183  tmp_value = value;
184  jac.apply_eigenvectors(last_dir, tmp_value, value);
185  });
186 
187  // Convert coordinates on face to local entity coordinates and store reconstructed values
188  internal::MultiIndexProvider<MultiArrayType> multi_indices(reconstructed_values);
189  for (const auto& multi_index : multi_indices) {
190  IntersectionLocalCoordType quadrature_point;
191  for (size_t ii = 0; ii < dimDomain; ++ii)
192  if (ii != dd)
193  quadrature_point[ii < dd ? ii : ii - 1] = quadrature_[multi_index[ii]].position();
194  reconstructed_values_map.emplace(
195  intersections[2 * dd + multi_index[dd]].geometryInInside().global(quadrature_point),
196  reconstructed_values(multi_index));
197  } // multi_indices
198  }
199  } // dd
200  } // void apply_local(...)

218  const Quadrature1dType& quadrature,
219  const MultiArrayType& stencil,
220  MultiArrayType& reconstructed_values,

226  std::copy_n(stencil.shape(), dimDomain, new_shape.begin());
227  new_shape[dir] = quadrature.size();

230  // get array_views corresponding to the left, center and right values

238  ranges[ii] = IndexRangeType(0, static_cast<long>(stencil.shape()[ii]));
239  ranges[dir] = IndexRangeType(0);

245 
246  // calculate slopes
247  thread_local SliceType slope;
248  std::array<size_t, dimDomain - 1> slope_sizes;
249  std::copy_n(u_entity.shape(), dimDomain - 1, slope_sizes.begin());
250  slope.resize(slope_sizes);
251  internal::MultiIndexProvider<decltype(u_left)> multi_indices(u_left);
252  // Todo: adapt to new slopes
253  // for (const auto& multi_index : multi_indices) {
254  // slope(multi_index) = slope_limiter_.get(u_left(multi_index), u_entity(multi_index),
255  // u_right(multi_index), eigenvectors);
256  // }
257 
258  // calculate reconstructed values
259  for (size_t ii = 0; ii < dimDomain; ++ii)
260  ranges[ii] = IndexRangeType(0, static_cast<long>(new_shape[ii]));
261  for (size_t ii = 0; ii < quadrature.size(); ++ii) {
262  ranges[dir] = IndexRangeType(static_cast<long>(ii));
263  auto reconstructed_ii = reconstructed_values[IndicesBuilderType::build(ranges)];
264  for (const auto& multi_index : multi_indices) {
265  reconstructed_ii(multi_index) = slope(multi_index);
266  reconstructed_ii(multi_index) *= (quadrature[ii].position() - 0.5);
267  reconstructed_ii(multi_index) += u_entity(multi_index);
268  }
269  } // ii (quadrature.size())
270  } // void linear_reconstruction(...)

283  bool fill_stencil(StencilType& stencil, const EntityType& entity)
284  {
285  const int dir = -2;
286  auto coords = center(stencil);

291 private:
292  bool fill_impl(StencilType& stencil, const EntityType& entity, const int dir, const CoordsType& coords)
293  {
294  bool ret = true;
295  const auto entity_index = grid_layer_.indexSet().index(entity);
296  stencil(coords) = source_values_[entity_index];
297  std::vector<int> boundary_dirs;
298  for (const auto& intersection : Dune::intersections(grid_layer_, entity)) {
299  const auto new_dir = intersection.indexInInside();
300  if (direction_allowed(dir, new_dir) && !end_of_stencil(stencil, new_dir, coords)) {
301  auto new_coords = coords;
302  if (intersection.boundary() && !intersection.neighbor()) { // boundary intersections
303  boundary_dirs.push_back(new_dir);
304  auto boundary_value = boundary_values_.local_function(entity)->evaluate(
305  intersection, entity.geometry().local(intersection.geometry().center()), source_values_[entity_index]);
306  while (!end_of_stencil(stencil, new_dir, new_coords)) {
307  next_coords_in_dir(new_dir, new_coords);
308  stencil(new_coords) = boundary_value;
309  }
310  } else if (intersection.neighbor()) { // inner and periodic intersections
311  const auto& outside = intersection.outside();
312  next_coords_in_dir(new_dir, new_coords);
313  ret = ret && fill_impl(stencil, outside, new_dir, new_coords);
314  } else if (!intersection.neighbor() && !intersection.boundary()) { // processor boundary
315  return false;
316  }
317  } // if (!end_of_stencil(...))
318  } // intersections
319 
320  assert(boundary_dirs.size() <= dimDomain);
321  if (boundary_dirs.size() > 1) {
322  auto new_coords = coords;
323  next_coords_in_dir(boundary_dirs[0], new_coords);
324  const auto& boundary_value = stencil(new_coords);
325 
326  std::for_each(stencil.data(),
327  stencil.data() + stencil.num_elements(),
328  [&boundary_value](boost::optional<VectorType>& value) {
329  if (!value)
330  value = boundary_value;
331  });
332  } // if (boundary_dirs.size() > 1)
333  return ret;
334  }
335 
336  // get next coords in direction dir (increase or decrease coords in that direction)
337  static void next_coords_in_dir(const int dir, CoordsType& coords)
338  {
339  assert(dir >= 0.);
340  size_t udir = static_cast<size_t>(dir);
341  udir % 2 ? coords[udir / 2]++ : coords[udir / 2]--;
342  }
343 
344  // Direction is allowed if end of stencil is not reached and direction is not visited by another iterator.
345  // Iterators never change direction, they may only spawn new iterators in the directions that have a higher
346  // index (i.e. iterators that walk in x direction will spawn iterators going in y and z direction,
347  // iterators going in y direction will only spawn iterators in z-direction and z iterators only walk
348  // without emitting new iterators).
349  static bool direction_allowed(const int dir, const int new_dir)
350  {
351  return new_dir == dir || new_dir / 2 > dir / 2;
352  }
353 
354  static bool end_of_stencil(const StencilType& stencil, const int dir, const CoordsType& coords)
355  {
356  assert(dir >= 0.);
357  size_t udir = static_cast<size_t>(dir);
358  return coords[udir / 2] == stencil.shape()[udir / 2] - 1 || coords[udir / 2] == 0;
359  }

463  template <class SourceType, class RangeType>
464  void apply(const SourceType& source, RangeType& range, const XT::Common::Parameter& param) const
465  {
466  static_assert(is_reconstructed_localizable_function<RangeType>::value,
467  "RangeType has to be derived from ReconstructedLocalizableFunction!");
468  // evaluate cell averages
469  const auto& grid_layer = source.space().grid_layer();
470  const auto& index_set = grid_layer.indexSet();
471  std::vector<VectorType> source_values(index_set.size(0));
472  for (const auto& entity : Dune::elements(grid_layer)) {
473  const auto entity_index = index_set.index(entity);
474  const auto local_source = source.local_function(entity);
475  source_values[entity_index] = local_source->evaluate(entity.geometry().local(entity.geometry().center()));
476  }
477 
478  // do reconstruction
479  auto local_reconstruction_operator =
480  LocalLinearReconstructionOperator<AnalyticalFluxType,
481  BoundaryValueType,
482  typename SourceType::SpaceType::GridLayerType,
483  JacobianWrapperType>(source_values,
484  analytical_flux_,
485  boundary_values_,
486  slope_,
487  grid_layer,
488  param,
489  quadrature_1d_,
490  range,
491  jacobian_wrapper_);
492  auto walker = XT::Grid::Walker<typename SourceType::SpaceType::GridLayerType>(grid_layer);
493  walker.append(local_reconstruction_operator);
494  walker.walk(true);
495  } // void apply(...)
