29 template <class DiscreteFunctionType, class RhsEvaluationType, class MatrixType>
30 class MatrixExponentialFunctor
31  : public XT::Grid::Functor::Codim0<typename DiscreteFunctionType::SpaceType::GridLayerType>
32 {
33  typedef typename XT::Grid::Functor::Codim0<typename DiscreteFunctionType::SpaceType::GridLayerType> BaseType;
34  typedef typename RhsEvaluationType::RangeFieldType FieldType;
35  typedef typename RhsEvaluationType::RangeType RangeType;
36  static const size_t dimRange = RhsEvaluationType::dimRange;
37 
38 public:
39  using typename BaseType::EntityType;
40 
41  MatrixExponentialFunctor(DiscreteFunctionType& solution,
42  const double t,
43  const double dt,
44  const RhsEvaluationType& rhs_evaluation,
45  const std::vector<MatrixType>& matrix_exponentials,
46  const std::vector<MatrixType>& matrix_exponential_integrals)
47 
48  : solution_(solution)
49  , t_(t)
50  , dt_(dt)
51  , rhs_evaluation_(rhs_evaluation)
52  , matrix_exponentials_(matrix_exponentials)
53  , int_exp_mAdt_b_(matrix_exponential_integrals.size())
54  {
55  for (size_t ii = 0; ii < matrix_exponential_integrals.size(); ++ii) {
56  const auto& b = rhs_evaluation_.values()[ii]->b();
57  matrix_exponential_integrals[ii].mv(b, int_exp_mAdt_b_[ii]);
58  }
59  }
60 
61  // Solves d_t u(t) = A u(t) + b locally on each entity
62  // Multiplying by exp(-At) we get
63  // (see https://en.wikipedia.org/wiki/Matrix_exponential#Linear_differential_equations)
64  // d_t (exp(-At)u(t)) = exp(-At) b
65  // By integrating over (0, dt) wrt. t, we get
66  // u(dt) = exp(Adt)(u(0) + (\int_0^{dt} exp(-At)) b)
67  virtual void apply_local(const EntityType& entity)
68  {
69  auto solution_local = solution_.local_discrete_function(entity);
70 
71  // get u
72  const auto center = entity.geometry().local(entity.geometry().center());
73  const auto u0 = solution_local->evaluate(center);
74 
75  const size_t subdomain = rhs_evaluation_.subdomain(entity);
76  const auto& exp_Adt = matrix_exponentials_[subdomain];
77 
78  // calculate solution u = exp(A dt) (u0 + int_exp_mAdt b)
79  FieldVector<FieldType, dimRange> ret;
80  const auto u = int_exp_mAdt_b_[subdomain] + u0;
81  exp_Adt.mv(u, ret);
82 
83  // write to return vector
84  auto& local_vector = solution_local->vector();
85  for (size_t ii = 0; ii < dimRange; ++ii)
86  local_vector.set(ii, ret[ii]);
87  }
88 
89 private:
90  DiscreteFunctionType& solution_;
91  const double t_;
92  const double dt_;
93  const RhsEvaluationType& rhs_evaluation_;
94  const std::vector<MatrixType>& matrix_exponentials_;
95  std::vector<RangeType> int_exp_mAdt_b_;
96 };

142  virtual RangeFieldType step(const RangeFieldType dt, const RangeFieldType max_dt) override final
143  {
144  const RangeFieldType actual_dt = std::min(dt, max_dt);
145  auto& t = current_time();
146  auto& u_n = current_solution();
147  calculate_matrix_exponentials(actual_dt);
148  MatrixExponentialFunctor<DiscreteFunctionType, AffineCheckerboardType, SparseMatrixType> functor(
149  u_n, t, actual_dt, evaluation_, matrix_exponentials_, matrix_exponential_integrals_);

151  assembler.append(functor);
152  assembler.assemble(true);

161 private:
162  void calculate_matrix_exponentials(const RangeFieldType& actual_dt)
163  {
164  if (XT::Common::FloatCmp::ne(actual_dt, last_dt_)) {
165  size_t num_threads = std::min(XT::Common::threadManager().max_threads(), num_subdomains_);
166  std::vector<std::set<size_t>> decomposition(num_threads);
167  for (size_t ii = 0; ii < num_subdomains_; ++ii)
168  decomposition[ii % num_threads].insert(ii);
169 
170  std::vector<std::thread> threads(num_threads);
171  // Launch a group of threads
172  for (size_t ii = 0; ii < num_threads; ++ii)
173  threads[ii] = std::thread(&ThisType::calculate_in_thread, this, actual_dt, decomposition[ii]);
174  // Join the threads with the main thread
175  for (size_t ii = 0; ii < num_threads; ++ii)
176  threads[ii].join();
177  }

186  void get_matrix_exponential(size_t index, const RangeFieldType& dt)
187  {
188  const auto& affine_function = *(evaluation_.values()[index]);
189  assert(affine_function.A().size() == 1 && "Not implemented for dimRangeCols > 1!");
190  auto A = affine_function.A()[0].operator std::unique_ptr<FieldMatrixType>();
191 
192  // calculate matrix exponential exp(A*dt)
193  auto Adt = XT::Common::make_unique<DenseMatrixType>(*A);
194  *Adt *= dt;
195  // get pointer to the underlying array of the FieldMatrix
196  RangeFieldType* Adt_array = &((*Adt)[0][0]);
197  const double* exp_Adt_array = r8mat_expm1(dimRange, Adt_array);
198  auto& exp_Adt = *Adt; // we do not need Adt anymore, use it as storage for exp_Adt
199  std::copy_n(exp_Adt_array, dimRange * dimRange, &(exp_Adt[0][0]));
200  delete[] exp_Adt_array;
201  matrix_exponentials_[index] = SparseMatrixType(exp_Adt, true);
202 
203  // calculate \int_0^{dt} exp(-At) dt
204  // see https://math.stackexchange.com/questions/658276/integral-of-matrix-exponential
205  // if A is invertible, the integral is -A^{-1}(exp(-Adt)-I)
206  // in general, it is the power series dt*(I + (-Adt)/(2!) + (-Adt)^2/(3!) + ... + (-Adt)^{n-1}/(n!) + ...)
207  auto A_inverse = XT::Common::make_unique<DenseMatrixType>(*A);
208  auto& int_exp_mAdt = *Adt; // use Adt's storage again
209  try {
210  // For 1x1, 2x2 and 3x3 matrices, dune-common only checks whether the matrix is actually invertible if
211  // DUNE_FMatrix_WITH_CHECKING is set. We always want to check to
212  // be able to catch the error, so we have to copy the checks over from dune/common/fmatrix.hh.
213  if (dimRange == 1) {
214  if (fvmeta::absreal((*A)[0][0]) < FMatrixPrecision<>::absolute_limit())
215  DUNE_THROW(FMatrixError, "matrix is singular");
216  } else if (dimRange == 2) {
217  RangeFieldType det = (*A)[0][0] * (*A)[1][1] - (*A)[0][1] * (*A)[1][0];
218  if (fvmeta::absreal(det) < FMatrixPrecision<>::absolute_limit())
219  DUNE_THROW(FMatrixError, "matrix is singular");
220  } else if (dimRange == 3) {
221  RangeFieldType t4 = (*A)[0][0] * (*A)[1][1];
222  RangeFieldType t6 = (*A)[0][0] * (*A)[1][2];
223  RangeFieldType t8 = (*A)[0][1] * (*A)[1][0];
224  RangeFieldType t10 = (*A)[0][2] * (*A)[1][0];
225  RangeFieldType t12 = (*A)[0][1] * (*A)[2][0];
226  RangeFieldType t14 = (*A)[0][2] * (*A)[2][0];
227  RangeFieldType det = (t4 * (*A)[2][2] - t6 * (*A)[2][1] - t8 * (*A)[2][2] + t10 * (*A)[2][1] + t12 * (*A)[1][2]
228  - t14 * (*A)[1][1]);
229  if (fvmeta::absreal(det) < FMatrixPrecision<>::absolute_limit())
230  DUNE_THROW(FMatrixError, "matrix is singular");
231  }
232  A_inverse->invert();
233  *A_inverse *= -1.;
234 
235  // calculate matrix exponential exp(-A*dt)
236  auto* mAdt = A.get();
237  *mAdt *= -dt;
238  // get pointer to the underlying array of the FieldMatrix
239  double* mAdt_array = &((*mAdt)[0][0]);
240  const double* exp_mAdt_array = r8mat_expm1(dimRange, mAdt_array);
241  std::copy_n(exp_mAdt_array, dimRange * dimRange, &(int_exp_mAdt[0][0]));
242  delete[] exp_mAdt_array;
243  for (size_t ii = 0; ii < dimRange; ++ii)
244  int_exp_mAdt[ii][ii] -= 1.;
245  int_exp_mAdt.leftmultiply(*A_inverse);
246  } catch (Dune::FMatrixError&) { // A not invertible
247  auto* minus_A = A.get();
248  *minus_A *= -1.;
249  const double* int_exp_mAdt_array = r8mat_expm_integral(dimRange, &((*minus_A)[0][0]), dt);
250  std::copy_n(int_exp_mAdt_array, dimRange * dimRange, &(int_exp_mAdt[0][0]));
251  delete[] int_exp_mAdt_array;
252  } // catch(...)
253  matrix_exponential_integrals_[index] = SparseMatrixType(int_exp_mAdt, true);
254  } // void get_matrix_exponential(...)
