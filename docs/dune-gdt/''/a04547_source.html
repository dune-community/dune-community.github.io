126 template <class GL, class FiniteElementType>
127 class FixedOrderMultipleCodimMultipleGeomTypeMapper
128  : public MapperInterface<internal::FixedOrderMultipleCodimMultipleGeomTypeMapperTraits<GL, FiniteElementType>>
129 {
130 public:
131  using Traits = internal::FixedOrderMultipleCodimMultipleGeomTypeMapperTraits<GL, FiniteElementType>;
132 
133 private:
134  using ThisType = FixedOrderMultipleCodimMultipleGeomTypeMapper<GL, FiniteElementType>;
135  using BaseType = MapperInterface<Traits>;
136  using D = typename GL::ctype;
137  static const constexpr size_t d = GL::dimension;
138 
139 public:
140  using typename BaseType::EntityType;
141  using typename BaseType::BackendType;
142 
143  FixedOrderMultipleCodimMultipleGeomTypeMapper(
144  const GL& grid_layer,
145  const std::shared_ptr<std::map<GeometryType, std::shared_ptr<FiniteElementType>>>& finite_elements)
146  : finite_elements_(finite_elements)
147  , mapper_(nullptr)
148  {
149  std::set<GeometryType> all_DoF_attached_geometry_types;
150  // collect all entities (for all codims) which are used to attach DoFs to
151  for (auto&& geometry_type : grid_layer.indexSet().types(0)) {
152  // get the finite element for this geometry type
153  const auto finite_element_search_result = finite_elements_->find(geometry_type);
154  if (finite_element_search_result == finite_elements_->end())
155  DUNE_THROW(mapper_error, "Missing finite element for the required geometry type " << geometry_type << "!");
156  const auto& finite_element = *finite_element_search_result->second;
157  // loop over all keys of this finite element
158  const auto& reference_element = ReferenceElements<D, d>::general(geometry_type);
159  const auto& coeffs = finite_element.localCoefficients();
160  for (unsigned int ii = 0; ii < coeffs.size(); ++ii) {
161  const auto& local_key = coeffs.localKey(ii);
162  if (local_key.index() != 0) // Would require twisting of DoFs and possibly more knowledge from the FE
163  DUNE_THROW(mapper_error, "This case is not covered yet, when we have more than one DoF per (sub)entity!");
164  // find the (sub)entity for this key
165  const auto sub_entity = local_key.subEntity();
166  const auto codim = local_key.codim();
167  const auto& subentity_geometry_type = reference_element.type(sub_entity, codim);
168  // and add the respective geometry type
169  all_DoF_attached_geometry_types.insert(subentity_geometry_type);
170  }
171  }
172  if (all_DoF_attached_geometry_types.size() == 0)
173  DUNE_THROW(mapper_error, "This must not happen, the finite elements report no DoFs attached to (sub)entities!");
174  mapper_ = std::make_shared<BackendType>(
175  grid_layer,
176  std::move(typename Traits::template GeometryTypeLayout<d>(std::move(all_DoF_attached_geometry_types))));
177  } // ... FixedOrderMultipleCodimMultipleGeomTypeMapper(...)
178 
179  FixedOrderMultipleCodimMultipleGeomTypeMapper(const ThisType&) = default;
180  FixedOrderMultipleCodimMultipleGeomTypeMapper(ThisType&&) = default;
181  FixedOrderMultipleCodimMultipleGeomTypeMapper& operator=(const ThisType&) = delete;
182  FixedOrderMultipleCodimMultipleGeomTypeMapper& operator=(ThisType&&) = delete;
183 
184  const BackendType& backend() const
185  {
186  return *mapper_;
187  }
188 
189  size_t size() const
190  {
191  return mapper_->size();
192  }

194  size_t maxNumDofs() const
195  {
196  size_t ret = 0;
197  for (const auto& geometry_type_and_finite_element_pair : *finite_elements_) {
198  const auto& finite_element = *geometry_type_and_finite_element_pair.second;
199  ret = std::max(ret, XT::Common::numeric_cast<size_t>(finite_element.size()));
200  }
201  return ret;
202  } // ... maxNumDofs(...)
203 
204  size_t numDofs(const EntityType& entity) const
205  {
206  const auto finite_element_search_result = finite_elements_->find(entity.geometry().type());
207  if (finite_element_search_result == finite_elements_->end())
208  DUNE_THROW(XT::Common::Exceptions::internal_error,
209  "This must not happen after the checks in the ctor, the grid layer did not report all geometry types!"
210  "\n entity.geometry().type() = "
211  << entity.geometry().type());
212  const auto& finite_element = *finite_element_search_result->second;
213  return finite_element.size();
214  } // ... numDofs(...)

225  void globalIndices(const EntityType& entity, DynamicVector<size_t>& ret) const
226  {
227  const auto finite_element_search_result = finite_elements_->find(entity.geometry().type());
228  if (finite_element_search_result == finite_elements_->end())
229  DUNE_THROW(XT::Common::Exceptions::internal_error,
230  "This must not happen after the checks in the ctor, the grid layer did not report all geometry types!"
231  "\n entity.geometry().type() = "
232  << entity.geometry().type());
233  const auto& finite_element = *finite_element_search_result->second;
234  const auto& local_coefficients = finite_element.localCoefficients();
235  const auto local_size = local_coefficients.size();
236  if (ret.size() < local_size)
237  ret.resize(local_size, 0);
238  for (unsigned int ii = 0; ii < local_size; ++ii) {
239  const auto& local_key = local_coefficients.localKey(ii);
240  // No need to assert local_key.index() == 0, has been checked in the ctor!
241  ret[ii] = mapper_->subIndex(entity, local_key.subEntity(), local_key.codim());
242  }
243  } // ... globalIndices(...)
244 
245  size_t mapToGlobal(const EntityType& entity, const size_t local_index) const
246  {
247  const auto finite_element_search_result = finite_elements_->find(entity.geometry().type());
248  if (finite_element_search_result == finite_elements_->end())
249  DUNE_THROW(XT::Common::Exceptions::internal_error,
250  "This must not happen after the checks in the ctor, the grid layer did not report all geometry types!"
251  "\n entity.geometry().type() = "
252  << entity.geometry().type());
253  const auto& finite_element = *finite_element_search_result->second;
254  const auto& local_coefficients = finite_element.localCoefficients();
255  if (local_index >= local_coefficients.size())
256  DUNE_THROW(Exception,
257  "finite_element.localCoefficients().size() = " << local_coefficients.size() << "\n local_index = "
258  << local_index);
259  assert(local_index <= std::numeric_limits<unsigned int>::max());
260  const auto& local_key = local_coefficients.localKey(static_cast<unsigned int>(local_index));
261  // No need to assert local_key.index() == 0, has been checked in the ctor!
262  return mapper_->subIndex(entity, local_key.subEntity(), local_key.codim());
263  } // ... mapToGlobal(...)
264 
265 private:
266  const std::shared_ptr<std::map<GeometryType, std::shared_ptr<FiniteElementType>>> finite_elements_;
267  std::shared_ptr<BackendType> mapper_;
268 }; // class FixedOrderMultipleCodimMultipleGeomTypeMapper

344 template <class GL, class FiniteElementType>
345 class FixedOrderScalarDiscontinuousMapper
346  : public MapperInterface<internal::FixedOrderScalarDiscontinuousMapperTraits<GL, FiniteElementType>>
347 {
348 public:
349  using Traits = internal::FixedOrderScalarDiscontinuousMapperTraits<GL, FiniteElementType>;
350 
351 private:
352  using ThisType = FixedOrderScalarDiscontinuousMapper<GL, FiniteElementType>;
353  using BaseType = MapperInterface<Traits>;
354  using D = typename GL::ctype;
355  static const constexpr size_t d = GL::dimension;
356 
357 public:
358  using typename BaseType::EntityType;
359  using typename BaseType::BackendType;
360 
361  FixedOrderScalarDiscontinuousMapper(
362  const GL& grid_layer,
363  const std::shared_ptr<std::map<GeometryType, std::shared_ptr<FiniteElementType>>>& finite_elements)
364  : finite_elements_(finite_elements)
365  , mapper_(new BackendType(grid_layer))
366  , offset_(new std::vector<size_t>(mapper_->size()))
367  , max_num_dofs_(0)
368  , size_(0)
369  {
370  for (auto&& element : elements(grid_layer)) {
371  (*offset_)[mapper_->subIndex(element, 0, 0)] = size_;
372  const auto finite_element_search_result = finite_elements_->find(element.geometry().type());
373  if (finite_element_search_result == finite_elements_->end())
374  DUNE_THROW(mapper_error,
375  "Missing finite element for the required geometry type " << element.geometry().type() << "!");
376  const auto& finite_element = *finite_element_search_result->second;
377  size_ += finite_element.size();
378  max_num_dofs_ = std::max(max_num_dofs_, XT::Common::numeric_cast<size_t>(finite_element.size()));
379  }
380  } // ... FixedOrderScalarDiscontinuousMapper(...)
381 
382  FixedOrderScalarDiscontinuousMapper(const ThisType&) = default;
383  FixedOrderScalarDiscontinuousMapper(ThisType&&) = default;
384  FixedOrderScalarDiscontinuousMapper& operator=(const ThisType&) = delete;
385  FixedOrderScalarDiscontinuousMapper& operator=(ThisType&&) = delete;
386 
387  const BackendType& backend() const
388  {
389  return *mapper_;
390  }
391 
392  size_t size() const
393  {
394  return size_;
395  }
396 
397  size_t maxNumDofs() const
398  {
399  return max_num_dofs_;
400  }

403  {

408  << "\n entity.geometry().type() = "

415  typename std::enable_if<cd != EntityType::codimension, size_t>::type
416  numDofs(const Entity<cd, EntityType::dimension, GridImp, EntityImp>& /*entity*/) const
417  {
418  return 0;
419  }

421  using BaseType::globalIndices;
422 
423  void globalIndices(const EntityType& entity, DynamicVector<size_t>& ret) const
424  {
425  const size_t offset = (*offset_)[mapper_->subIndex(entity, 0, 0)];
426  const auto local_size = numDofs(entity);
427  if (ret.size() < local_size)
428  ret.resize(local_size, 0);
429  for (size_t ii = 0; ii < local_size; ++ii)
430  ret[ii] = offset + ii;
431  } // ... globalIndices(...)
432 
433  size_t mapToGlobal(const EntityType& entity, const size_t local_index) const
434  {
435  if (local_index >= numDofs(entity))
436  DUNE_THROW(Exception, "numDofs(entity) = " << numDofs(entity) << "\n local_index = " << local_index);
437  const size_t offset = (*offset_)[mapper_->subIndex(entity, 0, 0)];
438  return offset + local_index;
439  }

443  const std::shared_ptr<BackendType> mapper_;
444  std::shared_ptr<std::vector<size_t>> offset_;
445  size_t max_num_dofs_;
446  size_t size_;
447 }; // class FixedOrderScalarDiscontinuousMapper
