81  template <int cd, class GridImp, template <int, int, class> class EntityImp>
82  static size_t numDofs(const ThisType& self, const Entity<cd, EntityType::dimension, GridImp, EntityImp>& entity)
83  {
84  const size_t block = find_block_of(self, entity);
85  if (self.backend()[block] == nullptr)
86  DUNE_THROW(InvalidStateException, "You did not provide a local space for block " << block << "!");
87  return self.backend()[block]->mapper().numDofs(entity);
88  }

90  static void globalIndices(const ThisType& self, const Comdim0EntityType& entity, Dune::DynamicVector<size_t>& ret)
91  {
92  const size_t block = find_block_of(self, entity);
93  if (self.backend()[block] == nullptr)
94  DUNE_THROW(InvalidStateException, "You did not provide a local space for block " << block << "!");
95  self.backend()[block]->mapper().globalIndices(entity, ret);
96  const size_t num_dofs = self.backend()[block]->mapper().numDofs(entity);
97  assert(ret.size() >= num_dofs);
98  for (size_t ii = 0; ii < num_dofs; ++ii)
99  ret[ii] += self.global_start_indices_[block];
100  }
101 
102  static size_t mapToGlobal(const ThisType& self, const Comdim0EntityType& entity, const size_t& localIndex)
103  {
104  const size_t block = find_block_of(self, entity);
105  if (self.backend()[block] == nullptr)
106  DUNE_THROW(InvalidStateException, "You did not provide a local space for block " << block << "!");
107  const size_t block_local_index = self.backend()[block]->mapper().mapToGlobal(entity, localIndex);
108  return self.global_start_indices_[block] + block_local_index;
109  }

111  private:
112  template <int cd, class GridImp, template <int, int, class> class EntityImp>
113  static size_t find_block_of(const ThisType& self,
114  const Entity<cd, EntityType::dimension, GridImp, EntityImp>& entity)
115  {
116  const auto global_entity_index = self.global_grid_part_->indexSet().index(entity);
117  const auto result = self.entity_to_subdomain_map_->find(global_entity_index);
118 #ifndef NDEBUG
119  if (result == self.entity_to_subdomain_map_->end())
120  DUNE_THROW(XT::Common::Exceptions::internal_error,
121  "Entity " << global_entity_index
122  << " of the global grid part was not found in the dd subdomain grid!");
123 #endif // NDEBUG
124  const size_t subdomain = result->second;
125 #ifndef NDEBUG
126  if (subdomain >= self.num_blocks_)
127  DUNE_THROW(XT::Common::Exceptions::internal_error,
128  "The DD subdomains grid is corrupted!\nIt reports Entity " << global_entity_index
129  << " to be in subdomain "
130  << subdomain
131  << " while only having "
132  << self.num_blocks_
133  << " subdomains!");
134 #endif // NDEBUG
135  return subdomain;
136  } // ... find_block_of(...)
