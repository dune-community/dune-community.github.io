209  template <class R>
210  void evaluate(
211  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>& local_diffusion_factor,
212  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1>&
213  local_diffusion_factor_hat,
214  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain, dimDomain>&
215  local_diffusion_tensor,
216  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, dimDomain>&
217  local_diffusive_flux,
218  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& test_base,
219  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1>& ansatz_base,
220  const Dune::FieldVector<DomainFieldType, dimDomain>& local_point,
221  Dune::DynamicMatrix<R>& ret) const
222  {
223  typedef FieldVector<R, dimDomain> DomainType;
224  DomainType left_sum(0);
225  DomainType right_sum(0);
226  // evaluate local functions
227  const auto diffusion_factor_value = local_diffusion_factor.evaluate(local_point);
228  const auto diffusion_factor_hat_value = local_diffusion_factor_hat.evaluate(local_point);
229  typedef XT::Common::FieldMatrix<R, dimDomain, dimDomain> TensorType;
230  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(local_point);
231  const TensorType diffusion_value = diffusion_tensor_value * diffusion_factor_value;
232  // TODO: there is no documented way to assert that the inversion was successfull, so find one or check the matrix
233  // beforehand
234  TensorType one_over_diffusion_hat_value = diffusion_tensor_value;
235  one_over_diffusion_hat_value.invert();
236  one_over_diffusion_hat_value /= diffusion_factor_hat_value;
237  const auto diffusive_flux_value = local_diffusive_flux.evaluate(local_point);
238  // evaluate test gradient
239  const size_t rows = test_base.size();
240  const auto test_gradients = test_base.jacobian(local_point);
241  assert(test_gradients.size() == rows);
242  // evaluate ansatz gradient
243  const size_t cols = ansatz_base.size();
244  const auto ansatz_gradients = ansatz_base.jacobian(local_point);
245  assert(ansatz_gradients.size() == rows);
246  // compute
247  assert(ret.rows() >= rows);
248  assert(ret.cols() >= cols);
249  for (size_t ii = 0; ii < rows; ++ii) {
250  left_sum = diffusion_value * test_gradients[ii];
251  left_sum += diffusive_flux_value;
252  auto& retRow = ret[ii];
253  for (size_t jj = 0; jj < cols; ++jj) {
254  right_sum = diffusion_value * ansatz_gradients[jj];
255  right_sum += diffusive_flux_value;
256  retRow[jj] = (one_over_diffusion_hat_value * left_sum) * right_sum;
257  }
258  }
259  } // ... evaluate(...)
