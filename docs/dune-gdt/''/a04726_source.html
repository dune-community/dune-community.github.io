242  template <class R, size_t r>
243  void
244  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, r, 1>& localFunction,
245  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, r, 1>& testBase,
246  const Dune::FieldVector<DomainFieldType, dimDomain>& localPoint,
247  Dune::DynamicVector<R>& ret) const
248  {
249  ret *= 0.0;
250  // evaluate local function
251  const auto functionValue = localFunction.evaluate(localPoint, param_);
252  // evaluate test base
253  const size_t size = testBase.size();
254  const auto testValues = testBase.evaluate(localPoint);
255  // compute product
256  assert(ret.size() >= size);
257  for (size_t ii = 0; ii < size; ++ii) {
258  ret[ii] = functionValue * testValues[ii];
259  }
260  } // ... evaluate(...)

266  template <class R, size_t r>
267  void
268  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localFunction,
269  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, r, 1>& testBase,
270  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, r, 1>& ansatzBase,
271  const Dune::FieldVector<DomainFieldType, dimDomain>& localPoint,
272  Dune::DynamicMatrix<R>& ret) const
273  {
274  ret *= 0.0;
275  // evaluate local function
276  const auto functionValue = localFunction.evaluate(localPoint, param_);
277  // evaluate bases
278  const auto rows = testBase.size();
279  const auto cols = ansatzBase.size();
280  const auto testValues = testBase.evaluate(localPoint);
281  const auto ansatzValues = ansatzBase.evaluate(localPoint);
282  // compute product
283  assert(ret.rows() >= rows);
284  assert(ret.cols() >= cols);
285  for (size_t ii = 0; ii < rows; ++ii) {
286  auto& retRow = ret[ii];
287  for (size_t jj = 0; jj < cols; ++jj) {
288  retRow[jj] = functionValue * (testValues[ii] * ansatzValues[jj]);
289  }
290  }
291  } // ... evaluate(...)
