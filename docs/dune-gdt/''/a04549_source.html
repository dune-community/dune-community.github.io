49 template <size_t ii, bool is_product_space = false>
50 struct visualize_helper
51 {
52  template <class DiscreteFunctionType>
53  static void visualize_factor(const std::string filename_prefix,
54  const std::string filename_suffix,
55  const bool subsampling,
56  const VTK::OutputType vtk_output_type,
57  const DiscreteFunctionType& discrete_function)
58  {
59  static_assert(ii == 0, "Space is not a product space, so there is no factor other than 0.");

61  subsampling,
62  vtk_output_type);
63  }

66 template <size_t ii>
67 struct visualize_helper<ii, true>
68 {
69  template <class DiscreteFunctionType>
70  static void visualize_factor(const std::string filename_prefix,
71  const std::string filename_suffix,
72  const bool subsampling,
73  const VTK::OutputType vtk_output_type,
74  const DiscreteFunctionType& discrete_function)
75  {
76  static_assert(ii < DiscreteFunctionType::SpaceType::num_factors, "This factor does not exist.");
77  const auto& space = discrete_function.space();
78  const auto& factor_space = space.template factor<ii>();
79  typename DiscreteFunctionType::VectorType factor_vector(factor_space.mapper().size());
80  const auto it_end = space.grid_layer().template end<0>();
81  for (auto it = space.grid_layer().template begin<0>(); it != it_end; ++it) {
82  const auto& entity = *it;
83  for (size_t jj = 0; jj < factor_space.mapper().numDofs(entity); ++jj)
84  factor_vector.set_entry(factor_space.mapper().mapToGlobal(entity, jj),
85  discrete_function.vector().get_entry(space.mapper().mapToGlobal(ii, entity, jj)));
86  }
87  ConstDiscreteFunction<
88  typename XT::Common::tuple_element<ii, typename DiscreteFunctionType::SpaceType::SpaceTupleType>::type,
89  typename DiscreteFunctionType::VectorType>
90  factor_discrete_function(factor_space, factor_vector);
91  factor_discrete_function.visualize(filename_prefix + "_factor_" + Dune::XT::Common::to_string(ii) + "_"
92  + filename_suffix,
93  subsampling,
94  vtk_output_type);
95  }
96 };

100 template <size_t index, size_t N>
101 struct static_for_loop

106  const bool subsampling,
107  const VTK::OutputType vtk_output_type,
108  const DiscreteFunctionType& discrete_function)
109  {
110  static_for_loop<index, N / 2>::visualize(
111  filename_prefix, filename_suffix, subsampling, vtk_output_type, discrete_function);
112  static_for_loop<index + N / 2, N - N / 2>::visualize(
113  filename_prefix, filename_suffix, subsampling, vtk_output_type, discrete_function);
114  }
