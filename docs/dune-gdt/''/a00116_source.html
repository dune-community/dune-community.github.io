151 template <class GridLayerType, class LocalizableFunctionType>
152 class LocalMomentModelBoundaryValue
153  : public LocalBoundaryValueInterface<GridLayerType, typename LocalizableFunctionType::RangeType>
154 {
155  using BaseType = LocalBoundaryValueInterface<GridLayerType, typename LocalizableFunctionType::RangeType>;
156  using RangeFieldType = typename LocalizableFunctionType::RangeFieldType;
157 
158 public:
159  using MatrixType = typename Dune::XT::LA::CommonSparseOrDenseMatrixCsr<RangeFieldType>;
160  using typename BaseType::BoundaryInfoType;
161  using typename BaseType::DomainType;
162  using typename BaseType::IntersectionType;
163  using typename BaseType::RangeType;
164  using BaseType::dimDomain;
165 
166  LocalMomentModelBoundaryValue(
167  const BoundaryInfoType& boundary_info,
168  const XT::Common::FieldVector<MatrixType, 2 * dimDomain>& reflection_matrices,
169  std::unique_ptr<typename LocalizableFunctionType::LocalfunctionType>&& local_dirichlet_boundary_values)
170  : BaseType(boundary_info)
171  , reflection_matrices_(reflection_matrices)
172  , local_dirichlet_boundary_values_(std::move(local_dirichlet_boundary_values))
173  {
174  }
175 
176  virtual RangeType evaluate(const IntersectionType& intersection,
177  const DomainType& x,
178  const RangeType& u,
179  const XT::Common::Parameter& /*param*/ = {}) const override
180  {
181  RangeType ret(0);
182  const auto& boundary_type = boundary_info_.type(intersection);
183  if (boundary_type == XT::Grid::absorbing_boundary)
184  ret = u;
185  else if (boundary_type == XT::Grid::dirichlet_boundary)
186  ret = local_dirichlet_boundary_values_->evaluate(x);
187  else if (boundary_type == XT::Grid::reflecting_boundary) {
188  assert(intersection.indexInInside() >= 0);
189  const auto direction = static_cast<size_t>(intersection.indexInInside());
190  reflection_matrices_[direction].mv(u, ret);
191  } else
192  DUNE_THROW(Dune::NotImplemented,
193  "Boundary types other than Absorbing, Dirichlet or Reflecting are not implemented!");
194  return ret;
195  }
196 
197 private:
198  using BaseType::boundary_info_;
199  const Dune::FieldVector<MatrixType, 2 * dimDomain>& reflection_matrices_;
200  const std::unique_ptr<typename LocalizableFunctionType::LocalfunctionType> local_dirichlet_boundary_values_;
201 };

204 template <class GridLayerType, class BasisfunctionType, class LocalizableFunctionType>
205 class MomentModelBoundaryValue
206  : public LocalizableBoundaryValueInterface<GridLayerType, typename BasisfunctionType::RangeType>
207 {
208  static_assert(std::is_same<typename BasisfunctionType::RangeType, typename LocalizableFunctionType::RangeType>::value,
209  "RangeTypes have to match!");
210  using BaseType = LocalizableBoundaryValueInterface<GridLayerType, typename BasisfunctionType::RangeType>;
211 
212 public:
213  using typename BaseType::BoundaryInfoType;
214  using typename BaseType::DomainType;
215  using typename BaseType::EntityType;
216  using typename BaseType::LocalBoundaryValueInterfaceType;
217  using typename BaseType::RangeType;
218  using LocalBoundaryValueType = LocalMomentModelBoundaryValue<GridLayerType, LocalizableFunctionType>;
219  using MatrixType = typename LocalBoundaryValueType::MatrixType;
220  static const size_t dimDomain = BasisfunctionType::dimDomain;
221  static const size_t dimRange = BasisfunctionType::dimRange;
222 
223 
224  MomentModelBoundaryValue(const BoundaryInfoType& boundary_info,
225  const BasisfunctionType& basis_functions,
226  const LocalizableFunctionType& dirichlet_boundary_values)
227  : BaseType(boundary_info)
228  , dirichlet_boundary_values_(dirichlet_boundary_values)
229  {
230  for (size_t ii = 0; ii < dimDomain; ++ii) {
231  DomainType n(0);
232  n[ii] = -1.;
233  reflection_matrices_[2 * ii] = MatrixType(basis_functions.reflection_matrix(n));
234  n[ii] = 1.;
235  reflection_matrices_[2 * ii + 1] = MatrixType(basis_functions.reflection_matrix(n));
236  }
237  }
238 
239  virtual std::unique_ptr<LocalBoundaryValueInterfaceType> local_function(const EntityType& entity) const override
240  {
241  return std::make_unique<LocalBoundaryValueType>(
242  boundary_info_, reflection_matrices_, dirichlet_boundary_values_.local_function(entity));
243  }
244 
245 private:
246  using BaseType::boundary_info_;
247  Dune::FieldVector<MatrixType, 2 * dimDomain> reflection_matrices_;
248  const LocalizableFunctionType& dirichlet_boundary_values_;
249 };
