72  template <class E, class D, size_t d, class R, size_t r, size_t rC, class RangeSpaceType, class VectorType>
73  typename std::enable_if<StaticCheck<E, D, d, R, r, rC, RangeSpaceType, VectorType>::value
74  && !is_fv_space<RangeSpaceType>::value,
75  void>::type
76  apply(const XT::Functions::LocalizableFunctionInterface<E, D, d, R, r, rC>& source,
77  LocalDiscreteFunction<RangeSpaceType, VectorType>& local_range) const
78  {
79  // create local L2 operator
80  typedef XT::Functions::ConstantFunction<E, D, d, R, 1> OneType;
81  const OneType one(1.); // <- is not actually used, just needed for the product evaluation
82  const LocalVolumeIntegralOperator<LocalProductIntegrand<OneType>,
83  typename RangeSpaceType::BaseFunctionSetType,
84  typename RangeSpaceType::BaseFunctionSetType,
85  R>
86  local_l2_operator(over_integrate_, one, param_);
87  // and functional
88  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, r, rC> SourceType;
89  const LocalVolumeIntegralFunctional<LocalProductIntegrand<SourceType>,
90  typename RangeSpaceType::BaseFunctionSetType,
91  R>
92  local_l2_functional(over_integrate_, source, param_);
93  // create local lhs and rhs
94  const auto& local_basis = local_range.basis();
95  const size_t size = local_basis.size();
96  XT::LA::CommonDenseMatrix<R> local_matrix(size, size);
97  XT::LA::CommonDenseVector<R> local_vector(size);
98  // assemble
99  local_l2_operator.apply2(local_basis, local_basis, local_matrix);
100  local_l2_functional.apply(local_basis, local_vector.backend());
101  // solve
102  XT::LA::CommonDenseVector<R> local_solution(size);
103  try {
104  XT::LA::Solver<XT::LA::CommonDenseMatrix<R>>(local_matrix).apply(local_vector, local_solution);
105  } catch (XT::LA::Exceptions::linear_solver_failed& ee) {
106  DUNE_THROW(projection_error,
107  "L2 projection failed because a local matrix could not be inverted!\n\n"
108  << "This was the original error: "
109  << ee.what());
110  }
111  // set local DoFs
112  auto& local_range_vector = local_range.vector();
113  assert(local_range_vector.size() == local_solution.size());
114  for (size_t ii = 0; ii < local_range_vector.size(); ++ii)
115  local_range_vector.set(ii, local_solution.get_entry(ii));
116  } // ... apply(...)

118  template <class E, class D, size_t d, class R, size_t r, size_t rC, class RangeSpaceType, class VectorType>
119  typename std::enable_if<StaticCheck<E, D, d, R, r, rC, RangeSpaceType, VectorType>::value
120  && is_fv_space<RangeSpaceType>::value,
121  void>::type
122  apply(const XT::Functions::LocalizableFunctionInterface<E, D, d, R, r, rC>& source,
123  LocalDiscreteFunction<RangeSpaceType, VectorType>& local_range) const
124  {
125  // create local L2 volume integral functional
126  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, r, rC> SourceType;
127  const LocalVolumeIntegralFunctional<LocalFVProductIntegrand<SourceType>,
128  typename RangeSpaceType::BaseFunctionSetType,
129  R>
130  local_l2_functional(over_integrate_, source, param_);
131  Dune::DynamicVector<R> local_vector(local_range.basis().size(), 0.);
132  const auto& entity = local_range.entity();
133  local_l2_functional.apply(local_range.basis(), local_vector);
134  local_vector /= entity.geometry().volume();
135  auto& local_range_vector = local_range.vector();
136  for (size_t ii = 0; ii < local_range_vector.size(); ++ii)
137  local_range_vector.set(ii, local_vector[ii]);
138  } // ... apply(...) for FV spaces
