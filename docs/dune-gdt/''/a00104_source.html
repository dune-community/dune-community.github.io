97  // and do the work (see below)

99  size_t kk = 0;
100  for (size_t ii = 0; ii < lagrange_points.size(); ++ii) {

102  // evaluate source function // which is not the case anymore.
103  const auto& lagrange_point = lagrange_points[ii];
104  const auto source_value = local_source->evaluate(lagrange_point, param_);
105  // set DoFs

107  local_DoF_vector.set(kk, source_value[jj]);
108  } else
