316  template <class GL, class S, size_t d, size_t r, size_t rC>
317  typename std::enable_if<XT::Grid::is_layer<GL>::value, PatternType>::type
318  compute_face_and_volume_pattern(const GL& grd_layr, const SpaceInterface<S, d, r, rC>& ansatz_space) const
319  {
320  // prepare
321  PatternType pattern(mapper().size());
322  Dune::DynamicVector<size_t> global_rows(mapper().maxNumDofs(), 0);
323  Dune::DynamicVector<size_t> global_cols(ansatz_space.mapper().maxNumDofs(), 0);
324  for (const auto& entity : elements(grd_layr)) {
325  const auto test_base_entity = base_function_set(entity);
326  const auto ansatz_base_entity = ansatz_space.base_function_set(entity);
327  mapper().globalIndices(entity, global_rows);
328  ansatz_space.mapper().globalIndices(entity, global_cols);
329  // compute entity/entity
330  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
331  for (size_t jj = 0; jj < ansatz_base_entity.size(); ++jj) {
332  pattern.insert(global_rows[ii], global_cols[jj]);
333  }
334  }
335  // walk the intersections
336  const auto intersection_it_end = grd_layr.iend(entity);
337  for (auto intersection_it = grd_layr.ibegin(entity); intersection_it != intersection_it_end; ++intersection_it) {
338  const auto& intersection = *intersection_it;
339  // get the neighbour
340  if (intersection.neighbor() && !intersection.boundary()) {
341  const auto neighbour = intersection.outside();
342  // get the basis
343  const auto ansatz_base_neighbour = ansatz_space.base_function_set(neighbour);
344  ansatz_space.mapper().globalIndices(neighbour, global_cols);
345  // compute entity/neighbour
346  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
347  for (size_t jj = 0; jj < ansatz_base_neighbour.size(); ++jj) {
348  pattern.insert(global_rows[ii], global_cols[jj]);
349  }
350  }
351  } // get the neighbour
352  } // walk the intersections
353  } // walk the grid layer
354  pattern.sort();
355  return pattern;
