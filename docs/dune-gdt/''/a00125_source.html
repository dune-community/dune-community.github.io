22 template <class SourceType, class RangeType, class AnalyticalFluxType, class BasisfunctionType>
23 class LocalMomentRegularizer : public XT::Grid::Functor::Codim0<typename SourceType::SpaceType::GridLayerType>
24 {
25  // stencil is (i-r, i+r) in all dimensions, where r = polOrder + 1

51  void apply_local(const EntityType& entity)
52  {
53  const auto x_in_inside_coords = entity.geometry().local(entity.geometry().center());
54  auto u = source_.local_function(entity)->evaluate(x_in_inside_coords, param_);
55  assert(dynamic_cast<const EntropyFluxType*>(&analytical_flux_) != nullptr
56  && "analytical_flux_ has to be derived from EntropyBasedLocalFlux");
57  const auto* entropy_flux = dynamic_cast<const EntropyFluxType*>(&analytical_flux_);
58  const auto& basis_functions = entropy_flux->basis_functions();
59  thread_local auto vector_indices = source_.space().mapper().globalIndices(entity);
60  source_.space().mapper().globalIndices(entity, vector_indices);
61  basis_functions.ensure_min_density(u, min_acceptable_density_);
62  auto entropy_local_func = entropy_flux->derived_local_function(entity);
63  const auto s = entropy_local_func->get_alpha(x_in_inside_coords, u, param_, true).second;
64  // if regularization was needed, we also need to replace u_n in that cell by its regularized version
65  if (s > 0.) {
66  if (!filename_.empty()) {
67  static std::mutex outfile_lock;
68  outfile_lock.lock();
69  std::ofstream outfile(filename_, std::ios_base::app);
70  outfile << param_.get("t")[0];
71  for (size_t ii = 0; ii < AnalyticalFluxType::dimDomain; ++ii)
72  outfile << " " << entity.geometry().center()[ii];
73  outfile << " " << s << " ";
74  outfile << XT::Common::to_string(u, 15) << std::endl;
75  outfile_lock.unlock();
76  }
77  const auto u_iso_scaled = basis_functions.u_iso() * basis_functions.density(u);
78  for (size_t ii = 0; ii < vector_indices.size(); ++ii)
79  range_.vector().set_entry(vector_indices[ii], (1 - s) * u[ii] + s * u_iso_scaled[ii]);
80  } else {
81  for (size_t ii = 0; ii < vector_indices.size(); ++ii)
82  range_.vector().set_entry(vector_indices[ii], u[ii]);
83  } // if (s > 0)
84  std::string key = "center_results_are_intersection_results";
85  if (param_.has_key(key) && param_.get(key)[0])
86  entropy_local_func->center_results_to_intersections(source_.space().grid_layer());
87  } // void apply_local(...)
