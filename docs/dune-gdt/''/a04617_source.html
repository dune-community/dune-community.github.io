166  void solve(DiscreteSolutionType& solution) const
167  {
168  try {
169  const auto& problem = test_case_.problem();
170  // get analytical flux, initial and boundary values
171  const AnalyticalFluxType& analytical_flux = problem.flux();
172  const InitialValueType& initial_values = problem.initial_values();
173  const auto& dirichlet_boundary_values = problem.boundary_values();
174  const auto boundary_info = XT::Grid::AllDirichletBoundaryInfo<IntersectionType>();
175  const BoundaryValueType boundary_values(boundary_info, dirichlet_boundary_values);
176  const RhsType& rhs = problem.rhs();
177 
178  // create a discrete function for the solution
179  DiscreteFunctionType u(*fv_space_, "solution");
180 
181  // project initial values
182  project_l2(initial_values, u, true);
183 
184  RangeFieldType t_end = test_case_.t_end();
185  const RangeFieldType CFL = problem.CFL();
186 
187  // calculate dx and choose initial dt
188  Dune::XT::Grid::Dimensions<typename SpaceType::GridLayerType> dimensions(fv_space_->grid_layer());
189  RangeFieldType dx = dimensions.entity_width.max();
190  if (dimDomain == 2)
191  dx /= std::sqrt(2);
192  if (dimDomain == 3)
193  dx /= std::sqrt(3);
194  RangeFieldType dt = CFL * dx;
195 
196  // create operators
197  const ConstantFunctionType dx_function(dx);
198  std::unique_ptr<AdvectionOperatorType> advection_op =
199  AdvectionOperatorCreatorType::create(analytical_flux, boundary_values, dx_function);
200 
201  MinmodSlope<RangeType, typename ReconstructionOperatorType::MatrixType> slope;
202  ReconstructionOperatorType reconstruction_op(analytical_flux, boundary_values, slope);
203 
204  AdvectionWithReconstructionOperatorType advection_with_reconstruction_op(*advection_op, reconstruction_op);
205 
206  const auto& fv_op = choose_fv_operator(*advection_op, advection_with_reconstruction_op);
207 
208  RhsOperatorType rhs_op(rhs);
209 
210  // create timestepper
211  OperatorTimeStepperType timestepper_op(fv_op, u, -1.0);
212 
213  // do the time steps
214  const size_t num_save_steps = 100;
215  solution.clear();
216  if (problem.has_non_zero_rhs()) {
217  // use fractional step method
218  RhsOperatorTimeStepperType timestepper_rhs(rhs_op, u);
219  TimeStepperType timestepper(timestepper_op, timestepper_rhs);
220  timestepper.solve(t_end, dt, num_save_steps, solution);
221  } else {
222  timestepper_op.solve(t_end, dt, num_save_steps, solution);
223  }
224 
225  } catch (Dune::Exception& e) {
226  std::cerr << "Dune reported: " << e.what() << std::endl;
227  std::abort();
228  }
229  } // void solve(...)
