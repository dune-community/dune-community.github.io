71  template <class SourceType, class IntersectionType, class SpaceType, class VectorType>
72  void apply(const SourceType& source,
73  const IntersectionType& intersection,
74  LocalDiscreteFunction<SpaceType, VectorType>& local_range_entity,
75  LocalDiscreteFunction<SpaceType, VectorType>& local_range_neighbor,
76  const XT::Common::Parameter& /*mu*/ = {}) const
77  {
78  const auto entity = intersection.inside();

80  const auto local_source_entity = source.local_function(entity);
81  const auto local_source_neighbor = source.local_function(neighbor);
82  const auto local_functions_tuple_entity = numerical_flux_.local_functions(entity);
83  const auto local_functions_tuple_neighbor = numerical_flux_.local_functions(neighbor);
84  ResultType result(0);
85  for (const auto& quad_point : quadrature_) {
86  result += ResultType(numerical_flux_.evaluate(local_functions_tuple_entity,
87  local_functions_tuple_neighbor,
88  *local_source_entity,
89  *local_source_neighbor,
90  intersection,
91  quad_point.position()))
92  * quad_point.weight() * intersection.geometry().integrationElement(quad_point.position());
93  }
94  local_range_entity.vector().add(result * (1.0 / entity.geometry().volume()));
95  local_range_neighbor.vector().add(result * (-1.0 / neighbor.geometry().volume()));
96  }

123  void apply(const SourceType& source,
124  const IntersectionType& intersection,
125  LocalDiscreteFunction<SpaceType, VectorType>& local_range_entity) const
126  {
127  const auto entity = intersection.inside();
128  const auto local_source_entity = source.local_function(entity);
129  const auto local_functions_tuple = numerical_flux_.local_functions(entity);
130  ResultType result(0);
131  for (const auto& quad_point : quadrature_) {
132  result += ResultType(numerical_flux_.evaluate(
133  local_functions_tuple, *local_source_entity, intersection, quad_point.position()))
134  * quad_point.weight() * intersection.geometry().integrationElement(quad_point.position());
135  }
136  result /= entity.geometry().volume();
137  local_range_entity.vector().add(result);
138  }
