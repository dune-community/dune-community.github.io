59  virtual void apply_local(const EntityType& entity)
60  {
61  auto solution_local = solution_.local_discrete_function(entity);
62 
63  // get u
64  const auto center = entity.geometry().local(entity.geometry().center());
65  const auto u0 = solution_local->evaluate(center);
66  const auto sigma_a = sigma_a_.local_function(entity)->evaluate(center)[0];
67  const auto sigma_s = sigma_s_.local_function(entity)->evaluate(center)[0];
68  const auto Q = Q_.local_function(entity)->evaluate(center)[0];
69  const auto exp_sigma_a = std::exp(-sigma_a * dt_);
70  const auto exp_sigma_s = std::exp(-sigma_s * dt_);
71  const auto u_iso = u_iso_ * basis_functions_.density(u0);
72 
73  auto u = exp_sigma_a * (u0 * exp_sigma_s + u_iso * (1 - exp_sigma_s));
74  if (!XT::Common::is_zero(sigma_a))
75  u += basis_integrated_ * (1 - exp_sigma_a) / sigma_a * Q;
76  else
77  u += basis_integrated_ * dt_ * Q;
78 
79  // write to return vector
80  auto& local_vector = solution_local->vector();
81  for (size_t ii = 0; ii < dimRange; ++ii)
82  local_vector.set(ii, u[ii]);
83  }

87  DiscreteFunctionType& solution_;
88  const double dt_;

90  const CheckerboardType& sigma_s_;
91  const CheckerboardType& Q_;
