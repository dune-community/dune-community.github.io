182  // get permutation instead of sorting directly to be able to sort two vectors the same way

185  template <typename T, typename Compare>
186  std::vector<std::size_t> get_sort_permutation(const std::vector<T>& vec, const Compare& compare)
187  {
188  std::vector<std::size_t> p(vec.size());
189  std::iota(p.begin(), p.end(), 0);
190  std::sort(p.begin(), p.end(), [&](std::size_t i, std::size_t j) { return compare(vec[i], vec[j]); });

194  template <typename T>
195  void apply_permutation_in_place(std::vector<T>& vec, const std::vector<std::size_t>& p)
196  {
197  std::vector<bool> done(vec.size());
198  for (std::size_t i = 0; i < vec.size(); ++i) {
199  if (done[i]) {
200  continue;
201  }
202  done[i] = true;
203  std::size_t prev_j = i;
204  std::size_t j = p[i];
205  while (i != j) {
206  std::swap(vec[prev_j], vec[j]);
207  done[j] = true;
208  prev_j = j;
209  j = p[j];
210  }
211  }
212  }

214  // Joins duplicate quadpoints, vectors have to be sorted!
215  void join_duplicate_quadpoints(std::vector<DomainType>& quad_points, std::vector<RangeFieldType>& quad_weights)
216  {
217  // Index of first quad_point of several quad_points with the same position
218  size_t curr_index = 0;
219  std::vector<size_t> indices_to_remove;
220  for (size_t ll = 1; ll < quad_weights.size(); ++ll) {
221  if (XT::Common::FloatCmp::eq(quad_points[curr_index], quad_points[ll])) {
222  quad_weights[curr_index] += quad_weights[ll];
223  indices_to_remove.push_back(ll);
224  } else {
225  curr_index = ll;
226  }
227  } // ll
228  assert(indices_to_remove.size() < std::numeric_limits<int>::max());
229  // remove duplicate points, from back to front to avoid invalidating indices
230  for (int ll = static_cast<int>(indices_to_remove.size()) - 1; ll >= 0; --ll) {
231  quad_points.erase(quad_points.begin() + indices_to_remove[ll]);
232  quad_weights.erase(quad_weights.begin() + indices_to_remove[ll]);
233  }
234  }

367  // The ClpSimplex structure seems to get corrupted sometimes (maybe some problems with infs/NaNs?), so we
368  // reinitialize it if the stopping conditions is always false
369  void setup_linear_program(const bool reinitialize) const
370  {
371  if (!*lp_ || reinitialize) {
372  // We start with creating a model with dimRange rows and num_quad_points columns */
373  constexpr int num_rows = static_cast<int>(dimRange);
374  assert(quad_points_.size() < std::numeric_limits<int>::max());
375  int num_cols = static_cast<int>(quad_points_.size()); /* variables are x_1, ..., x_{num_quad_points} */
376  *lp_ = std::make_unique<ClpSimplex>(false);
377  auto& lp = **lp_;
378  // set number of rows
379  lp.resize(num_rows, 0);
380 
381  // Clp wants the row indices that are non-zero in each column. We have a dense matrix, so provide all indices
382  // 0..num_rows
383  std::array<int, num_rows> row_indices;
384  for (int ii = 0; ii < num_rows; ++ii)
385  row_indices[static_cast<size_t>(ii)] = ii;
386 
387  // set columns for quadrature points
388  for (int ii = 0; ii < num_cols; ++ii) {
389  const auto v_i = basis_functions_.evaluate(quad_points_[static_cast<size_t>(ii)]);
390  // First argument: number of elements in column
391  // Second/Third argument: indices/values of column entries
392  // Fourth/Fifth argument: lower/upper column bound, i.e. lower/upper bound for x_i. As all x_i should be
393  // positive, set to 0/inf, which is the default.
394  // Sixth argument: Prefactor in objective for x_i, this is 0 for all x_i, which is also the default;
395  lp.addColumn(num_rows, row_indices.data(), &(v_i[0]));
396  }
397 
398  // silence lp
399  lp.setLogLevel(0);
400  } // if (!lp_)
401  }

614  AlphaReturnType get_alpha(const DomainType& x_local,
615  const StateRangeType& u,
616  const XT::Common::Parameter& param,
617  const bool regularize) const
618  {
619  const bool boundary = bool(param.get("boundary")[0]);
620  // get initial multiplier and basis matrix from last time step
621  AlphaReturnType ret;
622  mutex_.lock();
623  if (boundary)
624  cache_.set_capacity(cache_size + dimDomain);
625 
626  // rescale u such that the density <psi> is 1
627  RangeFieldType density = basis_functions_.density(u);
628  if (!(density > 0.) || std::isinf(density)) {
629  mutex_.unlock();
630  DUNE_THROW(Dune::MathError, "Negative, inf or NaN density!");
631  }
632  VectorType u_prime = u / density;
633  VectorType alpha_iso = basis_functions_.alpha_iso();
634  VectorType v, u_eps_diff, beta_in;
635  RangeFieldType first_error_cond, second_error_cond, tau_prime;
636 
637  // if value has already been calculated for these values, skip computation
638  const auto cache_iterator = cache_.find_closest(u_prime);
639  if (cache_iterator != cache_.end() && XT::Common::FloatCmp::eq(cache_iterator->first, u_prime, 1e-14, 1e-14)) {
640  const auto alpha_prime = cache_iterator->second;
641  ret.first = alpha_prime + alpha_iso * std::log(density);
642  ret.second = 0.;
643  alpha_storage_[x_local] = ret.first;
644  mutex_.unlock();
645  return ret;
646  } else {
647  StateRangeType u_iso = basis_functions_.u_iso();
648  const RangeFieldType dim_factor = is_full_moment_basis<BasisfunctionType>::value ? 1. : std::sqrt(dimDomain);
649  tau_prime = std::min(tau_ / ((1 + dim_factor * u_prime.two_norm()) * density + dim_factor * tau_), tau_);
650 
651  // define further variables
652  VectorType g_k, beta_out;
653  beta_in = cache_iterator != cache_.end() ? cache_iterator->second : alpha_iso;
654  thread_local auto T_k = XT::Common::make_unique<MatrixType>();
655 
656  const auto& r_sequence = regularize ? r_sequence_ : std::vector<RangeFieldType>{0.};
657  const auto r_max = r_sequence.back();
658  for (const auto& r : r_sequence) {
659  // regularize u
660  v = u_prime;
661  if (r > 0) {
662  beta_in = alpha_iso;
663  VectorType r_times_u_iso = u_iso;
664  r_times_u_iso *= r;
665  v *= 1 - r;
666  v += r_times_u_iso;
667  }
668  *T_k = T_minus_one_;
669  // calculate T_k u
670  VectorType v_k = v;
671  // calculate values of basis p = S_k m
672  thread_local BasisValuesMatrixType P_k(M_.backend(), false, 0., 0);
673  std::copy_n(M_.data(), M_.rows() * M_.cols(), P_k.data());
674  // calculate f_0
675  RangeFieldType f_k = calculate_scalar_integral(beta_in, P_k);
676  f_k -= beta_in * v_k;
677 
678  thread_local auto H = XT::Common::make_unique<MatrixType>(0.);
679 
680  int pure_newton = 0;
681  for (size_t kk = 0; kk < k_max_; ++kk) {
682  // exit inner for loop to increase r if too many iterations are used or cholesky decomposition fails
683  if (kk > k_0_ && r < r_max)
684  break;
685  try {
686  change_basis(beta_in, v_k, P_k, *T_k, g_k, beta_out, *H);
687  } catch (const Dune::MathError&) {
688  if (r < r_max)
689  break;
690  mutex_.unlock();
691  const std::string err_msg =
692  "Failed to converge for " + XT::Common::to_string(u) + " with density "
693  + XT::Common::to_string(density) + " and multiplier " + XT::Common::to_string(beta_in)
694  + " at position " + XT::Common::to_string(entity().geometry().center())
695  + " due to errors in change_basis! Last u_eps_diff = " + XT::Common::to_string(u_eps_diff)
696  + ", first_error_cond = " + XT::Common::to_string(first_error_cond) + ", second_error_cond = "
697  + XT::Common::to_string(second_error_cond) + ", tau_prime = " + XT::Common::to_string(tau_prime);
698  DUNE_THROW(MathError, err_msg);
699  }
700  // calculate descent direction d_k;
701  VectorType d_k = g_k;
702  d_k *= -1;
703  // Calculate stopping criteria (in original basis). Variables with _k are in current basis, without k in
704  // original basis.
705  VectorType alpha_tilde;
706  XT::LA::solve_lower_triangular_transposed(*T_k, alpha_tilde, beta_out);
707  VectorType u_alpha_tilde;
708  calculate_vector_integral(alpha_tilde, M_, M_, u_alpha_tilde);
709  VectorType g_alpha_tilde = u_alpha_tilde - v;
710  auto density_tilde = basis_functions_.density(u_alpha_tilde);
711  if (!(density_tilde > 0.) || std::isinf(density_tilde))
712  break;
713  const auto alpha_prime = alpha_tilde - alpha_iso * std::log(density_tilde);
714  VectorType u_alpha_prime;
715  calculate_vector_integral(alpha_prime, M_, M_, u_alpha_prime);
716  u_eps_diff = v - u_alpha_prime * (1 - epsilon_gamma_);
717  VectorType d_alpha_tilde;
718  XT::LA::solve_lower_triangular_transposed(*T_k, d_alpha_tilde, d_k);
719  first_error_cond = g_alpha_tilde.two_norm();
720  second_error_cond = std::exp(d_alpha_tilde.one_norm() + std::abs(std::log(density_tilde)));
721  if (first_error_cond < tau_prime && 1 - epsilon_gamma_ < second_error_cond
722  && realizability_helper_.is_realizable(u_eps_diff, kk == static_cast<size_t>(0.8 * k_0_))) {
723  ret.first = alpha_prime + alpha_iso * std::log(density);
724  ret.second = r;
725  cache_.insert(v, alpha_prime);
726  alpha_storage_[x_local] = ret.first;
727  goto outside_all_loops;
728  } else {
729  RangeFieldType zeta_k = 1;
730  beta_in = beta_out;
731  // backtracking line search
732  // while (pure_newton >= 2 || zeta_k > epsilon_ * beta_out.two_norm() / d_k.two_norm() * 100.) {
733  while (pure_newton >= 2 || zeta_k > epsilon_ * beta_out.two_norm() / d_k.two_norm()) {
734  VectorType beta_new = d_k;
735  beta_new *= zeta_k;
736  beta_new += beta_out;
737  RangeFieldType f = calculate_scalar_integral(beta_new, P_k);
738  f -= beta_new * v_k;
739  if (pure_newton >= 2 || XT::Common::FloatCmp::le(f, f_k + xi_ * zeta_k * (g_k * d_k))) {
740  beta_in = beta_new;
741  f_k = f;
742  pure_newton = 0;
743  break;
744  }
745  zeta_k = chi_ * zeta_k;
746  } // backtracking linesearch while
747  if (zeta_k <= epsilon_ * beta_out.two_norm() / d_k.two_norm())
748  ++pure_newton;
749  } // else (stopping conditions)
750  } // k loop (Newton iterations)
751  } // r loop (Regularization parameter)
752  mutex_.unlock();
753  const std::string err_msg =
754  "Failed to converge for " + XT::Common::to_string(u) + " with density " + XT::Common::to_string(density)
755  + " and multiplier " + XT::Common::to_string(beta_in) + " at position "
756  + XT::Common::to_string(entity().geometry().center()) + " due to too many iterations! Last u_eps_diff = "
757  + XT::Common::to_string(u_eps_diff) + ", first_error_cond = " + XT::Common::to_string(first_error_cond)
758  + ", second_error_cond = " + XT::Common::to_string(second_error_cond) + ", tau_prime = "
759  + XT::Common::to_string(tau_prime);
760  DUNE_THROW(MathError, err_msg);
761  } // else ( value has not been calculated before )
762 
763  outside_all_loops:
764  mutex_.unlock();
765  return ret;
766  }

897  void calculate_hessian(const VectorType& alpha, const BasisValuesMatrixType& M, MatrixType& H) const
898  {
899  std::fill(H.begin(), H.end(), 0.);
900  auto& work_vec = working_storage();
901  calculate_scalar_products(alpha, M, work_vec);
902  apply_exponential(work_vec);
903  const size_t num_quad_points = quad_weights_.size();
904  // matrix is symmetric, we only use lower triangular part
905  for (size_t ll = 0; ll < num_quad_points; ++ll) {
906  auto factor_ll = work_vec[ll] * quad_weights_[ll];
907  const auto* basis_ll = M.get_ptr(ll);
908  for (size_t ii = 0; ii < dimRange; ++ii) {
909  auto* H_row = &(H[ii][0]);
910  const auto factor_ll_ii = basis_ll[ii] * factor_ll;
911  if (!XT::Common::is_zero(factor_ll_ii)) {
912  for (size_t kk = 0; kk <= ii; ++kk) {
913  H_row[kk] += basis_ll[kk] * factor_ll_ii;
914  } // kk
915  }
916  } // ii
917  } // ll

920  // J = df/dalpha is the derivative of the flux with respect to alpha.
921  // As F = (f_1, f_2, f_3) is matrix-valued
922  // (div f = \sum_{i=1}^d \partial_{x_i} f_i = \sum_{i=1}^d \partial_{x_i} < v_i m \hat{psi}(alpha) > is
923  // vector-valued),
924  // the derivative is the vector of matrices (df_1/dalpha, df_2/dalpha, ...)
925  // this function returns the dd-th matrix df_dd/dalpha of J
926  // assumes work_vecs already contains the needed exp(alpha * m) values
927  void calculate_J(const BasisValuesMatrixType& M,
928  Dune::FieldMatrix<RangeFieldType, dimRange, StateType::dimRange>& J_dd,
929  const size_t dd) const
930  {
931  assert(dd < dimRangeCols);
932  const auto& work_vecs = working_storage();
933  std::fill(J_dd.begin(), J_dd.end(), 0);
934  const size_t num_quad_points = quad_points_.size();
935  for (size_t ll = 0; ll < num_quad_points; ++ll) {
936  const auto factor_ll = work_vecs[ll] * quad_points_[ll][dd] * quad_weights_[ll];
937  const auto* basis_ll = M.get_ptr(ll);
938  for (size_t ii = 0; ii < dimRange; ++ii) {
939  const auto factor_ll_ii = factor_ll * basis_ll[ii];
940  if (!XT::Common::is_zero(factor_ll_ii)) {
941  for (size_t kk = 0; kk <= ii; ++kk)
942  J_dd[ii][kk] += basis_ll[kk] * factor_ll_ii;
943  }
944  } // ii
945  } // ll
946  // symmetric update for upper triangular part of J
947  for (size_t mm = 0; mm < dimRange; ++mm)
948  for (size_t nn = mm + 1; nn < dimRange; ++nn)
949  J_dd[mm][nn] = J_dd[nn][mm];

1400  AlphaReturnType get_alpha(const DomainType& x_local,
1401  const StateRangeType& u_in,
1402  const XT::Common::Parameter& param,
1403  const bool regularize) const
1404  {
1405  const bool boundary = static_cast<bool>(param.get("boundary")[0]);
1406  // get initial multiplier and basis matrix from last time step
1407  AlphaReturnType ret;
1408  StateRangeType v_in;
1409  mutex_.lock();
1410  if (boundary)
1411  cache_.set_capacity(cache_size + dimDomain);
1412 
1413  // rescale u such that the density <psi> is 1
1414  RangeFieldType density = basis_functions_.density(u_in);
1415  if (!(density > 0. || !(basis_functions_.min_density(u_in) > 0.)) || std::isinf(density)) {
1416  mutex_.unlock();
1417  DUNE_THROW(Dune::MathError, "Negative, inf or NaN density!");
1418  }
1419  StateRangeType u_prime_in = u_in / density;
1420  StateRangeType alpha_iso_in = basis_functions_.alpha_iso();
1421  BlockVectorType alpha_iso = alpha_iso_in;
1422 
1423  // if value has already been calculated for these values, skip computation
1424  const auto cache_iterator = cache_.find_closest(u_prime_in);
1425  if (cache_iterator != cache_.end() && XT::Common::FloatCmp::eq(cache_iterator->first, u_prime_in, 1e-14, 1e-14)) {
1426  const auto alpha_prime = cache_iterator->second;
1427  ret.first = alpha_prime + alpha_iso * std::log(density);
1428  ret.second = 0.;
1429  alpha_storage_[x_local] = ret.first;
1430  mutex_.unlock();
1431  return ret;
1432  } else {
1433  RangeFieldType tau_prime = std::min(
1434  tau_ / ((1 + std::sqrt(dimRange) * u_prime_in.two_norm()) * density + std::sqrt(dimRange) * tau_), tau_);
1435 
1436  // calculate moment vector for isotropic distribution
1437  StateRangeType u_iso_in = basis_functions_.u_iso();
1438 
1439  // define further variables
1440  BlockVectorType g_k, beta_in, beta_out, v;
1441  thread_local auto T_k = XT::Common::make_unique<BlockMatrixType>();
1442  beta_in = cache_iterator != cache_.end() ? cache_iterator->second : alpha_iso;
1443 
1444  const auto& r_sequence = regularize ? r_sequence_ : std::vector<RangeFieldType>{0.};
1445  const auto r_max = r_sequence.back();
1446  for (const auto& r : r_sequence) {
1447  // regularize u
1448  v_in = u_prime_in;
1449  if (r > 0.) {
1450  beta_in = alpha_iso;
1451  v_in = v_in * (1 - r) + u_iso_in * r;
1452  }
1453  v = v_in;
1454  *T_k = T_minus_one_;
1455  // calculate T_k u
1456  BlockVectorType v_k = v;
1457  // calculate values of basis p = S_k m
1458  thread_local BasisValuesMatrixType P_k(XT::LA::CommonDenseMatrix<RangeFieldType>(0, 0, 0., 0));
1459  copy_basis_matrix(M_, P_k);
1460  // calculate f_0
1461  RangeFieldType f_k = calculate_scalar_integral(beta_in, P_k) - beta_in * v_k;
1462 
1463  thread_local auto H = XT::Common::make_unique<BlockMatrixType>(0.);
1464 
1465  int pure_newton = 0;
1466  for (size_t kk = 0; kk < k_max_; ++kk) {
1467  // exit inner for loop to increase r if too many iterations are used or cholesky decomposition fails
1468  if (kk > k_0_ && r < r_max)
1469  break;
1470  try {
1471  change_basis(beta_in, v_k, P_k, *T_k, g_k, beta_out, *H);
1472  } catch (const Dune::MathError&) {
1473  if (r < r_max)
1474  break;
1475  mutex_.unlock();
1476  // std::cerr << "Failed to converge for " << XT::Common::to_string(u_in, 15) << " with
1477  // density "
1478  // << XT::Common::to_string(density, 15) << " at position "
1479  // << XT::Common::to_string(entity().geometry().center(), 15) << " due to too many
1480  // iterations!"
1481  // << std::endl;
1482  DUNE_THROW(Dune::MathError, "Failure to converge!");
1483  }
1484 
1485  // calculate descent direction d_k;
1486  BlockVectorType d_k = g_k;
1487  d_k *= -1;
1488 
1489  // Calculate stopping criteria (in original basis). Variables with _k are in current basis, without k in
1490  // original basis.
1491  BlockVectorType alpha_tilde, u_alpha_tilde, u_alpha_prime, d_alpha_tilde, g_alpha_tilde;
1492  // convert everything to original basis
1493  for (size_t jj = 0; jj < num_blocks; ++jj) {

1496  } // jj

1500  if (!(density_tilde > 0.) || !(basis_functions_.min_density(u_alpha_tilde) > 0.)
1501  || std::isinf(density_tilde))
1502  break;
1503  const auto alpha_prime = alpha_tilde - alpha_iso * std::log(density_tilde);
1504  calculate_vector_integral(alpha_prime, M_, M_, u_alpha_prime);
1505  auto u_eps_diff = v - u_alpha_prime * (1 - epsilon_gamma_);
1506  if (g_alpha_tilde.two_norm() < tau_prime
1507  && 1 - epsilon_gamma_ < std::exp(d_alpha_tilde.one_norm() + std::abs(std::log(density_tilde)))
1508  && helper<dimDomain>::is_realizable(u_eps_diff, basis_functions_)) {
1509  ret.first = alpha_prime + alpha_iso * std::log(density);
1510  ret.second = r;
1511  cache_.insert(v_in, alpha_prime);
1512  alpha_storage_[x_local] = ret.first;
1513  goto outside_all_loops;
1514  } else {

1517  // backtracking line search
1518  while (pure_newton >= 2 || zeta_k > epsilon_ * beta_out.two_norm() / d_k.two_norm()) {

1523  f_k = f;
1524  pure_newton = 0;
1525  break;

1531  } // else (stopping conditions)
1532  } // k loop (Newton iterations)
1533  } // r loop (Regularization parameter)
1534  mutex_.unlock();

1536  // << XT::Common::to_string(density, 15) << " at position "
1537  // << XT::Common::to_string(entity().geometry().center(), 15) << " due to too many iterations!"
1538  // << std::endl;
1539  DUNE_THROW(MathError, "Failed to converge");
1540  } // else ( value has not been calculated before )
1541 
1542  outside_all_loops:
