140  template <class IntersectionType>
141  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
142  const LocalfunctionTupleType& local_functions_tuple_neighbor,
143  const IntersectionType& intersection,
144  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
145  const DomainType& x_in_inside_coords,
146  const DomainType& x_in_outside_coords,
147  const RangeType& u_i,
148  const RangeType& u_j) const
149  {
150  // find direction of unit outer normal
151  size_t direction = intersection.indexInInside() / 2;
152 
153  const auto& local_flux_inside = std::get<0>(local_functions_tuple_entity);
154  const auto& local_flux_outside = std::get<0>(local_functions_tuple_neighbor);
155  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
156 
157  if (use_local_) {
158  if (!analytical_flux_.is_affine() || !(max_derivative_calculated()[direction])) {
159  if (!jacobian_inside()) {
160  jacobian_inside() = XT::Common::make_unique<JacobianRangeType>();
161  jacobian_outside() = XT::Common::make_unique<JacobianRangeType>();
162  }
163  DomainFieldType max_derivative(0);
164  helper<dimDomain>::get_jacobian(
165  direction, *local_flux_inside, x_in_inside_coords, u_i, *jacobian_inside(), param_inside_);
166  helper<dimDomain>::get_jacobian(
167  direction, *local_flux_outside, x_in_outside_coords, u_j, *jacobian_outside(), param_outside_);
168 
169  static XT::Common::Configuration eigensolver_options = create_eigensolver_options();
170  const auto eigen_solver_inside = EigenSolverType((*jacobian_inside())[direction], eigensolver_options);
171  const auto eigen_solver_outside = EigenSolverType((*jacobian_outside())[direction], eigensolver_options);
172  const auto& eigenvalues_inside = eigen_solver_inside.real_eigenvalues();
173  const auto& eigenvalues_outside = eigen_solver_outside.real_eigenvalues();
174  for (size_t jj = 0; jj < dimRange; ++jj)
175  max_derivative =
176  std::max({std::abs(eigenvalues_inside[jj]), std::abs(eigenvalues_outside[jj]), max_derivative});
177  lambda_ij()[direction] = 1. / max_derivative;
178  if (analytical_flux_.is_affine()) {
179  jacobian_inside() = nullptr;
180  jacobian_outside() = nullptr;
181  }
182  max_derivative_calculated()[direction] = true;
183  }
184  } else if (lambda_provided_) {
185  lambda_ij()[direction] = lambda_[direction];
186  } else {
187  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
188  lambda_ij()[direction] = dt_ / dx;
189  } // if (use_local)
190 
191  // calculate flux evaluation as
192  // ret[kk] = (f_u_i[kk] + f_u_j[kk])*n_ij*0.5 - (u_j - u_i)[kk]*1.0/(num_neighbors*lambda_ij)
193  auto second_part = u_j;
194  second_part -= u_i;
195  second_part /= lambda_ij()[direction] * 2 * alpha_;
196  auto ret = local_flux_inside->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
197  ret += local_flux_outside->evaluate_col(direction, x_in_outside_coords, u_j, param_outside_);
198  ret *= n_ij[direction] * 0.5;
199  ret -= second_part;
200  return ret;
201  } // ... evaluate(...)
