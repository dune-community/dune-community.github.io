224 template <class OperatorImp, class DiscreteFunctionImp, TimeStepperMethods method = TimeStepperMethods::dormand_prince>
225 class AdaptiveRungeKuttaTimeStepper : public TimeStepperInterface<DiscreteFunctionImp>
226 {
227  typedef TimeStepperInterface<DiscreteFunctionImp> BaseType;
228  typedef typename internal::AdaptiveButcherArrayProvider<typename BaseType::RangeFieldType, method>
229  ButcherArrayProviderType;
230 
231 public:
232  typedef OperatorImp OperatorType;
233  typedef DiscreteFunctionImp DiscreteFunctionType;
234 
235  typedef typename DiscreteFunctionType::DomainFieldType DomainFieldType;
236  typedef typename DiscreteFunctionType::RangeFieldType RangeFieldType;
237  typedef typename Dune::DynamicMatrix<RangeFieldType> MatrixType;
238  typedef typename Dune::DynamicVector<RangeFieldType> VectorType;
239  typedef typename std::vector<std::pair<RangeFieldType, DiscreteFunctionType>> SolutionType;
240 
255  AdaptiveRungeKuttaTimeStepper(const OperatorType& op,
256  DiscreteFunctionType& initial_values,
257  const RangeFieldType r = 1.0,
258  const double t_0 = 0.0,
259  const RangeFieldType tol = 1e-4,
260  const RangeFieldType scale_factor_min = 0.2,
261  const RangeFieldType scale_factor_max = 5,
262  const MatrixType& A = ButcherArrayProviderType::A(),
263  const VectorType& b_1 = ButcherArrayProviderType::b_1(),
264  const VectorType& b_2 = ButcherArrayProviderType::b_2(),
265  const VectorType& c = ButcherArrayProviderType::c())
266  : BaseType(t_0, initial_values)
267  , op_(op)
268  , r_(r)
269  , tol_(tol)
270  , scale_factor_min_(scale_factor_min)
271  , scale_factor_max_(scale_factor_max)
272  , u_tmp_(BaseType::current_solution())
273  , A_(A)
274  , b_1_(b_1)
275  , b_2_(b_2)
276  , c_(c)
277  , b_diff_(b_2_ - b_1_)
278  , num_stages_(A_.rows())
279  {
280  assert(Dune::XT::Common::FloatCmp::gt(tol_, 0.0));
281  assert(Dune::XT::Common::FloatCmp::le(scale_factor_min_, 1.0));
282  assert(Dune::XT::Common::FloatCmp::ge(scale_factor_max_, 1.0));
283  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
284  assert(b_1_.size() == A_.rows());
285  assert(b_2_.size() == A_.rows());
286  assert(c_.size() == A_.rows());
287 #ifndef NDEBUG
288  for (size_t ii = 0; ii < A_.rows(); ++ii) {
289  for (size_t jj = ii; jj < A_.cols(); ++jj) {
290  assert(Dune::XT::Common::FloatCmp::eq(A_[ii][jj], 0.0)
291  && "A has to be a lower triangular matrix with 0 on the main diagonal");
292  }
293  }
294 #endif // NDEBUG
295  // store as many discrete functions as needed for intermediate stages
296  for (size_t ii = 0; ii < num_stages_; ++ii) {
297  stages_k_.emplace_back(current_solution());
298  }
299  } // constructor AdaptiveRungeKuttaTimeStepper
300 
301  using BaseType::current_solution;
302  using BaseType::current_time;
303 
304  RangeFieldType step(const RangeFieldType dt, const RangeFieldType max_dt)
305  {
306  RangeFieldType actual_dt = std::min(dt, max_dt);
307  RangeFieldType mixed_error = std::numeric_limits<RangeFieldType>::max();
308  RangeFieldType time_step_scale_factor = 1.0;
309 
310  auto& t = current_time();
311  auto& u_n = current_solution();
312 
313  while (Dune::XT::Common::FloatCmp::gt(mixed_error, tol_)) {
314  actual_dt *= time_step_scale_factor;
315  size_t first_stage_to_compute = 0;
316  if (last_stage_of_previous_step_) {
317  stages_k_[0].vector() = last_stage_of_previous_step_->vector();
318  first_stage_to_compute = 1;
319  }
320 
321  for (size_t ii = first_stage_to_compute; ii < num_stages_; ++ii) {
322  stages_k_[ii].vector() *= RangeFieldType(0);
323  u_tmp_.vector() = u_n.vector();
324  for (size_t jj = 0; jj < ii; ++jj)
325  u_tmp_.vector() += stages_k_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
326  op_.apply(u_tmp_, stages_k_[ii], t + actual_dt * c_[ii]);
327  }
328 
329  // compute error vector
330  u_tmp_.vector() = stages_k_[0].vector() * b_diff_[0];
331  for (size_t ii = 1; ii < num_stages_; ++ii)
332  u_tmp_.vector() += stages_k_[ii].vector() * b_diff_[ii];
333  u_tmp_.vector() *= actual_dt * r_;
334 
335  // calculate u at timestep n+1
336  for (size_t ii = 0; ii < num_stages_; ++ii)
337  u_n.vector() += stages_k_[ii].vector() * (actual_dt * r_ * b_1_[ii]);
338 
339  // scale error, use absolute error if norm is less than 0.01 and relative error else
340  auto& diff_vector = u_tmp_.vector();
341  for (size_t ii = 0; ii < diff_vector.size(); ++ii) {
342  if (std::abs(u_n.vector()[ii]) > 0.01)
343  diff_vector[ii] /= std::abs(u_n.vector()[ii]);
344  }
345  mixed_error = diff_vector.sup_norm();
346  // scale dt to get the estimated optimal time step length, TODO: adapt formula
347  time_step_scale_factor =
348  std::min(std::max(0.9 * std::pow(tol_ / mixed_error, 1.0 / 5.0), scale_factor_min_), scale_factor_max_);
349 
350  if (mixed_error > tol_) { // go back from u at timestep n+1 to timestep n
351  for (size_t ii = 0; ii < num_stages_; ++ii)
352  u_n.vector() += stages_k_[ii].vector() * (-1.0 * r_ * actual_dt * b_1_[ii]);
353  }
354  } // while (mixed_error > tol_)
355 
356  if (!last_stage_of_previous_step_)
357  last_stage_of_previous_step_ = Dune::XT::Common::make_unique<DiscreteFunctionType>(u_n);
358  last_stage_of_previous_step_->vector() = stages_k_[num_stages_ - 1].vector();
359 
360  t += actual_dt;
361 
362  return actual_dt * time_step_scale_factor;
363  } // ... step(...)
364 
365 private:
366  const OperatorType& op_;
367  const RangeFieldType r_;
368  const RangeFieldType tol_;
369  const RangeFieldType scale_factor_min_;
370  const RangeFieldType scale_factor_max_;
371  DiscreteFunctionType u_tmp_;
372  const MatrixType A_;
373  const VectorType b_1_;
374  const VectorType b_2_;
375  const VectorType c_;
376  const VectorType b_diff_;
377  std::vector<DiscreteFunctionType> stages_k_;
378  const size_t num_stages_;
379  std::unique_ptr<DiscreteFunctionType> last_stage_of_previous_step_;
380 }; // class AdaptiveRungeKuttaTimeStepper
