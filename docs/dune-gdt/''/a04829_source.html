61 class LaxWendroffFluxImplementation
62 {

78  explicit LaxWendroffFluxImplementation(const AnalyticalFluxType& analytical_flux,
79  XT::Common::Parameter param,
80  const RangeFieldType alpha = dimDomain,
81  const bool boundary = false)
82  : analytical_flux_(analytical_flux)
83  , param_inside_(param)
84  , param_outside_(param)
85  , dt_(param.get("dt")[0])
86  , alpha_(alpha)
87  {
88  param_inside_.set("boundary", {0.}, true);
89  param_outside_.set("boundary", {double(boundary)}, true);
90  }

92  template <class IntersectionType>
93  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
94  const LocalfunctionTupleType& local_functions_tuple_neighbor,
95  const IntersectionType& intersection,
96  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
97  const DomainType& x_in_inside_coords,
98  const DomainType& x_in_outside_coords,
99  const RangeType& u_i,
100  const RangeType& u_j) const
101  {
102  // find direction of unit outer normal
103  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
104  size_t direction = intersection.indexInInside() / 2;
105 
106  const auto& local_flux_inside = std::get<0>(local_functions_tuple_entity);
107  const auto& local_flux_outside = std::get<0>(local_functions_tuple_neighbor);
108 
109  // calculate flux evaluation as
110  // ret = f((u_i + u_j)*0.5 - (f_u_j - f_u_i)*0.5*dt/dx*dimDomain)*n_ij[direction]
111  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
112  RangeType ret = u_i;
113  ret += u_j;
114  ret *= 0.5;
115  RangeType second_part = local_flux_outside->evaluate_col(direction, x_in_outside_coords, u_j, param_outside_);
116  second_part -= local_flux_inside->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
117  second_part *= n_ij[direction] * 0.5 * dt_ / dx * alpha_;
118  ret -= second_part;
119  ret = local_flux_inside->evaluate_col(direction, x_in_inside_coords, ret, param_inside_);
120  ret *= n_ij[direction];
121  return ret;
122  } // ... evaluate(...)
123 
124  const AnalyticalFluxType& analytical_flux() const
125  {
126  return analytical_flux_;
127  }
128 
129 private:
130  const AnalyticalFluxType& analytical_flux_;
131  XT::Common::Parameter param_inside_;
132  XT::Common::Parameter param_outside_;
133  const double dt_;
134  const RangeFieldType alpha_;
135 }; // class LaxWendroffFluxImplementation<...>
