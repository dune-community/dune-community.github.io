186  virtual RangeFieldType solve(const RangeFieldType t_end,
187  const RangeFieldType initial_dt,
188  const size_t num_save_steps,
189  const size_t num_output_steps,
190  const bool save_solution,
191  const bool visualize,
192  const bool write_discrete,
193  const bool write_exact,
194  const std::string prefix,
195  DiscreteSolutionType& sol,
196  const VisualizerType& visualizer,
197  const StringifierType& stringifier,
198  const LocalizableFunctionType& exact_solution)
199  {
200  RangeFieldType dt = initial_dt;
201  RangeFieldType t = current_time();
202  assert(Dune::XT::Common::FloatCmp::ge(t_end, t));
203  size_t time_step_counter = 0;
204 
205  const RangeFieldType save_interval = (t_end - t) / num_save_steps;
206  const RangeFieldType output_interval = (t_end - t) / num_output_steps;
207  RangeFieldType next_save_time = t + save_interval > t_end ? t_end : t + save_interval;
208  RangeFieldType next_output_time = t + output_interval > t_end ? t_end : t + output_interval;
209  size_t save_step_counter = 1;
210 
211  // save/visualize initial solution
212  if (save_solution)
213  sol.insert(std::make_pair(t, current_solution()));
214  write_files(visualize,
215  write_discrete,
216  write_exact,
217  current_solution(),
218  exact_solution,
219  prefix,
220  0,
221  t,
222  stringifier,
223  visualizer);
224 
225  while (Dune::XT::Common::FloatCmp::lt(t, t_end)) {
226  RangeFieldType max_dt = dt;
227  // match saving times and t_end exactly
228  if (Dune::XT::Common::FloatCmp::gt(t + dt, t_end))
229  max_dt = t_end - t;
230  if (Dune::XT::Common::FloatCmp::gt(t + dt, next_save_time) && num_save_steps != size_t(-1))
231  max_dt = std::min(next_save_time - t, max_dt);
232 
233  // do a timestep
234  dt = step(dt, max_dt);
235  t = current_time();
236 
237  // augment time step counter
238  ++time_step_counter;
239 
240  // check if data should be written in this timestep (and write)
241  if (Dune::XT::Common::FloatCmp::ge(t, next_save_time) || num_save_steps == size_t(-1)) {
242  if (save_solution)
243  sol.insert(sol.end(), std::make_pair(t, current_solution()));
244  write_files(visualize,
245  write_discrete,
246  write_exact,
247  current_solution(),
248  exact_solution,
249  prefix,
250  save_step_counter,
251  t,
252  stringifier,
253  visualizer);
254  next_save_time += save_interval;
255  ++save_step_counter;
256  }
257  if (num_output_steps && (Dune::XT::Common::FloatCmp::ge(t, next_output_time) || num_output_steps == size_t(-1))) {
258  std::cout << "time step " << time_step_counter << " done, time =" << t << ", current dt= " << dt << std::endl;
259  next_output_time += output_interval;
260  }
261  } // while (t < t_end)
262 
263  return dt;
264  } // ... solve(...)

385  static void write_to_textfile(const LocalizableFunctionType& u_n,
386  const GridLayerType& grid_layer,
387  const std::string& prefix,
388  const size_t step,
389  const RangeFieldType t,
390  const StringifierType& stringifier)
391  {
392  // write one file per MPI rank
393  std::ofstream rankfile(rankfile_name(prefix, grid_layer.comm().rank(), step));
394  for (const auto& entity : elements(grid_layer, Dune::Partitions::interiorBorder)) {
395  const auto local_func = u_n.local_function(entity);
396  const auto entity_center = entity.geometry().center();
397  for (const auto& intersection : Dune::intersections(grid_layer, entity)) {
398  auto position = intersection.geometry().center();
399  assert(position.size() == dimDomain);
400  // avoid ambiguity at interface
401  for (size_t ii = 0; ii < dimDomain; ++ii)
402  if (position[ii] < entity_center[ii])
403  position[ii] += 1e-6 * (entity_center[ii] - position[ii]);
404  const auto val = local_func->evaluate(entity.geometry().local(position), {"t", t});
405  for (size_t ii = 0; ii < dimDomain; ++ii)
406  rankfile << XT::Common::to_string(position[ii], 15) << " ";
407  rankfile << stringifier(val) << std::endl;
408  }
409  }
410  rankfile.close();
411  // Wait till files on all ranks are written
412  grid_layer.comm().barrier();
413  // Merge files
414  if (grid_layer.comm().rank() == 0) {
415  const std::string mergedfile_name = prefix + "_" + XT::Common::to_string(step) + ".txt";
416  std::remove(mergedfile_name.c_str());
417  std::ofstream merged_file(mergedfile_name, std::ios_base::binary | std::ios_base::app);
418  for (int ii = 0; ii < grid_layer.comm().size(); ++ii) {
419  const std::string rankfile_to_merge_name = rankfile_name(prefix, ii, step);
420  std::ifstream rankfile_to_merge(rankfile_to_merge_name, std::ios_base::binary);
421  merged_file << rankfile_to_merge.rdbuf();
422  rankfile_to_merge.close();
423  std::remove(rankfile_to_merge_name.c_str());
424  } // ii
425  merged_file.close();
426  } // if (rank == 0)
427  } // void write_to_textfile()
