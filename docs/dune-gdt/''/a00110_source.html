79  , boundary_integrand_(diffusion_factor_, diffusion_tensor_)

86  // The local lambda which does the actual work in the local operator:
87  // lmbd_source is src
88  // local_range is the local_discrete_function of rng on the current entity
89  [&](const auto& lmbd_source, auto& local_range) {
90  const auto& entity = local_range.entity();
91  const auto local_DoF_indices = local_range.space().local_DoF_indices(entity);
92  const auto local_diffusion_factor = diffusion_factor_.local_function(entity);
93  const auto local_diffusion_tensor = diffusion_tensor_.local_function(entity);
94  const auto local_source = lmbd_source.local_function(entity);
95  const auto local_basis = local_range.space().base_function_set(entity);
96  const auto local_constant_one = one_.local_function(entity);
97  // walk the intersections
98  const auto intersection_it_end = this->grid_layer().iend(entity);
99  for (auto intersection_it = this->grid_layer().ibegin(entity); intersection_it != intersection_it_end;
100  ++intersection_it) {
101  const auto& intersection = *intersection_it;
102  if (intersection.neighbor() && !intersection.boundary()) {
103  const auto neighbor = intersection.outside();
104  if (this->grid_layer().indexSet().index(entity) < this->grid_layer().indexSet().index(neighbor)) {
105  const auto local_diffusion_factor_neighbor = diffusion_factor_.local_function(neighbor);
106  const auto local_diffusion_tensor_neighbor = diffusion_tensor_.local_function(neighbor);
107  const auto local_source_neighbor = lmbd_source.local_function(neighbor);
108  const auto local_constant_one_neighbor = one_.local_function(neighbor);
109  const size_t local_intersection_index = intersection.indexInInside();
110  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
111  // do a face quadrature
112  R lhs = 0;
113  R rhs = 0;
114  const size_t integrand_order = inner_integrand_.order(*local_diffusion_factor,
115  *local_diffusion_tensor,
116  *local_diffusion_factor_neighbor,
117  *local_diffusion_tensor_neighbor,
118  *local_constant_one,
119  *local_source,
120  *local_constant_one_neighbor,
121  *local_source_neighbor);
122  for (const auto& quadrature_point : QuadratureRules<D, d - 1>::rule(
123  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_))) {
124  const auto& xx_intersection = quadrature_point.position();
125  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
126  const auto normal = intersection.unitOuterNormal(xx_intersection);
127  const R integration_factor = intersection.geometry().integrationElement(xx_intersection);
128  const R weigth = quadrature_point.weight();
129  // evalaute
130  local_basis.evaluate(xx_entity, tmp_basis_values_);
131  const auto& basis_value = tmp_basis_values_[local_DoF_index];
132  tmp_matrix_en_en_ *= 0.0;
133  tmp_matrix_en_ne_ *= 0.0;
134  inner_integrand_.evaluate(*local_diffusion_factor,
135  *local_diffusion_tensor,
136  *local_diffusion_factor_neighbor,
137  *local_diffusion_tensor_neighbor,
138  *local_constant_one,
139  *local_source,
140  *local_constant_one_neighbor,
141  *local_source_neighbor,
142  intersection,
143  xx_intersection,
144  tmp_matrix_en_en_, // <- we are interested in this one
145  tmp_matrix_,
146  tmp_matrix_en_ne_, // <- and this one
147  tmp_matrix_);
148  // compute integrals
149  assert(tmp_matrix_en_en_.rows() >= 1);
150  assert(tmp_matrix_en_en_.cols() >= 1);
151  assert(tmp_matrix_en_ne_.rows() >= 1);
152  assert(tmp_matrix_en_ne_.cols() >= 1);
153  lhs += integration_factor * weigth * (basis_value * normal);
154  rhs += integration_factor * weigth * (tmp_matrix_en_en_[0][0] + tmp_matrix_en_ne_[0][0]);
155  } // do a face quadrature
156  // set DoF
157  if (XT::Common::isnan(rhs) || XT::Common::isinf(rhs))
158  local_range.vector().set(local_DoF_index, 0.);
159  else
160  local_range.vector().set(local_DoF_index, rhs / lhs);
161  }
162  } else if (!intersection.neighbor()) {
163  const size_t local_intersection_index = intersection.indexInInside();
164  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
165  // do a face quadrature
166  R lhs = 0;
167  R rhs = 0;
168  const size_t integrand_order = boundary_integrand_.order(
169  *local_diffusion_factor, *local_diffusion_tensor, *local_source, *local_constant_one);
170  for (const auto& quadrature_point : QuadratureRules<D, d - 1>::rule(
171  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_))) {
172  const auto xx_intersection = quadrature_point.position();
173  const auto normal = intersection.unitOuterNormal(xx_intersection);
174  const R integration_factor = intersection.geometry().integrationElement(xx_intersection);
175  const R weigth = quadrature_point.weight();
176  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
177  // evalaute
178  local_basis.evaluate(xx_entity, tmp_basis_values_);
179  const auto& basis_value = tmp_basis_values_[local_DoF_index];
180  tmp_matrix_ *= 0.0;
181  boundary_integrand_.evaluate(*local_diffusion_factor,
182  *local_diffusion_tensor,
183  *local_constant_one,
184  *local_source,
185  intersection,
186  xx_intersection,
187  tmp_matrix_);

190  assert(tmp_matrix_.cols() >= 1);
191  lhs += integration_factor * weigth * (basis_value * normal);

194  // set DoF

196  local_range.vector().set(local_DoF_index, 0.);
197  else

199  } else

506  for (auto intersection_it = grid_layer_.ibegin(entity); intersection_it != intersection_it_end;
507  ++intersection_it) {
508  const auto& intersection = *intersection_it;
509  if (intersection.neighbor() && !intersection.boundary()) {
510  const auto neighbor = intersection.outside();
511  if (grid_layer_.indexSet().index(entity) < grid_layer_.indexSet().index(neighbor)) {
512  const auto local_diffusion_neighbor = diffusion_.local_function(neighbor);
513  const auto local_source_neighbor = source.local_function(neighbor);
514  const auto local_constant_one_neighbor = constant_one.local_function(neighbor);
515  const size_t local_intersection_index = intersection.indexInInside();
516  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
517  // do a face quadrature
518  FieldType lhs = 0;
519  FieldType rhs = 0;
520  const size_t integrand_order = inner_evaluation.order(*local_diffusion,
521  *local_diffusion_neighbor,
522  *local_constant_one,
523  *local_source,
524  *local_constant_one_neighbor,
525  *local_source_neighbor);
526  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
527  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
528  const auto quadrature_it_end = quadrature.end();
529  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
530  const auto& xx_intersection = quadrature_it->position();
531  xx_entity = intersection.geometryInInside().global(xx_intersection);
532  normal = intersection.unitOuterNormal(xx_intersection);
533  const auto integration_factor = intersection.geometry().integrationElement(xx_intersection);
534  const auto weigth = quadrature_it->weight();
535  // evalaute
536  local_basis.evaluate(xx_entity, basis_values);
537  const auto& basis_value = basis_values[local_DoF_index];
538  tmp_matrix_en_en *= 0.0;
539  tmp_matrix_en_ne *= 0.0;
540  inner_evaluation.evaluate(*local_diffusion,
541  *local_diffusion_neighbor,
542  *local_constant_one,
543  *local_source,
544  *local_constant_one_neighbor,
545  *local_source_neighbor,
546  intersection,
547  xx_intersection,
548  tmp_matrix_en_en, // <- we are interested in this one
549  tmp_matrix,
550  tmp_matrix_en_ne, // <- and this one
551  tmp_matrix);
552  // compute integrals
553  assert(tmp_matrix_en_en.rows() >= 1);
554  assert(tmp_matrix_en_en.cols() >= 1);
555  assert(tmp_matrix_en_ne.rows() >= 1);
556  assert(tmp_matrix_en_ne.cols() >= 1);
557  lhs += integration_factor * weigth * (basis_value * normal);
558  rhs += integration_factor * weigth * (tmp_matrix_en_en[0][0] + tmp_matrix_en_ne[0][0]);
559  } // do a face quadrature
560  // set DoF
561  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
562  // and make sure we are the first to do so
563  assert(!(range_vector[global_DoF_index] < infinity));
564  range_vector[global_DoF_index] = rhs / lhs;
565  }
566  } else if (intersection.boundary() && !intersection.neighbor()) {
567  const size_t local_intersection_index = intersection.indexInInside();
568  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
569  // do a face quadrature
570  FieldType lhs = 0;
571  FieldType rhs = 0;
572  const size_t integrand_order =
573  boundary_evaluation.order(*local_diffusion, *local_source, *local_constant_one);
574  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
575  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
576  const auto quadrature_it_end = quadrature.end();
577  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
578  const auto xx_intersection = quadrature_it->position();
579  normal = intersection.unitOuterNormal(xx_intersection);
580  const auto integration_factor = intersection.geometry().integrationElement(xx_intersection);
581  const auto weigth = quadrature_it->weight();
582  xx_entity = intersection.geometryInInside().global(xx_intersection);
583  // evalaute
584  local_basis.evaluate(xx_entity, basis_values);
585  const auto& basis_value = basis_values[local_DoF_index];
586  tmp_matrix *= 0.0;
587  boundary_evaluation.evaluate(
588  *local_diffusion, *local_constant_one, *local_source, intersection, xx_intersection, tmp_matrix);
589  // compute integrals
590  assert(tmp_matrix.rows() >= 1);
591  assert(tmp_matrix.cols() >= 1);
592  lhs += integration_factor * weigth * (basis_value * normal);
593  rhs += integration_factor * weigth * tmp_matrix[0][0];
594  } // do a face quadrature
595  // set DoF
596  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
597  // and make sure we are the first to do so
598  assert(!(range_vector[global_DoF_index] < infinity));
599  range_vector[global_DoF_index] = rhs / lhs;
600  } else
601  DUNE_THROW(XT::Common::Exceptions::internal_error, "Unknown intersection type!");
602  } // walk the intersections
603  } // walk the grid
604  } // ... apply(...)
