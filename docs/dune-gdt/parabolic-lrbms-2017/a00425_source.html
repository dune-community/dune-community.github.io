256  AdaptiveRungeKuttaTimeStepper(const OperatorType& op,
257  const DiscreteFunctionType& initial_values,
258  const RangeFieldType r = 1.0,
259  const double t_0 = 0.0,
260  const RangeFieldType tol = 1e-4,
261  const TimeFieldType scale_factor_min = 0.2,
262  const TimeFieldType scale_factor_max = 5,
263  const MatrixType& A = ButcherArrayProviderType::A(),
264  const VectorType& b_1 = ButcherArrayProviderType::b_1(),
265  const VectorType& b_2 = ButcherArrayProviderType::b_2(),
266  const TimeVectorType& c = ButcherArrayProviderType::c())
267  : BaseType(t_0, initial_values)
268  , op_(op)
269  , r_(r)
270  , tol_(tol)
271  , scale_factor_min_(scale_factor_min)
272  , scale_factor_max_(scale_factor_max)
273  , u_tmp_(BaseType::current_solution())
274  , A_(A)
275  , b_1_(b_1)
276  , b_2_(b_2)
277  , c_(c)
278  , b_diff_(b_2_ - b_1_)
279  , num_stages_(A_.rows())
280  {
281  assert(Dune::XT::Common::FloatCmp::gt(tol_, 0.0));
282  assert(Dune::XT::Common::FloatCmp::le(scale_factor_min_, 1.0));
283  assert(Dune::XT::Common::FloatCmp::ge(scale_factor_max_, 1.0));
284  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
285  assert(b_1_.size() == A_.rows());
286  assert(b_2_.size() == A_.rows());
287  assert(c_.size() == A_.rows());
288 #ifndef NDEBUG
289  for (size_t ii = 0; ii < A_.rows(); ++ii) {
290  for (size_t jj = ii; jj < A_.cols(); ++jj) {
291  assert(Dune::XT::Common::FloatCmp::eq(A_[ii][jj], 0.0)
292  && "A has to be a lower triangular matrix with 0 on the main diagonal");
293  }
294  }
295 #endif // NDEBUG
296  // store as many discrete functions as needed for intermediate stages
297  for (size_t ii = 0; ii < num_stages_; ++ii) {
298  u_intermediate_stages_.emplace_back(current_solution());
299  }
300  } // constructor AdaptiveRungeKuttaTimeStepper
301 
302  using BaseType::current_solution;
303  using BaseType::current_time;

305  TimeFieldType step(const TimeFieldType dt, const TimeFieldType max_dt)
306  {
307  TimeFieldType actual_dt = std::min(dt, max_dt);
308  RangeFieldType mixed_error = std::numeric_limits<RangeFieldType>::max();
309  TimeFieldType time_step_scale_factor = 1.0;
310 
311  auto& t = current_time();
312  auto& u_n = current_solution();
313 
314  while (Dune::XT::Common::FloatCmp::gt(mixed_error, tol_)) {
315  actual_dt *= time_step_scale_factor;
316  size_t first_stage_to_compute = 0;
317  if (last_stage_of_previous_step_) {
318  u_intermediate_stages_[0].vector() = last_stage_of_previous_step_->vector();
319  first_stage_to_compute = 1;
320  }
321 
322  for (size_t ii = first_stage_to_compute; ii < num_stages_; ++ii) {
323  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
324  u_tmp_.vector() = u_n.vector();
325  for (size_t jj = 0; jj < ii; ++jj)
326  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
327  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
328  }
329 
330  // compute error vector
331  u_tmp_.vector() = u_intermediate_stages_[0].vector() * b_diff_[0];
332  for (size_t ii = 1; ii < num_stages_; ++ii)
333  u_tmp_.vector() += u_intermediate_stages_[ii].vector() * b_diff_[ii];
334  u_tmp_.vector() *= actual_dt * r_;
335 
336  // calculate u at timestep n+1
337  for (size_t ii = 0; ii < num_stages_; ++ii)
338  u_n.vector() += u_intermediate_stages_[ii].vector() * (actual_dt * r_ * b_1_[ii]);
339 
340  // scale error, use absolute error if norm is less than 0.01 and relative error else
341  auto& diff_vector = u_tmp_.vector();
342  for (size_t ii = 0; ii < diff_vector.size(); ++ii) {
343  if (std::abs(u_n.vector()[ii]) > 0.01)
344  diff_vector[ii] /= std::abs(u_n.vector()[ii]);
345  }
346  mixed_error = diff_vector.sup_norm();
347  // scale dt to get the estimated optimal time step length, TODO: adapt formula
348  time_step_scale_factor =
349  std::min(std::max(0.9 * std::pow(tol_ / mixed_error, 1.0 / 5.0), scale_factor_min_), scale_factor_max_);
350 
351  if (mixed_error > tol_) { // go back from u at timestep n+1 to timestep n
352  for (size_t ii = 0; ii < num_stages_; ++ii)
353  u_n.vector() += u_intermediate_stages_[ii].vector() * (-1.0 * r_ * actual_dt * b_1_[ii]);
354  }
355  } // while (mixed_error > tol_)
356 
357  if (!last_stage_of_previous_step_)
358  last_stage_of_previous_step_ = Dune::XT::Common::make_unique<DiscreteFunctionType>(u_n);
359  last_stage_of_previous_step_->vector() = u_intermediate_stages_[num_stages_ - 1].vector();
360 
361  t += actual_dt;
362 
363  return actual_dt * time_step_scale_factor;
364  } // ... step(...)
