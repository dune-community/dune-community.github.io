52  : public MapperInterface<internal::RestrictedMapperTraits<UnrestrictedSpace, RestrictionGridLayer>>
53 {

61  RestrictedMapper(const UnrestrictedSpace& unrestricted_space, RestrictionGridLayer restriction_grid_layer)
62  : unrestricted_space_(unrestricted_space)
63  , grid_layer_(restriction_grid_layer)
64  , max_num_dofs_(0)
65  , map_to_restricted_()
66  , map_to_unrestricted_()
67  {
68  std::set<size_t> restricted_indices;
69  DynamicVector<size_t> unrestricted_indices(backend().maxNumDofs(), 0);
70  for (auto&& entity : elements(grid_layer_)) {
71  max_num_dofs_ = std::max(max_num_dofs_, backend().numDofs(entity));
72  backend().globalIndices(entity, unrestricted_indices);
73  for (size_t ii = 0; ii < backend().numDofs(entity); ++ii)
74  restricted_indices.insert(unrestricted_indices[ii]);
75  }
76  map_to_unrestricted_ = std::vector<size_t>(restricted_indices.size());
77  size_t counter = 0;
78  for (const auto& index : restricted_indices) {
79  map_to_restricted_[index] = counter;
80  map_to_unrestricted_[counter] = index;
81  ++counter;
82  }
83  if (map_to_restricted_.size() != map_to_unrestricted_.size())
84  DUNE_THROW(InvalidStateException, "This should not happen!");
85  }

94  void restrict(const XT::LA::VectorInterface<U>& unrestricted_vector,
95  XT::LA::VectorInterface<V>& restricted_vector) const
96  {
97  if (unrestricted_vector.size() != backend().size())
98  DUNE_THROW(XT::Common::Exceptions::shapes_do_not_match,
99  "unrestricted_vector.size() = " << unrestricted_vector.size()

103  DUNE_THROW(XT::Common::Exceptions::shapes_do_not_match,
