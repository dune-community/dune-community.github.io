179 template <class OperatorImp,
180  class DiscreteFunctionImp,
181  class SolverImp,
182  class TimeFieldImp = double,
183  RosenbrockTimeStepperMethods method = RosenbrockTimeStepperMethods::GRK4T>
184 class RosenbrockTimeStepper : public TimeStepperInterface<DiscreteFunctionImp, TimeFieldImp>
185 {
186  typedef TimeStepperInterface<DiscreteFunctionImp, TimeFieldImp> BaseType;
187  typedef typename internal::RosenbrockButcherArrayProvider<typename BaseType::RangeFieldType, TimeFieldImp, method>
188  ButcherArrayProviderType;

192  using typename BaseType::TimeFieldType;
193  using typename BaseType::DomainFieldType;
194  using typename BaseType::RangeFieldType;
195  using typename BaseType::SolutionType;
196 
197  typedef OperatorImp OperatorType;
198  typedef SolverImp SolverType;
199  typedef typename Dune::DynamicMatrix<RangeFieldType> MatrixType;
200  typedef typename Dune::DynamicVector<RangeFieldType> VectorType;
201  typedef typename Dune::DynamicVector<TimeFieldType> TimeVectorType;

203  using BaseType::current_solution;
204  using BaseType::current_time;

221  RosenbrockTimeStepper(const OperatorType& op,
222  const DiscreteFunctionType& initial_values,
223  const RangeFieldType r = 1.0,
224  const double t_0 = 0.0,
225  const RangeFieldType tol = 1e-4,
226  const TimeFieldType scale_factor_min = 0.2,
227  const TimeFieldType scale_factor_max = 5,
228  const MatrixType& A = ButcherArrayProviderType::A(),
229  const VectorType& b_1 = ButcherArrayProviderType::b_1(),
230  const VectorType& b_2 = ButcherArrayProviderType::b_2(),
231  const TimeVectorType& c = ButcherArrayProviderType::c(),
232  const MatrixType& Gamma = ButcherArrayProviderType::Gamma())
233  : BaseType(t_0, initial_values)
234  , op_(op)
235  , r_(r)
236  , tol_(tol)
237  , scale_factor_min_(scale_factor_min)
238  , scale_factor_max_(scale_factor_max)
239  , u_tmp_(BaseType::current_solution())
240  , A_(A)
241  , m_1_(b_1)
242  , m_2_(b_2)
243  , c_(c)
244  , d_(Gamma.rows())
245  , Gamma_(Gamma)
246  , m_diff_(b_1)
247  , num_stages_(A_.rows())
248  , gamma_ii_equal_for_all_i_(true)
249  {
250  assert(Dune::XT::Common::FloatCmp::gt(tol_, 0.0));
251  assert(Dune::XT::Common::FloatCmp::le(scale_factor_min_, 1.0));
252  assert(Dune::XT::Common::FloatCmp::ge(scale_factor_max_, 1.0));
253  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
254  assert(Gamma_.rows() == Gamma_.cols() && "Gamma has to be a square matrix");
255  assert(Gamma_.rows() == A_.rows() && "Sizes of A and Gamma have to match!");
256  assert(b_1.size() == A_.rows());
257  assert(b_2.size() == A_.rows());
258  assert(c_.size() == A_.rows());
259 #ifndef NDEBUG
260  for (size_t ii = 0; ii < A_.rows(); ++ii) {
261  TimeFieldType c_calculated = 0;
262  for (size_t jj = 0; jj < ii; ++jj)
263  c_calculated += A_[ii][jj];
264  assert(Dune::XT::Common::FloatCmp::eq(c_calculated, c_[ii]));
265  for (size_t jj = ii; jj < A_.cols(); ++jj) {
266  assert(Dune::XT::Common::FloatCmp::eq(A_[ii][jj], 0.0)
267  && "A has to be a lower triangular matrix with 0 on the main diagonal!");
268  if (jj == ii)
269  assert(Dune::XT::Common::FloatCmp::ne(Gamma_[ii][jj], 0.0)
270  && "The diagonal entries of Gamma must not vanish!");
271  else
272  assert(Dune::XT::Common::FloatCmp::eq(Gamma_[ii][jj], 0.0) && "Gamma has to be a lower triangular matrix!");
273  }
274  }
275 #endif // NDEBUG
276  // store as many discrete functions as needed for intermediate stages
277  for (size_t ii = 0; ii < num_stages_; ++ii) {
278  u_intermediate_stages_.emplace_back(current_solution());
279  }
280  // transform variables for faster calculations
281  // C_ = diag(1/gamma_11, ... , 1/gamma_nn) - Gamma^(-1), A_ = A*Gamma^(-1), m = (b_1,...,b_n)*Gamma^(-1), see e.g.
282  // Hairer, Wanner (1996), Solving ordinary differential equations II: Stiff and differential-algebraic problems,
283  // pp 119ff
284  auto Gamma_inv = Gamma_;
285  Gamma_inv.invert();
286  C_ = Gamma_inv;
287  C_ *= -1.0;
288  for (size_t ii = 0; ii < C_.rows(); ++ii)
289  C_[ii][ii] += 1.0 / (Gamma_[ii][ii]);
290  A_.rightmultiply(Gamma_inv);
291  Gamma_inv.mtv(b_1, m_1_);
292  Gamma_inv.mtv(b_2, m_2_);
293 
294  auto gamma = Gamma[0][0];
295  for (size_t ii = 0; ii < Gamma.rows(); ++ii) {
296  if (Dune::XT::Common::FloatCmp::ne(gamma, Gamma_[ii]))
297  gamma_ii_equal_for_all_i_ = false;
298  d_[ii] = 0.0;
299  for (size_t jj = 0; jj <= ii; ++jj)
300  d_[ii] += Gamma[ii][jj];
301  }
302  } // constructor RosenbrockTimeStepper
303 
304  TimeFieldType step(const TimeFieldType dt, const TimeFieldType max_dt)
305  {
306  TimeFieldType actual_dt = std::min(dt, max_dt);
307  RangeFieldType mixed_error = std::numeric_limits<RangeFieldType>::max();
308  TimeFieldType time_step_scale_factor = 1.0;
309 
310  auto& t = current_time();
311  auto& u_n = current_solution();
312 
313  while (Dune::XT::Common::FloatCmp::gt(mixed_error, tol_)) {
314  actual_dt *= time_step_scale_factor;
315 
316  for (size_t ii = 0; ii < num_stages_; ++ii) {
317  // GRK4A and GRK4T use the same coefficients for the third and fourth stage
318  if ((method == RosenbrockTimeStepperMethods::GRK4A || method == RosenbrockTimeStepperMethods::GRK4T) && ii == 3)
319  u_intermediate_stages_[ii].vector() = u_intermediate_stages_[ii - 1].vector();
320  else {
321  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
322  u_tmp_.vector() = u_n.vector();
323  for (size_t jj = 0; jj < ii; ++jj)
324  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * A_[ii][jj];
325  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
326  u_intermediate_stages_[ii].vector() *= r_;
327  }
328 
329  // TODO: If ii == 0, calculate (negative of) jacobian of L and partielle_Ableitung_von_L_nach_t
330  //
331  //
332  //
333 
334  // if gamma is the same for all i, we only need to calculate the matrix in the first step
335  if ((gamma_ii_equal_for_all_i_ && ii == 0) || !gamma_ii_equal_for_all_i_) {
336  // create solver
337  system_matrix_ = jacobian_ * -1.0 * r_;
338  for (size_t row = 0; row < system_matrix_.rows(); ++row)
339  system_matrix_.add_to_entry(row, row, 1.0 / (Gamma_[ii][ii] * actual_dt));
340  solver_ = Dune::XT::Common::make_unique<SolverType>(system_matrix_);
341  }
342  for (size_t jj = 0; jj < ii; ++jj)
343  u_intermediate_stages_[ii].vector().axpy(C_[ii][jj] / actual_dt, u_intermediate_stages_[jj].vector());
344  u_intermediate_stages_[ii].vector() += partielle_Ableitung_von_L_nach_t_ * (r_ * actual_dt * d_[ii]);
345  // solve, TODO: maybe need to do a copy before applying solver?
346  solver_->apply(u_intermediate_stages_[ii].vector(), u_intermediate_stages_[ii].vector());
347  }
348 
349  // compute error vector
350  u_tmp_.vector() = u_intermediate_stages_[0].vector() * m_diff_[0];
351  for (size_t ii = 1; ii < num_stages_; ++ii)
352  u_tmp_.vector() += u_intermediate_stages_[ii].vector() * m_diff_[ii];
353 
354  // calculate u at timestep n+1
355  for (size_t ii = 0; ii < num_stages_; ++ii)
356  u_n.vector() += u_intermediate_stages_[ii].vector() * (m_1_[ii]);
357 
358  // scale error, use absolute error if norm is less than 0.01 and relative error else
359  auto& diff_vector = u_tmp_.vector();
360  for (size_t ii = 0; ii < diff_vector.size(); ++ii) {
361  if (std::abs(u_n.vector()[ii]) > 0.01)
362  diff_vector[ii] /= std::abs(u_n.vector()[ii]);
363  }
364  mixed_error = diff_vector.sup_norm();
365  // scale dt to get the estimated optimal time step length, TODO: adapt formula
366  time_step_scale_factor =
367  std::min(std::max(0.9 * std::pow(tol_ / mixed_error, 1.0 / 4.0), scale_factor_min_), scale_factor_max_);
368 
369  if (mixed_error > tol_) { // go back from u at timestep n+1 to timestep n
370  for (size_t ii = 0; ii < num_stages_; ++ii)
371  u_n.vector() += u_intermediate_stages_[ii].vector() * (-1.0 * m_1_[ii]);
372  }
373  } // while (mixed_error > tol_)
374 
375  t += actual_dt;
376 
377  return actual_dt * time_step_scale_factor;
378  } // ... step(...)
