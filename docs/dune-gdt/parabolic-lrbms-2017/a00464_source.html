417  void solve(DiscreteSolutionType& solution) const
418  {
419 #if HAVE_EIGEN
420  try {
421  // get analytical flux, initial and boundary values
422  const std::shared_ptr<const AnalyticalFluxType> analytical_flux = problem().flux();
423  const std::shared_ptr<const InitialValueType> initial_values = problem().initial_values();
424  const std::shared_ptr<const BoundaryValueType> boundary_values = problem().boundary_values();
425  const std::shared_ptr<const RHSType> rhs = problem().rhs();
426 
427  // create a discrete function for the solution
428  DiscreteFunctionType u(*fv_space_, "solution");
429 
430  // project initial values
431  project(*initial_values, u);
432 
433  RangeFieldType t_end = test_case_.t_end();
434  const RangeFieldType CFL = problem().CFL();
435 
436  // calculate dx and choose initial dt
437  Dune::XT::Grid::Dimensions<typename SpaceType::GridLayerType> dimensions(fv_space_->grid_layer());
438  RangeFieldType dx = dimensions.entity_width.max();
439  if (dimDomain == 2)
440  dx /= std::sqrt(2);
441  RangeFieldType dt = CFL * dx;
442 
443  // create operators
444  const ConstantFunctionType dx_function(dx);
445  AdvectionOperatorType advection_operator =
446  internal::AdvectionOperatorCreator<AdvectionOperatorType, numerical_flux>::create(
447  *analytical_flux, *boundary_values, dx_function, dt, linear);
448  RHSOperatorType rhs_operator(*rhs);
449 
450  // create timestepper
451  OperatorTimeStepperType timestepper_op(advection_operator, u, -1.0);
452 
453  // do the time steps
454  const size_t num_save_steps = 100;
455  solution.clear();
456  if (problem().has_non_zero_rhs()) {
457  // use fractional step method
458  RHSOperatorTimeStepperType timestepper_rhs(rhs_operator, u);
459  TimeStepperType timestepper(timestepper_op, timestepper_rhs);
460  timestepper.solve(t_end, dt, num_save_steps, solution);
461  } else {
462  timestepper_op.solve(t_end, dt, num_save_steps, solution);
463  }
464 
465  } catch (Dune::Exception& e) {
466  std::cerr << "Dune reported: " << e.what() << std::endl;
467  std::abort();
468  }
469 #else // HAVE_EIGEN
470  (void)solution; // silence warning during compilation
471  static_assert(AlwaysFalse<DiscreteSolutionType>::value, "You are missing eigen!");
472 #endif // HAVE_EIGEN
473  }
