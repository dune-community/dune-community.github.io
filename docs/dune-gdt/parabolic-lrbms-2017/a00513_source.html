233  }
234 #endif // NDEBUG

253  virtual TimeFieldType step(const TimeFieldType dt, const TimeFieldType max_dt) override final
254  {
255  const TimeFieldType actual_dt = std::min(dt, max_dt);
256  auto& t = current_time();
257  auto& u_n = current_solution();
258  // calculate stages
259  for (size_t ii = 0; ii < num_stages_; ++ii) {
260  u_intermediate_stages_[ii].vector() *= RangeFieldType(0);
261  u_tmp_.vector() = u_n.vector();
262  for (size_t jj = 0; jj < ii; ++jj)
263  u_tmp_.vector() += u_intermediate_stages_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
264  op_.apply(u_tmp_, u_intermediate_stages_[ii], t + actual_dt * c_[ii]);
265  }
266 
267  // calculate value of u at next time step
268  for (size_t ii = 0; ii < num_stages_; ++ii)
269  u_n.vector() += u_intermediate_stages_[ii].vector() * (r_ * actual_dt * b_[ii]);
270 
271  // augment time
272  t += actual_dt;
273 
274  return dt;
275  } // ... step(...)

277  const std::pair<bool, TimeFieldType>
278  find_suitable_dt(const TimeFieldType initial_dt,
279  const TimeFieldType dt_refinement_factor = 2,
280  const RangeFieldType treshold = 0.9 * std::numeric_limits<RangeFieldType>::max(),
281  const size_t max_steps_per_dt = 20,
282  const size_t max_refinements = 20)
283  {
284  auto& t = current_time();
285  auto& u_n = current_solution();
286  assert(treshold > 0);
287  // save current state
288  DiscreteFunctionType initial_u_n = u_n;
289  TimeFieldType initial_t = t;
290  // start with initial dt
291  TimeFieldType current_dt = initial_dt;
292  size_t num_refinements = 0;
293  while (num_refinements < max_refinements) {
294  std::cout << "Trying time step length dt = " << current_dt << "... " << std::flush;
295  bool unlikely_value_occured = false;
296  size_t num_steps = 0;
297  // do max_steps_per_dt time steps...
298  while (!unlikely_value_occured) {
299  step(current_dt);
300  ++num_steps;
301  // ... unless there is a value above threshold
302  for (size_t kk = 0; kk < u_n.vector().size(); ++kk) {
303  if (std::abs(u_n.vector()[kk]) > treshold) {
304  unlikely_value_occured = true;
305  std::cout << "failed" << std::endl;
306  break;
307  }
308  }
309  // if we are able to do max_steps_per_dt time steps with this dt, we accept this dt
310  if (num_steps == max_steps_per_dt) {
311  std::cout << "looks fine" << std::endl;
312  u_n.vector() = initial_u_n.vector();
313  t = initial_t;
314  return std::make_pair(bool(true), current_dt);
315  }
316  }
317  // if there was a value above threshold start over with smaller dt
318  u_n.vector() = initial_u_n.vector();
319  t = initial_t;
320  current_dt /= dt_refinement_factor;
321  ++num_refinements;
322  }
323  return std::make_pair(bool(false), current_dt);
324  }
