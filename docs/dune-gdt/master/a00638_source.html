110  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
111  {
112  if (ret.size() < dimRange)
113  ret.resize(dimRange);
114  const size_t base = dimRange * backend_.index(entity);
115  for (size_t ii = 0; ii < dimRange; ++ii)
116  ret[ii] = base + ii;
117  } // ... globalIndices(...)
118 
119  using InterfaceType::globalIndices;
120 
121  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
122  {
123  assert(localIndex < dimRange);
124  return (dimRange * backend_.index(entity)) + localIndex;
125  }
126

128  const BackendType& backend_;
129 }; // class FvMapper< ..., rangeDim, 1 >
