118  template <class IntersectionType>
119  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple,
120  const IntersectionType& intersection,
121  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
122  const DomainType& x_in_inside_coords,
123  const DomainType& /*x_in_outside_coords*/,
124  const RangeType& u_i,
125  const RangeType& u_j) const
126  {
127  size_t direction = intersection.indexInInside() / 2;
128  // get unit outer normal
129  const auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
130  assert(XT::Common::FloatCmp::eq(std::abs(n_ij[direction]), 1.));
131 
132  // intialize jacobians
133  initialize_jacobians(direction, local_functions_tuple, x_in_inside_coords, u_i, u_j);
134 
135  // get jump at the intersection
136  const RangeType delta_u = u_i - u_j;
137  // calculate waves
138  RangeType waves(0);
139  n_ij[direction] > 0 ? jacobian_neg()[direction].mtv(delta_u, waves) : jacobian_pos()[direction].mtv(delta_u, waves);
140  // calculate flux
141  const auto& local_flux = std::get<0>(local_functions_tuple);
142  RangeType ret = local_flux->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
143  ret -= waves;
144  ret *= n_ij[direction];
145  return ret;
146  } // RangeType evaluate(...) const

159  // solve R^T F = P^T D A^T where is P is the permutation matrix and D is the diagonal eigenvalue matrix
160  static void solve_upper_triangular_transposed(CscSparseMatrixType& F,
161  const SparseMatrixType& R,
162  const FieldVector<int, dimRange>& permutations,
163  const StateRangeType& eigenvalues,
164  const SparseMatrixType& A)
165  {
166  F.clear();
167  StateRangeType rhs(0.), x(0.);
168  for (size_t cc = 0; cc < dimRange; ++cc) {
169  // apply permutations and scaling to column of A^T and store in vector
170  std::fill(rhs.begin(), rhs.end(), 0.);
171  for (size_t kk = A.outer_index_ptr()[cc]; kk < A.outer_index_ptr()[cc + 1]; ++kk) {
172  size_t index = A.inner_index_ptr()[kk];
173  auto eigval = eigenvalues[index];
174  if (XT::Common::FloatCmp::ne(eigval, 0.))
175  rhs[permutations[index]] = A.entries()[kk] * eigval;
176  } // kk
177  XT::LA::solve_upper_triangular_transposed(R, x, rhs);
178  F.start_column();
179  for (size_t rr = 0; rr < dimRange; ++rr)
180  if (XT::Common::FloatCmp::ne(x[rr], 0.))
181  F.push_entry(rr, x[rr]);
182  F.end_column();
183  } // cc
184  } // void solve_upper_triangular_transposed(...)

200  // use simple linearized Riemann solver, LeVeque p.316
201  void initialize_jacobians(const size_t direction,
202  const LocalfunctionTupleType& local_functions_tuple,
203  const DomainType& x_local,
204  const RangeType& u_i,
205  const RangeType& u_j) const
206  {
207  if (local_initialization_counts_[direction] != initialization_count_) {
208  // calculate jacobian as jacobian(0.5*(u_i+u_j)
209  RangeType u_mean = u_i + u_j;
210  u_mean *= RangeFieldType(0.5);
211  const auto& local_flux = std::get<0>(local_functions_tuple);
212  thread_local std::unique_ptr<MatrixType> jacobian;
213  thread_local RangeType tau;
214  if (!jacobian) {
215  jacobian = XT::Common::make_unique<MatrixType>();
216  }
217  helper<dimDomain>::get_jacobian(direction, *local_flux, x_local, u_mean, *jacobian, param_inside_);
218  // get matrix of eigenvectors A and eigenvalues
219  static auto eigensolver_options = create_eigensolver_options();
220  const auto eigen_solver = EigenSolverType(*jacobian, eigensolver_options);
221  const auto eigenvalues = eigen_solver.real_eigenvalues();
222  StateRangeType eigvals_neg(0.);
223  StateRangeType eigvals_pos(0.);
224  for (size_t ii = 0; ii < eigenvalues.size(); ++ii) {
225  if (eigenvalues[ii] < 0.)
226  eigvals_neg[ii] = eigenvalues[ii];
227  else
228  eigvals_pos[ii] = eigenvalues[ii];
229  }
230  thread_local std::unique_ptr<MatrixType> eigenvectors_dense = XT::Common::make_unique<MatrixType>();
231  *eigenvectors_dense = eigen_solver.real_eigenvectors();
232  thread_local SparseMatrixType eigenvectors(dimRange, dimRange, size_t(0));
233  eigenvectors = *eigenvectors_dense;
234  // calculate QR decomposition with column pivoting A = QRP^T
235  FieldVector<int, dimRange> permutations;
236  std::cout << "eigvecs = " << XT::Common::to_string(*eigenvectors_dense) << std::endl;
237  XT::LA::qr(*eigenvectors_dense, tau, permutations);
238  static auto upper_triangular_pattern =
239  XT::LA::triangular_pattern(dimRange, dimRange, XT::Common::MatrixPattern::upper_triangular);
240  thread_local SparseMatrixType R(dimRange, dimRange, size_t(0));
241  R.assign(*eigenvectors_dense, upper_triangular_pattern);
242  std::cout << "R = " << XT::Common::to_string(R) << std::endl;
243  const auto Q = XT::LA::calculate_q_from_qr(*eigenvectors_dense, tau);
244  std::cout << "Q = " << XT::Common::to_string(Q) << std::endl;
245  std::cout << "tau = " << XT::Common::to_string(tau) << std::endl;
246  std::cout << "permutations = " << XT::Common::to_string(permutations) << std::endl;
247  FieldVector<int, dimRange> inverse_permutations;
248  for (size_t ii = 0; ii < dimRange; ++ii)
249  inverse_permutations[permutations[ii]] = ii;
250  // we want to compute C_{+,-} = A D_{+,-} A^{-1}, where D_+ is the diagonal matrix containing only positive
251  // eigenvalues and D_- contains only negative eigenvalues. Using the QR decomposition, we get
252  // C = A D A^{-1} = A D (QRP^T)^{-1} = A D P R^{-1} Q^T. For the transpose of C, we get
253  // C^T = Q R^{-T} P^T D A^T. We calculate C^T by first solving (column-wise)
254  // R^T F = P^T D A^T and then computing C^T = Q F;
255  thread_local CscSparseMatrixType F_neg(dimRange, dimRange, size_t(0));
256  solve_upper_triangular_transposed(F_neg, R, inverse_permutations, eigvals_neg, eigenvectors);
257  jacobian_neg()[direction] = Q;
258  jacobian_pos()[direction].deep_copy(jacobian_neg()[direction]);
259  jacobian_neg()[direction].rightmultiply(F_neg);
260  auto& F_pos = F_neg;
261  solve_upper_triangular_transposed(F_pos, R, inverse_permutations, eigvals_pos, eigenvectors);
262  jacobian_pos()[direction].rightmultiply(F_pos);
263  if (analytical_flux_.is_affine()) {
264  jacobian = nullptr;
265  ++local_initialization_counts_[direction];
266  }
267  } // if (local_initialization_counts_[direction] != initialization_count_)
268  } // void calculate_jacobians(...)
