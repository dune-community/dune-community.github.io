54  : public XT::Grid::internal::Codim0Object<GridLayerType>

58  typedef XT::Grid::internal::Codim0Object<GridLayerType> BaseType;
59  typedef DirichletConstraints<XT::Grid::extract_intersection_t<GridLayerType>> ConstraintsType;
60 
61 public:
62  using typename BaseType::EntityType;
63 
64  ConstraintsWrapper(const XT::Common::PerThreadValue<const TestSpaceType>& test_space,
65  const XT::Common::PerThreadValue<const AnsatzSpaceType>& ansatz_space,
66  const XT::Grid::ApplyOn::WhichEntity<GridLayerType>* where,
67  ConstraintsType& constraints)
68  : test_space_(test_space)
69  , ansatz_space_(ansatz_space)
70  , where_(where)
71  , constraints_(constraints)
72  , thread_local_constraints_(constraints_.boundary_info(), constraints_.size())
73  {
74  }
75 
76  bool apply_on(const GridLayerType& gv, const EntityType& entity) const override final
77  {
78  return where_->apply_on(gv, entity);
79  }
80

82  {
83  test_space_->local_constraints(*ansatz_space_, entity, *thread_local_constraints_);
84  }
85 
86  void finalize() override final
87  {
88  DUNE_UNUSED std::lock_guard<std::mutex> mutex_guard(constraints_.mutex_);
89  constraints_.dirichlet_DoFs_.insert(thread_local_constraints_->dirichlet_DoFs_.begin(),
90  thread_local_constraints_->dirichlet_DoFs_.end());
91  }
