61  std::vector<size_t> local_DoF_indices_2dsimplex_order0(const EntityType& entity) const
62  {
63  static_assert(dimDomain == 2, "Not implemented!");
64  static_assert(polOrder == 0, "Not implemented!");
65  // prepare
66  const auto num_intersections = entity.subEntities(1);
67  std::vector<size_t> local_DoF_index_of_vertex(num_intersections, std::numeric_limits<size_t>::infinity());
68  std::vector<size_t> local_DoF_index_of_intersection(num_intersections, std::numeric_limits<size_t>::infinity());
69  typedef typename BaseFunctionSetType::DomainType DomainType;
70  std::vector<DomainType> vertices(num_intersections, DomainType(0));
71  std::vector<bool> lies_on_intersection(num_intersections, false);
72  DomainType corner(0);
73  typedef typename BaseFunctionSetType::RangeType RangeType;
74  const RangeType one(1);
75  std::vector<RangeType> basis_values(num_intersections, one);
76  const auto basis = this->base_function_set(entity);
77  assert(basis.size() == num_intersections);
78  const auto geometry = entity.geometry();
79  const auto& reference_element = ReferenceElements<DomainFieldType, dimDomain>::general(geometry.type());
80  // find the basis function index that corresponds to each vertex of the entity
81  // (find the basis function that evaluates to zero at the vertex, and nonzero at the other ones)
82  // therefore we walk the vertices
83  assert(num_intersections == entity.subEntities(dimDomain));
84  for (size_t vv = 0; vv < num_intersections; ++vv) {
85  const auto vertex = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(vv));
86  // get the vertex coordinates
87  vertices[vv] = vertex.geometry().center();
88  const auto& vertex_entity = reference_element.position(boost::numeric_cast<int>(vv), dimDomain);
89  // evalaute the basis
90  basis.evaluate(vertex_entity, basis_values);
91  // and find the basis that evaluates zero here
92  size_t zeros = 0;
93  size_t nonzeros = 0;
94  for (size_t ii = 0; ii < num_intersections; ++ii) {
95  // we would like to check against 0, but there is a bug in dune-commons FloatCmp
96  if (XT::Common::FloatCmp::eq(basis_values[ii] + one, one)) {
97  // this is a candidate for the basis function we are looking for
98  local_DoF_index_of_vertex[vv] = ii;
99  ++zeros;
100  } else
101  ++nonzeros;
102  }
103  // make sure there was only one candidate
104  if (zeros != 1 || nonzeros != (num_intersections - 1))
105  DUNE_THROW(XT::Common::Exceptions::internal_error,
106  "This must not happen for RTN0 in 2d!\n"
107  << " zeros = "
108  << zeros
109  << "\n"
110  << " nonzeros = "
111  << nonzeros);
112  } // walk the vertices
113  // so from here on we have the local DoF index that corresponds to each vertex vv in local_DoF_index_of_vertex[vv]
114  // now we need to find the intersection opposite to this vertex
115  // therefore we walk the intersections
116  size_t intersection_counter = 0;
117  const auto intersection_it_end = this->grid_layer().iend(entity);
118  for (auto intersection_it = this->grid_layer().ibegin(entity); intersection_it != intersection_it_end;
119  ++intersection_it) {
120  const auto& intersection = *intersection_it;
121  const auto intersection_geometry = intersection.geometry();
122  const size_t local_intersection_index = intersection.indexInInside();
123  // make sure this index has not been already taken by another intersection
124  assert(local_DoF_index_of_intersection[local_intersection_index] == std::numeric_limits<size_t>::infinity());
125  // walk the corners of the intersection
126  for (int cc = 0; cc < intersection_geometry.corners(); ++cc) {
127  corner = intersection_geometry.corner(boost::numeric_cast<int>(cc));
128  // check which vertices lie on the intersection
129  for (size_t vv = 0; vv < num_intersections; ++vv)
130  if (XT::Common::FloatCmp::eq(vertices[vv], corner))
131  lies_on_intersection[vv] = true;
132  } // walk the corners of the intersection
133  // now see if we find a vertex that does not lie on the intersection
134  size_t found = 0;
135  size_t missed = 0;
136  for (size_t vv = 0; vv < num_intersections; ++vv) {
137  if (!(lies_on_intersection[vv])) {
138  // this is a good candidate
139  // so the local DoF id that corresponds to this vertex is the one that corresponds to the intersection
140  local_DoF_index_of_intersection[local_intersection_index] = local_DoF_index_of_vertex[vv];
141  ++found;
142  } else
143  ++missed;
144  // and clear for the next intersection
145  lies_on_intersection[vv] = false;
146  } // walk the vertices of this entity
147  // make sure there was only one candidate
148  if (found != 1 || missed != (num_intersections - 1))
149  DUNE_THROW(XT::Common::Exceptions::internal_error,
150  "This must not happen for RTN0 in 2d!\n"
151  << " found = "
152  << found
153  << "\n"
154  << " missed = "
155  << missed);
156  ++intersection_counter;
157  } // walk the intersection
158  assert(intersection_counter == num_intersections);
159  return local_DoF_index_of_intersection;
160  } // ... local_DoF_indices_2dsimplex_order0(...)
