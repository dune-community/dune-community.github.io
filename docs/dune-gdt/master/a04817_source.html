82  void
83  apply2(const TestBaseType& test_base, const AnsatzBaseType& ansatz_base, DynamicMatrix<FieldType>& ret) const override
84  {
85  const auto& entity = ansatz_base.entity();
86  const auto local_functions = integrand_.localFunctions(entity);
87  // create quadrature
88  const size_t integrand_order = integrand_.order(local_functions, test_base, ansatz_base) + over_integrate_;
89  const auto& quadrature = QuadratureRules<D, d>::rule(entity.type(), boost::numeric_cast<int>(integrand_order));
90  // prepare storage
91  const size_t rows = test_base.size();
92  const size_t cols = ansatz_base.size();
93  ret *= 0.0;
94  assert(ret.rows() >= rows);
95  assert(ret.cols() >= cols);
96  DynamicMatrix<FieldType> integrand_eval(rows, cols, 0.); // \todo: make mutable member, after SMP refactor
97  // loop over all quadrature points
98  for (const auto& quadrature_point : quadrature) {
99  const auto xx = quadrature_point.position();
100  // integration factors
101  const auto integration_factor = entity.geometry().integrationElement(xx);
102  const auto quadrature_weight = quadrature_point.weight();
103  // evaluate the integrand
104  integrand_.evaluate(local_functions, test_base, ansatz_base, xx, integrand_eval);
105  // compute integral
106  for (size_t ii = 0; ii < rows; ++ii) {
107  const auto& integrand_eval_row = integrand_eval[ii];
108  auto& ret_row = ret[ii];
109  for (size_t jj = 0; jj < cols; ++jj)
110  ret_row[jj] += integrand_eval_row[jj] * integration_factor * quadrature_weight;
111  } // compute integral
112  } // loop over all quadrature points
113  } // ... apply2(...)
114 
115 private:
116  const BinaryEvaluationType integrand_;
117  const size_t over_integrate_;

199  void apply2(const TestBaseEntityType& test_base_en,
200  const AnsatzBaseEntityType& ansatz_base_en,
201  const TestBaseNeighborType& test_base_ne,
202  const AnsatzBaseNeighborType& ansatz_base_ne,
203  const IntersectionType& intersection,
204  DynamicMatrix<FieldType>& ret_en_en,
205  DynamicMatrix<FieldType>& ret_ne_ne,
206  DynamicMatrix<FieldType>& ret_en_ne,
207  DynamicMatrix<FieldType>& ret_ne_en) const override
208  {
209  // local inducing function
210  const auto& entity = test_base_en.entity();
211  const auto& neighbor = test_base_ne.entity();
212  const auto local_functions_en = integrand_.localFunctions(entity);
213  const auto local_functions_ne = integrand_.localFunctions(neighbor);
214  // quadrature
215  const size_t integrand_order =
216  integrand_.order(
217  local_functions_en, local_functions_ne, test_base_en, ansatz_base_en, test_base_ne, ansatz_base_ne)
218  + over_integrate_;
219  const auto& quadrature =
220  QuadratureRules<D, d - 1>::rule(intersection.type(), boost::numeric_cast<int>(integrand_order));
221  // check matrices
222  ret_en_en *= 0.0;
223  ret_ne_ne *= 0.0;
224  ret_en_ne *= 0.0;
225  ret_ne_en *= 0.0;
226  const size_t rows_en = test_base_en.size();
227  const size_t cols_en = ansatz_base_en.size();
228  const size_t rows_ne = test_base_ne.size();
229  const size_t cols_ne = ansatz_base_ne.size();
230  assert(ret_en_en.rows() >= rows_en);
231  assert(ret_en_en.cols() >= cols_en);
232  assert(ret_ne_ne.rows() >= rows_ne);
233  assert(ret_ne_ne.cols() >= cols_ne);
234  assert(ret_en_ne.rows() >= rows_en);
235  assert(ret_en_ne.cols() >= cols_ne);
236  assert(ret_ne_en.rows() >= rows_en);
237  assert(ret_ne_en.cols() >= cols_en);
238  // \todo: make mutable member, after SMP refactor
239  DynamicMatrix<FieldType> integrand_eval_en_en(rows_en, cols_en, 0.);
240  DynamicMatrix<FieldType> integrand_eval_ne_ne(rows_ne, cols_ne, 0.);
241  DynamicMatrix<FieldType> integrand_eval_en_ne(rows_en, cols_ne, 0.);
242  DynamicMatrix<FieldType> integrand_eval_ne_en(rows_ne, cols_en, 0.);
243  // loop over all quadrature points
244  for (const auto& quadrature_point : quadrature) {
245  const auto xx = quadrature_point.position();
246  const auto integration_factor = intersection.geometry().integrationElement(xx);
247  const auto quadrature_weight = quadrature_point.weight();
248  // evaluate local
249  integrand_.evaluate(local_functions_en,
250  local_functions_ne,
251  test_base_en,
252  ansatz_base_en,
253  test_base_ne,
254  ansatz_base_ne,
255  intersection,
256  xx,
257  integrand_eval_en_en,
258  integrand_eval_ne_ne,
259  integrand_eval_en_ne,
260  integrand_eval_ne_en);
261  // compute integrals
262  // loop over all entity test basis functions
263  for (size_t ii = 0; ii < rows_en; ++ii) {
264  auto& ret_en_en_row = ret_en_en[ii];
265  auto& ret_en_ne_row = ret_en_ne[ii];
266  const auto& integrand_eval_en_en_row = integrand_eval_en_en[ii];
267  const auto& integrand_eval_en_ne_row = integrand_eval_en_ne[ii];
268  // loop over all entity ansatz basis functions
269  for (size_t jj = 0; jj < cols_en; ++jj) {
270  ret_en_en_row[jj] += integrand_eval_en_en_row[jj] * integration_factor * quadrature_weight;
271  }
272  // loop over all neighbor ansatz basis functions
273  for (size_t jj = 0; jj < cols_ne; ++jj) {
274  ret_en_ne_row[jj] += integrand_eval_en_ne_row[jj] * integration_factor * quadrature_weight;
275  }
276  }
277  // loop over all neighbor test basis functions
278  for (size_t ii = 0; ii < rows_ne; ++ii) {
279  auto& ret_ne_ne_row = ret_ne_ne[ii];
280  auto& ret_ne_en_row = ret_ne_en[ii];
281  const auto& integrand_eval_ne_ne_row = integrand_eval_ne_ne[ii];
282  const auto& integrand_eval_ne_en_row = integrand_eval_ne_en[ii];
283  // loop over all neighbor ansatz basis functions
284  for (size_t jj = 0; jj < cols_ne; ++jj) {
285  ret_ne_ne_row[jj] += integrand_eval_ne_ne_row[jj] * integration_factor * quadrature_weight;
286  }
287  // loop over all entity ansatz basis functions
288  for (size_t jj = 0; jj < cols_en; ++jj) {
289  ret_ne_en_row[jj] += integrand_eval_ne_en_row[jj] * integration_factor * quadrature_weight;
290  }
291  }
292  } // loop over all quadrature points
293  } // void apply(...) const

297  const size_t over_integrate_;
