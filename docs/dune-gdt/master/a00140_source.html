60  MatrixSolveFunctor(const std::vector<MatrixType>& matrices,

66  {

69  virtual void apply_local(const EntityType& entity)
70  {
71  // get mapper
72  const auto& mapper = rhs_.space().mapper();
73 
74  // copy rhs to DynamicVector
75  DynamicVector<typename MatrixType::value_type> local_solution(mapper.numDofs(entity), 0.);
76  DynamicVector<typename MatrixType::value_type> local_rhs(local_solution.size(), 0.);
77  const auto& rhs_vector = rhs_.vector();
78  auto& solution_vector = solution_.vector();
79  const auto global_indices = mapper.globalIndices(entity);
80  for (size_t ii = 0; ii < local_rhs.size(); ++ii)
81  local_rhs[ii] = rhs_vector.get_entry(global_indices[ii]);
82  // solve
83  matrices_[rhs_.space().grid_layer().indexSet().index(entity)].solve(local_solution, local_rhs);
84  // write solution
85  for (size_t ii = 0; ii < local_rhs.size(); ++ii)
86  solution_vector.set_entry(global_indices[ii], local_solution[ii]);
87  }
88 
89 private:
90  const std::vector<MatrixType>& matrices_;
91  const DiscreteFunctionType& rhs_;
92  DiscreteFunctionType& solution_;
93 };
