88  std::vector<DomainType> lagrange_points_order_1(const EntityType& entity) const
89  {
90  // check
91  static_assert(polOrder == 1, "Not tested for higher polynomial orders!");
92  if (dimRange != 1)
93  DUNE_THROW(NotImplemented, "Does not work for higher dimensions");
94  assert(this->grid_layer().indexSet().contains(entity));
95  // get the basis and reference element
96  const auto basis = this->base_function_set(entity);
97  typedef typename BaseType::BaseFunctionSetType::RangeType RangeType;
98  std::vector<RangeType> tmp_basis_values(basis.size(), RangeType(0));
99  const auto& reference_element = ReferenceElements<DomainFieldType, dimDomain>::general(entity.type());
100  const auto num_vertices = reference_element.size(dimDomain);
101  assert(num_vertices >= 0);
102  assert(boost::numeric_cast<size_t>(num_vertices) == basis.size() && "This should not happen with polOrder 1!");
103  // prepare return vector
104  std::vector<DomainType> local_vertices(num_vertices, DomainType(0));
105  // loop over all vertices
106  for (auto ii : Dune::XT::Common::value_range(num_vertices)) {
107  // get the local coordinate of the iith vertex
108  const auto local_vertex = reference_element.position(ii, dimDomain);
109  // evaluate the basefunctionset
110  basis.evaluate(local_vertex, tmp_basis_values);
111  // find the basis function that evaluates to one here (has to be only one!)
112  size_t ones = 0;
113  size_t zeros = 0;
114  size_t failures = 0;
115  for (size_t jj = 0; jj < basis.size(); ++jj) {
116  if (std::abs((tmp_basis_values)[jj][0] - RangeFieldType(1)) < compare_tolerance_) {
117  local_vertices[jj] = local_vertex;
118  ++ones;
119  } else if (std::abs((tmp_basis_values)[jj][0]) < compare_tolerance_)
120  ++zeros;
121  else
122  ++failures;
123  }
124  assert(ones == 1 && zeros == (basis.size() - 1) && failures == 0 && "This must not happen for polOrder 1!");
125  }
126  return local_vertices;
127  } // ... lagrange_points_order_1(...)

129  std::set<size_t> local_dirichlet_DoFs_order_1(
130  const EntityType& entity,
131  const XT::Grid::BoundaryInfo<XT::Grid::extract_intersection_t<GridLayerType>>& boundaryInfo) const
132  {
133  static_assert(polOrder == 1, "Not tested for higher polynomial orders!");
134  static const XT::Grid::DirichletBoundary dirichlet{};
135  if (dimRange != 1)
136  DUNE_THROW(NotImplemented, "Does not work for higher dimensions");
137  // check
138  assert(this->grid_layer().indexSet().contains(entity));
139  if (!entity.hasBoundaryIntersections())
140  return std::set<size_t>();
141  // prepare
142  std::set<size_t> localDirichletDofs;
143  std::vector<DomainType> dirichlet_vertices;
144  // get all dirichlet vertices of this entity, therefore
145  // * loop over all intersections
146  const auto intersection_it_end = this->grid_layer().iend(entity);
147  for (auto intersection_it = this->grid_layer().ibegin(entity); intersection_it != intersection_it_end;
148  ++intersection_it) {
149  // only work on dirichlet ones
150  const auto& intersection = *intersection_it;
151  // actual dirichlet intersections + process boundaries for parallel runs
152  if (boundaryInfo.type(intersection) == dirichlet || (!intersection.neighbor() && !intersection.boundary())) {
153  // and get the vertices of the intersection
154  const auto geometry = intersection.geometry();
155  for (auto cc : Dune::XT::Common::value_range(geometry.corners()))
156  dirichlet_vertices.emplace_back(entity.geometry().local(geometry.corner(cc)));
157  } // only work on dirichlet ones
158  } // loop over all intersections
159  // find the corresponding basis functions
160  const auto basis = this->base_function_set(entity);
161  typedef typename BaseType::BaseFunctionSetType::RangeType RangeType;
162  std::vector<RangeType> tmp_basis_values(basis.size(), RangeType(0));
163  for (size_t cc = 0; cc < dirichlet_vertices.size(); ++cc) {
164  // find the basis function that evaluates to one here (has to be only one!)
165  basis.evaluate(dirichlet_vertices[cc], tmp_basis_values);
166  size_t ones = 0;
167  size_t zeros = 0;
168  size_t failures = 0;
169  for (size_t jj = 0; jj < basis.size(); ++jj) {
170  if (std::abs(tmp_basis_values[jj][0] - RangeFieldType(1)) < compare_tolerance_) {
171  localDirichletDofs.insert(jj);
172  ++ones;
173  } else if (std::abs(tmp_basis_values[jj][0]) < compare_tolerance_)
174  ++zeros;
175  else
176  ++failures;
177  }
178  // asserts valid for polorder 1 only
179  assert(ones == 1);
180  assert(zeros == (basis.size() - 1));
181  assert(failures == 0);
182  }
183  return localDirichletDofs;
184  } // ... local_dirichlet_DoFs_order_1(...)

186  std::set<size_t> local_dirichlet_DoFs_simplicial_lagrange_elements(
187  const EntityType& entity,
188  const XT::Grid::BoundaryInfo<XT::Grid::extract_intersection_t<GridLayerType>>& boundaryInfo) const
189  {
190  if (!entity.type().isSimplex())
191  DUNE_THROW(NotImplemented, "Only implemented for simplex elements!");
192  // check
193  assert(this->grid_layer().indexSet().contains(entity));
194  // prepare
195  std::set<size_t> localDirichletDofs;
196  std::vector<DomainType> dirichlet_vertices;
197  // get all dirichlet vertices of this entity, therefore
198  // * loop over all intersections
199  const auto intersection_it_end = this->grid_layer().iend(entity);
200  for (auto intersection_it = this->grid_layer().ibegin(entity); intersection_it != intersection_it_end;
201  ++intersection_it) {
202  const auto& intersection = *intersection_it;
203  std::vector<StuffDomainType> dirichlet_vertices_intersection;
204  // only work on dirichlet ones, actual dirichlet intersections + process boundaries for parallel runs
205  if (boundaryInfo.dirichlet(intersection) || (!intersection.neighbor() && !intersection.boundary())) {
206  // and get the vertices of the intersection
207  const auto geometry = intersection.geometry();
208  for (auto cc : Dune::XT::Common::value_range(geometry.corners())) {
209  dirichlet_vertices_intersection.emplace_back(entity.geometry().local(geometry.corner(cc)));
210  dirichlet_vertices.emplace_back(entity.geometry().local(geometry.corner(cc)));
211  }
212  } // only work on dirichlet ones
213  // for higher polynomial orders, add points associated to DoFs within the intersection
214  // calculate by using lagrange grid {x = \sum_{j=0}^d \lambda_j a_j | \sum_j lambda_j = 1}, where a_j are the
215  // vertices of the entity and \lambda_j \in {\frac{m}{polOrder} | m = 0, ... , polOrder}
216  std::vector<double> possible_coefficients(polOrder < 1 ? 0 : polOrder - 1);
217  for (int m = 0; m < polOrder; ++m)
218  possible_coefficients[m] = m / polOrder;
219  std::set<std::vector<double>> possible_coefficient_vectors;
220  possible_convex_combination_coefficients(
221  possible_coefficient_vectors, possible_coefficients, dirichlet_vertices_intersection.size());
222  for (const auto& coefficient_vector : possible_coefficient_vectors)
223  dirichlet_vertices.emplace_back(std::inner_product(dirichlet_vertices_intersection.begin(),
224  dirichlet_vertices_intersection.end(),
225  coefficient_vector.begin(),
226  StuffDomainType(0)));
227  } // loop over all intersections
228 
229  // find the corresponding basis functions
230  const auto basis = this->base_function_set(entity);
231  typedef typename BaseType::BaseFunctionSetType::RangeType RangeType;
232  std::vector<RangeType> tmp_basis_values(basis.size(), RangeType(0));
233  for (size_t cc = 0; cc < dirichlet_vertices.size(); ++cc) {
234  // find the basis function that evaluates to one here (has to be only one per range dimension!)
235  basis.evaluate(dirichlet_vertices[cc], tmp_basis_values);
236  size_t ones = 0;
237  size_t zeros = 0;
238  size_t failures = 0;
239  for (size_t jj = 0; jj < basis.size(); ++jj) {
240  for (size_t rr = 0; rr < dimRange; ++rr) {
241  if (std::abs(tmp_basis_values[jj][rr] - RangeFieldType(1)) < compare_tolerance_) {
242  localDirichletDofs.insert(jj);
243  ++ones;
244  } else if (std::abs(tmp_basis_values[jj][rr]) < compare_tolerance_)
245  ++zeros;
246  else
247  ++failures;
248  }
249  }
250  assert(ones == dimRange && zeros == ((basis.size() - 1) * dimRange) && failures == 0 && "This must not happen!");
251  }
252  return localDirichletDofs;
253  } // ... local_dirichlet_DoFs_simplicial_lagrange_elements(...)

306  void possible_convex_combination_coefficients(std::set<std::vector<double>>& vectors_in,
307  const std::vector<double>& possible_coefficients,
308  const size_t final_size) const
309  {
310  if (vectors_in.empty()) {
311  if (final_size != 0)
312  for (const auto& coeff : possible_coefficients)
313  vectors_in.insert(std::vector<double>(1, coeff));
314  } else if (vectors_in.begin()->size() != final_size) {
315  std::set<std::vector<double>> vectors_out;
316  for (auto& vec : vectors_in) {
317  for (const auto& coeff : possible_coefficients) {
318  if ((vec.size() != final_size - 1
319  && Dune::XT::Common::FloatCmp::le(std::accumulate(vec.begin(), vec.end(), 0.0) + coeff, 1.0))
320  || (vec.size() == final_size - 1
321  && Dune::XT::Common::FloatCmp::eq(std::accumulate(vec.begin(), vec.end(), 0.0) + coeff, 1.0))) {
322  std::vector<double> vec_copy = vec;
323  vec_copy.push_back(coeff);
324  vectors_out.insert(vec_copy);
325  }
326  }
327  }
328  vectors_in = vectors_out;
329  possible_convex_combination_coefficients(vectors_in, possible_coefficients, final_size);
330  } // if (...)
331  } // ... possible_convex_combination_coefficients(...)
