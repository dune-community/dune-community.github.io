250  AlphaReturnType get_alpha(const DomainType& /*x_local*/,
251  const StateRangeType& u,
252  const XT::Common::Parameter& param,
253  const bool regularize = true) const
254  {
255  const bool boundary = bool(param.get("boundary")[0]);
256  if (boundary)
257  cache_.increase_capacity(2 * cache_size);
258  // get initial multiplier and basis matrix from last time step
259  AlphaReturnType ret;
260 
261  // if value has already been calculated for these values, skip computation
262  mutex_.lock();
263  auto cache_iterator = cache_.lower_bound(u);
264  if (cache_iterator != cache_.end() && XT::Common::FloatCmp::eq(cache_iterator->first, u)) {
265  ret = cache_iterator->second;
266  mutex_.unlock();
267  return ret;
268  } else {
269  StateRangeType u_iso, alpha_iso;
270  std::tie(u_iso, alpha_iso) = basis_functions_.calculate_isotropic_distribution(u);
271 
272  // define further variables
273  VectorType g_k, beta_in, beta_out;
274  beta_in = cache_iterator != cache_.end() ? cache_iterator->second.first : alpha_iso;
275  static thread_local std::unique_ptr<MatrixType> T_k = XT::Common::make_unique<MatrixType>();
276 
277  const auto& r_sequence = regularize ? r_sequence_ : std::vector<RangeFieldType>{0.};
278  const auto r_max = r_sequence.back();
279  for (const auto& r : r_sequence) {
280  // regularize u
281  VectorType r_times_u_iso = u_iso;
282  r_times_u_iso *= r;
283  VectorType v = u;
284  v *= 1 - r;
285  v += r_times_u_iso;
286  if (r > 0)
287  beta_in = alpha_iso;
288  *T_k = T_minus_one_;
289  // calculate T_k u
290  VectorType v_k = v;
291  // calculate values of basis p = S_k m
292  static thread_local BasisValuesMatrixType P_k(M_.size());
293  P_k = M_;
294  // calculate f_0
295  RangeFieldType f_k(0);
296  for (size_t ll = 0; ll < quadrature_.size(); ++ll)
297  f_k += quadrature_[ll].weight() * std::exp(beta_in * P_k[ll]);
298  f_k -= beta_in * v_k;
299 
300  for (size_t kk = 0; kk < k_max_; ++kk) {
301  // exit inner for loop to increase r if to many iterations are used or cholesky decomposition fails
302  if (kk > k_0_ && r < r_max)
303  break;
304  try {
305  change_basis(beta_in, v_k, P_k, *T_k, g_k, beta_out);
306  } catch (const Dune::MathError&) {
307  if (r < r_max)
308  break;
309  mutex_.unlock();
310  DUNE_THROW(Dune::MathError, "Failure to converge!");
311  }
312  // calculate current error
313  VectorType error(0), tmp_m, Tinv_m;
314  for (size_t ll = 0; ll < quadrature_.size(); ++ll) {
315  XT::LA::solve_lower_triangular(*T_k, Tinv_m, M_[ll]);
316  tmp_m = M_[ll];
317  tmp_m *= std::exp(beta_out * Tinv_m) * quadrature_[ll].weight();
318  error += tmp_m;
319  }
320  error -= v;
321  // calculate descent direction d_k;
322  VectorType d_k = g_k;
323  d_k *= -1;
324  VectorType T_k_inv_transp_d_k;
325  XT::LA::solve_lower_triangular_transposed(*T_k, T_k_inv_transp_d_k, d_k);
326  if (error.two_norm() < tau_ && std::exp(5 * T_k_inv_transp_d_k.one_norm()) < 1 + epsilon_gamma_) {
327  XT::LA::solve_lower_triangular_transposed(*T_k, ret.first, beta_out);
328  ret.second = r;
329  goto outside_all_loops;
330  } else {
331  RangeFieldType zeta_k = 1;
332  beta_in = beta_out;
333  // backtracking line search
334  while (zeta_k > epsilon_ * beta_out.two_norm() / d_k.two_norm()) {
335  RangeFieldType f(0);
336  VectorType beta_new = d_k;
337  beta_new *= zeta_k;
338  beta_new += beta_out;
339  for (size_t ll = 0; ll < quadrature_.size(); ++ll)
340  f += quadrature_[ll].weight() * std::exp(beta_new * P_k[ll]);
341  f -= beta_new * v_k;
342  if (XT::Common::FloatCmp::le(f, f_k + xi_ * zeta_k * (g_k * d_k))) {
343  beta_in = beta_new;
344  f_k = f;
345  break;
346  }
347  zeta_k = chi_ * zeta_k;
348  } // backtracking linesearch while
349  } // else (stopping conditions)
350  } // k loop (Newton iterations)
351  } // r loop (Regularization parameter)
352 
353  mutex_.unlock();
354  DUNE_THROW(MathError, "Failed to converge");
355 
356  outside_all_loops:
357  // store values as initial conditions for next time step on this entity
358  cache_.insert(u, ret);
359  mutex_.unlock();
360  } // else ( value has not been calculated before )
361 
362  return ret;
363  }

499  // calculates A = A B^{-1}. B is assumed to be symmetric positive definite.
500  static void calculate_A_Binv(MatrixType& A, MatrixType& B, bool L_calculated = false)
501  {
502  // if B = LL^T, then we have to calculate ret = A (L^T)^{-1} L^{-1} = C L^{-1}
503  // calculate B = LL^T first
504  if (!L_calculated)
505  XT::LA::cholesky(B);
506  VectorType tmp_vec;
507  for (size_t ii = 0; ii < dimRange; ++ii) {
508  // calculate C = A (L^T)^{-1} and store in B
509  XT::LA::solve_lower_triangular(B, tmp_vec, A[ii]);
510  // calculate ret = C L^{-1}
511  XT::LA::solve_lower_triangular_transposed(B, A[ii], tmp_vec);
512  } // ii
513  } // void calculate_A_Binv(...)

515  void calculate_hessian(const VectorType& alpha, const BasisValuesMatrixType& M, MatrixType& H) const
516  {
517  std::fill(H.begin(), H.end(), 0);
518  VectorType m_times_factor;
519  for (size_t ll = 0; ll < quadrature_.size(); ++ll) {
520  const auto& m = M[ll];
521  m_times_factor = m;
522  m_times_factor *= std::exp(alpha * M[ll]) * quadrature_[ll].weight();
523  // add m m^T * factor
524  // matrix is symmetric, we only use lower triangular part
525  for (size_t ii = 0; ii < dimRange; ++ii)
526  for (size_t jj = 0; jj <= ii; ++jj)
527  H[ii][jj] += m[ii] * m_times_factor[jj];
528  } // quadrature points for loop
529  } // void calculate_hessian(...)

531  // J = df/dalpha is the derivative of the flux with respect to alpha.
532  // As F = (f_1, f_2, f_3) is matrix-valued
533  // (div f = \sum_{i=1}^d \partial_{x_i} f_i = \sum_{i=1}^d \partial_{x_i} < v_i m \hat{psi}(alpha) > is
534  // vector-valued),
535  // the derivative is the vector of matrices (df_1/dalpha, df_2/dalpha, ...)
536  // this function returns the dd-th matrix df_dd/dalpha of J
537  void calculate_J(const VectorType& alpha,
538  const BasisValuesMatrixType& M,
539  Dune::FieldMatrix<double, dimRange, StateType::dimRange>& J_dd,
540  const size_t dd) const
541  {
542  assert(dd < dimRangeCols);
543  std::fill(J_dd.begin(), J_dd.end(), 0);
544  VectorType m_times_factor;
545  for (size_t ll = 0; ll < quadrature_.size(); ++ll) {
546  const auto& m = M[ll];
547  const auto& v = quadrature_[ll].position();
548  m_times_factor = m;
549  m_times_factor *= v[dd] * std::exp(alpha * M[ll]) * quadrature_[ll].weight();
550  // add m m^T * factor
551  for (size_t ii = 0; ii < dimRange; ++ii)
552  for (size_t jj = ii; jj < dimRange; ++jj)
553  J_dd[ii][jj] += m[ii] * m_times_factor[jj];
554  } // quadrature points for loop
555  // symmetric update for lower triangular part of J
556  for (size_t rr = 0; rr < dimRange; ++rr)
557  for (size_t cc = 0; cc < rr; ++cc)
558  J_dd[rr][cc] = J_dd[cc][rr];
559  } // void calculate_J(...)

561  void change_basis(const VectorType& beta_in,
562  VectorType& v_k,
563  BasisValuesMatrixType& P_k,
564  MatrixType& T_k,
565  VectorType& g_k,
566  VectorType& beta_out) const
567  {
568  thread_local auto H = XT::Common::make_unique<MatrixType>(0);
569  calculate_hessian(beta_in, P_k, *H);
570  XT::LA::cholesky(*H);
571  const auto& L = *H;
572  VectorType Linv_P;
573  for (size_t ll = 0; ll < P_k.size(); ++ll) {
574  XT::LA::solve_lower_triangular(L, Linv_P, P_k[ll]);
575  P_k[ll] = Linv_P;
576  }
577  T_k.rightmultiply(L);
578  L->mtv(beta_in, beta_out);
579 
580  VectorType v_k_tmp;
581  XT::LA::solve_lower_triangular(L, v_k_tmp, v_k);
582  v_k = v_k_tmp;
583  g_k *= 0.;
584  for (size_t ll = 0; ll < quadrature_.size(); ++ll) {
585  auto contribution = P_k[ll];
586  contribution *= std::exp(beta_out * P_k[ll]) * quadrature_[ll].weight();
587  g_k += contribution;
588  }
589  g_k -= v_k;
590  } // void change_basis(...)

947  AlphaReturnType get_alpha(const DomainType& /*x_local*/,
948  const StateRangeType& u_in,
949  const XT::Common::Parameter& param,
950  const bool regularize = true) const
951  {
952  const bool boundary = bool(param.get("boundary")[0]);
953  if (boundary)
954  cache_.increase_capacity(2 * cache_size);
955  // get initial multiplier and basis matrix from last time step
956  AlphaReturnType ret;
957  StateRangeType u_out;
958 
959  // if value has already been calculated for these values, skip computation
960  mutex_.lock();
961  auto cache_iterator = cache_.lower_bound(u_in);
962  if (cache_iterator != cache_.end() && XT::Common::FloatCmp::eq(cache_iterator->first, u_in)) {
963  ret = cache_iterator->second;
964  mutex_.unlock();
965  return ret;
966  } else {
967  StateRangeType u_iso_in, alpha_iso_in;
968  std::tie(u_iso_in, alpha_iso_in) = basis_functions_.calculate_isotropic_distribution(u_in);
969  VectorType u, u_iso, alpha_iso;
970  for (size_t jj = 0; jj < num_blocks; ++jj) {
971  const auto offset = jj * block_size;
972  for (size_t ii = 0; ii < block_size; ++ii) {
973  u[jj][ii] = u_in[offset + ii];
974  u_iso[jj][ii] = u_iso_in[offset + ii];
975  alpha_iso[jj][ii] = alpha_iso_in[offset + ii];
976  } // ii
977  } // jj
978 
979  // define further variables
980  VectorType g_k, beta_in, beta_out;
981  thread_local auto T_k = XT::Common::make_unique<MatrixType>();
982  beta_in = cache_iterator != cache_.end() ? cache_iterator->second.first : alpha_iso;
983 
984  const auto& r_sequence = regularize ? r_sequence_ : std::vector<RangeFieldType>{0.};
985  const auto r_max = r_sequence.back();
986  for (const auto& r : r_sequence) {
987  // normalize u
988  VectorType r_times_u_iso = u_iso;
989  r_times_u_iso *= r;
990  VectorType v = u;
991  v *= 1 - r;
992  v += r_times_u_iso;
993  if (r > 0)
994  beta_in = alpha_iso;
995  *T_k = T_minus_one_;
996  // calculate T_k u
997  VectorType v_k = v;
998  // calculate values of basis p = S_k m
999  static thread_local BasisValuesMatrixType P_k(M_);
1000  P_k = M_;
1001  // calculate f_0
1002  RangeFieldType f_k(0);
1003  for (size_t jj = 0; jj < num_blocks; ++jj)
1004  for (size_t ll = 0; ll < quadrature_[jj].size(); ++ll)
1005  f_k += quadrature_[jj][ll].weight() * std::exp(beta_in[jj] * P_k[jj][ll]);
1006  f_k -= beta_in * v_k;
1007 
1008  for (size_t kk = 0; kk < k_max_; ++kk) {
1009  // exit inner for loop to increase r if to many iterations are used or cholesky decomposition fails
1010  if (kk > k_0_ && r < r_max)
1011  break;
1012  try {
1013  change_basis(beta_in, v_k, P_k, *T_k, g_k, beta_out);
1014  } catch (const Dune::MathError&) {
1015  if (r < r_max)
1016  break;
1017  mutex_.unlock();
1018  DUNE_THROW(Dune::MathError, "Failure to converge!");
1019  }
1020  // calculate current error
1021  VectorType error(FieldVector<RangeFieldType, block_size>(0.));
1022  FieldVector<RangeFieldType, block_size> tmp_vec;
1023  for (size_t jj = 0; jj < num_blocks; ++jj) {
1024  for (size_t ll = 0; ll < quadrature_[jj].size(); ++ll) {
1025  XT::LA::solve_lower_triangular((*T_k)[jj], tmp_vec, M_[jj][ll]);
1026  auto m_times_factor = M_[jj][ll];
1027  m_times_factor *= std::exp(beta_out[jj] * tmp_vec) * quadrature_[jj][ll].weight();
1028  error[jj] += m_times_factor;
1029  } // ll
1030  error[jj] -= v[jj];
1031  } // jj
1032  // calculate descent direction d_k;
1033  VectorType d_k = g_k;
1034  d_k *= -1.;
1035  VectorType T_k_inv_transp_d_k;
1036  for (size_t jj = 0; jj < num_blocks; ++jj)
1037  XT::LA::solve_lower_triangular_transposed((*T_k)[jj], T_k_inv_transp_d_k[jj], d_k[jj]);
1038  if (two_norm(error) < tau_ && std::exp(5 * one_norm(T_k_inv_transp_d_k)) < 1 + epsilon_gamma_) {
1039  for (size_t jj = 0; jj < num_blocks; ++jj)
1040  XT::LA::solve_lower_triangular_transposed((*T_k)[jj], ret.first[jj], beta_out[jj]);
1041  for (size_t jj = 0; jj < num_blocks; ++jj)
1042  for (size_t ii = 0; ii < block_size; ++ii)
1043  u_out[jj * block_size + ii] = v[jj][ii];
1044  ret.second = r;
1045  goto outside_all_loops;
1046  } else {
1047  RangeFieldType zeta_k = 1;
1048  beta_in = beta_out;
1049  // backtracking line search
1050  while (zeta_k > epsilon_ * two_norm(beta_out) / two_norm(d_k)) {
1051  RangeFieldType f(0);
1052  VectorType beta_new = d_k;
1053  beta_new *= zeta_k;
1054  beta_new += beta_out;
1055  for (size_t jj = 0; jj < num_blocks; ++jj) {
1056  for (size_t ll = 0; ll < quadrature_[jj].size(); ++ll)
1057  f += quadrature_[jj][ll].weight() * std::exp(beta_new[jj] * P_k[jj][ll]);
1058  f -= beta_new[jj] * v_k[jj];
1059  }
1060  if (XT::Common::FloatCmp::le(f, f_k + xi_ * zeta_k * (g_k * d_k))) {
1061  beta_in = beta_new;
1062  f_k = f;
1063  break;
1064  }
1065  zeta_k = chi_ * zeta_k;
1066  } // backtracking linesearch while
1067  } // else (stopping conditions)
1068  } // k loop (Newton iterations)
1069  } // r loop (Regularization parameter)
1070 
1071  mutex_.unlock();
1072  DUNE_THROW(MathError, "Failed to converge");
1073 
1074  outside_all_loops:
1075  // store values as initial conditions for next time step on this entity
1076  cache_.insert(u_in, ret);
1077  mutex_.unlock();
1078  } // else ( value has not been calculated before )
1079  return ret;

1949  AlphaReturnType get_alpha(const DomainType& /*x_local*/,
1950  const StateRangeType& u,
1951  const XT::Common::Parameter& param,
1952  const bool regularize = true) const
1953  {
1954  const bool boundary = bool(param.get("boundary")[0]);
1955  AlphaReturnType ret;
1956  if (boundary)
1957  cache_.increase_capacity(2 * cache_size);
1958 
1959  mutex_.lock();
1960  auto cache_iterator = cache_.lower_bound(u);
1961  if (cache_iterator != cache_.end() && XT::Common::FloatCmp::eq(cache_iterator->first, u)) {
1962  ret = cache_iterator->second;
1963  mutex_.unlock();
1964  return ret;
1965  } else {
1966  // define further variables
1967  thread_local std::unique_ptr<MatrixType> H_k = XT::Common::make_unique<MatrixType>();
1968 
1969  // calculate moment vector for isotropic distribution
1970  StateRangeType u_iso, alpha_iso;
1971  std::tie(u_iso, alpha_iso) = basis_functions_.calculate_isotropic_distribution(u);
1972  StateRangeType alpha_k = cache_iterator != cache_.end() ? cache_iterator->second.first : alpha_iso;
1973 
1974  const auto& r_sequence = regularize ? r_sequence_ : std::vector<RangeFieldType>{0.};
1975  const auto r_max = r_sequence.back();
1976  for (const auto& r : r_sequence) {
1977  // get initial alpha
1978  if (r > 0)
1979  alpha_k = alpha_iso;
1980 
1981  // normalize u
1982  StateRangeType r_times_u_iso(u_iso);
1983  r_times_u_iso *= r;
1984  StateRangeType v = u;
1985  v *= 1 - r;
1986  v += r_times_u_iso;
1987 
1988  // calculate f_0
1989  RangeFieldType f_k = calculate_f(alpha_k, v);
1990 
1991  for (size_t kk = 0; kk < k_max_; ++kk) {
1992  // exit inner for loop to increase r if too many iterations are used
1993  if (kk > k_0_ && r < r_max)
1994  break;
1995 
1996  // calculate gradient g
1997  StateRangeType g_k = calculate_gradient(alpha_k, v);
1998  // calculate Hessian H
1999  calculate_hessian(alpha_k, *H_k);
2000  // calculate descent direction d_k;
2001  StateRangeType d_k(0);
2002  StateRangeType minus_g_k(g_k);
2003  minus_g_k *= -1;
2004  try {
2005  XT::LA::cholesky(*H_k);
2006  const auto& L = *H_k;
2007  StateRangeType tmp_vec;
2008  XT::LA::solve_lower_triangular(L, tmp_vec, minus_g_k);
2009  XT::LA::solve_lower_triangular_transposed(L, d_k, tmp_vec);
2010  } catch (const Dune::MathError&) {
2011  if (r < r_max) {
2012  break;
2013  } else {
2014  mutex_.unlock();
2015  DUNE_THROW(Dune::MathError, "Failure to converge!");
2016  }
2017  }
2018  if (g_k.two_norm() < tau_ && std::exp(5 * d_k.one_norm()) < 1 + epsilon_gamma_) {
2019  ret = std::make_pair(alpha_k, r);
2020  goto outside_all_loops;
2021  } else {
2022  RangeFieldType zeta_k = 1;
2023  // backtracking line search
2024  while (zeta_k > epsilon_ * alpha_k.two_norm() / d_k.two_norm()) {
2025  // calculate alpha_new = alpha_k + zeta_k d_k
2026  auto alpha_new = d_k;
2027  alpha_new *= zeta_k;
2028  alpha_new += alpha_k;

2030  if (XT::Common::FloatCmp::le(f_new, f_k + xi_ * zeta_k * (g_k * d_k))) {
2031  alpha_k = alpha_new;
2032  f_k = f_new;
2033  break;
2034  }
2035  zeta_k = chi_ * zeta_k;
2036  } // backtracking linesearch while
2037  } // else (stopping conditions)
2038  } // k loop (Newton iterations)
2039  } // r loop (Regularization parameter)
2040 
2041  mutex_.unlock();

2043

2045  cache_.insert(u, ret);
2046  mutex_.unlock();
2047  } // else ( value has not been calculated before )
2048
