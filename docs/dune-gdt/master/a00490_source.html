129 template <class NumericalCouplingFluxType,
130  class NumericalBoundaryFluxType,
131  size_t polOrder,
132  SlopeLimiters slope_limiter,
133  class RealizabilityLimiterType>
134 struct AdvectionOperatorApplier
135 {
136  template <class AnalyticalFluxType,
137  class BoundaryValueType,
138  class SourceType,
139  class RangeType,
140  class DomainFieldType,
141  class... LocalOperatorArgTypes>
142  static void
143  apply(const AnalyticalFluxType& analytical_flux,
144  const BoundaryValueType& boundary_values,
145  const SourceType& source,
146  RangeType& range,
147  const XT::Common::Parameter& param,
148  bool is_linear,
149  const Dune::QuadratureRule<DomainFieldType, 1> quadrature1d,
150  const Dune::QuadratureRule<DomainFieldType, BoundaryValueType::dimDomain - 1> intersection_quadrature,
151  const std::shared_ptr<RealizabilityLimiterType>& realizability_limiter,
152  LocalOperatorArgTypes&&... local_operator_args)
153  {
154  typedef typename SourceType::SpaceType::GridLayerType GridLayerType;
155  typedef typename BoundaryValueType::DomainType DomainType;
156  const GridLayerType& grid_layer = source.space().grid_layer();
157 
158  // evaluate cell averages
159  std::vector<typename BoundaryValueType::RangeType> source_values(grid_layer.indexSet().size(0));
160  for (const auto& entity : Dune::elements(grid_layer)) {
161  const auto& entity_index = grid_layer.indexSet().index(entity);
162  const auto& local_source = source.local_function(entity);
163  source_values[entity_index] = local_source->evaluate(entity.geometry().local(entity.geometry().center()));
164  }
165 
166  // do reconstruction
167  typedef std::vector<std::map<DomainType, typename BoundaryValueType::RangeType, XT::Common::FieldVectorLess>>
168  ReconstructedValuesType;
169  ReconstructedValuesType reconstructed_values(grid_layer.size(0));
170 
171  auto local_reconstruction_operator =
172  LocalReconstructionFvOperator<GridLayerType, AnalyticalFluxType, BoundaryValueType, polOrder, slope_limiter>(
173  source_values,
174  analytical_flux,
175  boundary_values,
176  grid_layer,
177  param,
178  is_linear,
179  quadrature1d,
180  reconstructed_values);
181  auto walker = XT::Grid::Walker<GridLayerType>(grid_layer);
182  walker.append(local_reconstruction_operator);
183  walker.walk(true);
184 
185  // communicate reconstructed values
186  typedef ReconstructionDataHandle<ReconstructedValuesType, GridLayerType> DataHandleType;
187  DataHandleType reconstruction_data_handle(reconstructed_values, grid_layer);
188  grid_layer.template communicate<DataHandleType>(
189  reconstruction_data_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
190 
191  // realizability limiting
192  if (realizability_limiter) {
193  realizability_limiter->set_source(&source);
194  realizability_limiter->set_reconstructed_values(&reconstructed_values);
195  walker.clear();
196  walker.append(*realizability_limiter);
197  walker.walk(true);
198  }
199

201  DomainFieldType,
202  BoundaryValueType::dimDomain,
203  typename AnalyticalFluxType::RangeFieldType,
204  BoundaryValueType::dimRange>
205  ReconstructedLocalizableFunctionType;
206  const ReconstructedLocalizableFunctionType reconstructed_function(grid_layer, reconstructed_values);

406  static Dune::QuadratureRule<DomainFieldType, dimDomain - 1> get(const OnedQuadratureType& quadrature_1d)
407  {
408  Dune::QuadratureRule<DomainFieldType, dimDomain - 1> ret;
409  for (size_t ii = 0; ii < quadrature_1d.size(); ++ii)
410  for (size_t jj = 0; jj < quadrature_1d.size(); ++jj)
411  ret.push_back(Dune::QuadraturePoint<DomainFieldType, dimDomain - 1>(
412  {quadrature_1d[ii].position()[0], quadrature_1d[jj].position()[0]},
413  quadrature_1d[ii].weight() * quadrature_1d[jj].weight()));
414  return ret;
415  }
