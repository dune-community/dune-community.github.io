123  template <class IntersectionType>
124  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple,
125  const IntersectionType& intersection,
126  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
127  const DomainType& x_in_inside_coords,
128  const DomainType& /*x_in_outside_coords*/,
129  const RangeType& u_i,
130  const RangeType& u_j) const
131  {
132  size_t direction = intersection.indexInInside() / 2;
133  // get unit outer normal
134  const auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
135  assert(XT::Common::FloatCmp::eq(std::abs(n_ij[direction]), 1.));
136 
137  // intialize jacobians
138  initialize_jacobians(direction, local_functions_tuple, x_in_inside_coords, u_i, u_j);
139 
140  // get jump at the intersection
141  const RangeType delta_u = u_i - u_j;
142  // calculate waves
143  RangeType waves(0);
144  n_ij[direction] > 0 ? jacobian_neg()[direction].mv(delta_u, waves) : jacobian_pos()[direction].mv(delta_u, waves);
145  // calculate flux
146  const auto& local_flux = std::get<0>(local_functions_tuple);
147  RangeType ret = local_flux->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
148  ret -= waves;
149  ret *= n_ij[direction];
150  return ret;
151  } // RangeType evaluate(...) const

163 private:
164  // use simple linearized Riemann solver, LeVeque p.316
165  void initialize_jacobians(const size_t direction,
166  const LocalfunctionTupleType& local_functions_tuple,
167  const DomainType& x_local,
168  const RangeType& u_i,
169  const RangeType& u_j) const
170  {
171  if (local_initialization_counts_[direction] != initialization_count_) {
172  // calculate jacobian as jacobian(0.5*(u_i+u_j)
173  RangeType u_mean = u_i + u_j;
174  u_mean *= RangeFieldType(0.5);
175  const auto& local_flux = std::get<0>(local_functions_tuple);
176  if (!jacobian())
177  jacobian() = XT::Common::make_unique<JacobianRangeType>();
178  helper<dimDomain>::get_jacobian(direction, local_flux, x_local, u_mean, *(jacobian()), param_inside_);
179  XT::Common::Configuration eigensolver_options(
180  {"type", "check_for_inf_nan", "check_evs_are_real", "check_evs_are_positive", "check_eigenvectors_are_real"},
181  {EigenSolverType::types()[1], "1", "1", "0", "1"});
182  const auto eigen_solver = EigenSolverType((*(jacobian()))[direction]);
183  const auto eigenvalues = eigen_solver.eigenvalues(eigensolver_options);
184  const auto eigenvectors = eigen_solver.real_eigenvectors_as_matrix(eigensolver_options);
185  auto eigenvectors_inverse = std::make_shared<MatrixType>(*eigenvectors);
186  eigenvectors_inverse->invert();
187  if (is_linear_)
188  jacobian() = nullptr;
189 
190  auto jacobian_neg_dense = XT::Common::make_unique<MatrixType>(0);
191  auto jacobian_pos_dense = XT::Common::make_unique<MatrixType>(0);
192  for (size_t rr = 0; rr < dimRange; ++rr)
193  for (size_t cc = 0; cc < dimRange; ++cc)
194  for (size_t kk = 0; kk < dimRange; ++kk)
195  if (XT::Common::FloatCmp::lt(eigenvalues[kk].real(), 0.))
196  (*jacobian_neg_dense)[rr][cc] +=
197  (*eigenvectors)[rr][kk] * (*eigenvectors_inverse)[kk][cc] * eigenvalues[kk].real();
198  else
199  (*jacobian_pos_dense)[rr][cc] +=
200  (*eigenvectors)[rr][kk] * (*eigenvectors_inverse)[kk][cc] * eigenvalues[kk].real();
201  jacobian_neg()[direction] = SparseMatrixType(*jacobian_neg_dense, true);
202  jacobian_pos()[direction] = SparseMatrixType(*jacobian_pos_dense, true);
203  if (is_linear_)
204  ++local_initialization_counts_[direction];
205  } // if (local_initialization_counts_[direction] != initialization_count_)
206  } // void calculate_jacobians(...)
