76  const auto& rtn0_space = range.space();
77  auto& range_vector = range.vector();
78  const FieldType infinity = std::numeric_limits<FieldType>::infinity();
79  for (size_t ii = 0; ii < range_vector.size(); ++ii)
80  range_vector[ii] = infinity;
81  const LocalEllipticIpdgIntegrands::Inner<DiffusionFactorType, DiffusionTensorType> inner_evaluation(
82  diffusion_factor_, diffusion_tensor_);
83  const LocalEllipticIpdgIntegrands::BoundaryLHS<DiffusionFactorType, DiffusionTensorType> boundary_evaluation(
84  diffusion_factor_, diffusion_tensor_);
85  const XT::Functions::ConstantFunction<EntityType, DomainFieldType, dimDomain, FieldType, 1> constant_one(1);
86  DomainType normal(0);
87  DomainType xx_entity(0);
88  DynamicMatrix<FieldType> tmp_matrix(1, 1, 0);
89  DynamicMatrix<FieldType> tmp_matrix_en_en(1, 1, 0);
90  DynamicMatrix<FieldType> tmp_matrix_en_ne(1, 1, 0);
91  std::vector<typename DunePdelabRtSpaceWrapper<GL, 0, FieldType, dimDomain>::BaseFunctionSetType::RangeType>
92  basis_values(rtn0_space.mapper().maxNumDofs(),
93  typename DunePdelabRtSpaceWrapper<GL, 0, FieldType, dimDomain>::BaseFunctionSetType::RangeType(0));
94  // walk the grid
95  const auto entity_it_end = grid_layer_.template end<0>();
96  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
97  const auto& entity = *entity_it;
98  const auto local_DoF_indices = rtn0_space.local_DoF_indices(entity);
99  const auto global_DoF_indices = rtn0_space.mapper().globalIndices(entity);
100  assert(global_DoF_indices.size() == local_DoF_indices.size());
101  const auto local_diffusion_factor = diffusion_factor_.local_function(entity);
102  const auto local_diffusion_tensor = diffusion_tensor_.local_function(entity);
103  const auto local_source = source.local_function(entity);
104  const auto local_basis = rtn0_space.base_function_set(entity);
105  const auto local_constant_one = constant_one.local_function(entity);
106  // walk the intersections
107  const auto intersection_it_end = grid_layer_.iend(entity);
108  for (auto intersection_it = grid_layer_.ibegin(entity); intersection_it != intersection_it_end;
109  ++intersection_it) {
110  const auto& intersection = *intersection_it;
111  if (intersection.neighbor() && !intersection.boundary()) {
112  const auto neighbor = intersection.outside();
113  if (grid_layer_.indexSet().index(entity) < grid_layer_.indexSet().index(neighbor)) {
114  const auto local_diffusion_factor_neighbor = diffusion_factor_.local_function(neighbor);
115  const auto local_diffusion_tensor_neighbor = diffusion_tensor_.local_function(neighbor);
116  const auto local_source_neighbor = source.local_function(neighbor);
117  const auto local_constant_one_neighbor = constant_one.local_function(neighbor);
118  const size_t local_intersection_index = intersection.indexInInside();
119  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
120  // do a face quadrature
121  FieldType lhs = 0;
122  FieldType rhs = 0;
123  const size_t integrand_order = inner_evaluation.order(*local_diffusion_factor,
124  *local_diffusion_tensor,
125  *local_diffusion_factor_neighbor,
126  *local_diffusion_tensor_neighbor,
127  *local_constant_one,
128  *local_source,
129  *local_constant_one_neighbor,
130  *local_source_neighbor);
131  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
132  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
133  const auto quadrature_it_end = quadrature.end();
134  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
135  const auto& xx_intersection = quadrature_it->position();
136  xx_entity = intersection.geometryInInside().global(xx_intersection);
137  normal = intersection.unitOuterNormal(xx_intersection);
138  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
139  const FieldType weigth = quadrature_it->weight();
140  // evalaute
141  local_basis.evaluate(xx_entity, basis_values);
142  const auto& basis_value = basis_values[local_DoF_index];
143  tmp_matrix_en_en *= 0.0;
144  tmp_matrix_en_ne *= 0.0;
145  inner_evaluation.evaluate(*local_diffusion_factor,
146  *local_diffusion_tensor,
147  *local_diffusion_factor_neighbor,
148  *local_diffusion_tensor_neighbor,
149  *local_constant_one,
150  *local_source,
151  *local_constant_one_neighbor,
152  *local_source_neighbor,
153  intersection,
154  xx_intersection,
155  tmp_matrix_en_en, // <- we are interested in this one
156  tmp_matrix,
157  tmp_matrix_en_ne, // <- and this one
158  tmp_matrix);
159  // compute integrals
160  assert(tmp_matrix_en_en.rows() >= 1);
161  assert(tmp_matrix_en_en.cols() >= 1);
162  assert(tmp_matrix_en_ne.rows() >= 1);
163  assert(tmp_matrix_en_ne.cols() >= 1);
164  lhs += integration_factor * weigth * (basis_value * normal);
165  rhs += integration_factor * weigth * (tmp_matrix_en_en[0][0] + tmp_matrix_en_ne[0][0]);
166  } // do a face quadrature
167  // set DoF
168  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
169  // and make sure we are the first to do so
170  assert(!(range_vector[global_DoF_index] < infinity));
171  range_vector[global_DoF_index] = rhs / lhs;
172  }
173  } else if (intersection.boundary() && !intersection.neighbor()) {
174  const size_t local_intersection_index = intersection.indexInInside();
175  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
176  // do a face quadrature
177  FieldType lhs = 0;
178  FieldType rhs = 0;
179  const size_t integrand_order = boundary_evaluation.order(
180  *local_diffusion_factor, *local_diffusion_tensor, *local_source, *local_constant_one);
181  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
182  intersection.type(), boost::numeric_cast<int>(integrand_order + over_integrate_));
183  const auto quadrature_it_end = quadrature.end();
184  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
185  const auto xx_intersection = quadrature_it->position();
186  normal = intersection.unitOuterNormal(xx_intersection);
187  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
188  const FieldType weigth = quadrature_it->weight();
189  xx_entity = intersection.geometryInInside().global(xx_intersection);
190  // evalaute
191  local_basis.evaluate(xx_entity, basis_values);
192  const auto& basis_value = basis_values[local_DoF_index];
193  tmp_matrix *= 0.0;
194  boundary_evaluation.evaluate(*local_diffusion_factor,
195  *local_diffusion_tensor,
196  *local_constant_one,
197  *local_source,
198  intersection,
199  xx_intersection,
200  tmp_matrix);
201  // compute integrals
202  assert(tmp_matrix.rows() >= 1);
203  assert(tmp_matrix.cols() >= 1);
204  lhs += integration_factor * weigth * (basis_value * normal);
205  rhs += integration_factor * weigth * tmp_matrix[0][0];
206  } // do a face quadrature
207  // set DoF

209  // and make sure we are the first to do so
