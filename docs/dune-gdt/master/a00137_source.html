44  limit(const VectorType& slope_left, const VectorType& slope_right, const VectorType& /*slope_center*/)
45  {
46  VectorType ret(0.);
47  for (size_t ii = 0; ii < slope_left.size(); ++ii)
48  if (slope_left[ii] * slope_right[ii] > 0) // check for equal sign
49  ret[ii] = (std::abs(slope_left[ii]) < std::abs(slope_right[ii])) ? slope_left[ii] : slope_right[ii];

69  template <class VectorType>
70  static VectorType maxmod(const VectorType& slope_left, const VectorType& slope_right)
71  {
72  VectorType ret(0.);
73  for (size_t ii = 0; ii < slope_left.size(); ++ii) {
74  const auto slope_left_abs = std::abs(slope_left[ii]);
75  const auto slope_right_abs = std::abs(slope_right[ii]);
76  if (slope_left_abs > slope_right_abs && slope_left[ii] * slope_right[ii] > 0)
77  ret[ii] = slope_left[ii];
78  else if (Dune::XT::Common::FloatCmp::le(slope_left_abs, slope_right_abs) && slope_left[ii] * slope_right[ii] > 0)
79  ret[ii] = slope_right[ii];
80  else
81  ret[ii] = 0.0;
82  }
83  return ret;
84  }
85 };
