72  explicit LaxWendroffFluxImplementation(const AnalyticalFluxType& analytical_flux,
73  XT::Common::Parameter param,
74  const bool is_linear = false,
75  const RangeFieldType alpha = dimDomain,
76  const bool boundary = false)
77  : analytical_flux_(analytical_flux)
78  , param_inside_(param)
79  , param_outside_(param)
80  , dt_(param.get("dt")[0])
81  , is_linear_(is_linear)
82  , alpha_(alpha)
83  {
84  param_inside_.set("boundary", {0.}, true);
85  param_outside_.set("boundary", {double(boundary)}, true);

88  template <class IntersectionType>
89  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
90  const LocalfunctionTupleType& local_functions_tuple_neighbor,
91  const IntersectionType& intersection,
92  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
93  const DomainType& x_in_inside_coords,
94  const DomainType& x_in_outside_coords,
95  const RangeType& u_i,
96  const RangeType& u_j) const
97  {
98  // find direction of unit outer normal
99  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
100  size_t direction = intersection.indexInInside() / 2;
101 
102  const auto& local_flux_inside = std::get<0>(local_functions_tuple_entity);
103  const auto& local_flux_outside = std::get<0>(local_functions_tuple_neighbor);
104 
105  // calculate flux evaluation as
106  // ret = f((u_i + u_j)*0.5 - (f_u_j - f_u_i)*0.5*dt/dx*dimDomain)*n_ij[direction]
107  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
108  RangeType ret = u_i;
109  ret += u_j;
110  ret *= 0.5;
111  RangeType second_part = local_flux_outside->evaluate_col(direction, x_in_outside_coords, u_j, param_outside_);
112  second_part -= local_flux_inside->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
113  second_part *= n_ij[direction] * 0.5 * dt_ / dx * alpha_;
114  ret -= second_part;
115  ret = local_flux_inside->evaluate_col(direction, x_in_inside_coords, ret, param_inside_);
116  ret *= n_ij[direction];
117  return ret;
118  } // ... evaluate(...)
119 
120  const AnalyticalFluxType& analytical_flux() const
121  {
122  return analytical_flux_;
123  }
124 
125 private:
126  const AnalyticalFluxType& analytical_flux_;
127  XT::Common::Parameter param_inside_;
128  XT::Common::Parameter param_outside_;
129  const double dt_;
130  const bool is_linear_;
131  const RangeFieldType alpha_;
132 }; // class LaxWendroffFluxImplementation<...>
