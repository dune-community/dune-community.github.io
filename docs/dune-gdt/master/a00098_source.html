61  template <class E, class D, size_t d, class R, size_t r, size_t rC, class RS, class V>
62  void apply(const XT::Functions::LocalizableFunctionInterface<E, D, d, R, r, rC>& source,
63  LocalDiscreteFunction<RS, V>& local_range) const
64  {
65  const auto& entity = local_range.entity();
66  if (!entity.hasBoundaryIntersections())
67  return;
68  const auto local_dirichlet_DoFs = local_range.space().local_dirichlet_DoFs(entity, boundary_info_);
69  if (local_dirichlet_DoFs.size() == 0)
70  return;
71  const auto lagrange_points = local_range.space().lagrange_points(entity);
72  const auto local_source = source.local_function(entity);
73  auto& local_range_DoF_vector = local_range.vector();
74  assert(lagrange_points.size() == local_range_DoF_vector.size());
75  for (const size_t& local_DoF_id : local_dirichlet_DoFs)
76  local_range_DoF_vector.set(local_DoF_id, local_source->evaluate(lagrange_points[local_DoF_id]));
77  } // ... apply(...)
