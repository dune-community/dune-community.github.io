117  void apply_local(const EntityType& entity)
118  {
119  auto& local_reconstructed_values = (*reconstructed_values_)[index_set_->index(entity)];
120 
121  // get cell average
122  const RangeType& u_bar =
123  source_->local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
124 
125  // vector to store thetas for each local reconstructed value
126  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
127 
128  size_t ll = -1;
129  for (const auto& pair : local_reconstructed_values) {
130  ++ll;
131  // rescale u_l, u_bar
132  auto u_l = pair.second;
133  auto u_bar_minus_u_l = u_bar - u_l;
134  const auto factor = basis_functions_.realizability_limiter_max(u_l, u_bar);
135  u_l /= factor;
136  u_bar_minus_u_l /= factor;
137 
138  for (const auto& coeffs : *plane_coefficients_) {
139  const RangeType& a = coeffs.first;
140  const RangeFieldType& b = coeffs.second;
141  RangeFieldType theta_li = (b - a * u_l) / (a * u_bar_minus_u_l);
142  if (XT::Common::FloatCmp::ge(theta_li, -epsilon_) && XT::Common::FloatCmp::le(theta_li, 1.))
143  thetas[ll] = std::max(thetas[ll], theta_li);
144  } // coeffs
145  } // ll
146  for (auto& theta : thetas)
147  theta = std::min(epsilon_ + theta, 1.);
148 
149  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
150  if (XT::Common::FloatCmp::ne(theta_entity, 0.)) {
151  for (auto& pair : local_reconstructed_values) {
152  auto& u = pair.second;
153  auto u_scaled = u;
154  u_scaled *= (1 - theta_entity);
155  auto u_bar_scaled = u_bar;
156  u_bar_scaled *= theta_entity;
157  u = u_scaled + u_bar_scaled;
158  }
159  }
160  } // void apply_local(...)

162 private:
163  // calculate half space representation of realizable set
164  std::shared_ptr<const PlaneCoefficientsType> calculate_plane_coefficients()
165  {
166  using orgQhull::Qhull;
167  Qhull qhull;
168  std::vector<FieldVector<RangeFieldType, dimRange>> points(quadrature_.size() + 1);
169  points[0] = FieldVector<RangeFieldType, dimRange>(0);
170  size_t ii = 1;
171  for (const auto& quad_point : quadrature_)
172  points[ii++] = basis_functions_.evaluate(quad_point.position());
173 
174  qhull.runQhull("Realizable set", int(dimRange), int(points.size()), &(points[0][0]), "Qt T1");
175  // qhull.outputQhull("n");
176  const auto facet_end = qhull.endFacet();
177  std::vector<std::pair<RangeType, RangeFieldType>> plane_coefficients(qhull.facetList().count());
178  ii = 0;
179  for (auto facet = qhull.beginFacet(); facet != facet_end; facet = facet.next(), ++ii) {
180  for (size_t jj = 0; jj < dimRange; ++jj)
181  plane_coefficients[ii].first[jj] = *(facet.hyperplane().coordinates() + jj);
182  plane_coefficients[ii].second = -facet.hyperplane().offset();
183  }
184  return std::make_shared<const PlaneCoefficientsType>(plane_coefficients);
185  }

222 class LPLocalRealizabilityLimiter
223  : public XT::Grid::Functor::Codim0<typename DiscreteFunctionType::SpaceType::GridLayerType>
224 {

226  typedef typename DiscreteFunctionType::EntityType EntityType;

261  void apply_local(const EntityType& entity)
262  {
263  auto& local_reconstructed_values = (*reconstructed_values_)[index_set_->index(entity)];
264 
265  // get cell average
266  const RangeType& u_bar =
267  source_->local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
268 
269  // vector to store thetas for each local reconstructed value
270  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
271 
272  size_t ll = -1;
273  for (const auto& pair : local_reconstructed_values) {
274  ++ll;
275  auto u_l = pair.second;
276  if (XT::Common::FloatCmp::eq(u_bar, u_l)) {
277  thetas[ll] = 0.;
278  continue;
279  }
280 
281  // solve LP:
282  // min \theta s.t.
283  // (\sum x_i v_i) + \theta (u - \bar{u}) = u
284  // x_i, \theta >= 0
285  auto theta = solve_linear_program(u_bar, u_l);
286  if (theta > -epsilon_)
287  theta = theta + epsilon_;
288  if (theta > 1.)
289  theta = 1.;
290  else if (theta < 0)
291  theta = 0.;
292  thetas[ll] = theta;
293  } // ll
294 
295  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
296  if (XT::Common::FloatCmp::ne(theta_entity, 0.)) {
297  for (auto& pair : local_reconstructed_values) {
298  auto& u = pair.second;
299  auto u_scaled = u;
300  u_scaled *= 1 - theta_entity;
301  auto u_bar_scaled = u_bar;
302  u_bar_scaled *= theta_entity;
303  u = u_scaled + u_bar_scaled;
304  }
305  }
306  } // void apply_local(...)

308 private:
309  RangeFieldType solve_linear_program(const RangeType& u_bar, const RangeType& u_l)
310  {
311  typename lpsolve::lprec* lp;
312 
313  auto u_l_minus_u_bar = u_l - u_bar;
314 
315  // We start with creating a model with dimRange+1 rows and num_quad_points+1 columns */
316  constexpr int num_rows = int(dimRange);
317  int num_cols = int(quadrature_.size() + 1); /* variables are x_1, ..., x_{num_quad_points}, theta */
318  lp = lpsolve::make_lp(num_rows, num_cols);
319  if (!lp)
320  DUNE_THROW(Dune::MathError, "Couldn't construct linear program");
321 
322  /* let us name our variables. Not required, but can be useful for debugging */
323  std::vector<char> name(5);
324  for (int ii = 1; ii <= num_cols - 1; ++ii) {
325  std::string ii_string = XT::Common::to_string(ii);
326  std::string name_string = "x";
327  for (size_t jj = 5 - ii_string.size(); jj > 0; --jj)
328  name_string += "0";
329  name_string += ii_string;
330  std::copy(name_string.begin(), name_string.end(), name.begin());
331  lpsolve::set_col_name(lp, ii, name.data());
332  }
333 
334  name = {'t', 'h', 'e', 't', 'a', ' '};
335  lpsolve::set_col_name(lp, num_cols, name.data());
336 
337  // In the call to set_column, the first entry (row 0) is the value of the objective function
338  // (c_i in the objective function c^T x), the other entries are the entries of the i-th column
339  // in the constraints matrix. The 0-th column is the rhs vector. The entry (0, 0) corresponds
340  // to the initial value of the objective function.
341  std::array<REAL, num_rows + 1> column;
342 
343  // set rhs (column 0)
344  column[0] = 0.;
345  std::copy(u_l.begin(), u_l.end(), column.begin() + 1);
346  lpsolve::set_rh_vec(lp, column.data());
347  lpsolve::set_rh(lp, 0, column[0]);
348 
349  // set columns for quadrature points
350  column[0] = 0.;
351  for (int ii = 0; ii < int(M_.size()); ++ii) {
352  const auto& v_i = M_[ii];
353  std::copy(v_i.begin(), v_i.end(), column.begin() + 1);
354  lpsolve::set_column(lp, ii + 1, column.data());
355  }
356 
357  // set theta column
358  column[0] = 1.;
359  std::copy(u_l_minus_u_bar.begin(), u_l_minus_u_bar.end(), column.begin() + 1);
360  lpsolve::set_column(lp, num_cols, column.data());
361 
362  // set all contraints to equality constraints
363  for (int ii = 1; ii <= num_rows; ++ii)
364  lpsolve::set_constr_type(lp, ii, EQ);
365 
366  // Set bounds for variables. 0 <= x <= inf is the default for all variables.
367  // The bounds for theta should be [0,1], but we allow allow theta to be -0.1
368  // so we can check if u_l is on the boundary and if so, move it a little away
369  // from the boundary
370  lpsolve::set_bounds(lp, num_cols, -0.1, 1.);
371 
372  default_basis(lp);
373 
374  /* I only want to see important messages on screen while solving */
375  lpsolve::set_verbose(lp, IMPORTANT);
376 
377  /* Now let lpsolve calculate a solution */
378  const auto solve_status = lpsolve::solve(lp);
379  RangeFieldType theta = lpsolve::get_objective(lp);
380  if (theta > -0.09) {
381  lpsolve::write_LP(lp, stdout);
382  static size_t counter = 0;
383  std::cout << counter++ << ": theta: " << XT::Common::to_string(theta, 15) << std::endl;
384  }
385  if (solve_status != OPTIMAL) {
386  std::cout << solve_status << std::endl;
387  DUNE_THROW(Dune::MathError, "An unexpected error occured while solving the linear program");
388  }
389 
390  /* free allocated memory */
391  if (lp)
392  lpsolve::delete_lp(lp);
393 
394  return theta;
395  }
