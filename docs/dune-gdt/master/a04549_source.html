40 template <class DiffusionFactorType,
41  typename DiffusionTensorType, // may be void
42  class GridLayer,
43  class Range,
44  class Source = Range,
45  class Field = typename Range::RangeFieldType>
46 class EllipticLocalizableProduct : public LocalizableProductBase<GridLayer, Range, Source, Field>
47 {
48  typedef LocalizableProductBase<GridLayer, Range, Source, Field> BaseType;
49 
50 public:
51  // Usually, we only have to hold the data functions for the local operator and perfect forward the rest of the
52  // arguments to BaseType. Here it is a bit more complicated, since DiffusionTensorType might be void (and
53  // DiffusionFactorType is the the only diffusion). To handle this case we require the enable_if hacks below to
54  // disable half of the ctors if DiffusionTensorType is void. In addition we require each ctor twice, once with
55  // over_integrate and once without (since the perfect forwarding does not allow for default arguments).
56 
57  template <typename DiffusionImp // This ctor is only enabled if we are given a single diffusion data function.
58  ,
59  typename = typename std::enable_if<(std::is_same<DiffusionTensorType, void>::value)
60  && (std::is_same<DiffusionImp, DiffusionFactorType>::value)
61  && sizeof(DiffusionImp)>::type,
62  class... Args>
63  explicit EllipticLocalizableProduct(const DiffusionImp& diffusion, Args&&... args)
64  : BaseType(std::forward<Args>(args)...)
65  , local_elliptic_operator_(diffusion, BaseType::parameter())
66  {
67  this->append(local_elliptic_operator_);
68  }
69 
70  template <typename DiffusionImp // This ctor is only enabled if we are given a single diffusion data function.
71  ,
72  typename = typename std::enable_if<(std::is_same<DiffusionTensorType, void>::value)
73  && (std::is_same<DiffusionImp, DiffusionFactorType>::value)
74  && sizeof(DiffusionImp)>::type,
75  class... Args>
76  explicit EllipticLocalizableProduct(const size_t over_integrate, const DiffusionImp& diffusion, Args&&... args)
77  : BaseType(std::forward<Args>(args)...)
78  , local_elliptic_operator_(over_integrate, diffusion, BaseType::parameter())
79  {
80  this->append(local_elliptic_operator_);
81  }
82 
83  template <typename DiffusionFactorImp, // This ctor is only enabled if we are given two diffusion data functions
84  typename DiffusionTensorImp, // (factor and tensor).
85  typename = typename std::enable_if<(!std::is_same<DiffusionTensorType, void>::value)
86  && (std::is_same<DiffusionFactorImp, DiffusionFactorType>::value)
87  && sizeof(DiffusionFactorImp)>::type,
88  class... Args>
89  explicit EllipticLocalizableProduct(const DiffusionFactorImp& diffusion_factor,
90  const DiffusionTensorImp& diffusion_tensor,
91  Args&&... args)
92  : BaseType(std::forward<Args>(args)...)
93  , local_elliptic_operator_(diffusion_factor, diffusion_tensor, BaseType::parameter())
94  {
95  this->append(local_elliptic_operator_);
96  }
97 
98  template <typename DiffusionFactorImp, // This ctor is only enabled if we are given two diffusion data functions
99  typename DiffusionTensorImp, // (factor and tensor).
100  typename = typename std::enable_if<(!std::is_same<DiffusionTensorType, void>::value)
101  && (std::is_same<DiffusionFactorImp, DiffusionFactorType>::value)
102  && sizeof(DiffusionFactorImp)>::type,
103  class... Args>
104  explicit EllipticLocalizableProduct(const size_t over_integrate,
105  const DiffusionFactorImp& diffusion_factor,
106  const DiffusionTensorImp& diffusion_tensor,
107  Args&&... args)
108  : BaseType(std::forward<Args>(args)...)
109  , local_elliptic_operator_(over_integrate, diffusion_factor, diffusion_tensor, BaseType::parameter())
110  {
111  this->append(local_elliptic_operator_);
112  }
113 
114 private:
115  const LocalVolumeIntegralOperator<LocalEllipticIntegrand<DiffusionFactorType, DiffusionTensorType>,
116  typename Range::LocalfunctionType,
117  typename Source::LocalfunctionType,
118  Field>
119  local_elliptic_operator_;
120 }; // class EllipticLocalizableProduct
