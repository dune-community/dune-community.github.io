148  template <class T, class S, class V>
149  void redirect_apply(
150  const CgSpaceInterface<T, dimDomain, dimDomain, 1>& /*space*/,
151  const XT::Functions::LocalizableFunctionInterface<EntityType, DomainFieldType, dimDomain, FieldType, 1, 1>&
152  source,
153  DiscreteFunction<S, V>& range,
154  const XT::Common::Parameter param) const
155  {
156  typedef typename XT::LA::Container<FieldType, V::sparse_matrix_type>::MatrixType MatrixType;
157  MatrixType lhs(
158  range.space().mapper().size(), range.space().mapper().size(), range.space().compute_volume_pattern());
159  V rhs(range.space().mapper().size());
160 
161  // walk the grid
162  const auto entity_it_end = grid_layer_.template end<0>();
163  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
164  const auto& entity = *entity_it;
165  const auto local_function = function_.local_function(entity);
166  const auto local_source = source.local_function(entity);
167  const auto basis = range.space().base_function_set(entity);
168  // do a volume quadrature
169  const size_t integrand_order =
170  std::max(local_function->order() + ssize_t(local_source->order()) - 1, basis.order()) + basis.order();
171  const auto& quadrature =
172  QuadratureRules<DomainFieldType, dimDomain>::rule(entity.type(), boost::numeric_cast<int>(integrand_order));
173  const auto quadrature_it_end = quadrature.end();
174  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
175  const auto xx = quadrature_it->position();
176  const auto quadrature_weight = quadrature_it->weight();
177  const auto integration_element = entity.geometry().integrationElement(xx);
178  const ValueType function_value = local_function->evaluate(xx, param);
179  const auto source_gradient = local_source->jacobian(xx, param);
180  const auto basis_value = basis.evaluate(xx, param);
181  for (size_t ii = 0; ii < basis.size(); ++ii) {
182  const size_t global_ii = range.space().mapper().mapToGlobal(entity, ii);
183  rhs.add_to_entry(global_ii,
184  integration_element * quadrature_weight * -1.0
185  * ((function_value * source_gradient[0]) * basis_value[ii]));
186  for (size_t jj = 0; jj < basis.size(); ++jj) {
187  const size_t global_jj = range.space().mapper().mapToGlobal(entity, jj);
188  lhs.add_to_entry(
189  global_ii, global_jj, integration_element * quadrature_weight * (basis_value[ii] * basis_value[jj]));
190  }
191  }
192  } // do a volume quadrature
193  } // walk the grid
194 
195  // solve
196  try {
197  XT::LA::make_solver(lhs, range.space().dof_communicator()).apply(rhs, range.vector());
198  } catch (XT::LA::Exceptions::linear_solver_failed& ee) {
199  DUNE_THROW(operator_error,
200  "Application of the Darcy operator failed because a matrix could not be inverted!\n\n"
201  << "This was the original error: "
202  << ee.what());
203  }
204  } // ... redirect_apply(...)

206  template <class T, class S, class V>
207  void redirect_apply(
208  const RtSpaceInterface<T, dimDomain, dimDomain, 1>& /*space*/,
209  const XT::Functions::LocalizableFunctionInterface<EntityType, DomainFieldType, dimDomain, FieldType, 1>& source,
210  DiscreteFunction<S, V>& range,
211  const XT::Common::Parameter param) const
212  {
213  static_assert(RtSpaceInterface<T, dimDomain, 1>::polOrder == 0, "Untested!");
214  const auto& rtn0_space = range.space();
215  auto& range_vector = range.vector();
216  const auto infinity = std::numeric_limits<FieldType>::infinity();
217  for (size_t ii = 0; ii < range_vector.size(); ++ii)
218  range_vector[ii] = infinity;
219  // walk the grid
220  const auto entity_it_end = grid_layer_.template end<0>();
221  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
222  const auto& entity = *entity_it;
223  const auto local_DoF_indices = rtn0_space.local_DoF_indices(entity);
224  const auto global_DoF_indices = rtn0_space.mapper().globalIndices(entity);
225  assert(global_DoF_indices.size() == local_DoF_indices.size());
226  const auto local_function = function_.local_function(entity);
227  const auto local_source = source.local_function(entity);
228  const auto local_basis = rtn0_space.base_function_set(entity);
229  // walk the intersections
230  const auto intersection_it_end = grid_layer_.iend(entity);
231  for (auto intersection_it = grid_layer_.ibegin(entity); intersection_it != intersection_it_end;
232  ++intersection_it) {
233  const auto& intersection = *intersection_it;
234  if (intersection.neighbor() && !intersection.boundary()) {
235  const auto neighbor = intersection.outside();
236  if (grid_layer_.indexSet().index(entity) < grid_layer_.indexSet().index(neighbor)) {
237  const auto local_function_neighbor = function_.local_function(neighbor);
238  const auto local_source_neighbor = source.local_function(neighbor);
239  const size_t local_intersection_index = intersection.indexInInside();
240  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
241  // do a face quadrature
242  FieldType lhs = 0;
243  FieldType rhs = 0;
244  const size_t integrand_order = local_function->order();
245  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
246  intersection.type(), boost::numeric_cast<int>(integrand_order));
247  const auto quadrature_it_end = quadrature.end();
248  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
249  const auto xx_intersection = quadrature_it->position();
250  const auto normal = intersection.unitOuterNormal(xx_intersection);
251  const FieldType integration_factor = intersection.geometry().integrationElement(xx_intersection);
252  const FieldType weight = quadrature_it->weight();
253  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
254  const auto xx_neighbor = intersection.geometryInOutside().global(xx_intersection);
255  // evaluate
256  ValueType function_value = local_function->evaluate(xx_entity, param);
257  function_value *= 0.5;
258  ValueType function_value_neighbor = local_function_neighbor->evaluate(xx_neighbor, param);
259  function_value_neighbor *= 0.5;
260  function_value += function_value_neighbor;
261  auto source_gradient = local_source->jacobian(xx_entity, param)[0];
262  source_gradient *= 0.5;
263  auto source_gradient_neighbor = local_source_neighbor->jacobian(xx_neighbor, param)[0];
264  source_gradient_neighbor *= 0.5;
265  source_gradient += source_gradient_neighbor;
266  const auto basis_values = local_basis.evaluate(xx_entity, param);
267  const auto basis_value = basis_values[local_DoF_index];
268  // compute integrals
269  lhs += integration_factor * weight * (basis_value * normal);
270  rhs += integration_factor * weight * -1.0 * compute_value(function_value, source_gradient, normal);
271  } // do a face quadrature
272  // set DoF
273  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
274  assert(!(range_vector[global_DoF_index] < infinity));
275  range_vector[global_DoF_index] = rhs / lhs;
276  }
277  } else if (intersection.boundary() && !intersection.neighbor()) {
278  const size_t local_intersection_index = intersection.indexInInside();
279  const size_t local_DoF_index = local_DoF_indices[local_intersection_index];
280  // do a face quadrature
281  FieldType lhs = 0;
282  FieldType rhs = 0;
283  const size_t integrand_order = local_function->order();
284  const auto& quadrature = QuadratureRules<DomainFieldType, dimDomain - 1>::rule(
285  intersection.type(), boost::numeric_cast<int>(integrand_order));
286  const auto quadrature_it_end = quadrature.end();
287  for (auto quadrature_it = quadrature.begin(); quadrature_it != quadrature_it_end; ++quadrature_it) {
288  const auto xx_intersection = quadrature_it->position();
289  const auto normal = intersection.unitOuterNormal(xx_intersection);
290  const auto integration_factor = intersection.geometry().integrationElement(xx_intersection);
291  const auto weight = quadrature_it->weight();
292  const auto xx_entity = intersection.geometryInInside().global(xx_intersection);
293  // evalaute
294  const ValueType function_value = local_function->evaluate(xx_entity, param);
295  const auto source_gradient = local_source->jacobian(xx_entity, param)[0];
296  const auto basis_values = local_basis.evaluate(xx_entity, param);
297  const auto basis_value = basis_values[local_DoF_index];
298  // compute integrals
299  lhs += integration_factor * weight * (basis_value * normal);
300  rhs += integration_factor * weight * -1.0 * compute_value(function_value, source_gradient, normal);
301  } // do a face quadrature
302  // set DoF
303  const size_t global_DoF_index = global_DoF_indices[local_DoF_index];
304  assert(!(range_vector[global_DoF_index] < infinity));
305  range_vector[global_DoF_index] = rhs / lhs;
306  } else
307  DUNE_THROW(XT::Common::Exceptions::internal_error, "Unknown intersection type!");
308  } // walk the intersections
309  } // walk the grid
310  } // ... redirect_apply(...)

313  R compute_value(const FieldVector<R, 1>& function_value,
314  const FieldVector<R, d>& source_gradient,
315  const FieldVector<R, d>& normal) const
316  {
317  return function_value * (source_gradient * normal);
