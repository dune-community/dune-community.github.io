270  const XT::Functions::

275 
276  template <class R, size_t rT, size_t rCT, size_t rA, size_t rCA>
277  void evaluate(
278  const LocalfunctionTupleType& local_functions_tuple,
279  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, rT, rCT>& test_base,
280  const XT::Functions::
281  LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, rA, rCA>& /*ansatzBase*/,
282  const Dune::FieldVector<DomainFieldType, dimDomain>& x_local,
283  Dune::DynamicMatrix<R>& ret) const
284  {
285  const auto& entity = test_base.entity();
286  const auto u = std::get<1>(local_functions_tuple)->evaluate(x_local);
287  ret = rhs_evaluation_.jacobian(u, entity, x_local);
288  ret *= factor_ / std::get<2>(local_functions_tuple);
289  for (size_t ii = 0; ii < ret.size(); ++ii)
290  ret[ii][ii] += 1.;
291  }
292 
293 private:
294  const RhsEvaluationType& rhs_evaluation_;
295  const SourceType& source_;
296  const double factor_;
297 }; // class LocalFvRhsNewtonIntegrand
