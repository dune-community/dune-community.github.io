93  explicit KineticFluxImplementation(const AnalyticalFluxType* analytical_flux,
94  const XT::Common::Parameter& param,
95  const BasisfunctionType& basis_functions,
96  const bool boundary)
97  : analytical_flux_(analytical_flux)
98  , param_inside_(param)
99  , param_outside_(param)
100  , basis_functions_(basis_functions)
101  {
102  param_inside_.set("boundary", {0.}, true);
103  param_outside_.set("boundary", {double(boundary)}, true);

106  template <class IntersectionType>
107  RangeType evaluate(const IntersectionType& intersection,
108  const EntityType& entity,
109  const EntityType& neighbor,
110  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
111  const DomainType& x_in_inside_coords,
112  const DomainType& x_in_outside_coords,
113  const RangeType& u_i,
114  const RangeType& u_j) const
115  {
116  // find direction of unit outer normal
117  size_t direction = intersection.indexInInside() / 2;
118  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
119  if (dynamic_cast<const EntropyFluxType*>(analytical_flux_) != nullptr) {
120  return dynamic_cast<const EntropyFluxType*>(analytical_flux_)
121  ->evaluate_kinetic_flux(entity,
122  x_in_inside_coords,
123  u_i,
124  neighbor,
125  x_in_outside_coords,
126  u_j,
127  n_ij,
128  direction,
129  param_inside_,
130  param_outside_);
131  } else {
132  static auto flux_matrices = initialize_flux_matrices(basis_functions_);
133  RangeType ret(0);
134  auto tmp_vec = ret;
135  const auto& inner_flux_matrix = flux_matrices[direction][n_ij[direction] > 0 ? 1 : 0];
136  const auto& outer_flux_matrix = flux_matrices[direction][n_ij[direction] > 0 ? 0 : 1];
137  inner_flux_matrix.mv(u_i, tmp_vec);
138  outer_flux_matrix.mv(u_j, ret);
139  ret += tmp_vec;
140  ret *= n_ij[direction];
141  return ret;
142  }
143  } // ... evaluate(...)
144 
145  const AnalyticalFluxType& analytical_flux() const
146  {
147  return analytical_flux_;
148  }
149 
150 private:
151  static FieldVector<FieldVector<SparseMatrixType, 2>, dimDomain>
152  initialize_flux_matrices(const BasisfunctionType& basis_functions)
153  {
154  // calculate < v_i b b^T >_- M^{-1} and < v_i b b^T >_+ M^{-1}
155  auto kinetic_flux_matrices = basis_functions.kinetic_flux_matrices();
156  auto mass_matrix = basis_functions.mass_matrix();
157  auto flux_matrices_dense = kinetic_flux_matrices;
158  FieldVector<FieldVector<SparseMatrixType, 2>, dimDomain> flux_matrices(
159  FieldVector<SparseMatrixType, 2>(SparseMatrixType(dimRange, dimRange, size_t(0))));
160  for (size_t dd = 0; dd < dimDomain; ++dd) {
161  for (size_t kk = 0; kk < 2; ++kk) {
162  for (size_t rr = 0; rr < dimRange; ++rr)
163  mass_matrix.solve(flux_matrices_dense[dd][kk][rr], kinetic_flux_matrices[dd][kk][rr]);
164  flux_matrices[dd][kk] = flux_matrices_dense[dd][kk];
165  } // kk
166  } // dd
167  return flux_matrices;
168  }
169 
170  const AnalyticalFluxType* analytical_flux_;
171  XT::Common::Parameter param_inside_;
172  XT::Common::Parameter param_outside_;
173  const BasisfunctionType& basis_functions_;
174 }; // class KineticFluxImplementation<...>
