116  template <class SourceType, class RangeType>
117  void apply_p1_dg(const SourceType& source, RangeType& range) const
118  {
119  range.vector() *= 0.;
120  // data structures we need
121  // * a map from a global vertex index to global DoF indices
122  // given a vertex, one obtains a set of all global DoF ids, which are associated with this vertex
123  std::map<size_t, std::set<size_t>> global_vertex_id_to_global_DoF_id_map;
124  // * a map from a global DoF index to the global index of its associated vertex
125  std::vector<size_t> global_DoF_id_to_global_vertex_id_map(range.space().mapper().size());
126  // * a set to hold the global id of all boundary vertices
127  std::set<size_t> boundary_vertices;
128 
129  const auto entity_it_end = grid_layer_.template end<0>();
130  // walk the grid to create the maps explained above and to find the boundary vertices
131  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
132  const auto& entity = *entity_it;
133  const size_t num_vertices = entity.subEntities(dimDomain);
134  const auto basis = range.space().base_function_set(entity);
135  if (basis.size() != num_vertices)
136  DUNE_THROW(XT::Common::Exceptions::internal_error, "basis.size() = " << basis.size());
137 
138  // loop over all vertices of the entitity, to find their associated global DoF indices
139  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
140  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id);
141  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
142  const auto vertex_center = vertex.geometry().center();
143  // find the local basis function which corresponds to this vertex
144  const auto basis_values = basis.evaluate(entity.geometry().local(vertex_center));
145  if (basis_values.size() != num_vertices)
146  DUNE_THROW(XT::Common::Exceptions::internal_error, "basis_values.size() = " << basis_values.size());
147  size_t ones = 0;
148  size_t zeros = 0;
149  size_t failures = 0;
150  size_t local_DoF_index = 0;
151  for (size_t ii = 0; ii < basis.size(); ++ii) {
152  if (std::abs(basis_values[ii][0] - 1.0) < 1e-14) {
153  local_DoF_index = ii;
154  ++ones;
155  } else if (std::abs(basis_values[ii][0] - 0.0) < 1e-14)
156  ++zeros;
157  else
158  ++failures;
159  }
160  if (ones != 1 || zeros != (basis.size() - 1) || failures > 0) {
161  std::stringstream ss;
162  ss << "ones = " << ones << ", zeros = " << zeros << ", failures = " << failures
163  << ", num_vertices = " << num_vertices << ", entity " << grid_layer_.indexSet().index(entity)
164  << ", vertex " << local_vertex_id << ": [ " << vertex_center << "], ";
165  XT::Common::print(basis_values, "basis_values", ss);
166  DUNE_THROW(XT::Common::Exceptions::internal_error, ss.str());
167  }
168  // now we know that the local DoF index of this vertex is ii
169  const size_t global_DoF_index = range.space().mapper().mapToGlobal(entity, local_DoF_index);
170  global_DoF_id_to_global_vertex_id_map[global_DoF_index] = global_vertex_id;
171  global_vertex_id_to_global_DoF_id_map[global_vertex_id].insert(global_DoF_index);
172  } // loop over all vertices
173 
174  static const constexpr auto dirichlet = XT::Grid::DirichletBoundary();
175  // in order to determine the boundary vertices, we need to
176  // loop over all intersections
177  const auto intersectionEndIt = grid_layer_.iend(entity);
178  for (auto intersectionIt = grid_layer_.ibegin(entity); intersectionIt != intersectionEndIt; ++intersectionIt) {
179  const auto& intersection = *intersectionIt;
180  if (boundary_info_.type(intersection) == dirichlet) {
181  const auto& intersection_geometry = intersection.geometry();
182  for (auto local_intersection_corner_id : XT::Common::value_range(intersection_geometry.corners())) {
183  const auto global_intersection_corner = intersection_geometry.corner(local_intersection_corner_id);
184  // now, we need to find the entity's vertex this intersection's corner point equals to, so we
185  // loop over all vertices of the entity
186  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
187  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id);
188  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
189  const auto vertex_center = vertex.geometry().center();
190  if (XT::Common::FloatCmp::eq(global_intersection_corner, vertex_center))
191  boundary_vertices.insert(global_vertex_id);
192  } // loop over all vertices of the entity
193  } // loop over all intersection corners
194  } // if (intersection.boundary() && !intersection.neighbor())
195  } // loop over all intersections
196  } // walk the grid for the first time
197 
198  // walk the grid for the second time
199  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
200  const auto& entity = *entity_it;
201  const auto num_vertices = entity.subEntities(dimDomain);
202  // get the local functions
203  const auto local_source = source.local_function(entity);
204 
205  // * loop over all local DoFs
206  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
207  const size_t global_DoF_index =
208  range.space().mapper().mapToGlobal(entity, XT::Common::numeric_cast<size_t>(local_vertex_id));
209  const size_t global_vertex_id = global_DoF_id_to_global_vertex_id_map[global_DoF_index];
210  // if we are on the domain boundary
211  if (boundary_vertices.count(global_vertex_id)) {
212  // set the dof to zero (we have dirichlet zero)
213  range.vector().set_entry(global_DoF_index, FieldType(0));
214  } else {
215  // do the oswald projection
216  const size_t num_DoFS_per_vertex = global_vertex_id_to_global_DoF_id_map[global_vertex_id].size();
217  // * get the source value
218  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id).geometry().center();
219  const FieldType source_value = local_source->evaluate(entity.geometry().local(vertex));
220  // * and add it to all target DoFs
221  for (size_t target_global_DoF_id : global_vertex_id_to_global_DoF_id_map[global_vertex_id])
222  range.vector().add_to_entry(target_global_DoF_id, source_value / num_DoFS_per_vertex);
223  } // if (boundary_vertices.find(global_vertex_id))
224  } // loop over all local DoFs
225  } // walk the grid for the second time
226  } // ... apply_p1_dg(...)
227
