153  virtual RangeFieldType solve(const RangeFieldType t_end,
154  const RangeFieldType initial_dt,
155  const size_t num_save_steps,
156  const bool save_solution,
157  const bool output_progress,
158  const bool visualize,
159  const std::string filename_prefix,
160  SolutionType& sol,
161  const VisualizerType& visualizer)
162  {
163  RangeFieldType dt = initial_dt;
164  RangeFieldType t = current_time();
165  assert(Dune::XT::Common::FloatCmp::ge(t_end, t));
166  size_t time_step_counter = 0;
167 
168  const RangeFieldType save_interval = (t_end - t) / num_save_steps;
169  RangeFieldType next_save_time = t + save_interval > t_end ? t_end : t + save_interval;
170  size_t save_step_counter = 1;
171 
172  // save/visualize initial solution
173  if (save_solution)
174  sol.insert(std::make_pair(t, current_solution()));
175  if (visualize)
176  visualizer(current_solution(), filename_prefix, 0);
177 
178  while (Dune::XT::Common::FloatCmp::lt(t, t_end)) {
179  RangeFieldType max_dt = dt;
180  // match saving times and t_end exactly
181  if (Dune::XT::Common::FloatCmp::gt(t + dt, t_end))
182  max_dt = t_end - t;
183  if (Dune::XT::Common::FloatCmp::gt(t + dt, next_save_time) && num_save_steps != size_t(-1))
184  max_dt = std::min(next_save_time - t, max_dt);
185 
186  // do a timestep
187  dt = step(dt, max_dt);
188  t = current_time();
189 
190  // augment time step counter
191  ++time_step_counter;
192 
193  // check if data should be written in this timestep (and write)
194  if (Dune::XT::Common::FloatCmp::ge(t, next_save_time) || num_save_steps == size_t(-1)) {
195  if (save_solution)
196  sol.insert(sol.end(), std::make_pair(t, current_solution()));
197  if (visualize)
198  visualizer(current_solution(), filename_prefix, save_step_counter);
199  if (output_progress)
200  std::cout << "time step " << time_step_counter << " done, time =" << t << ", current dt= " << dt << std::endl;
201  next_save_time += save_interval;
202  ++save_step_counter;
203  }
204  } // while (t < t_end)
205 
206  return dt;
207  } // ... solve(...)
