214  template <bool is_dg = (SP::space_type == SpaceType::dg), bool anything = false>
215  struct projector
216  {
217  template <class V>
218  static V project(const type& self, const std::vector<V>& local_vectors, const std::vector<size_t>& subdomains)
219  {
220  V neighborhood_vector(self.mapper().size(), 0.);
221  assert(local_vectors.size() == subdomains.size() && "This should not happen, blame the caller!");
222  for (size_t ii = 0; ii < local_vectors.size(); ++ii) {
223  const auto subdomain = subdomains[ii];
224  const auto& local_vector = local_vectors[ii];
225  for (size_t jj = 0; jj < local_vector.size(); ++jj)
226  neighborhood_vector[self.mapper().mapToGlobal(subdomain, jj)] = local_vector[jj];
227  }
228  return neighborhood_vector;
229  }
230  }; // struct projector<true, ...>
231 
232  template <bool anything>
233  struct projector<false, anything>
234  {
235  template <class V>
236  static V project(const type&, const std::vector<V>&, const std::vector<size_t>&)
237  {
238  static_assert(AlwaysFalse<V>::value, "Not implemented for non DG spaces");
239  }
240  };

250  c.def("project_onto_neighborhood",
251  [](const type& self, const std::vector<V>& local_vectors, const std::vector<ssize_t>& neighborhood) {
252  if (local_vectors.size() != neighborhood.size())
253  DUNE_THROW(XT::Common::Exceptions::shapes_do_not_match,
254  "local_vectors.size(): " << local_vectors.size() << "\n neighborhood.size(): "
255  << neighborhood.size());
256  std::vector<size_t> subdomains(neighborhood.size());
257  for (size_t ii = 0; ii < neighborhood.size(); ++ii)
258  subdomains[ii] = XT::Common::numeric_cast<size_t>(neighborhood[ii]);
259  std::vector<std::shared_ptr<const S>> neighborhood_spaces(self.dd_grid().size(), nullptr);
260  for (const auto& subdomain : subdomains) {
261  if (self.backend()[subdomain] == nullptr)
262  DUNE_THROW(XT::Common::Exceptions::you_are_using_this_wrong,
263  "This BlockSpace (restricted to a neighborhood) does not have a local space for subdomain "
264  << subdomain
265  << "!");
266  neighborhood_spaces[subdomain] = self.backend()[subdomain];
267  }
268  const type neighborhood_space(self.dd_grid(), neighborhood_spaces);
269  return projector<>::project(neighborhood_space, local_vectors, subdomains);
270  },
