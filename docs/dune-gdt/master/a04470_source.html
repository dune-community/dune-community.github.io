230  using BaseType::evaluate;
231 
232  void evaluate(const DomainType& xx,
233  std::vector<RangeType>& ret,
234  const XT::Common::Parameter& /*mu*/ = {}) const override final
235  {
236  assert(this->is_a_valid_point(xx));
237  // evaluate shape functions
238  finite_element_.localBasis().evaluateFunction(xx, ret);
239  // flip and scale shape functions to ensure
240  // - continuity of normal component and
241  // - to ensure basis*integrationElementNormal = 1
242  for (size_t ii = 0; ii < finite_element_.localBasis().size(); ++ii)
243  ret[ii] *= switches_[ii];
244  // apply piola transformation
245  const auto J_T = this->entity().geometry().jacobianTransposed(xx);
246  const auto det_J_T = std::abs(J_T.determinant());
247  RangeType tmp_value;
248  for (size_t ii = 0; ii < finite_element_.localBasis().size(); ++ii) {
249  J_T.mtv(ret[ii], tmp_value);
250  tmp_value /= det_J_T;
251  ret[ii] = tmp_value;
252  }
253  } // ... evaluate(...)
254 
255  using BaseType::jacobian;
256 
257  void jacobian(const DomainType& xx,
258  std::vector<JacobianRangeType>& ret,
259  const XT::Common::Parameter& /*mu*/ = {}) const override final
260  {
261  assert(this->is_a_valid_point(xx));
262  // evaluate jacobian of shape functions
263  finite_element_.localBasis().evaluateJacobian(xx, ret);
264  // flip and scale shape functions to ensure
265  // - continuity of normal component and
266  // - to ensure basis*integrationElementNormal = 1
267  for (size_t ii = 0; ii < finite_element_.localBasis().size(); ++ii)
268  ret[ii] *= switches_[ii];
269  // apply piola transformation
270  const auto J_T = this->entity().geometry().jacobianTransposed(xx);
271  const auto det_J_T = std::abs(J_T.determinant());
272  const auto J_inv_T = this->entity().geometry().jacobianInverseTransposed(xx);
273  auto tmp_jacobian_row = ret[0][0];
274  for (size_t ii = 0; ii < finite_element_.localBasis().size(); ++ii) {
275  for (size_t jj = 0; jj < d; ++jj) {
276  J_inv_T.mtv(ret[ii][jj], tmp_jacobian_row);
277  J_T.mtv(tmp_jacobian_row, ret[ii][jj]);
278  ret[ii][jj] /= det_J_T;
279  }
280  }
281  } // ... jacobian(...)

285  const std::vector<R>& switches_;

290 class RaviartThomasSpace
291  : public RtSpaceInterface<internal::RaviartThomasSpaceTraits<GL, p, R>, GL::dimension, GL::dimension>
292 {
293 public:
294  using Traits = internal::RaviartThomasSpaceTraits<GL, p, R>;
295 
296 private:
297  using BaseType = RtSpaceInterface<Traits, GL::dimension, GL::dimension>;
298  using ThisType = RaviartThomasSpace<GL, p, R>;
299  using D = typename GL::ctype;
300  static const constexpr size_t d = BaseType::dimDomain;
301  using FiniteElementType = typename BaseType::BackendType;
302  typedef typename Traits::DofCommunicationChooserType DofCommunicationChooserType;
303 
304 public:
305  using typename BaseType::GridLayerType;
306  using typename BaseType::BackendType;
307  using typename BaseType::EntityType;
308  using typename BaseType::MapperType;
309  using typename BaseType::BaseFunctionSetType;
310  typedef typename Traits::DofCommunicatorType DofCommunicatorType;
311 
312  RaviartThomasSpace(GridLayerType grd_lr)
313  : grid_layer_(grd_lr)
314  , communicator_(DofCommunicationChooserType::create(grid_layer_))
315  , backend_(0)
316  , finite_elements_(new std::map<GeometryType, std::shared_ptr<FiniteElementType>>())
317  , geometry_to_local_DoF_indices_map_(new std::map<GeometryType, std::vector<size_t>>())
318  , entity_indices_(new ZeroOrderScalarDiscontinuousMapper<GL>(grid_layer_)) // <- We need unique indices for codim 0
319  , switches_(new std::vector<std::vector<R>>(entity_indices_->size())) // entities: this cannot be achieved by
320  , mapper_(nullptr) // the grid layers index set for mixed grids.
321  , communicator_prepared_(false)
322  {
323  // create finite elements
324  for (auto&& geometry_type : grid_layer_.indexSet().types(0)) {
325  auto finite_element = Traits::template ChooseLocalFiniteElement<>::create(geometry_type, p);
326  finite_elements_->insert(std::make_pair(geometry_type, finite_element));
327  // this is only valid for p0 elements
328  geometry_to_local_DoF_indices_map_->insert(
329  std::make_pair(geometry_type, std::vector<size_t>(finite_element->size())));
330  }
331  // compute local-key-to-intersection relationship
332  for (const auto& geometry_type_and_finite_element_ptr : *finite_elements_) {
333  const auto& geometry_type = geometry_type_and_finite_element_ptr.first;
334  const auto& finite_element = *geometry_type_and_finite_element_ptr.second;
335  const auto& coeffs = finite_element.localCoefficients();
336  auto& local_key_to_intersection_map = geometry_to_local_DoF_indices_map_->at(geometry_type);
337  for (size_t ii = 0; ii < coeffs.size(); ++ii) {
338  const auto& local_key = coeffs.localKey(ii);
339  if (local_key.index() != 0)
340  DUNE_THROW(XT::Common::Exceptions::internal_error, "This must not happen for p0!");
341  if (local_key.codim() != 1)
342  DUNE_THROW(XT::Common::Exceptions::internal_error, "This must not happen for p0!");
343  local_key_to_intersection_map[ii] = local_key.subEntity();
344  }
345  }
346  // compute scaling to ensure basis*integrationElementNormal = 1
347  std::map<GeometryType, std::vector<R>> geometry_to_scaling_factors_map;
348  for (const auto& geometry_type_and_finite_element_ptr : *finite_elements_) {
349  const auto& geometry_type = geometry_type_and_finite_element_ptr.first;
350  const auto& finite_element = *geometry_type_and_finite_element_ptr.second;
351  const auto& shape_functions = finite_element.localBasis();
352  std::vector<typename BaseFunctionSetType::RangeType> shape_functions_evaluations(shape_functions.size());
353  const auto& reference_element = ReferenceElements<D, d>::general(geometry_type);
354  const auto num_intersections = reference_element.size(1);
355  geometry_to_scaling_factors_map.insert(std::make_pair(geometry_type, std::vector<R>(num_intersections, R(1.))));
356  auto& scaling_factors = geometry_to_scaling_factors_map.at(geometry_type);
357  const auto& DoF_indices = geometry_to_local_DoF_indices_map_->at(geometry_type);
358  for (auto&& intersection_index : XT::Common::value_range(num_intersections)) {
359  const auto& normal = reference_element.integrationOuterNormal(intersection_index);
360  const auto& intersection_center = reference_element.position(intersection_index, 1);
361  const auto DoF_index = DoF_indices[intersection_index];
362  shape_functions.evaluateFunction(intersection_center, shape_functions_evaluations);
363  scaling_factors[intersection_index] /= shape_functions_evaluations[DoF_index] * normal;
364  }
365  }
366  // compute switches (as signs of the scaling factor) to ensure continuity of the normal component
367  for (auto&& entity : elements(grid_layer_)) {
368  const auto geometry_type = entity.geometry().type();
369  const auto& finite_element = *finite_elements_->at(geometry_type);
370  const auto& coeffs = finite_element.localCoefficients();
371  const auto entity_index = entity_indices_->mapToGlobal(entity, 0);
372  (*switches_)[entity_index] = geometry_to_scaling_factors_map.at(geometry_type);
373  auto& local_switches = switches_->at(entity_index);
374  for (auto&& intersection : intersections(grid_layer_, entity)) {
375  if (intersection.neighbor() && entity_index < entity_indices_->mapToGlobal(intersection.outside(), 0)) {
376  const auto intersection_index = XT::Common::numeric_cast<unsigned int>(intersection.indexInInside());
377  for (size_t ii = 0; ii < coeffs.size(); ++ii) {
378  const auto& local_key = coeffs.localKey(ii);
379  const auto DoF_subentity_index = local_key.subEntity();
380  if (local_key.codim() == 1 && DoF_subentity_index == intersection_index)
381  local_switches[DoF_subentity_index] *= -1.;
382  }
383  }
384  }
385  }
386  // create mapper
387  mapper_ = std::make_shared<MapperType>(grid_layer_, finite_elements_);
388  } // RaviartThomasSpace(...)
