108  typedef EntropyBasedLocalFlux<BasisfunctionType, GridLayerType, StateType, dimQuadrature> EntropyFluxType;

110  explicit KineticFluxImplementation(const AnalyticalFluxType* analytical_flux,
111  const XT::Common::Parameter& param,
112  const BasisfunctionType& basis_functions,
113  const bool boundary)
114  : analytical_flux_(analytical_flux)
115  , param_inside_(param)
116  , param_outside_(param)
117  , basis_functions_(basis_functions)
118  {
119  param_inside_.set("boundary", {0.}, true);
120  param_outside_.set("boundary", {double(boundary)}, true);
121  }

123  template <class IntersectionType>
124  RangeType evaluate(const IntersectionType& intersection,
125  const EntityType& entity,
126  const EntityType& neighbor,
127  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
128  const DomainType& x_in_inside_coords,
129  const DomainType& x_in_outside_coords,
130  const RangeType& u_i,
131  const RangeType& u_j) const
132  {
133  // find direction of unit outer normal
134  size_t direction = intersection.indexInInside() / 2;
135  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
136  if (dynamic_cast<const EntropyFluxType*>(analytical_flux_) != nullptr) {
137  return dynamic_cast<const EntropyFluxType*>(analytical_flux_)
138  ->evaluate_kinetic_flux(entity,
139  x_in_inside_coords,
140  u_i,
141  neighbor,
142  x_in_outside_coords,
143  u_j,
144  n_ij,
145  direction,
146  param_inside_,
147  param_outside_);
148  } else {
149  static auto flux_matrices = initialize_flux_matrices(basis_functions_);
150  RangeType ret(0);
151  auto tmp_vec = ret;
152  const auto& inner_flux_matrix = flux_matrices[direction][n_ij[direction] > 0 ? 1 : 0];
153  const auto& outer_flux_matrix = flux_matrices[direction][n_ij[direction] > 0 ? 0 : 1];
154  inner_flux_matrix.mv(u_i, tmp_vec);
155  outer_flux_matrix.mv(u_j, ret);
156  ret += tmp_vec;
157  ret *= n_ij[direction];
158  return ret;
159  }
160  } // ... evaluate(...)
161 
162  const AnalyticalFluxType& analytical_flux() const
163  {
164  return analytical_flux_;
165  }
166 
167 private:
168  static FieldVector<FieldVector<SparseMatrixType, 2>, dimDomain>
169  initialize_flux_matrices(const BasisfunctionType& basis_functions)
170  {
171  // calculate < v_i b b^T >_- M^{-1} and < v_i b b^T >_+ M^{-1}
172  auto kinetic_flux_matrices = basis_functions.kinetic_flux_matrices();
173  auto mass_matrix = basis_functions.mass_matrix();
174  auto flux_matrices_dense = kinetic_flux_matrices;
175  FieldVector<FieldVector<SparseMatrixType, 2>, dimDomain> flux_matrices(
176  FieldVector<SparseMatrixType, 2>(SparseMatrixType(dimRange, dimRange, size_t(0))));
177  for (size_t dd = 0; dd < dimDomain; ++dd) {
178  for (size_t kk = 0; kk < 2; ++kk) {
179  for (size_t rr = 0; rr < dimRange; ++rr)
180  mass_matrix.solve(flux_matrices_dense[dd][kk][rr], kinetic_flux_matrices[dd][kk][rr]);
181  flux_matrices[dd][kk] = flux_matrices_dense[dd][kk];
182  } // kk
183  } // dd
184  return flux_matrices;
185  }
186 
187  const AnalyticalFluxType* analytical_flux_;
188  XT::Common::Parameter param_inside_;
189  XT::Common::Parameter param_outside_;
190  const BasisfunctionType& basis_functions_;
191 }; // class KineticFluxImplementation<...>
