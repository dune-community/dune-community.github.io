37 template <class ProductGridLayer, class InterpolationGridLayerType>
38 class NonconformityProduct
39  : public LocalizableProductBase<ProductGridLayer,
40  XT::Functions::
41  LocalizableFunctionInterface<XT::Grid::extract_entity_t<ProductGridLayer>,
42  typename ProductGridLayer::ctype,
43  ProductGridLayer::dimension,
44  double,
45  1>>
46 {
47  static_assert(XT::Grid::is_layer<ProductGridLayer>::value, "");
48  static_assert(XT::Grid::is_layer<InterpolationGridLayerType>::value, "");
49  typedef XT::Grid::extract_entity_t<ProductGridLayer> E;
50  static_assert(std::is_same<XT::Grid::extract_entity_t<InterpolationGridLayerType>, E>::value, "");
51  typedef typename ProductGridLayer::ctype D;
52  static const constexpr size_t d = ProductGridLayer::dimension;
53  typedef double R;
54 
55 public:
56  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, 1> ScalarFunctionType;
57  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, d, d> TensorFunctionType;
58 
59 private:
60  typedef LocalizableProductBase<ProductGridLayer, ScalarFunctionType> BaseType;
61  typedef NonconformityProduct<ProductGridLayer, InterpolationGridLayerType> ThisType;
62  typedef LocalVolumeIntegralOperator<LocalLambdaBinaryVolumeIntegrand<E>,
63  typename ScalarFunctionType::LocalfunctionType>
64  LocalProductType;
65  typedef DiscontinuousLagrangeSpace<InterpolationGridLayerType, 1, R> DgSpaceType;
66  typedef DiscreteFunction<DgSpaceType> DiscreteFunctionType;
67 
68 public:
69  using typename BaseType::GridLayerType;
70 
71  NonconformityProduct(GridLayerType product_grid_layer,
72  InterpolationGridLayerType interpolation_grid_layer,
73  const XT::Grid::BoundaryInfo<XT::Grid::extract_intersection_t<InterpolationGridLayerType>>&
74  interpolation_boundary_info,
75  const ScalarFunctionType& lambda,
76  const TensorFunctionType& kappa,
77  const ScalarFunctionType& u,
78  const ScalarFunctionType& v,
79  const size_t over_integrate = 2)
80  : BaseType(product_grid_layer, u, v)
81  , interpolation_grid_layer_(interpolation_grid_layer)
82  , lambda_(lambda)
83  , kappa_(kappa)
84  , dg_space_(interpolation_grid_layer_)
85  , interpolated_u_(dg_space_)
86  , interpolated_v_(dg_space_)
87  , over_integrate_(over_integrate)
88  , local_product_(
89  // the order lambda
90  [&](const auto& local_u, const auto& local_v) {
91  const auto& entity = local_u.entity();
92  const auto local_lambda = lambda_.local_function(entity);
93  const auto local_kappa = kappa_.local_function(entity);
94  return local_lambda->order() + local_kappa->order()
95  + size_t(std::max(ssize_t(local_u.order()) - 1, ssize_t(0))
96  + std::max(ssize_t(local_v.order()) - 1, ssize_t(0)))
97  + over_integrate_;
98  },
99  // the evaluate lambda
100  [&](const auto& local_u, const auto& local_v, const auto& local_point, auto& ret) {
101  const auto& entity = local_u.entity();
102  XT::Common::FieldMatrix<R, d, d> diffusion = kappa_.local_function(entity)->evaluate(local_point);
103  diffusion *= lambda_.local_function(entity)->evaluate(local_point);
104  const auto grad_u = local_u.jacobian(local_point).at(0)[0];
105  const auto grad_interpolated_u = interpolated_u_.local_function(entity)->jacobian(local_point)[0];
106  const auto grad_v = local_v.jacobian(local_point).at(0)[0];
107  const auto grad_interpolated_v = interpolated_v_.local_function(entity)->jacobian(local_point)[0];
108  ret[0][0] = (diffusion * (grad_u - grad_interpolated_u)) * (grad_v - grad_interpolated_v);
109  })
110  {
111  OswaldInterpolationOperator<InterpolationGridLayerType> oswald_interpolation(interpolation_grid_layer_,
112  interpolation_boundary_info);
113  oswald_interpolation.apply(this->range(), interpolated_u_);
114  oswald_interpolation.apply(this->source(), interpolated_v_);
115  this->append(local_product_);
116  }
117 
118  NonconformityProduct(const ThisType&) = delete;
119  NonconformityProduct(ThisType&&) = delete;
120 
121 private:
122  const InterpolationGridLayerType interpolation_grid_layer_;
123  const ScalarFunctionType& lambda_;
124  const TensorFunctionType& kappa_;
125  const DgSpaceType dg_space_;
126  DiscreteFunctionType interpolated_u_;
127  DiscreteFunctionType interpolated_v_;
128  const size_t over_integrate_;
129  const LocalProductType local_product_;
130 }; // class NonconformityProduct

202 template <class ProductGridLayer, class ReconstructionGridLayer>
203 class ResidualProduct
204  : internal::ResidualProductBase<ProductGridLayer, ReconstructionGridLayer>,
205  public LocalizableProductBase<ProductGridLayer,
206  XT::Functions::
207  LocalizableFunctionInterface<XT::Grid::extract_entity_t<ProductGridLayer>,
208  typename ProductGridLayer::ctype,
209  ProductGridLayer::dimension,
210  double,
211  1>>
212 {
213  typedef internal::ResidualProductBase<ProductGridLayer, ReconstructionGridLayer> ResidualProductBaseType;
214  typedef LocalizableProductBase<ProductGridLayer,
215  XT::Functions::
216  LocalizableFunctionInterface<XT::Grid::extract_entity_t<ProductGridLayer>,
217  typename ProductGridLayer::ctype,
218  ProductGridLayer::dimension,
219  double,
220  1>>
221  LocalizableProductBaseType;
222 
223 public:
224  using typename ResidualProductBaseType::ScalarFunctionType;
225  using typename ResidualProductBaseType::TensorFunctionType;
226 
227 private:
228  using typename ResidualProductBaseType::E;
229  using typename ResidualProductBaseType::R;
230  typedef LocalVolumeIntegralOperator<LocalLambdaBinaryVolumeIntegrand<E>,
231  typename ScalarFunctionType::LocalfunctionType>
232  LocalProductType;
233 
234 public:
235  ResidualProduct(ProductGridLayer product_grid_layer,
236  ReconstructionGridLayer reconstruction_grid_layer,
237  const ScalarFunctionType& lambda,
238  const TensorFunctionType& kappa,
239  const ScalarFunctionType& f,
240  const ScalarFunctionType& u,
241  const ScalarFunctionType& v,
242  const double& poincare_constant = 1.0 / (M_PIl * M_PIl),
243  const size_t over_integrate = 2)
244  : ResidualProductBaseType(reconstruction_grid_layer, lambda, kappa, f, u, v)
245  , LocalizableProductBaseType(product_grid_layer,
246  this->f_minus_divergence_of_reconstructed_u_,
247  this->f_minus_divergence_of_reconstructed_v_)
248  , lambda_(lambda)
249  , kappa_(kappa)
250  , poincare_constant_(poincare_constant)
251  , over_integrate_(over_integrate)
252  , local_product_(
253  // the order lambda
254  [&](const auto& local_f_minus_divergence_of_reconstructed_u,
255  const auto& local_f_minus_divergence_of_reconstructed_v) {
256  return local_f_minus_divergence_of_reconstructed_u.order()
257  + local_f_minus_divergence_of_reconstructed_v.order() + over_integrate_;
258  },
259  // the evaluate lambda
260  [&](const auto& local_f_minus_divergence_of_reconstructed_u,
261  const auto& local_f_minus_divergence_of_reconstructed_v,
262  const auto& local_point,
263  auto& ret) {
264  const auto& entity = local_f_minus_divergence_of_reconstructed_u.entity();
265  // we need the min_ev for this entity, so we just evaluate in one point
266  const auto center = entity.geometry().local(entity.geometry().center());
267  auto diffusion = kappa_.local_function(entity)->evaluate(center);
268  diffusion *= lambda_.local_function(entity)->evaluate(center);
269  const auto min_ev = XT::LA::make_eigen_solver(
270  diffusion, XT::Common::Configuration{{"assert_positive_eigenvalues"}, {1e-15}})
271  .min_eigenvalues(1)
272  .at(0);
273  const auto h = XT::Grid::entity_diameter(entity);
274  ret[0][0] = (poincare_constant_ / min_ev) * h * h
275  * local_f_minus_divergence_of_reconstructed_u.evaluate(local_point).at(0)[0]
276  * local_f_minus_divergence_of_reconstructed_v.evaluate(local_point).at(0)[0];
277  })
278  {
279  this->append(local_product_);
280  }
281 
282 private:
283  const ScalarFunctionType& lambda_;
284  const TensorFunctionType& kappa_;
285  const double poincare_constant_;
286  const size_t over_integrate_;
287  const LocalProductType local_product_;
288 }; // class ResidualProduct

291 template <class ProductGridLayer, class ReconstructionGridLayer>
292 class DiffusiveFluxProduct
293  : public LocalizableProductBase<ProductGridLayer,
294  XT::Functions::
295  LocalizableFunctionInterface<XT::Grid::extract_entity_t<ProductGridLayer>,
296  typename ProductGridLayer::ctype,
297  ProductGridLayer::dimension,
298  double,
299  1>>
300 {
301  static_assert(XT::Grid::is_layer<ProductGridLayer>::value, "");
302  static_assert(XT::Grid::is_layer<ReconstructionGridLayer>::value, "");
303  typedef XT::Grid::extract_entity_t<ProductGridLayer> E;
304  static_assert(std::is_same<XT::Grid::extract_entity_t<ReconstructionGridLayer>, E>::value, "");
305  typedef typename ProductGridLayer::ctype D;
306  static const constexpr size_t d = ProductGridLayer::dimension;
307  typedef double R;
308  typedef DiffusiveFluxProduct<ProductGridLayer, ReconstructionGridLayer> ThisType;
309 
310 public:
311  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, 1> ScalarFunctionType;
312  typedef XT::Functions::LocalizableFunctionInterface<E, D, d, R, d, d> TensorFunctionType;
313 
314 private:
315  typedef RaviartThomasSpace<ReconstructionGridLayer, 0, R> RtSpaceType;
316  typedef DiscreteFunction<RtSpaceType> FluxReconstructionType;
317  typedef LocalizableProductBase<ProductGridLayer, XT::Functions::LocalizableFunctionInterface<E, D, d, R, 1>> BaseType;
318  typedef LocalVolumeIntegralOperator<LocalLambdaBinaryVolumeIntegrand<E>,
319  typename ScalarFunctionType::LocalfunctionType>
320  LocalProductType;
321 
322 public:
323  DiffusiveFluxProduct(ProductGridLayer product_grid_layer,
324  ReconstructionGridLayer reconstruction_grid_layer,
325  const ScalarFunctionType& lambda,
326  const TensorFunctionType& kappa,
327  const ScalarFunctionType& u,
328  const ScalarFunctionType& v,
329  const size_t over_integrate = 2)
330  : BaseType(product_grid_layer, u, v)
331  , lambda_(lambda)
332  , kappa_(kappa)
333  , rt_space_(reconstruction_grid_layer)
334  , reconstructed_u_(rt_space_)
335  , reconstructed_v_(rt_space_)
336  , over_integrate_(over_integrate)
337  , local_product_(
338  // the order lambda
339  [&](const auto& local_u, const auto& local_v) {
340  const auto& entity = local_u.entity();
341  const size_t diffusion_order =
342  lambda_.local_function(entity)->order() + kappa_.local_function(entity)->order();
343  return 3 * diffusion_order + size_t(std::max(ssize_t(local_u.order()) - 1, ssize_t(0)))
344  + size_t(std::max(ssize_t(local_v.order()) - 1, ssize_t(0))) + over_integrate_;
345  },
346  // the evaluate lambda
347  [&](const auto& local_u, const auto& local_v, const auto& local_point, auto& ret) {
348  const auto& entity = local_u.entity();
349  XT::Common::FieldMatrix<R, d, d> diffusion = kappa_.local_function(entity)->evaluate(local_point);
350  diffusion *= lambda_.local_function(entity)->evaluate(local_point);
351  XT::Common::FieldMatrix<R, d, d> one_over_diffusion = diffusion;
352  one_over_diffusion.invert(); // there is no documented way to assert that the inversion was successfull
353  const auto grad_u = local_u.jacobian(local_point).at(0)[0];
354  const auto grad_v = local_v.jacobian(local_point).at(0)[0];
355  const auto val_reconstructed_u = reconstructed_u_.local_function(entity)->evaluate(local_point);
356  const auto val_reconstructed_v = reconstructed_v_.local_function(entity)->evaluate(local_point);
357  ret[0][0] = (one_over_diffusion * ((diffusion * grad_u) + val_reconstructed_u)) // clang-format off
358  * ((diffusion * grad_v) + val_reconstructed_v); // clang-format on
359  })
360  {
361  DiffusiveFluxReconstructionOperator<ReconstructionGridLayer, ScalarFunctionType, TensorFunctionType>
362  flux_reconstruction(reconstruction_grid_layer, lambda, kappa);
363  flux_reconstruction.apply(u, reconstructed_u_);
364  flux_reconstruction.apply(v, reconstructed_v_);
365  this->append(local_product_);
366  }
367 
368  DiffusiveFluxProduct(const ThisType&) = delete;
369  DiffusiveFluxProduct(ThisType&&) = delete;
370 
371 private:
372  const ScalarFunctionType& lambda_;
373  const TensorFunctionType& kappa_;
374  const RtSpaceType rt_space_;
375  FluxReconstructionType reconstructed_u_;
376  FluxReconstructionType reconstructed_v_;
377  const size_t over_integrate_;
378  const LocalProductType local_product_;
379 }; // class DiffusiveFluxProduct
