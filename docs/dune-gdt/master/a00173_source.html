319  template <class IntersectionType, class R>
320  void
321  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionEntity,
322  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, 2, R, 1, 1>& localFunctionNeighbor,
323  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseEntity,
324  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseEntity,
325  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& testBaseNeighbor,
326  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, 2, R, 1, 1>& ansatzBaseNeighbor,
327  const IntersectionType& intersection,
328  const Dune::FieldVector<DomainFieldType, 1>& localPoint,
329  Dune::DynamicMatrix<R>& entityEntityRet,
330  Dune::DynamicMatrix<R>& neighborNeighborRet,
331  Dune::DynamicMatrix<R>& entityNeighborRet,
332  Dune::DynamicMatrix<R>& neighborEntityRet) const
333  {
334  // clear ret
335  entityEntityRet *= 0.0;
336  neighborNeighborRet *= 0.0;
337  entityNeighborRet *= 0.0;
338  neighborEntityRet *= 0.0;
339  // convert local point (which is in intersection coordinates) to entity/neighbor coordinates
340  const auto localPointEn = intersection.geometryInInside().global(localPoint);
341  const auto localPointNe = intersection.geometryInOutside().global(localPoint);
342  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
343  // evaluate local function
344  const auto functionValueEn = localFunctionEntity.evaluate(localPointEn);
345  const auto functionValueNe = localFunctionNeighbor.evaluate(localPointNe);
346  // compute penalty (see Epshteyn, Riviere, 2007)
347  const auto max_polorder =
348  std::max(testBaseEntity.order(),
349  std::max(ansatzBaseEntity.order(), std::max(testBaseNeighbor.order(), ansatzBaseNeighbor.order())));
350  const R sigma = internal::inner_sigma(max_polorder);
351  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
352  // evaluate bases
353  // * entity
354  // * test
355  const auto rowsEn = testBaseEntity.size();
356  const auto testValuesEn = testBaseEntity.evaluate(localPointEn);
357  const auto testGradientsEn = testBaseEntity.jacobian(localPointEn);
358  // * ansatz
359  const auto colsEn = ansatzBaseEntity.size();
360  const auto ansatzValuesEn = ansatzBaseEntity.evaluate(localPointEn);
361  const auto ansatzGradientsEn = ansatzBaseEntity.jacobian(localPointEn);
362  // * neighbor
363  // * test
364  const auto rowsNe = testBaseNeighbor.size();
365  const auto testValuesNe = testBaseNeighbor.evaluate(localPointNe);
366  const auto testGradientsNe = testBaseNeighbor.jacobian(localPointNe);
367  // * ansatz
368  const auto colsNe = ansatzBaseNeighbor.size();
369  const auto ansatzValuesNe = ansatzBaseNeighbor.evaluate(localPointNe);
370  const auto ansatzGradientsNe = ansatzBaseNeighbor.jacobian(localPointNe);
371  // compute the evaluations
372  assert(entityEntityRet.rows() >= rowsEn);
373  assert(entityEntityRet.cols() >= colsEn);
374  assert(entityNeighborRet.rows() >= rowsEn);
375  assert(entityNeighborRet.cols() >= colsNe);
376  assert(neighborEntityRet.rows() >= rowsNe);
377  assert(neighborEntityRet.cols() >= colsEn);
378  assert(neighborNeighborRet.rows() >= rowsNe);
379  assert(neighborNeighborRet.cols() >= colsNe);
380  // loop over all entity test basis functions
381  for (size_t ii = 0; ii < rowsEn; ++ii) {
382  auto& entityEntityRetRow = entityEntityRet[ii];
383  auto& entityNeighborRetRow = entityNeighborRet[ii];
384  // loop over all entity ansatz basis functions
385  for (size_t jj = 0; jj < colsEn; ++jj) {
386  // consistency term
387  entityEntityRetRow[jj] +=
388  -0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesEn[ii];
389  // symmetry term
390  entityEntityRetRow[jj] +=
391  -0.5 * ansatzValuesEn[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
392  // penalty term
393  entityEntityRetRow[jj] += penalty * ansatzValuesEn[jj] * testValuesEn[ii];
394  } // loop over all entity ansatz basis functions
395  // loop over all neighbor ansatz basis functions
396  for (size_t jj = 0; jj < colsNe; ++jj) {
397  // consistency term
398  entityNeighborRetRow[jj] +=
399  -0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesEn[ii];
400  // symmetry term
401  entityNeighborRetRow[jj] +=
402  0.5 * ansatzValuesNe[jj] * functionValueEn * (testGradientsEn[ii][0] * unitOuterNormal);
403  // penalty term
404  entityNeighborRetRow[jj] += -1.0 * penalty * ansatzValuesNe[jj] * testValuesEn[ii];
405  } // loop over all neighbor ansatz basis functions
406  } // loop over all entity test basis functions
407  // loop over all neighbor test basis functions
408  for (size_t ii = 0; ii < rowsNe; ++ii) {
409  auto& neighborEntityRetRow = neighborEntityRet[ii];
410  auto& neighborNeighborRetRow = neighborNeighborRet[ii];
411  // loop over all entity ansatz basis functions
412  for (size_t jj = 0; jj < colsEn; ++jj) {
413  // consistency term
414  neighborEntityRetRow[jj] +=
415  0.5 * functionValueEn * (ansatzGradientsEn[jj][0] * unitOuterNormal) * testValuesNe[ii];
416  // symmetry term
417  neighborEntityRetRow[jj] +=
418  -0.5 * ansatzValuesEn[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
419  // penalty term
420  neighborEntityRetRow[jj] += -1.0 * penalty * ansatzValuesEn[jj] * testValuesNe[ii];
421  } // loop over all entity ansatz basis functions
422  // loop over all neighbor ansatz basis functions
423  for (size_t jj = 0; jj < colsNe; ++jj) {
424  // consistency term
425  neighborNeighborRetRow[jj] +=
426  0.5 * functionValueNe * (ansatzGradientsNe[jj][0] * unitOuterNormal) * testValuesNe[ii];
427  // symmetry term
428  neighborNeighborRetRow[jj] +=
429  0.5 * ansatzValuesNe[jj] * functionValueNe * (testGradientsNe[ii][0] * unitOuterNormal);
430  // penalty term
431  neighborNeighborRetRow[jj] += penalty * ansatzValuesNe[jj] * testValuesNe[ii];
432  } // loop over all neighbor ansatz basis functions
433  } // loop over all neighbor test basis functions
434  } // ... evaluate(...)

648  size_t order(
649  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLF, rCLF>& localDiffusion,
650  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, rLR, rCLR>& localDirichlet,
651  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, rT, rCT>& testBase)
652  const
653  {
654  const size_t testOrder = testBase.order();
655  const size_t testGradientOrder = boost::numeric_cast<size_t>(std::max(ssize_t(testOrder) - 1, ssize_t(0)));
656  const size_t diffusionOrder = localDiffusion.order();
657  const size_t dirichletOrder = localDirichlet.order();

665  template <class IntersectionType, class R>
666  void
667  evaluate(const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDiffusion,
668  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& localDirichlet,
669  const XT::Functions::LocalfunctionSetInterface<EntityType, DomainFieldType, dimDomain, R, 1, 1>& testBase,
670  const IntersectionType& intersection,
671  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& localPoint,
672  Dune::DynamicVector<R>& ret) const
673  {
674  // clear ret
675  ret *= 0.0;
676  // get local point (which is in intersection coordinates) in entity coordinates
677  const auto localPointEntity = intersection.geometryInInside().global(localPoint);
678  const auto unitOuterNormal = intersection.unitOuterNormal(localPoint);
679  // evaluate local functions
680  const auto diffusionValue = localDiffusion.evaluate(localPointEntity);
681  const auto dirichletValue = localDirichlet.evaluate(localPointEntity);
682  // compute penalty (see Epshteyn, Riviere, 2007)
683  const auto polorder = testBase.order();
684  const R sigma = internal::boundary_sigma(polorder);
685  const R penalty = sigma / std::pow(intersection.geometry().volume(), beta_);
686  // evaluate basis
687  const auto size = testBase.size();
688  const auto testValues = testBase.evaluate(localPointEntity);
689  const auto testGradients = testBase.jacobian(localPointEntity);
690  // compute
691  assert(ret.size() >= size);
692  // loop over all test basis functions
693  for (size_t ii = 0; ii < size; ++ii) {
694  // symmetry term
695  ret[ii] += -1.0 * dirichletValue * diffusionValue * (testGradients[ii][0] * unitOuterNormal);
696  // penalty term
697  ret[ii] += penalty * dirichletValue * testValues[ii];
698  } // loop over all test basis functions
699  } // ... evaluate(...)
