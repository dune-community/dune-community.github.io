55 class MustaFluxImplementation

72  explicit MustaFluxImplementation(const AnalyticalFluxType& analytical_flux,
73  XT::Common::Parameter param,
74  const size_t num_stages,
75  const bool boundary)
76  : analytical_flux_(analytical_flux)
77  , param_inside_(param)
78  , param_outside_(param)
79  , dt_(param.get("dt")[0])
80  , num_stages_(num_stages)

82  param_inside_.set("boundary", {0.}, true);

86  template <class IntersectionType>
87  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
88  const LocalfunctionTupleType& local_functions_tuple_neighbor,
89  const IntersectionType& intersection,
90  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
91  const DomainType& x_in_inside_coords,
92  const DomainType& x_in_outside_coords,
93  const RangeType& u_i,
94  const RangeType& u_j) const
95  {
96  // find direction of unit outer normal
97  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
98  size_t direction = intersection.indexInInside() / 2;
99  bool positive_dir = intersection.indexInInside() % 2;
100 
101  // get data on left and right entity
102  const RangeType& u_R = positive_dir ? u_j : u_i;
103  const RangeType& u_L = positive_dir ? u_i : u_j;
104  const DomainType& x_in_R_coords = positive_dir ? x_in_outside_coords : x_in_inside_coords;
105  const DomainType& x_in_L_coords = positive_dir ? x_in_inside_coords : x_in_outside_coords;
106  const auto& local_flux_R =
107  std::get<0>(positive_dir ? local_functions_tuple_neighbor : local_functions_tuple_entity);
108  const auto& local_flux_L =
109  std::get<0>(positive_dir ? local_functions_tuple_entity : local_functions_tuple_neighbor);
110  const auto& local_flux_inside = positive_dir ? local_flux_L : local_flux_R;
111  const auto& param_L = positive_dir ? param_outside_ : param_inside_;
112  const auto& param_R = positive_dir ? param_inside_ : param_outside_;
113 
114  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
115 
116  RangeType u_L_l = u_L;
117  RangeType u_R_l = u_R;
118  RangeType ret;
119  for (size_t ll = 0; ll <= num_stages_; ++ll)
120  calculate_stage(ret,
121  u_L_l,
122  u_R_l,
123  *local_flux_L,
124  *local_flux_R,
125  *local_flux_inside,
126  x_in_L_coords,
127  x_in_R_coords,
128  x_in_inside_coords,
129  param_L,
130  param_R,
131  dx,
132  direction,
133  ll != num_stages_);
134  ret *= n_ij[direction];
135  return ret;
136  }
137 
138  const AnalyticalFluxType& analytical_flux() const
139  {
140  return analytical_flux_;
141  }
142 
143 private:
144  void calculate_stage(RangeType& ret,
145  RangeType& u_L,
146  RangeType& u_R,
147  const AnalyticalFluxLocalfunctionType& local_flux_L,
148  const AnalyticalFluxLocalfunctionType& local_flux_R,
149  const AnalyticalFluxLocalfunctionType& local_flux_inside,
150  const DomainType& x_in_L_coords,
151  const DomainType& x_in_R_coords,
152  const DomainType& x_in_inside_coords,
153  const XT::Common::Parameter& param_L,
154  const XT::Common::Parameter& param_R,
155  const RangeFieldType& dx,
156  const size_t direction,
157  const bool update) const
158  {
159  // get data of current_stage
160  const RangeType f_L = local_flux_L.evaluate_col(direction, x_in_L_coords, u_L, param_L);
161  const RangeType f_R = local_flux_R.evaluate_col(direction, x_in_R_coords, u_R, param_R);
162 
163  // calculate u^{1/2}
164  RangeType u_half = u_L;
165  u_half += u_R;
166  RangeType f_diff = f_R;
167  f_diff -= f_L;
168  f_diff *= dx / dt_;
169  u_half -= f_diff;
170  u_half *= 0.5;
171 
172  // calculate f_M
173  const RangeType f_M = local_flux_inside.evaluate_col(direction, x_in_inside_coords, u_half, param_inside_);
174 
175  // calculate force flux
176  auto& f_force = f_diff;
177  f_force = f_M;
178  f_force *= 2.;
179  f_force += f_L;
180  f_force += f_R;
181  auto& u_diff = u_half;
182  u_diff = u_R;
183  u_diff -= u_L;
184  u_diff *= dx / dt_;
185  f_force -= u_diff;
186  f_force *= 0.25;
187 
188  // update data
189  if (update) {
190  auto update_L = f_force;
191  update_L -= f_L;
192  update_L *= dt_ / dx;
193  u_L -= update_L;
194  auto& update_R = update_L;
195  update_R = f_R;
196  update_R -= f_force;
197  update_R *= dt_ / dx;
198  u_R -= update_R;
199  } else {
200  ret = f_force;
201  }
202  }
203 
204  const AnalyticalFluxType& analytical_flux_;
205  XT::Common::Parameter param_inside_;
206  XT::Common::Parameter param_outside_;
207  const double dt_;
208  const size_t num_stages_;
209 }; // class MustaFluxImplementation<...>
