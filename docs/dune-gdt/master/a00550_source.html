394 class DisjointPartitioningGatherScatter
395 {
396 
397 public:
398  template <typename MessageBuffer, typename Entity, typename LocalView>
399  bool gather(MessageBuffer& buff, const Entity& e, LocalView& local_view) const
400  {
401  // We only gather from interior and border entities, so we can throw in our ownership
402  // claim without any further checks.
403  buff.write(_rank);
404 
405  return true;
406  }
407 
408  template <typename MessageBuffer, typename Entity, typename LocalView>
409  bool scatter(MessageBuffer& buff, std::size_t n, const Entity& e, LocalView& local_view) const
410  {
411  // Value used for DOFs with currently unknown rank.
412  const RankIndex unknown_rank = std::numeric_limits<RankIndex>::max();
413 
414  // We can only own this DOF if it is either on the interior or border partition.
415  const bool is_interior_or_border =
416  (e.partitionType() == Dune::InteriorEntity || e.partitionType() == Dune::BorderEntity);
417 
418  // Receive data.
419  RankIndex received_rank;
420  buff.read(received_rank);
421 
422  for (std::size_t i = 0; i < local_view.size(); ++i) {
423  // Get the currently stored owner rank for this DOF.
424  RankIndex current_rank = local_view[i];
425 
426  // We only gather from interior and border entities, so we need to make sure
427  // we relinquish any ownership claims on overlap and ghost entities on the
428  // receiving side. We also need to make sure not to overwrite any data already
429  // received, so we only blank the rank value if the currently stored value is
430  // equal to our own rank.
431  if (!is_interior_or_border && current_rank == _rank)
432  current_rank = unknown_rank;
433 
434  // Assign DOFs to minimum rank value.
435  local_view[i] = std::min(current_rank, received_rank);
436  }
437  return true;
438  }
