94  diffusion *= lambda_.local_function(entity)->evaluate(local_point);
95  const auto grad_u = local_u.jacobian(local_point).at(0)[0];
96  const auto grad_interpolated_u = interpolated_u_.local_function(entity)->jacobian(local_point)[0];

98  const auto grad_interpolated_v = interpolated_v_.local_function(entity)->jacobian(local_point)[0];
