55 class L2GlobalProjectionLocalizableOperator : public LocalizableOperatorBase<GridLayerImp, SourceImp, RangeImp>
56 {

76 public:
77  L2GlobalProjectionLocalizableOperator(const size_t over_integrate,
78  GridLayerType grd_vw,
79  const SourceType& src,
80  RangeType& rng,
81  const XT::Common::Parameter& /*param*/ = {})
82  : BaseType(grd_vw, src, rng)
83  , lhs_operator_(over_integrate, range_.space(), BaseType::grid_layer())
84  , rhs_functional_(over_integrate, source_, range_.space(), BaseType::grid_layer())
85  , solved_(false)
86  {
87  this->append(lhs_operator_);
88  this->append(rhs_functional_);
89  issue_warning(this->range().space());

92  L2GlobalProjectionLocalizableOperator(GridLayerType grd_vw,
93  const SourceType& src,
94  RangeType& rng,
95  const XT::Common::Parameter& param = {})
96  : L2GlobalProjectionLocalizableOperator(0, grd_vw, src, rng, param)
97  {
98  }
99 
100  void apply()
101  {
102  if (solved_)
103  return;
104  BaseType::apply();
105  try {
106  XT::LA::Solver<MatrixType>(lhs_operator_.matrix()).apply(rhs_functional_.vector(), range_.vector());
107  } catch (XT::LA::Exceptions::linear_solver_failed& ee) {
108  DUNE_THROW(projection_error,
109  "L2 projection failed because a global matrix could not be inverted!\n\n"
110  << "This was the original error: "
111  << ee.what());
112  }
113  } // ... apply(...)
