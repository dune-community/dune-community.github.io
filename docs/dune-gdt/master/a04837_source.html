59 template <class PdelabSpaceImp, size_t rangeDim>
60 class ProductDgMapper<PdelabSpaceImp, rangeDim, 1>
61  : public ProductMapperInterface<internal::ProductDgMapperTraits<PdelabSpaceImp, rangeDim, 1>>

99  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
100  {
101  if (ret.size() < numDofs(entity))
102  ret.resize(numDofs(entity));
103  const auto factor_num_dofs = factor_mapper_.numDofs(entity);
104  const auto factor_mapper_size = factor_mapper_.size();
105  const auto factor_global_indices = factor_mapper_.globalIndices(entity);
106  for (size_t ii = 0; ii < dimRange; ++ii) {
107  const auto factor_num_dofs_times_ii = factor_num_dofs * ii;
108  const auto factor_mapper_size_times_ii = factor_mapper_size * ii;
109  for (size_t jj = 0; jj < factor_num_dofs; ++jj) {
110  ret[factor_num_dofs_times_ii + jj] = factor_global_indices[jj] + factor_mapper_size_times_ii;
111  }
112  }
113  } // ... globalIndices(...)
114 
115  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
116  {
117  assert(localIndex < numDofs(entity));
118  size_t factor_index = 0;
119  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
120  while (localIndex >= factor_mapper_num_dofs) {
121  localIndex -= factor_mapper_num_dofs;
122  ++factor_index;
123  }
124  return factor_mapper_.mapToGlobal(entity, localIndex) + factor_index * factor_mapper_.size();
125  }
126 
127  using InterfaceType::globalIndices;

129  size_t numDofs(const size_t /*factor_index*/, const EntityType& entity) const
130  {

132  }

137  }

139  void globalIndices(const size_t factor_index, const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
140  {
141  assert(factor_index < dimRange);
142  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
143  if (ret.size() < factor_mapper_num_dofs)
144  ret.resize(factor_mapper_num_dofs);
145  const auto factor_mapper_size_times_factor_index = factor_mapper_.size() * factor_index;
146  const auto factor_mapper_global_indices = factor_mapper_.globalIndices(entity);
147  for (size_t jj = 0; jj < factor_mapper_num_dofs; ++jj)
148  ret[jj] = factor_mapper_global_indices[jj] + factor_mapper_size_times_factor_index;
149  } // ... globalIndices(...)
150 
151  size_t mapToGlobal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
152  {
153  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
154  assert(factor_index < dimRange);
155  return factor_mapper_.mapToGlobal(entity, local_index_in_factor) + factor_index * factor_mapper_.size();
156  }
157 
158  size_t mapToLocal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
159  {
160  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
161  assert(factor_index < dimRange);
162  return factor_mapper_.numDofs(entity) * factor_index + local_index_in_factor;
163  }
