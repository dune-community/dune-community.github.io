130 protected:
131  // returns u_i, u_j and x_in_inside_coords
132  template <class IntersectionType, size_t boundary_value_pos>
133  std::tuple<RangeType, RangeType, DomainType> get_values(
134  const LocalfunctionTupleType& local_functions_tuple,
135  const XT::Functions::LocalfunctionInterface<EntityType, DomainFieldType, dimDomain, RangeFieldType, dimRange, 1>&
136  local_source_entity,
137  const IntersectionType& intersection,
138  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords) const
139  {
140  std::tuple<RangeType, RangeType, DomainType> ret;
141  auto& u_i = std::get<0>(ret);
142  auto& u_j = std::get<1>(ret);
143  auto& x_in_inside_coords = std::get<2>(ret);
144  x_in_inside_coords = intersection.geometryInInside().global(x_in_intersection_coords);
145  u_i = local_source_entity.evaluate(x_in_inside_coords);
146  const auto& boundary_type = boundary_info_.type(intersection);
147  if (boundary_type == dirichlet_boundary_)
148  u_j = std::get<boundary_value_pos>(local_functions_tuple)->evaluate(x_in_inside_coords);
149  else if (boundary_type == reflecting_boundary_)
150  u_j = std::get<1>(boundary_values_).evaluate(intersection, u_i);
151  else if (boundary_type == absorbing_boundary_)
152  u_j = u_i;
153  else
154  DUNE_THROW(NotImplemented, "Can't handle boundarys other than dirichlet, reflecting and absorbing!");
155  return ret;
156  }
