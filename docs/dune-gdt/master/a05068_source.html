144  template <class IntersectionType>
145  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
146  const LocalfunctionTupleType& local_functions_tuple_neighbor,
147  const IntersectionType& intersection,
148  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
149  const DomainType& x_in_inside_coords,
150  const DomainType& x_in_outside_coords,
151  const RangeType& u_i,
152  const RangeType& u_j) const
153  {
154  // find direction of unit outer normal
155  size_t direction = intersection.indexInInside() / 2;
156 
157  const auto& local_flux_inside = std::get<0>(local_functions_tuple_entity);
158  const auto& local_flux_outside = std::get<0>(local_functions_tuple_neighbor);
159  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
160 
161  if (use_local_) {
162  if (!is_linear_ || !(max_derivative_calculated()[direction])) {
163  if (!jacobian_inside()) {
164  jacobian_inside() = XT::Common::make_unique<JacobianRangeType>();
165  jacobian_outside() = XT::Common::make_unique<JacobianRangeType>();
166  }
167  DomainFieldType max_derivative(0);
168  helper<dimDomain>::get_jacobian(
169  direction, local_flux_inside, x_in_inside_coords, u_i, *jacobian_inside(), param_inside_);
170  helper<dimDomain>::get_jacobian(
171  direction, local_flux_outside, x_in_outside_coords, u_j, *jacobian_outside(), param_outside_);
172  XT::Common::Configuration eigensolver_options({"type",
173  "check_for_inf_nan",
174  "check_evs_are_real",
175  "check_evs_are_positive",
176  "check_eigenvectors_are_real"},
177  {EigenSolverType::types()[0].c_str(), "1", "1", "0", "1"});
178  const auto eigen_solver_inside = EigenSolverType((*jacobian_inside())[direction]);
179  const auto eigen_solver_outside = EigenSolverType((*jacobian_outside())[direction]);
180  const auto& eigenvalues_inside = eigen_solver_inside.eigenvalues(eigensolver_options);
181  const auto& eigenvalues_outside = eigen_solver_outside.eigenvalues(eigensolver_options);
182  for (size_t jj = 0; jj < dimRange; ++jj)
183  max_derivative = std::max(
184  {std::abs(eigenvalues_inside[jj].real()), std::abs(eigenvalues_outside[jj].real()), max_derivative});
185  lambda_ij()[direction] = 1. / max_derivative;
186  if (is_linear_) {
187  jacobian_inside() = nullptr;
188  jacobian_outside() = nullptr;
189  }
190  max_derivative_calculated()[direction] = true;
191  }
192  } else if (lambda_provided_) {
193  lambda_ij()[direction] = lambda_[direction];
194  } else {
195  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
196  lambda_ij()[direction] = dt_ / dx;
197  } // if (use_local)
198 
199  // calculate flux evaluation as
200  // ret[kk] = (f_u_i[kk] + f_u_j[kk])*n_ij*0.5 - (u_j - u_i)[kk]*1.0/(num_neighbors*lambda_ij)
201  auto second_part = u_j;
202  second_part -= u_i;
203  second_part /= lambda_ij()[direction] * 2 * alpha_;
204  auto ret = local_flux_inside->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
205  ret += local_flux_outside->evaluate_col(direction, x_in_outside_coords, u_j, param_outside_);
206  ret *= n_ij[direction] * 0.5;
207  ret -= second_part;
208  return ret;
209  } // ... evaluate(...)

212  {
213  return analytical_flux_;
214  }

216  static void reset()
217  {
218  max_derivative_calculated() = false;
219  }
