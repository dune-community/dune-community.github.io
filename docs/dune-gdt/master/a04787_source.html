178  void solve(DiscreteSolutionType& solution) const
179  {
180  try {
181  reset_static_variables();
182 
183  const auto& problem = test_case_.problem();
184  // get analytical flux, initial and boundary values
185  const AnalyticalFluxType& analytical_flux = problem.flux();
186  const InitialValueType& initial_values = problem.initial_values();
187  const BoundaryValueType& boundary_values = problem.boundary_values();
188  const RhsType& rhs = problem.rhs();
189 
190  // create a discrete function for the solution
191  DiscreteFunctionType u(*fv_space_, "solution");
192 
193  // project initial values
194  project_l2(initial_values, u);
195 
196  RangeFieldType t_end = problem.t_end();
197  const RangeFieldType CFL = problem.CFL();
198 
199  // calculate dx and choose initial dt
200  Dune::XT::Grid::Dimensions<typename SpaceType::GridLayerType> dimensions(fv_space_->grid_layer());
201  RangeFieldType dx = dimensions.entity_width.max();
202  if (dimDomain == 2)
203  dx /= std::sqrt(2);
204  if (dimDomain == 3)
205  dx /= std::sqrt(3);
206  RangeFieldType dt = CFL * dx;
207 
208  // create operators
209  const ConstantFunctionType dx_function(dx);
210  std::unique_ptr<AdvectionOperatorType> advection_operator =
211  AdvectionOperatorCreatorType::create(analytical_flux, boundary_values, dx_function, linear);
212 
213  RhsOperatorType rhs_operator(rhs);
214 
215  // create timestepper
216  OperatorTimeStepperType timestepper_op(*advection_operator, u, -1.0);
217 
218  // do the time steps
219  const size_t num_save_steps = 100;
220  solution.clear();
221  if (problem.has_non_zero_rhs()) {
222  // use fractional step method
223  RhsOperatorTimeStepperType timestepper_rhs(rhs_operator, u);
224  // TimeStepperType timestepper(timestepper_rhs, timestepper_op);
225  TimeStepperType timestepper(timestepper_op, timestepper_rhs);
226  timestepper.solve(t_end, dt, num_save_steps, solution);
227  } else {
228  timestepper_op.solve(t_end, dt, num_save_steps, solution);
229  }
230 
231  } catch (Dune::Exception& e) {
232  std::cerr << "Dune reported: " << e.what() << std::endl;
233  std::abort();
234  }
235  } // void solve(...)
