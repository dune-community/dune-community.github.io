56  // we need to distinguish three cases here (since DiffusionTensorImp may be void):

87  static_assert(DiffusionTensorImp::dimRangeCols == DiffusionTensorImp::dimDomain,

258  template <class R, size_t r>
259  void evaluate(const XT::Functions::LocalfunctionInterface<E, D, d, R, 1, 1>& local_diffusion_factor,
260  const XT::Functions::LocalfunctionInterface<E, D, d, R, d, d>& local_diffusion_tensor,
261  const XT::Functions::LocalfunctionSetInterface<E, D, d, R, r, 1>& test_base,
262  const XT::Functions::LocalfunctionSetInterface<E, D, d, R, r, 1>& ansatz_base,
263  const Dune::FieldVector<D, d>& localPoint,
264  Dune::DynamicMatrix<R>& ret) const
265  {
266  typedef XT::Common::FieldMatrix<R, d, d> TensorType;
267  ret *= 0.0;
268  // evaluate local functions
269  const auto diffusion_factor_value = local_diffusion_factor.evaluate(localPoint, param_);
270  const TensorType diffusion_tensor_value = local_diffusion_tensor.evaluate(localPoint, param_);
271  const auto diffusion_value = diffusion_tensor_value * diffusion_factor_value;
272  // evaluate bases
273  const auto testGradients = test_base.jacobian(localPoint, param_);
274  const auto ansatzGradients = ansatz_base.jacobian(localPoint, param_);
275  // compute elliptic evaluation
276  const size_t rows = test_base.size();
277  const size_t cols = ansatz_base.size();
278  assert(ret.rows() >= rows);
279  assert(ret.cols() >= cols);
280  for (size_t ii = 0; ii < rows; ++ii) {
281  auto& retRow = ret[ii];
282  for (size_t jj = 0; jj < cols; ++jj) {
283  for (size_t rr = 0; rr < r; ++rr) {
284  retRow[jj] += (diffusion_value * ansatzGradients[jj][rr]) * testGradients[ii][rr];
285  }
286  }
287  }
288  } // ... evaluate(...)
