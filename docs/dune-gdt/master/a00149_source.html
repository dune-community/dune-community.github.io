101  template <class SourceType, class RangeType>
102  void apply_p1_dg(const SourceType& source, RangeType& range) const
103  {
104  range.vector() *= 0.;
105  // data structures we need
106  // * a map from a global vertex index to global DoF indices
107  // given a vertex, one obtains a set of all global DoF ids, which are associated with this vertex
108  std::map<size_t, std::set<size_t>> global_vertex_id_to_global_DoF_id_map;
109  // * a map from a global DoF index to the global index of its associated vertex
110  std::vector<size_t> global_DoF_id_to_global_vertex_id_map(range.space().mapper().size());
111  // * a set to hold the global id of all boundary vertices
112  std::set<size_t> boundary_vertices;
113 
114  const auto entity_it_end = grid_layer_.template end<0>();
115  // walk the grid to create the maps explained above and to find the boundary vertices
116  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
117  const auto& entity = *entity_it;
118  const size_t num_vertices = entity.subEntities(dimDomain);
119  const auto basis = range.space().base_function_set(entity);
120  if (basis.size() != num_vertices)
121  DUNE_THROW(XT::Common::Exceptions::internal_error, "basis.size() = " << basis.size());
122 
123  // loop over all vertices of the entitity, to find their associated global DoF indices
124  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
125  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id);
126  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
127  const auto vertex_center = vertex.geometry().center();
128  // find the local basis function which corresponds to this vertex
129  const auto basis_values = basis.evaluate(entity.geometry().local(vertex_center));
130  if (basis_values.size() != num_vertices)
131  DUNE_THROW(XT::Common::Exceptions::internal_error, "basis_values.size() = " << basis_values.size());
132  size_t ones = 0;
133  size_t zeros = 0;
134  size_t failures = 0;
135  size_t local_DoF_index = 0;
136  for (size_t ii = 0; ii < basis.size(); ++ii) {
137  if (std::abs(basis_values[ii][0] - 1.0) < 1e-14) {
138  local_DoF_index = ii;
139  ++ones;
140  } else if (std::abs(basis_values[ii][0] - 0.0) < 1e-14)
141  ++zeros;
142  else
143  ++failures;
144  }
145  if (ones != 1 || zeros != (basis.size() - 1) || failures > 0) {
146  std::stringstream ss;
147  ss << "ones = " << ones << ", zeros = " << zeros << ", failures = " << failures
148  << ", num_vertices = " << num_vertices << ", entity " << grid_layer_.indexSet().index(entity)
149  << ", vertex " << local_vertex_id << ": [ " << vertex_center << "], ";
150  XT::Common::print(basis_values, "basis_values", ss);
151  DUNE_THROW(XT::Common::Exceptions::internal_error, ss.str());
152  }
153  // now we know that the local DoF index of this vertex is ii
154  const size_t global_DoF_index = range.space().mapper().mapToGlobal(entity, local_DoF_index);
155  global_DoF_id_to_global_vertex_id_map[global_DoF_index] = global_vertex_id;
156  global_vertex_id_to_global_DoF_id_map[global_vertex_id].insert(global_DoF_index);
157  } // loop over all vertices
158 
159  static const constexpr auto dirichlet = XT::Grid::DirichletBoundary();
160  // in order to determine the boundary vertices, we need to
161  // loop over all intersections
162  const auto intersectionEndIt = grid_layer_.iend(entity);
163  for (auto intersectionIt = grid_layer_.ibegin(entity); intersectionIt != intersectionEndIt; ++intersectionIt) {
164  const auto& intersection = *intersectionIt;
165  if (boundary_info_.type(intersection) == dirichlet) {
166  const auto& intersection_geometry = intersection.geometry();
167  for (auto local_intersection_corner_id : XT::Common::value_range(intersection_geometry.corners())) {
168  const auto global_intersection_corner = intersection_geometry.corner(local_intersection_corner_id);
169  // now, we need to find the entity's vertex this intersection's corner point equals to, so we
170  // loop over all vertices of the entity
171  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
172  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id);
173  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
174  const auto vertex_center = vertex.geometry().center();
175  if (XT::Common::FloatCmp::eq(global_intersection_corner, vertex_center))
176  boundary_vertices.insert(global_vertex_id);
177  } // loop over all vertices of the entity
178  } // loop over all intersection corners
179  } // if (intersection.boundary() && !intersection.neighbor())
180  } // loop over all intersections
181  } // walk the grid for the first time
182 
183  // walk the grid for the second time
184  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
185  const auto& entity = *entity_it;
186  const auto num_vertices = entity.subEntities(dimDomain);
187  // get the local functions
188  const auto local_source = source.local_function(entity);
189 
190  // * loop over all local DoFs
191  for (auto local_vertex_id : XT::Common::value_range(num_vertices)) {
192  const size_t global_DoF_index =
193  range.space().mapper().mapToGlobal(entity, XT::Common::numeric_cast<size_t>(local_vertex_id));
194  const size_t global_vertex_id = global_DoF_id_to_global_vertex_id_map[global_DoF_index];
195  // if we are on the domain boundary
196  if (boundary_vertices.count(global_vertex_id)) {
197  // set the dof to zero (we have dirichlet zero)
198  range.vector().set_entry(global_DoF_index, FieldType(0));
199  } else {
200  // do the oswald projection
201  const size_t num_DoFS_per_vertex = global_vertex_id_to_global_DoF_id_map[global_vertex_id].size();
202  // * get the source value
203  const auto vertex = entity.template subEntity<dimDomain>(local_vertex_id).geometry().center();
204  const FieldType source_value = local_source->evaluate(entity.geometry().local(vertex));
205  // * and add it to all target DoFs
206  for (size_t target_global_DoF_id : global_vertex_id_to_global_DoF_id_map[global_vertex_id])
207  range.vector().add_to_entry(target_global_DoF_id, source_value / num_DoFS_per_vertex);
208  } // if (boundary_vertices.find(global_vertex_id))
209  } // loop over all local DoFs
210  } // walk the grid for the second time
211  } // ... apply_p1_dg(...)
212 
213  const GridLayerType& grid_layer_;
214  const XT::Grid::BoundaryInfo<XT::Grid::extract_intersection_t<GridLayerType>>& boundary_info_;
