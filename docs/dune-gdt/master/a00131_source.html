308 class BlockedJacobianWrapper
309  : public JacobianWrapperBase<AnalyticalFluxType,
310  FieldVector<FieldMatrix<typename AnalyticalFluxType::RangeFieldType,
311  block_size,
312  block_size>,

368  virtual void compute(const size_t dd) override
369  {
370  for (size_t jj = 0; jj < num_blocks; ++jj) {
371  if (block_size == 2) {
372  const auto& jac = jacobian(dd)[jj];
373  const auto trace = jac[0][0] + jac[1][1];
374  const auto det = jac[0][0] * jac[1][1] - jac[0][1] * jac[1][0];
375  const auto sqrt_val = std::sqrt(0.25 * trace * trace - det);
376  const auto eigval1 = 0.5 * trace + sqrt_val;
377  const auto eigval2 = 0.5 * trace - sqrt_val;
378  if (std::abs(jac[1][0]) > std::abs(jac[0][1])) {
379  if (XT::Common::FloatCmp::ne(jac[1][0], 0.)) {
380  eigenvectors_[dd][jj][0][0] = eigval1 - jac[1][1];
381  eigenvectors_[dd][jj][0][1] = eigval2 - jac[1][1];
382  eigenvectors_[dd][jj][1][0] = eigenvectors_[dd][jj][1][1] = jac[1][0];
383  } else {
384  eigenvectors_[dd][jj][0][0] = eigenvectors_[dd][jj][1][1] = 1.;
385  eigenvectors_[dd][jj][0][1] = eigenvectors_[dd][jj][1][0] = 0.;
386  }
387  } else {
388  if (XT::Common::FloatCmp::ne(jac[0][1], 0.)) {
389  eigenvectors_[dd][jj][1][0] = eigval1 - jac[0][0];
390  eigenvectors_[dd][jj][1][1] = eigval2 - jac[0][0];
391  eigenvectors_[dd][jj][0][0] = eigenvectors_[dd][jj][0][1] = jac[0][1];
392  } else {
393  eigenvectors_[dd][jj][0][0] = eigenvectors_[dd][jj][1][1] = 1.;
394  eigenvectors_[dd][jj][0][1] = eigenvectors_[dd][jj][1][0] = 0.;
395  }
396  }
397  } else {
398 #if HAVE_MKL || HAVE_LAPACKE
399  int info = XT::Common::Lapacke::dgeev_work(XT::Common::Lapacke::row_major(),
400  /*do_not_compute_left_eigenvectors: */ 'N',
401  /*compute_right_eigenvectors: */ 'V',
402  static_cast<int>(block_size),
403  &(jacobian(dd)[jj][0][0]),
404  static_cast<int>(block_size),
405  &(eigenvalues_[jj][0][0]),
406  &(dummy_complex_eigenvalues_[0]),
407  nullptr,
408  static_cast<int>(block_size),
409  &(eigenvectors_[dd][jj][0][0]),
410  static_cast<int>(block_size),
411  work_.data(),
412  static_cast<int>(work_.size()));
413  if (info != 0)
414  DUNE_THROW(Dune::XT::LA::Exceptions::eigen_solver_failed, "The lapack backend reported '" << info << "'!");
415 #else // HAVE_MKL || HAVE_LAPACKE
416  static XT::Common::Configuration eigensolver_options =
417  BaseType::template create_eigensolver_opts<LocalMatrixType>();
418  const auto eigensolver = EigenSolverType(jacobian(dd)[jj], &eigensolver_options);
419  eigenvectors_[dd][jj] = eigensolver.real_eigenvectors();
420 #endif // HAVE_MKL || HAVE_LAPACKE
421  } // else (block_size == 2)
422  QR_[dd][jj] = eigenvectors_[dd][jj];
423  XT::LA::qr(QR_[dd][jj], tau_[dd][jj], permutations_[dd][jj]);
424  } // jj
425  computed_[dd] = true;
426  }

784  last_dir = curr_dir;
785  } // dir

787  std::for_each(reconstructed_values.data(),
788  reconstructed_values.data() + reconstructed_values.num_elements(),
789  [&](RangeType& value) {

792  });

796  for (const auto& multi_index : multi_indices) {
797  IntersectionLocalCoordType quadrature_point;

802  intersections[2 * dd + multi_index[dd]].geometryInInside().global(quadrature_point),

804  } // multi_indices
805  }
806  } // dd

809  static void linear_reconstruction_1d(const FieldVector<RangeType, axis_size>& stencil,
810  FieldVector<RangeType, 2>& reconstructed_values)
811  {
812  const auto& u_left = stencil[0];
813  const auto& u_entity = stencil[1];
814  const auto& u_right = stencil[2];
815  const auto slope_left = u_entity - u_left;
816  const auto slope_center = 0.5 * (u_right - u_left);
817  const auto slope_right = u_right - u_entity;
818  auto slope = internal::ChooseLimiter<slope_limiter>::limit(slope_left, slope_right, slope_center);
819  slope *= 0.5;
820  reconstructed_values[0] = u_entity - slope;
821  reconstructed_values[1] = u_entity + slope;

824  static void linear_reconstruction(const size_t dir,
825  const Quadrature1dType& quadrature,
826  const MultiArrayType& stencil,
827  MultiArrayType& reconstructed_values)
828  {
829  // resize the reconstructed_values array
830  CoordsType new_shape;
831  std::copy_n(stencil.shape(), dimDomain, new_shape.begin());
832  new_shape[dir] = quadrature.size();
833  reconstructed_values.resize(new_shape);
834 
835  // get array_views corresponding to the left, center and right values
836  using IndexRangeType = typename MultiArrayType::index_range;
837  using RangeArrayType = XT::Common::FieldVector<IndexRangeType, dimDomain>;
838  using IndicesBuilderType = IndicesBuilder<RangeArrayType, dimDomain, dimDomain - 1>;
839 
840  assert(stencil.shape()[dir] == 3);
841  RangeArrayType ranges;
842  for (size_t ii = 0; ii < dimDomain; ++ii)
843  ranges[ii] = IndexRangeType(0, stencil.shape()[ii]);
844  ranges[dir] = IndexRangeType(0);
845  const auto u_left = stencil[IndicesBuilderType::build(ranges)];
846  ranges[dir] = IndexRangeType(1);
847  const auto u_entity = stencil[IndicesBuilderType::build(ranges)];
848  ranges[dir] = IndexRangeType(2);
849  const auto u_right = stencil[IndicesBuilderType::build(ranges)];
850 
851  // calculate slopes
852  thread_local SliceType slope;
853  std::array<size_t, dimDomain - 1> slope_sizes;
854  std::copy_n(u_entity.shape(), dimDomain - 1, slope_sizes.begin());
855  slope.resize(slope_sizes);
856  MultiIndexProvider<decltype(u_left)> multi_indices(u_left);
857  for (const auto& multi_index : multi_indices) {
858  const auto slope_left = u_entity(multi_index) - u_left(multi_index);
859  const auto slope_center = 0.5 * (u_right(multi_index) - u_left(multi_index));
860  const auto slope_right = u_right(multi_index) - u_entity(multi_index);
861  slope(multi_index) = internal::ChooseLimiter<slope_limiter>::limit(slope_left, slope_right, slope_center);
862  }
863 
864  // calculate reconstructed values
865  for (size_t ii = 0; ii < dimDomain; ++ii)
866  ranges[ii] = IndexRangeType(0, new_shape[ii]);
867  for (size_t ii = 0; ii < quadrature.size(); ++ii) {
868  ranges[dir] = IndexRangeType(ii);
869  auto reconstructed_ii = reconstructed_values[IndicesBuilderType::build(ranges)];
870  for (const auto& multi_index : multi_indices) {
871  reconstructed_ii(multi_index) = slope(multi_index);
872  reconstructed_ii(multi_index) *= (quadrature[ii].position() - 0.5);
873  reconstructed_ii(multi_index) += u_entity(multi_index);
874  }
875  } // ii (quadrature.size())
876  } // static void linear_reconstruction(...)

898  bool fill_impl(StencilType& stencil, const EntityType& entity, const int dir, const CoordsType& coords)
899  {
900  bool ret = true;
901  const auto entity_index = grid_layer_.indexSet().index(entity);
902  stencil(coords) = source_values_[entity_index];
903  std::vector<int> boundary_dirs;
904  for (const auto& intersection : Dune::intersections(grid_layer_, entity)) {
905  const auto new_dir = intersection.indexInInside();
906  if (direction_allowed(dir, new_dir) && !end_of_stencil(stencil, new_dir, coords)) {
907  auto new_coords = coords;
908  if (intersection.boundary() && !intersection.neighbor()) { // boundary intersections
909  boundary_dirs.push_back(new_dir);
910  auto boundary_value = boundary_values_.local_function(entity)->evaluate(
911  intersection, entity.geometry().local(intersection.geometry().center()), source_values_[entity_index]);
912  while (!end_of_stencil(stencil, new_dir, new_coords)) {
913  next_coords_in_dir(new_dir, new_coords);
914  stencil(new_coords) = boundary_value;
915  }
916  } else if (intersection.neighbor()) { // inner and periodic intersections
917  const auto& outside = intersection.outside();
918  next_coords_in_dir(new_dir, new_coords);
919  ret = ret && fill_impl(stencil, outside, new_dir, new_coords);
920  } else if (!intersection.neighbor() && !intersection.boundary()) { // processor boundary
921  return false;
922  }
923  } // if (!end_of_stencil(...))
924  } // intersections
925 
926  assert(boundary_dirs.size() <= dimDomain);
927  if (boundary_dirs.size() > 1) {
928  auto new_coords = coords;
929  next_coords_in_dir(boundary_dirs[0], new_coords);
930  const auto& boundary_value = stencil(new_coords);
931 
932  std::for_each(stencil.data(),
933  stencil.data() + stencil.num_elements(),
934  [&boundary_value](boost::optional<RangeType>& value) {
935  if (!value)
936  value = boundary_value;
937  });
938  } // if (boundary_dirs.size() > 1)
939  return ret;
940  }

942  // get next coords in direction dir (increase or decrease coords in that direction)
943  static void next_coords_in_dir(const int dir, CoordsType& coords)
944  {
945  dir % 2 ? coords[dir / 2]++ : coords[dir / 2]--;
946  }

951  // iterators going in y direction will only spawn iterators in z-direction and z iterators only walk
952  // without emitting new iterators).
953  static bool direction_allowed(const int dir, const int new_dir)
954  {
955  return new_dir == dir || new_dir / 2 > dir / 2;
956  }
