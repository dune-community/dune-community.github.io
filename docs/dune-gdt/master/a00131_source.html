61  void compute(const size_t dd)
62  {
63  static auto eigensolver_options = create_eigensolver_options();

65  eigenvectors_[dd] = eigensolver.real_eigenvectors();
66  eigenvalues_[dd] = eigensolver.real_eigenvalues();

163  using IndicesType = std::array<size_t, array_dim>;
164 
165  MultiIndexIterator(const MultiArrayType& multi_array, const IndicesType& indices)
166  : multi_array_(multi_array)
167  , indices_(indices)
168  {
169  }
170 
171  IndicesType& operator*()

173  return indices_;
174  }
175 
176  const IndicesType& operator*() const
177  {
178  return indices_;
179  }
180 
181  MultiIndexIterator& operator++()
182  {
183  size_t ii = 0;
184  for (; ii < array_dim; ++ii) {
185  indices_[ii]++;
186  if (indices_[ii] < (dim == 0 ? 1 : multi_array_.shape()[ii]))
187  break;
188  indices_[ii] = 0;
189  }
190  if (ii == array_dim)
191  std::copy_n(multi_array_.shape(), array_dim, indices_.begin());
192  return *this;
193  } // ... operator++()

196  {
197  MultiIndexIterator ret = *this;
198  this->operator++();
199  return ret;
200  } // ... operator++(int)
201 
202  bool operator==(const MultiIndexIterator& other) const
203  {
204  return indices_ == other.indices_;
205  }
206 
207  bool operator!=(const MultiIndexIterator& other) const
208  {
209  return indices_ != other.indices_;
210  }

299  static constexpr size_t dimDomain = BoundaryValueType::dimDomain;
300  static constexpr size_t dimRange = BoundaryValueType::dimRange;
301  static constexpr size_t axis_size = 3;

303  typedef typename GridLayerType::IndexSet IndexSetType;
304  typedef typename BoundaryValueType::DomainType DomainType;
305  typedef typename BoundaryValueType::DomainFieldType DomainFieldType;
306  typedef typename BoundaryValueType::RangeType RangeType;

308  typedef FieldMatrix<RangeFieldType, dimRange, dimRange> MatrixType;
309  typedef typename XT::LA::EigenSolver<MatrixType> EigenSolverType;
310  typedef typename XT::LA::EigenSolverOptions<MatrixType> EigenSolverOptionsType;
311  typedef Dune::QuadratureRule<DomainFieldType, 1> Quadrature1dType;
312  typedef typename GridLayerType::Intersection IntersectionType;
313  typedef FieldVector<IntersectionType, 2 * dimDomain> IntersectionVectorType;
314  typedef typename IntersectionType::Geometry::LocalCoordinate IntersectionLocalCoordType;
315  typedef typename AnalyticalFluxType::LocalfunctionType AnalyticalFluxLocalfunctionType;
316  typedef typename AnalyticalFluxLocalfunctionType::StateRangeType StateRangeType;
317  typedef typename XT::Grid::BoundaryInfo<IntersectionType> BoundaryInfoType;
318  using ReconstructedFunctionType =

347  void apply_local(const EntityType& entity)
348  {
349  auto& jac = *jacobian_wrapper_;
350  static const CoordsType stencil_sizes = []() {
351  CoordsType ret;
352  ret.fill(axis_size);
353  return ret;
354  }();
355  thread_local StencilType stencil(stencil_sizes);
356  bool valid = fill_stencil(stencil, entity);
357  // In a MPI parallel run, if entity is in overlap, we do not have to reconstruct
358  if (!valid)
359  return;
360  // get intersections
361  FieldVector<typename GridLayerType::Intersection, 2 * dimDomain> intersections;
362  for (const auto& intersection : Dune::intersections(grid_layer_, entity))
363  intersections[intersection.indexInInside()] = intersection;
364  const auto entity_index = grid_layer_.indexSet().index(entity);
365  auto& reconstructed_values_map = reconstructed_function_.values()[entity_index];
366 
367  // get jacobian
368  if (!jac.computed() || !analytical_flux_.is_affine()) {
369  const auto& u_entity = source_values_[entity_index];
370  const DomainType x_in_inside_coords = entity.geometry().local(entity.geometry().center());
371  analytical_flux_.local_function(entity)->partial_u(x_in_inside_coords, u_entity, *jac.jacobian(), param_);
372  jac.compute();
373  if (analytical_flux_.is_affine())
374  jac.jacobian() = nullptr;
375  }
376 
377  for (size_t dd = 0; dd < dimDomain; ++dd) {
378  // Transform values on stencil to characteristic variables of the current coordinate direction dd
379  thread_local MultiArrayType reconstructed_values(stencil_sizes);
380  thread_local auto tmp_multiarray = reconstructed_values;
381  tmp_multiarray.resize(stencil_sizes);
382  for (size_t ii = 0; ii < stencil.num_elements(); ++ii)
383  jac.apply_inverse_eigenvectors(dd, *stencil.data()[ii], tmp_multiarray.data()[ii]);
384 
385  size_t curr_dir = dd;
386  size_t last_dir = curr_dir;
387  RangeType tmp_value;
388  CoordsType current_sizes;
389  for (size_t dir = 0; dir < dimDomain; ++dir) {
390  curr_dir = (dd + dir) % dimDomain;
391  // Transform to characteristic variables of the current reconstruction direction.
392  if (dir > 0) {
393  std::copy_n(reconstructed_values.shape(), dimDomain, current_sizes.begin());
394  tmp_multiarray.resize(current_sizes);
395  tmp_multiarray = reconstructed_values;
396  std::for_each(
397  tmp_multiarray.data(), tmp_multiarray.data() + tmp_multiarray.num_elements(), [&](RangeType& value) {
398  jac.apply_eigenvectors(last_dir, value, tmp_value);
399  jac.apply_inverse_eigenvectors(curr_dir, tmp_value, value);
400  });
401  } // if (dir > 0)
402  // perform the actual reconstruction
403  const auto& curr_quadrature = dir > 0 ? quadrature_ : get_left_right_quadrature();
404  linear_reconstruction(curr_dir, curr_quadrature, tmp_multiarray, reconstructed_values);
405  last_dir = curr_dir;
406  } // dir
407  // convert back to non-characteristic variables
408  std::for_each(reconstructed_values.data(),
409  reconstructed_values.data() + reconstructed_values.num_elements(),
410  [&](RangeType& value) {
411  tmp_value = value;
412  jac.apply_eigenvectors(last_dir, tmp_value, value);
413  });
414 
415  // Convert coordinates on face to local entity coordinates and store reconstructed values
416  MultiIndexProvider<MultiArrayType> multi_indices(reconstructed_values);
417  for (const auto& multi_index : multi_indices) {
418  IntersectionLocalCoordType quadrature_point;
419  for (size_t ii = 0; ii < dimDomain; ++ii)
420  if (ii != dd)
421  quadrature_point[ii < dd ? ii : ii - 1] = quadrature_[multi_index[ii]].position();
422  reconstructed_values_map.insert(
423  std::make_pair(intersections[2 * dd + multi_index[dd]].geometryInInside().global(quadrature_point),
424  reconstructed_values(multi_index)));
425  } // multi_indices
426  } // dd
427  } // void apply_local(...)

429  static void linear_reconstruction(const size_t dir,
430  const Quadrature1dType& quadrature,
431  const MultiArrayType& stencil,
432  MultiArrayType& reconstructed_values)
433  {
434  // resize the reconstructed_values array
435  CoordsType new_shape;
436  std::copy_n(stencil.shape(), dimDomain, new_shape.begin());
437  new_shape[dir] = quadrature.size();
438  reconstructed_values.resize(new_shape);
439 
440  // get array_views corresponding to the left, center and right values
441  using IndexRangeType = typename MultiArrayType::index_range;
442  using RangeArrayType = XT::Common::FieldVector<IndexRangeType, dimDomain>;
443  using IndicesBuilderType = IndicesBuilder<RangeArrayType, dimDomain, dimDomain - 1>;
444 
445  assert(stencil.shape()[dir] == 3);
446  RangeArrayType ranges;
447  for (size_t ii = 0; ii < dimDomain; ++ii)
448  ranges[ii] = IndexRangeType(0, stencil.shape()[ii]);
449  ranges[dir] = IndexRangeType(0);
450  const auto u_left = stencil[IndicesBuilderType::build(ranges)];
451  ranges[dir] = IndexRangeType(1);
452  const auto u_entity = stencil[IndicesBuilderType::build(ranges)];
453  ranges[dir] = IndexRangeType(2);
454  const auto u_right = stencil[IndicesBuilderType::build(ranges)];
455 
456  // calculate slopes
457  thread_local SliceType slope;
458  std::array<size_t, dimDomain - 1> slope_sizes;
459  std::copy_n(u_entity.shape(), dimDomain - 1, slope_sizes.begin());
460  slope.resize(slope_sizes);
461  MultiIndexProvider<decltype(u_left)> multi_indices(u_left);
462  for (const auto& multi_index : multi_indices) {
463  const auto slope_left = u_entity(multi_index) - u_left(multi_index);
464  const auto slope_center = 0.5 * (u_right(multi_index) - u_left(multi_index));
465  const auto slope_right = u_right(multi_index) - u_entity(multi_index);
466  slope(multi_index) = internal::ChooseLimiter<slope_limiter>::limit(slope_left, slope_right, slope_center);
467  }
468 
469  // calculate reconstructed values
470  for (size_t ii = 0; ii < dimDomain; ++ii)
471  ranges[ii] = IndexRangeType(0, new_shape[ii]);
472  for (size_t ii = 0; ii < quadrature.size(); ++ii) {
473  ranges[dir] = IndexRangeType(ii);
474  auto reconstructed_ii = reconstructed_values[IndicesBuilderType::build(ranges)];
475  for (const auto& multi_index : multi_indices) {
476  reconstructed_ii(multi_index) = slope(multi_index);
477  reconstructed_ii(multi_index) *= (quadrature[ii].position() - 0.5);
478  reconstructed_ii(multi_index) += u_entity(multi_index);
479  }
480  } // ii (quadrature.size())
481  } // static void linear_reconstruction(...)
482 
483 private:
484  CoordsType center(const StencilType& stencil) const
485  {
486  CoordsType ret;
487  for (size_t ii = 0; ii < dimDomain; ++ii) {
488  assert(stencil.shape()[ii] % 2 && "Center not well-defined if one of the axis_sizes is even!");
489  ret[ii] = stencil.shape()[ii] / 2;
490  }
491  return ret;
492  }

494  bool fill_stencil(StencilType& stencil, const EntityType& entity)
495  {
496  const int dir = -2;
497  auto coords = center(stencil);
498  std::fill_n(stencil.data(), stencil.num_elements(), boost::none);
499  return fill_impl(stencil, entity, dir, coords);

502 private:
503  bool fill_impl(StencilType& stencil, const EntityType& entity, const int dir, const CoordsType& coords)
504  {
505  bool ret = true;
506  const auto& entity_index = grid_layer_.indexSet().index(entity);
507  stencil(coords) = source_values_[entity_index];
508  std::vector<int> boundary_dirs;
509  for (const auto& intersection : Dune::intersections(grid_layer_, entity)) {
510  const auto new_dir = intersection.indexInInside();
511  if (!end_of_stencil(stencil, new_dir, coords)) {
512  auto new_coords = coords;
513  if (intersection.boundary() && !intersection.neighbor()) { // boundary intersections
514  boundary_dirs.push_back(new_dir);
515  auto boundary_value = boundary_values_.local_function(entity)->evaluate(
516  intersection, entity.geometry().local(intersection.geometry().center()), source_values_[entity_index]);
517  while (!end_of_stencil(stencil, new_dir, new_coords)) {
518  next_coords_in_dir(new_dir, new_coords);
519  stencil(new_coords) = boundary_value;
520  }
521  } else if (intersection.neighbor() && direction_allowed(dir, new_dir)) { // inner and periodic intersections
522  const auto& outside = intersection.outside();
523  next_coords_in_dir(new_dir, new_coords);
524  ret = ret && fill_impl(stencil, outside, new_dir, new_coords);
525  } else if (direction_allowed(dir, new_dir) && !intersection.neighbor()
526  && !intersection.boundary()) { // processor boundary
527  return false;
528  }
529  } // if (!end_of_stencil(...))
530  } // intersections
531 
532  assert(boundary_dirs.size() <= dimDomain);
533  if (boundary_dirs.size() > 1) {
534  auto new_coords = coords;
535  next_coords_in_dir(boundary_dirs[0], new_coords);
536  const auto& boundary_value = stencil(new_coords);
537 
538  std::for_each(stencil.data(),
539  stencil.data() + stencil.num_elements(),
540  [&boundary_value](boost::optional<RangeType>& value) {
541  if (!value)
542  value = boundary_value;
543  });
544  } // if (boundary_dirs.size() > 1)
545  return ret;
546  }
547 
548  // get next coords in direction dir (increase or decrease coords in that direction)
549  static void next_coords_in_dir(const int dir, CoordsType& coords)
550  {
551  dir % 2 ? coords[dir / 2]++ : coords[dir / 2]--;
552  }
553 
554  // Direction is allowed if end of stencil is not reached and direction is not visited by another iterator.
555  // Iterators never change direction, they may only spawn new iterators in the directions that have a higher
556  // index (i.e. iterators that walk in x direction will spawn iterators going in y and z direction,
557  // iterators going in y direction will only spawn iterators in z-direction and z iterators only walk
558  // without emitting new iterators).
559  static bool direction_allowed(const int dir, const int new_dir)
560  {
561  return new_dir == dir || new_dir / 2 > dir / 2;
562  }
563 
564  bool end_of_stencil(const StencilType& stencil, const int dir, const CoordsType& coords)
565  {
566  return coords[dir / 2] == stencil.shape()[dir / 2] - 1 || coords[dir / 2] == 0;
567  }

866  const RangeType& x,

873  tmp_x[ll] = x[jj * block_size + ll];
874  XT::LA::solve_qr_factorized(QR[jj], tau[jj], permutations[jj], tmp_ret, tmp_x, &work);
875  for (size_t ll = 0; ll < block_size; ++ll)
876  ret[jj * block_size + ll] = tmp_ret[ll];
877  }
