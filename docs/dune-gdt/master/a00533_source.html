93  void globalIndices(const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
94  {
95  if (ret.size() < numDofs(entity))
96  ret.resize(numDofs(entity));
97  const auto factor_num_dofs = factor_mapper_.numDofs(entity);
98  const auto factor_mapper_size = factor_mapper_.size();
99  const auto factor_global_indices = factor_mapper_.globalIndices(entity);
100  for (size_t ii = 0; ii < dimRange; ++ii) {
101  const auto factor_num_dofs_times_ii = factor_num_dofs * ii;
102  const auto factor_mapper_size_times_ii = factor_mapper_size * ii;
103  for (size_t jj = 0; jj < factor_num_dofs; ++jj) {
104  ret[factor_num_dofs_times_ii + jj] = factor_global_indices[jj] + factor_mapper_size_times_ii;
105  }
106  }
107  } // ... globalIndices(...)
108 
109  size_t mapToGlobal(const EntityType& entity, const size_t& localIndex) const
110  {
111  assert(localIndex < numDofs(entity));
112  size_t factor_index = 0;
113  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
114  while (localIndex >= factor_mapper_num_dofs) {
115  localIndex -= factor_mapper_num_dofs;
116  ++factor_index;
117  }
118  return factor_mapper_.mapToGlobal(entity, localIndex) + factor_index * factor_mapper_.size();
119  }

133  void globalIndices(const size_t factor_index, const EntityType& entity, Dune::DynamicVector<size_t>& ret) const
134  {
135  assert(factor_index < dimRange);
136  const auto factor_mapper_num_dofs = factor_mapper_.numDofs(entity);
137  if (ret.size() < factor_mapper_num_dofs)
138  ret.resize(factor_mapper_num_dofs);
139  const auto factor_mapper_size_times_factor_index = factor_mapper_.size() * factor_index;
140  const auto factor_mapper_global_indices = factor_mapper_.globalIndices(entity);
141  for (size_t jj = 0; jj < factor_mapper_num_dofs; ++jj)
142  ret[jj] = factor_mapper_global_indices[jj] + factor_mapper_size_times_factor_index;
143  } // ... globalIndices(...)

145  size_t mapToGlobal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
146  {
147  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
148  assert(factor_index < dimRange);
149  return factor_mapper_.mapToGlobal(entity, local_index_in_factor) + factor_index * factor_mapper_.size();
150  }

152  size_t mapToLocal(const size_t factor_index, const EntityType& entity, const size_t& local_index_in_factor) const
153  {
154  assert(local_index_in_factor < factor_mapper_.numDofs(entity));
155  assert(factor_index < dimRange);
156  return factor_mapper_.numDofs(entity) * factor_index + local_index_in_factor;
157  }
