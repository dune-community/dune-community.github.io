360  template <class GL, class S, size_t d, size_t r, size_t rC>
361  typename std::enable_if<XT::Grid::is_layer<GL>::value, PatternType>::type
362  compute_face_and_volume_pattern(const GL& grd_layr, const SpaceInterface<S, d, r, rC>& ansatz_space) const
363  {
364  // prepare
365  PatternType pattern(mapper().size());
366  Dune::DynamicVector<size_t> global_rows(mapper().maxNumDofs(), 0);
367  Dune::DynamicVector<size_t> global_cols(ansatz_space.mapper().maxNumDofs(), 0);
368  for (const auto& entity : elements(grd_layr)) {
369  const auto test_base_entity = base_function_set(entity);
370  const auto ansatz_base_entity = ansatz_space.base_function_set(entity);
371  mapper().globalIndices(entity, global_rows);
372  ansatz_space.mapper().globalIndices(entity, global_cols);
373  // compute entity/entity
374  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
375  for (size_t jj = 0; jj < ansatz_base_entity.size(); ++jj) {
376  pattern.insert(global_rows[ii], global_cols[jj]);
377  }
378  }
379  // walk the intersections
380  const auto intersection_it_end = grd_layr.iend(entity);
381  for (auto intersection_it = grd_layr.ibegin(entity); intersection_it != intersection_it_end; ++intersection_it) {
382  const auto& intersection = *intersection_it;
383  // get the neighbour
384  if (intersection.neighbor() && !intersection.boundary()) {
385  const auto neighbour = intersection.outside();
386  // get the basis
387  const auto ansatz_base_neighbour = ansatz_space.base_function_set(neighbour);
388  ansatz_space.mapper().globalIndices(neighbour, global_cols);
389  // compute entity/neighbour
390  for (size_t ii = 0; ii < test_base_entity.size(); ++ii) {
391  for (size_t jj = 0; jj < ansatz_base_neighbour.size(); ++jj) {
392  pattern.insert(global_rows[ii], global_cols[jj]);
393  }
394  }
395  } // get the neighbour
396  } // walk the intersections
397  } // walk the grid layer
398  pattern.sort();
399  return pattern;
400  } // ... compute_face_and_volume_pattern(...)
