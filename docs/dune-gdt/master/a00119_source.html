120  void apply_local(const EntityType& entity)
121  {
122  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
123 
124  // get cell average
125  const RangeType u_bar =
126  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
127 
128  // vector to store thetas for each local reconstructed value
129  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), 0.);
130 
131  size_t ll = -1;
132  for (const auto& pair : local_reconstructed_values) {
133  ++ll;
134  const auto& u_l = pair.second;
135  for (size_t ii = 0; ii < u_l.size(); ++ii)
136  if (XT::Common::FloatCmp::ne(u_l[ii], u_bar[ii])) {
137  auto theta_ii = u_l[ii] / (u_l[ii] - u_bar[ii]) + epsilon_;
138  if (XT::Common::FloatCmp::le(theta_ii, 1.))
139  thetas[ll] = std::max(thetas[ll], theta_ii);
140  }
141  } // ll
142  for (auto& theta : thetas)
143  theta = std::min(theta, 1.);
144 
145  auto theta_entity_it = std::max_element(thetas.begin(), thetas.end());
146  auto theta_entity = theta_entity_it == thetas.end() ? 0. : *theta_entity_it;
147  if (theta_entity > 0.) {
148  for (auto& pair : local_reconstructed_values) {
149  auto& u = pair.second;
150  auto u_scaled = u;
151  u_scaled *= (1 - theta_entity);
152  auto u_bar_scaled = u_bar;
153  u_bar_scaled *= theta_entity;
154  u = u_scaled + u_bar_scaled;
155  }
156  }
157  } // void apply_local(...)
158 
159 private:
160  using BaseType::source_;
161  using BaseType::reconstructed_function_;
162  using BaseType::epsilon_;

256  {
257  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
258 
259  // get cell average
260  const RangeType u_bar =
261  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
262 
263  // vector to store thetas for each local reconstructed value
264  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
265 
266  size_t ll = -1;
267  for (const auto& pair : local_reconstructed_values) {
268  ++ll;
269  const auto& u_l = pair.second;
270  FieldVector<RangeFieldType, dimRange / 2> local_thetas(-epsilon_);
271  for (size_t ii = 0; ii < dimRange / 2; ++ii) {
272  const auto& u0 = u_l[2 * ii];
273  const auto& u1 = u_l[2 * ii + 1];
274  const auto& ubar0 = u_bar[2 * ii];
275  const auto& ubar1 = u_bar[2 * ii + 1];
276  const auto& vj = triangulation_[ii];
277  const auto& vjplus1 = triangulation_[ii];
278  FieldVector<RangeFieldType, 3> thetas_ii;
279  thetas_ii[0] = u0 / (ubar0 - u0);
280  thetas_ii[1] = (u0 * vj - u1) / ((ubar1 - u1) - (ubar0 - u0) * vj);
281  thetas_ii[2] = (u0 * vjplus1 - u1) / ((ubar1 - u1) - (ubar0 - u0) * vjplus1);
282  for (size_t kk = 0; kk < 3; ++kk) {
283  if (thetas_ii[kk] < -epsilon_ || thetas_ii[kk] > 1.)
284  thetas_ii[kk] = -epsilon_;
285  local_thetas[ii] = std::max(local_thetas[ii], thetas_ii[kk]);
286  } // kk
287  } // ii
288  thetas[ll] = *std::max_element(local_thetas.begin(), local_thetas.end());
289  } // ll
290  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
291  theta_entity = std::min(theta_entity + epsilon_, 1.);
292  if (theta_entity > 0.) {
293  std::cout << "Realizability limiter applied, theta = " << theta_entity << std::endl;
294  for (auto& pair : local_reconstructed_values) {
295  auto& u = pair.second;
296  auto u_scaled = u;
297  u_scaled *= (1 - theta_entity);
298  auto u_bar_scaled = u_bar;
299  u_bar_scaled *= theta_entity;
300  u = u_scaled + u_bar_scaled;
301  }
302  }
303  } // void apply_local(...)
304 
305 private:
306  using BaseType::source_;
307  using BaseType::reconstructed_function_;
308  using BaseType::epsilon_;
309  typename BasisfunctionImp::TriangulationType triangulation_;
310 }; // class DgLocalRealizabilityLimiter

507  continue;

551  {
552  plane_coefficients_ = std::make_shared<PlaneCoefficientsType>();
553  using orgQhull::Qhull;
554  FieldVector<std::vector<FieldVector<RangeFieldType, block_size>>, num_blocks> points;
555  FieldVector<QuadratureType, num_blocks> blocked_quadrature;
556  for (const auto& quad_point : quadrature_) {
557  const auto face_indices = basis_functions_.get_face_indices(quad_point.position());
558  const size_t num_adjacent_faces = face_indices.size();
559  for (const auto& kk : face_indices)
560  blocked_quadrature[kk].emplace_back(quad_point.position(), quad_point.weight() / num_adjacent_faces);
561  } // ii

642  void apply_local(const EntityType& entity)
643  {
644  auto& local_reconstructed_values = reconstructed_function_.local_values(entity);
645 
646  // get cell average
647  const RangeType u_bar =
648  source_.local_function(entity)->evaluate(entity.geometry().local(entity.geometry().center()));
649 
650  // vector to store thetas for each local reconstructed value
651  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
652 
653  size_t ll = -1;
654  for (const auto& pair : local_reconstructed_values) {
655  ++ll;
656  const auto& u_l = pair.second;
657  // if (XT::Common::FloatCmp::eq(u_bar, u_l) || basis_functions_.obviously_realizable(u_l)) {
658  if (XT::Common::FloatCmp::eq(u_bar, u_l)) {
659  thetas[ll] = 0.;
660  continue;
661  }
662 
663  // solve LP:
664  // min \theta s.t.
665  // (\sum x_i v_i) + \theta (u - \bar{u}) = u
666  // x_i, \theta >= 0
667  auto theta = solve_linear_program(u_bar, u_l);
668  theta = theta + epsilon_;
669  if (theta > 1.)
670  theta = 1.;
671  thetas[ll] = theta;
672  } // ll
673 
674  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
675  if (theta_entity > 0.)
676  limit_realizability(local_reconstructed_values, u_bar, theta_entity);
677  } // void apply_local(...)

679 private:
680  void limit_realizability(std::map<DomainType, RangeType, XT::Common::FieldVectorLess>& values,
681  const RangeType& u_bar,
682  const RangeFieldType theta)
683  {
684  for (auto& pair : values) {
685  auto& u = pair.second;
686  u = convex_combination(u, u_bar, theta);
687  }
688  }

690  RangeType convex_combination(const RangeType& u, const RangeType& u_bar, const RangeFieldType& theta)
691  {
692  if (XT::Common::FloatCmp::eq(theta, 0.))
693  return u;
694  else {
695  RangeType u_scaled = u;
696  u_scaled *= 1 - theta;
697  RangeType u_bar_scaled = u_bar;
698  u_bar_scaled *= theta;
699  return u_scaled + u_bar_scaled;
700  }
701  }
702 
703  // RangeFieldType solve_linear_program(const RangeType& u_bar, const RangeType& u_l, const size_t index)
704  RangeFieldType solve_linear_program(const RangeType& u_bar, const RangeType& u_l)
705  {
706  RangeFieldType theta;
707  const auto u_l_minus_u_bar = u_l - u_bar;
708 
709  // We start with creating a model with dimRange+1 rows and num_quad_points+1 columns */
710  constexpr int num_rows = int(dimRange);
711  int num_cols = int(quadrature_.size() + 1); /* variables are x_1, ..., x_{num_quad_points}, theta */
712  auto lp = XT::Common::lp_solve::make_lp(num_rows, num_cols);
713 
714  /* let us name our variables. Not required, but can be useful for debugging */
715  for (int ii = 1; ii <= num_cols - 1; ++ii) {
716  std::string ii_string = XT::Common::to_string(ii);
717  std::string name_string = "x";
718  for (size_t jj = 5 - ii_string.size(); jj > 0; --jj)
719  name_string += "0";
720  name_string += ii_string;
721  XT::Common::lp_solve::set_col_name(*lp, ii, name_string);
722  }
723 
724  std::string name_string = "theta ";
725  XT::Common::lp_solve::set_col_name(*lp, num_cols, name_string);
726 
727  // In the call to set_column, the first entry (row 0) is the value of the objective function
728  // (c_i in the objective function c^T x), the other entries are the entries of the i-th column
729  // in the constraints matrix. The 0-th column is the rhs vector. The entry (0, 0) corresponds
730  // to the initial value of the objective function.
731  std::array<double, num_rows + 1> column;
732 
733  // set rhs (column 0)
734  column[0] = 0.;
735  std::copy(u_l.begin(), u_l.end(), column.begin() + 1);
736  XT::Common::lp_solve::set_rh_vec(*lp, column.data());
737  XT::Common::lp_solve::set_rh(*lp, 0, column[0]);
738 
739  // set columns for quadrature points
740  column[0] = 0.;
741  for (int ii = 0; ii < int(basis_values_.size()); ++ii) {
742  const auto& v_i = basis_values_[ii];
743  std::copy(v_i.begin(), v_i.end(), column.begin() + 1);
744  XT::Common::lp_solve::set_column(*lp, ii + 1, column.data());
745  }
746 
747  // set theta column
748  column[0] = 1.;
749  std::copy(u_l_minus_u_bar.begin(), u_l_minus_u_bar.end(), column.begin() + 1);
750  XT::Common::lp_solve::set_column(*lp, num_cols, column.data());
751 
752  // set all contraints to equality constraints
753  for (int ii = 1; ii <= num_rows; ++ii)
754  XT::Common::lp_solve::set_constr_type(*lp, ii, XT::Common::lp_solve::eq());
755 
756  // Set bounds for variables. 0 <= x <= inf is the default for all variables.
757  // The bounds for theta should be [0,1], but we allow allow theta to be -0.1
758  // so we can check if u_l is on the boundary and if so, move it a little away
759  // from the boundary
760  XT::Common::lp_solve::set_bounds(*lp, num_cols, -0.1, 1.);
761 
762  // XT::Common::lp_solve::default_basis(lp);
763 
764  /* I only want to see important messages on screen while solving */
765  XT::Common::lp_solve::set_verbose(*lp, XT::Common::lp_solve::important());
766 
767  /* Now let lpsolve calculate a solution */
768  const auto solve_status = XT::Common::lp_solve::solve(*lp);
769  theta = XT::Common::lp_solve::get_objective(*lp);
770  if (solve_status != XT::Common::lp_solve::optimal()) {
771  XT::Common::lp_solve::write_LP(*lp, stdout);
772  DUNE_THROW(Dune::MathError, "An unexpected error occured while solving the linear program");
773  }
774 
775  return theta;
776  }
