68 {

84  explicit LaxWendroffFluxImplementation(const AnalyticalFluxType& analytical_flux,
85  XT::Common::Parameter param,
86  const bool is_linear = false,
87  const RangeFieldType alpha = dimDomain,
88  const bool boundary = false)
89  : analytical_flux_(analytical_flux)
90  , param_inside_(param)
91  , param_outside_(param)
92  , dt_(param.get("dt")[0])
93  , is_linear_(is_linear)
94  , alpha_(alpha)
95  {
96  param_inside_.set("boundary", {0.}, true);
97  param_outside_.set("boundary", {double(boundary)}, true);

100  template <class IntersectionType>
101  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple_entity,
102  const LocalfunctionTupleType& local_functions_tuple_neighbor,
103  const IntersectionType& intersection,
104  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
105  const DomainType& x_in_inside_coords,
106  const DomainType& x_in_outside_coords,
107  const RangeType& u_i,
108  const RangeType& u_j) const
109  {
110  // find direction of unit outer normal
111  auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
112  size_t direction = intersection.indexInInside() / 2;
113 
114  const auto& local_flux_inside = std::get<0>(local_functions_tuple_entity);
115  const auto& local_flux_outside = std::get<0>(local_functions_tuple_neighbor);
116 
117  // calculate flux evaluation as
118  // ret = f((u_i + u_j)*0.5 - (f_u_j - f_u_i)*0.5*dt/dx*dimDomain)*n_ij[direction]
119  const RangeFieldType dx = std::get<1>(local_functions_tuple_entity)->evaluate(x_in_inside_coords)[0];
120  RangeType ret = u_i;
121  ret += u_j;
122  ret *= 0.5;
123  RangeType second_part = local_flux_outside->evaluate_col(direction, x_in_outside_coords, u_j, param_outside_);
124  second_part -= local_flux_inside->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
125  second_part *= n_ij[direction] * 0.5 * dt_ / dx * alpha_;
126  ret -= second_part;
127  ret = local_flux_inside->evaluate_col(direction, x_in_inside_coords, ret, param_inside_);
128  ret *= n_ij[direction];
129  return ret;
130  } // ... evaluate(...)
131 
132  const AnalyticalFluxType& analytical_flux() const
133  {
134  return analytical_flux_;
135  }
136 
137 private:
138  const AnalyticalFluxType& analytical_flux_;
139  XT::Common::Parameter param_inside_;
140  XT::Common::Parameter param_outside_;
141  const double dt_;
142  const bool is_linear_;
143  const RangeFieldType alpha_;
144 }; // class LaxWendroffFluxImplementation<...>
