228 #endif // NDEBUG

247  virtual RangeFieldType step(const RangeFieldType dt, const RangeFieldType max_dt) override final
248  {
249  const RangeFieldType actual_dt = std::min(dt, max_dt);
250  auto& t = current_time();
251  auto& u_n = current_solution();
252  // calculate stages
253  for (size_t ii = 0; ii < num_stages_; ++ii) {
254  stages_k_[ii].vector() *= RangeFieldType(0);
255  u_i_.vector() = u_n.vector();
256  for (size_t jj = 0; jj < ii; ++jj)
257  u_i_.vector() += stages_k_[jj].vector() * (actual_dt * r_ * (A_[ii][jj]));
258  // TODO: provide actual_dt to op_. This leads to spurious oscillations in the Lax-Friedrichs flux
259  // because actual_dt/dx may become very small.
260  op_.apply(u_i_, stages_k_[ii], XT::Common::Parameter({{"t", {t + actual_dt * c_[ii]}}, {"dt", {dt}}}));
261  DataHandleType stages_k_ii_handle(stages_k_[ii]);
262  stages_k_[ii].space().grid_layer().template communicate<DataHandleType>(
263  stages_k_ii_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
264  }
265 
266  // calculate value of u at next time step
267  for (size_t ii = 0; ii < num_stages_; ++ii)
268  u_n.vector() += stages_k_[ii].vector() * (r_ * actual_dt * b_[ii]);
269 
270  // augment time
271  t += actual_dt;
272 
273  return dt;
274  } // ... step(...)

276  const std::pair<bool, RangeFieldType>
277  find_suitable_dt(const RangeFieldType initial_dt,
278  const RangeFieldType dt_refinement_factor = 2,
279  const RangeFieldType treshold = 0.9 * std::numeric_limits<RangeFieldType>::max(),
280  const size_t max_steps_per_dt = 20,
281  const size_t max_refinements = 20)
282  {
283  auto& t = current_time();
284  auto& u_n = current_solution();
285  assert(treshold > 0);
286  // save current state
287  DiscreteFunctionType initial_u_n = u_n;
288  RangeFieldType initial_t = t;
289  // start with initial dt
290  RangeFieldType current_dt = initial_dt;
291  size_t num_refinements = 0;
292  while (num_refinements < max_refinements) {
293  std::cout << "Trying time step length dt = " << current_dt << "... " << std::flush;
294  bool unlikely_value_occured = false;
295  size_t num_steps = 0;
296  // do max_steps_per_dt time steps...
297  while (!unlikely_value_occured) {
298  step(current_dt);
299  ++num_steps;
300  // ... unless there is a value above threshold
301  for (size_t kk = 0; kk < u_n.vector().size(); ++kk) {
302  if (std::abs(u_n.vector()[kk]) > treshold) {
303  unlikely_value_occured = true;
304  std::cout << "failed" << std::endl;
305  break;
306  }
307  }
308  // if we are able to do max_steps_per_dt time steps with this dt, we accept this dt
309  if (num_steps == max_steps_per_dt) {
310  std::cout << "looks fine" << std::endl;
311  u_n.vector() = initial_u_n.vector();
312  t = initial_t;
313  return std::make_pair(bool(true), current_dt);
314  }
315  }
316  // if there was a value above threshold start over with smaller dt
317  u_n.vector() = initial_u_n.vector();
318  t = initial_t;
319  current_dt /= dt_refinement_factor;
320  ++num_refinements;
321  }
322  return std::make_pair(bool(false), current_dt);
323  }
