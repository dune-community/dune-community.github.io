55 template <class GridLayerImp, class SourceImp, class RangeImp>
56 class L2GlobalProjectionLocalizableOperator : public LocalizableOperatorBase<GridLayerImp, SourceImp, RangeImp>
57 {
58  static_assert(XT::Functions::is_localizable_function<SourceImp>::value, "");
59  static_assert(is_discrete_function<RangeImp>::value, "");
60  typedef LocalizableOperatorBase<GridLayerImp, SourceImp, RangeImp> BaseType;
61 
62 public:
63  using typename BaseType::GridLayerType;
64  using typename BaseType::SourceType;
65  using typename BaseType::RangeType;
66  typedef typename RangeType::VectorType VectorType;
67  typedef
68  typename XT::LA::Container<typename RangeType::RangeFieldType, VectorType::Traits::sparse_matrix_type>::MatrixType
69 
70  MatrixType;
71 
72 private:
73  typedef typename RangeType::SpaceType SpaceType;
74  typedef L2MatrixOperator<SpaceType, MatrixType, GridLayerType> LhsOperatorType;
75  typedef L2VolumeVectorFunctional<SourceType, SpaceType, VectorType, GridLayerType> RhsFunctionalType;
76 
77 public:
78  L2GlobalProjectionLocalizableOperator(const size_t over_integrate,
79  GridLayerType grd_vw,
80  const SourceType& src,
81  RangeType& rng,
82  const XT::Common::Parameter& /*param*/ = {})
83  : BaseType(grd_vw, src, rng)
84  , lhs_operator_(over_integrate, range_.space(), BaseType::grid_layer())
85  , rhs_functional_(over_integrate, source_, range_.space(), BaseType::grid_layer())
86  , solved_(false)
87  {
88  this->append(lhs_operator_);
89  this->append(rhs_functional_);
90  issue_warning(this->range().space());
91  }
92 
93  L2GlobalProjectionLocalizableOperator(GridLayerType grd_vw,
94  const SourceType& src,
95  RangeType& rng,
96  const XT::Common::Parameter& param = {})
97  : L2GlobalProjectionLocalizableOperator(0, grd_vw, src, rng, param)
98  {
99  }
100 
101  void apply()
102  {
103  if (solved_)
104  return;
105  BaseType::apply();
106  try {
107  XT::LA::make_solver(lhs_operator_.matrix(), this->range().space().dof_communicator())
108  .apply(rhs_functional_.vector(), range_.vector());
109  } catch (XT::LA::Exceptions::linear_solver_failed& ee) {
110  DUNE_THROW(projection_error,
111  "L2 projection failed because a global matrix could not be inverted!\n\n"
112  << "This was the original error: "
113  << ee.what());
114  }
115  } // ... apply(...)
