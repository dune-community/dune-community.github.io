34 template <class SourceType, class AnalyticalFluxType>
35 class MomentRegularizer : public XT::Grid::Functor::Codim0<typename SourceType::SpaceType::GridLayerType>
36 {
37  // stencil is (i-r, i+r) in all dimensions, where r = polOrder + 1
38  typedef typename SourceType::SpaceType SpaceType;
39  typedef typename SpaceType::GridLayerType GridLayerType;
40  typedef typename GridLayerType::template Codim<0>::Entity EntityType;
41  typedef typename GridLayerType::IndexSet IndexSetType;
42 
43 public:
44  explicit MomentRegularizer(SourceType& source,
45  const AnalyticalFluxType& analytical_flux,
46  const XT::Common::Parameter& param)
47  : source_(source)
48  , analytical_flux_(analytical_flux)
49  , grid_layer_(source_.space().grid_layer())
50  , param_(param)
51  {
52  param_.set("boundary", {0.});
53  }
54 
55  void apply_local(const EntityType& entity)
56  {
57  const auto x_in_inside_coords = entity.geometry().local(entity.geometry().center());
58  const auto u = source_.local_function(entity)->evaluate(x_in_inside_coords, param_);
59  analytical_flux_.local_function(entity)->calculate_alpha(x_in_inside_coords, u, param_);
60 
61  // if regularization was needed, we also need to replace u_n in that cell by its regularized version
62  const auto& entity_index = grid_layer_.indexSet().index(entity);
63  const auto& s = analytical_flux_.regularization_parameters()[entity_index];
64  if (s > 0.) {
65  std::cout << "regularization in regularizer: time: " << param_.get("t")[0]
66  << ", entitycenter: " << XT::Common::to_string(entity.geometry().center())
67  << ", s: " << XT::Common::to_string(s, 15) << std::endl;
68  const auto& vector_indices = source_.space().mapper().globalIndices(entity);
69  const auto u_iso = analytical_flux_.get_isotropic_moment(u);
70  auto& source_vector = source_.vector();
71  for (size_t ii = 0; ii < vector_indices.size(); ++ii) {
72  const auto& vec_index = vector_indices[ii];
73  source_vector.set_entry(vec_index, (1 - s) * source_vector.get_entry(vec_index) + s * u_iso[ii]);
74  }
75  }
76  } // void apply_local(...)
77 
78 private:
79  SourceType& source_;
80  const AnalyticalFluxType& analytical_flux_;
81  const GridLayerType& grid_layer_;
82  XT::Common::Parameter param_;
83 }; // class MomentRegularizer<...>

116  void apply_local(const EntityType& entity)
117  {
118  const auto entity_index = index_set_.index(entity);
119  auto& local_reconstructed_values = reconstructed_values_[entity_index];
120  const RangeType& u_bar = source_values_[entity_index];
121  auto local_func = analytical_flux_.local_function(entity);
122 
123  auto limited_values = local_reconstructed_values;
124  bool finished = false;
125  const auto x_in_inside_coords = entity.geometry().local(entity.geometry().center());
126  const auto r_max = r_sequence_.back();
127  size_t solves = 0;
128  while (!finished) {
129  finished = true;
130  for (auto& pair : limited_values) {
131  auto& u = pair.second;
132  RangeFieldType r_used = 0.;
133  for (const auto& r : r_sequence_) {
134  r_used = r;
135  RangeType u_limited = convex_combination(u, u_bar, r);
136  try {
137  local_func->calculate_alpha(x_in_inside_coords, u_limited, param_);
138  ++solves;
139  break;
140  } catch (const Dune::MathError& e) {
141  if (r < r_max)
142  continue;
143  else
144  DUNE_THROW(Dune::MathError, "This was the original error:\n" + XT::Common::to_string(e.what()));
145  } // try ... catch ...
146  } // r
147  if (r_used > 0.) {
148  std::cout << "limited in solver with r = " << XT::Common::to_string(r_used)
149  << ", entity: " << XT::Common::to_string(entity.geometry().center()) << std::endl;
150  limit_realizability(limited_values, local_reconstructed_values, u_bar, r_used);
151  finished = false;
152  break;
153  }
154  } // reconstructed values
155  } // while (!finished)
156  if (solves > 2 * dimDomain)
157  std::cout << "Took: " << solves << std::endl;
158  local_reconstructed_values = limited_values;
159  } // void apply_local(...)

161 private:
162  void limit_realizability(std::map<DomainType, RangeType, XT::Common::FieldVectorLess>& limited_values,
163  const std::map<DomainType, RangeType, XT::Common::FieldVectorLess>& old_values,
164  const RangeType& u_bar,
165  const RangeFieldType theta)
166  {
167  for (auto& pair : limited_values) {
168  auto& u = pair.second;
169  const auto& u_old = old_values.at(pair.first);
170  u = convex_combination(u_old, u_bar, theta);
171  }
172  }
173 
174  RangeType convex_combination(const RangeType& u, const RangeType& u_bar, const RangeFieldType& theta)
175  {
176  if (XT::Common::FloatCmp::eq(theta, 0.))
177  return u;
178  else {
179  RangeType u_scaled = u;
180  u_scaled *= 1 - theta;
181  RangeType u_bar_scaled = u_bar;
182  u_bar_scaled *= theta;
183  return u_scaled + u_bar_scaled;
184  }
185  }
