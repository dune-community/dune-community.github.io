130 template <class NumericalCouplingFluxType,
131  class NumericalBoundaryFluxType,
132  size_t polOrder,
133  SlopeLimiters slope_limiter,
134  class RealizabilityLimiterType>
135 struct AdvectionOperatorApplier
136 {
137  template <class AnalyticalFluxType,
138  class BoundaryValueType,
139  class SourceType,
140  class RangeType,
141  class DomainFieldType,
142  class... LocalOperatorArgTypes>
143  static void
144  apply(const AnalyticalFluxType& analytical_flux,
145  const BoundaryValueType& boundary_values,
146  const SourceType& source,
147  RangeType& range,
148  const XT::Common::Parameter& param,
149  bool is_linear,
150  const Dune::QuadratureRule<DomainFieldType, 1> quadrature1d,
151  const Dune::QuadratureRule<DomainFieldType, BoundaryValueType::dimDomain - 1> intersection_quadrature,
152  const std::shared_ptr<RealizabilityLimiterType>& realizability_limiter,
153  LocalOperatorArgTypes&&... local_operator_args)
154  {
155  typedef typename SourceType::SpaceType::GridLayerType GridLayerType;
156  typedef typename BoundaryValueType::DomainType DomainType;
157  const GridLayerType& grid_layer = source.space().grid_layer();
158 
159  // evaluate cell averages
160  std::vector<typename BoundaryValueType::RangeType> source_values(grid_layer.indexSet().size(0));
161  for (const auto& entity : Dune::elements(grid_layer)) {
162  const auto& entity_index = grid_layer.indexSet().index(entity);
163  const auto& local_source = source.local_function(entity);
164  source_values[entity_index] = local_source->evaluate(entity.geometry().local(entity.geometry().center()));
165  }
166 
167  // do reconstruction
168  typedef std::vector<std::map<DomainType, typename BoundaryValueType::RangeType, XT::Common::FieldVectorLess>>
169  ReconstructedValuesType;
170  ReconstructedValuesType reconstructed_values(grid_layer.size(0));
171 
172  auto local_reconstruction_operator =
173  LocalReconstructionFvOperator<GridLayerType, AnalyticalFluxType, BoundaryValueType, polOrder, slope_limiter>(
174  source_values,
175  analytical_flux,
176  boundary_values,
177  grid_layer,
178  param,
179  is_linear,
180  quadrature1d,
181  reconstructed_values);
182  auto walker = XT::Grid::Walker<GridLayerType>(grid_layer);
183  walker.append(local_reconstruction_operator);
184  walker.walk(true);
185 
186  // communicate reconstructed values
187  typedef ReconstructionDataHandle<ReconstructedValuesType, GridLayerType> DataHandleType;
188  DataHandleType reconstruction_data_handle(reconstructed_values, grid_layer);
189  grid_layer.template communicate<DataHandleType>(
190  reconstruction_data_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
191 
192  // realizability limiting
193  if (realizability_limiter) {
194  realizability_limiter->set_source(&source);
195  realizability_limiter->set_reconstructed_values(&reconstructed_values);
196  walker.clear();
197  walker.append(*realizability_limiter);
198  walker.walk(true);
199  }
200 
201  typedef ReconstructedLocalizableFunction<GridLayerType,
202  DomainFieldType,
203  BoundaryValueType::dimDomain,
204  typename AnalyticalFluxType::RangeFieldType,
205  BoundaryValueType::dimRange>
206  ReconstructedLocalizableFunctionType;
207  const ReconstructedLocalizableFunctionType reconstructed_function(grid_layer, reconstructed_values);

406  {
407  static Dune::QuadratureRule<DomainFieldType, dimDomain - 1> get(const OnedQuadratureType& quadrature_1d)
408  {
409  Dune::QuadratureRule<DomainFieldType, dimDomain - 1> ret;
410  for (size_t ii = 0; ii < quadrature_1d.size(); ++ii)
411  for (size_t jj = 0; jj < quadrature_1d.size(); ++jj)
412  ret.push_back(Dune::QuadraturePoint<DomainFieldType, dimDomain - 1>(
413  {quadrature_1d[ii].position()[0], quadrature_1d[jj].position()[0]},
414  quadrature_1d[ii].weight() * quadrature_1d[jj].weight()));
415  return ret;
416  }
