161  virtual RangeFieldType solve(const RangeFieldType t_end,
162  const RangeFieldType initial_dt,
163  const size_t num_save_steps,
164  const bool save_solution,
165  const bool output_progress,
166  const bool visualize,
167  const bool write_to_file,
168  const std::string filename_prefix,
169  SolutionType& sol,
170  const VisualizerType& visualizer,
171  const StringifierType& stringifier)
172  {
173  RangeFieldType dt = initial_dt;
174  RangeFieldType t = current_time();
175  assert(Dune::XT::Common::FloatCmp::ge(t_end, t));
176  size_t time_step_counter = 0;
177 
178  const RangeFieldType save_interval = (t_end - t) / num_save_steps;
179  RangeFieldType next_save_time = t + save_interval > t_end ? t_end : t + save_interval;
180  size_t save_step_counter = 1;
181 
182  // save/visualize initial solution
183  if (save_solution)
184  sol.insert(std::make_pair(t, current_solution()));
185  if (visualize)
186  visualizer(current_solution(), filename_prefix, 0);
187  if (write_to_file)
188  write_to_textfile(current_solution(), filename_prefix, 0, stringifier);
189 
190  while (Dune::XT::Common::FloatCmp::lt(t, t_end)) {
191  RangeFieldType max_dt = dt;
192  // match saving times and t_end exactly
193  if (Dune::XT::Common::FloatCmp::gt(t + dt, t_end))
194  max_dt = t_end - t;
195  if (Dune::XT::Common::FloatCmp::gt(t + dt, next_save_time) && num_save_steps != size_t(-1))
196  max_dt = std::min(next_save_time - t, max_dt);
197 
198  // do a timestep
199  dt = step(dt, max_dt);
200  t = current_time();
201 
202  // augment time step counter
203  ++time_step_counter;
204 
205  // check if data should be written in this timestep (and write)
206  if (Dune::XT::Common::FloatCmp::ge(t, next_save_time) || num_save_steps == size_t(-1)) {
207  if (save_solution)
208  sol.insert(sol.end(), std::make_pair(t, current_solution()));
209  if (visualize)
210  visualizer(current_solution(), filename_prefix, save_step_counter);
211  if (write_to_file)
212  write_to_textfile(current_solution(), filename_prefix, save_step_counter, stringifier);
213  if (output_progress)
214  std::cout << "time step " << time_step_counter << " done, time =" << t << ", current dt= " << dt << std::endl;
215  next_save_time += save_interval;
216  ++save_step_counter;
217  }
218  } // while (t < t_end)
219 
220  return dt;
221  } // ... solve(...)

346  void write_to_textfile(const DiscreteFunctionType& u_n,
347  const std::string& filename_prefix,
348  const size_t step,
349  const StringifierType& stringifier)
350  {
351  const auto& grid_layer = u_n.space().grid_layer();
352  std::ofstream valuesfile(filename_prefix + "_values_rank_" + XT::Common::to_string(grid_layer.comm().rank()) + "_"
353  + XT::Common::to_string(step)
354  + ".txt");
355  for (const auto& entity : elements(grid_layer, Dune::Partitions::interiorBorder)) {
356  const auto local_func = u_n.local_function(entity);
357  const auto entity_center = entity.geometry().center();
358  const auto val = local_func->evaluate(entity.geometry().local(entity_center));
359  for (size_t ii = 0; ii < entity_center.size(); ++ii)
360  valuesfile << XT::Common::to_string(entity_center[ii], 15) << " ";
361  valuesfile << stringifier(val) << std::endl;
362  }
363  valuesfile.close();
364  grid_layer.comm().barrier();
365  if (grid_layer.comm().rank() == 0) {
366  std::remove((filename_prefix + "_values_" + XT::Common::to_string(step) + ".txt").c_str());
367  std::ofstream merged_valuesfile(filename_prefix + "_values_" + XT::Common::to_string(step) + ".txt",
368  std::ios_base::binary | std::ios_base::app);
369  for (int ii = 0; ii < grid_layer.comm().size(); ++ii) {
370  std::ifstream valuefile_rank(
371  filename_prefix + "_values_rank_" + XT::Common::to_string(ii) + "_" + XT::Common::to_string(step) + ".txt",
372  std::ios_base::binary);
373  merged_valuesfile << valuefile_rank.rdbuf();
374  valuefile_rank.close();
375  } // ii
376  merged_valuesfile.close();
377  } // if (rank == 0)
378  } // void write_to_textfile()
