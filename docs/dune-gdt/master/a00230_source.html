180  DiagonallyImplicitRungeKuttaTimeStepper(const OperatorType& op,
181  const DiscreteFunctionType& initial_values,
182  const RangeFieldType r = 1.0,
183  const RangeFieldType t_0 = 0.0,
184  const RangeFieldType beta = 1e-4,
185  const std::string solver_type = "",
186  const MatrixType& A = ButcherArrayProviderType::A(),
187  const VectorType& b = ButcherArrayProviderType::b(),
188  const VectorType& c = ButcherArrayProviderType::c())
189  : BaseType(t_0, initial_values)
190  , op_(op)
191  , r_(r)
192  , u_i_(BaseType::current_solution())
193  , u_i_expl_(u_i_)
194  , d_(u_i_)
195  , N_times_d_(u_i_)
196  , res_(u_i_)
197  , new_res_(u_i_)
198  , u_i_plus_alpha_d_(u_i_)
199  , solver_type_(solver_type)
200  , beta_(beta)
201  , newton_matrix_(u_i_.vector().size(), u_i_.vector().size(), this->newton_matrix_pattern(u_i_.space()))
202  , solver_(newton_matrix_, u_i_.space().dof_communicator())
203  , A_(A)
204  , b_(b)
205  , c_(c)
206  , num_stages_(A_.rows())
207  {
208  assert(A_.rows() == A_.cols() && "A has to be a square matrix");
209  assert(b_.size() == A_.rows());
210  assert(c_.size() == A_.rows());
211  bool lower_triangular = true;
212  bool diagonally_implicit = true;
213  for (size_t ii = 0; ii < A_.rows(); ++ii) {
214  for (size_t jj = ii; jj < A_.cols(); ++jj) {
215  if (Dune::XT::Common::FloatCmp::ne(A_[ii][jj], 0.0))
216  lower_triangular = false;
217  if (jj > ii && Dune::XT::Common::FloatCmp::ne(A_[ii][jj], 0.0))
218  diagonally_implicit = false;
219  }
220  }
221  if (lower_triangular)
222  DUNE_THROW(XT::Common::Exceptions::you_are_using_this_wrong,
223  "Matrix A is strictly lower triangular, use ExplicitRungeKuttaTimeStepper!");
224  if (!diagonally_implicit)
225  DUNE_THROW(Dune::NotImplemented, "Matrix A has entries above the diagonal, this is not supported yet!");
226  // store as many discrete functions as needed for intermediate stages
227  for (size_t ii = 0; ii < num_stages_; ++ii) {
228  stages_k_.emplace_back(u_i_);
229  }
230  } // constructor

232  RangeFieldType calculate_l2_norm(const DiscreteFunctionType& disc_func)
233  {
234  RangeFieldType norm = 0;
235  const auto& vector = disc_func.vector();
236  const auto& mapper = disc_func.space().mapper();
237  for (const auto& entity : Dune::elements(disc_func.space().grid_layer(), Dune::Partitions::interiorBorder))
238  for (const auto& index : mapper.globalIndices(entity))
239  norm += std::pow(vector[index], 2);
240  disc_func.space().grid_layer().comm().sum(norm);
241  norm = std::sqrt(norm);
242  return norm;
243  }

245  virtual RangeFieldType step(const RangeFieldType dt, const RangeFieldType max_dt) override final
246  {
247  const RangeFieldType actual_dt = std::min(dt, max_dt);
248  auto& t = current_time();
249  auto& u_n = current_solution();
250  // calculate stages
251  SolverMatrixDataHandleType newton_matrix_handle(newton_matrix_, u_n.space());
252  for (size_t ii = 0; ii < num_stages_; ++ii) {
253  // calculate u_expl
254  if (ii == 0)
255  u_i_expl_.vector() = u_n.vector();
256  else
257  u_i_expl_.vector() += stages_k_[ii - 1].vector() * (actual_dt * r_ * (A_[ii][ii - 1]));
258  DataHandleType stages_k_ii_handle(stages_k_[ii]);
259  // choose explicit or implicit treatment
260  if (XT::Common::FloatCmp::eq(A_[ii][ii], 0.0)) {
261  // stage is explicit
262  stages_k_[ii].vector() *= RangeFieldType(0);
263  op_.apply(
264  u_i_expl_, stages_k_[ii], XT::Common::Parameter({{"t", {t + actual_dt * c_[ii]}}, {"dt", {actual_dt}}}));
265  stages_k_[ii].space().grid_layer().template communicate<DataHandleType>(
266  stages_k_ii_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
267  } else {
268  // stage is implicit, Newton method with Armijo backtracking
269  // initial guess for u_i: u_i_expl_ + dt a_{ii} k_{i-1}
270  u_i_.vector() =
271  (ii == 0) ? u_n.vector() : u_i_expl_.vector() + stages_k_[ii - 1].vector() * (actual_dt * r_ * A_[ii][ii]);
272  // calculate residual with current iterate
273  stages_k_[ii].vector() *= RangeFieldType(0);
274  op_.apply(u_i_, stages_k_[ii], XT::Common::Parameter({{"t", {t + actual_dt * c_[ii]}}, {"dt", {actual_dt}}}));
275  stages_k_[ii].space().grid_layer().template communicate<DataHandleType>(
276  stages_k_ii_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
277  res_.vector() = u_i_expl_.vector() - u_i_.vector() + stages_k_[ii].vector() * (actual_dt * r_ * A_[ii][ii]);
278  // calculate norm of residual
279  RangeFieldType current_error = calculate_l2_norm(res_);
280  RangeFieldType first_error = current_error;
281  const double abs_tol = 1e-12;
282  const double rel_reduction = 1e-8;
283  // Newton loop
284  while (current_error > abs_tol && current_error > first_error * rel_reduction) {
285  // assemble Newton matrix N = I - dt * a_{ii} * J_L
286  newton_matrix_ *= 0;
287  op_.assemble_jacobian(newton_matrix_, u_i_, t + actual_dt * c_[ii]);
288  // u_i_.space().grid_layer().template communicate<SolverMatrixDataHandleType>(
289  // newton_matrix_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
290  newton_matrix_ *= -actual_dt * r_ * A_[ii][ii];
291  for (size_t kk = 0; kk < newton_matrix_.rows(); ++kk)
292  newton_matrix_.add_to_entry(kk, kk, 1.);
293  // solve N d = res
294  d_.vector() *= 0;
295  solver_.apply(res_.vector(), d_.vector(), solver_type_);
296 
297  // Armijo backtracking line search
298  RangeFieldType alpha = 2;
299  RangeFieldType error_threshold = abs_tol;
300  while (current_error > error_threshold) {
301  // decrease alpha and calculate current u_i
302  alpha /= 2;
303  // calculate error threshold with current alpha
304  N_times_d_.vector() *= 0;
305  newton_matrix_.mv(d_.vector(), N_times_d_.vector());
306  N_times_d_.vector() = res_.vector() - (N_times_d_.vector() * alpha * beta_);
307  error_threshold = calculate_l2_norm(N_times_d_);
308  // calculate residual with current alpha
309  u_i_plus_alpha_d_.vector() = u_i_.vector() + d_.vector() * alpha;
310  stages_k_[ii].vector() *= RangeFieldType(0);
311  op_.apply(u_i_plus_alpha_d_,
312  stages_k_[ii],
313  XT::Common::Parameter({{"t", {t + actual_dt * c_[ii]}}, {"dt", {actual_dt}}}));
314  stages_k_[ii].space().grid_layer().template communicate<DataHandleType>(
315  stages_k_ii_handle, Dune::InteriorBorder_All_Interface, Dune::ForwardCommunication);
316  new_res_.vector() = u_i_expl_.vector() - u_i_plus_alpha_d_.vector()
317  + stages_k_[ii].vector() * (actual_dt * r_ * A_[ii][ii]);
318  current_error = calculate_l2_norm(new_res_);
319  } // Armijo backtracking
320  u_i_.vector() = u_i_plus_alpha_d_.vector();
321  res_.vector() = new_res_.vector();
322  } // Newton
323  } // if (explicit) else (implicit)
324  } // loop over stages
325 
326  // calculate value of u at next time step
327  for (size_t ii = 0; ii < num_stages_; ++ii)
328  u_n.vector() += stages_k_[ii].vector() * (r_ * actual_dt * b_[ii]);
329 
330  // augment time
331  t += actual_dt;
332 
333  return dt;
334  } // ... step(...)
