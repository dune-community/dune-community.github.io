125  template <class IntersectionType>
126  RangeType evaluate(const LocalfunctionTupleType& local_functions_tuple,
127  const IntersectionType& intersection,
128  const Dune::FieldVector<DomainFieldType, dimDomain - 1>& x_in_intersection_coords,
129  const DomainType& x_in_inside_coords,
130  const DomainType& /*x_in_outside_coords*/,
131  const RangeType& u_i,
132  const RangeType& u_j) const
133  {
134  size_t direction = intersection.indexInInside() / 2;
135  // get unit outer normal
136  const auto n_ij = intersection.unitOuterNormal(x_in_intersection_coords);
137  assert(XT::Common::FloatCmp::eq(std::abs(n_ij[direction]), 1.));
138 
139  // intialize jacobians
140  initialize_jacobians(direction, local_functions_tuple, x_in_inside_coords, u_i, u_j);
141 
142  // get jump at the intersection
143  const RangeType delta_u = u_i - u_j;
144  // calculate waves
145  RangeType waves(0);
146  n_ij[direction] > 0 ? jacobian_neg()[direction].mtv(delta_u, waves) : jacobian_pos()[direction].mtv(delta_u, waves);
147  // calculate flux
148  const auto& local_flux = std::get<0>(local_functions_tuple);
149  RangeType ret = local_flux->evaluate_col(direction, x_in_inside_coords, u_i, param_inside_);
150  ret -= waves;
151  ret *= n_ij[direction];
152  return ret;
153  } // RangeType evaluate(...) const

166  // solve R^T F = P^T D A^T where is P is the permutation matrix and D is the diagonal eigenvalue matrix
167  static void solve_upper_triangular_transposed(CscSparseMatrixType& F,
168  const SparseMatrixType& R,
169  const FieldVector<size_t, dimRange>& permutations,
170  const StateRangeType& eigenvalues,
171  const SparseMatrixType& A)
172  {
173  F.clear();
174  StateRangeType rhs(0.);
175  for (size_t cc = 0; cc < dimRange; ++cc) {
176  // apply permutations and scaling to column of A^T and store in vector
177  std::fill(rhs.begin(), rhs.end(), 0.);
178  size_t end = A.row_pointers()[cc + 1];
179  for (size_t kk = A.row_pointers()[cc]; kk < end; ++kk) {
180  size_t index = A.column_indices()[kk];
181  auto eigval = eigenvalues[index];
182  if (XT::Common::FloatCmp::ne(eigval, 0.))
183  rhs[permutations[index]] = A.entries()[kk] * eigval;
184  } // kk
185  // forward solve
186  for (size_t ii = 0; ii < dimRange; ++ii) {
187  if (XT::Common::FloatCmp::ne(rhs[ii], 0.)) {
188  size_t kk = R.row_pointers()[ii];
189  const auto& diag_ii = R.entries()[kk];
190  const auto x_ii = rhs[ii] / diag_ii;
191  F.entries().push_back(x_ii);
192  F.row_indices().push_back(ii);
193  ++kk;
194  for (; kk < R.row_pointers()[ii + 1]; ++kk)
195  rhs[R.column_indices()[kk]] -= R.entries()[kk] * x_ii;
196  }
197  } // ii
198  F.column_pointers()[cc + 1] = F.row_indices().size();
199  } // cc
200  } // void solve_upper_triangular_transposed(...)

202  // use simple linearized Riemann solver, LeVeque p.316
203  void initialize_jacobians(const size_t direction,
204  const LocalfunctionTupleType& local_functions_tuple,
205  const DomainType& x_local,
206  const RangeType& u_i,
207  const RangeType& u_j) const
208  {
209  if (local_initialization_counts_[direction] != initialization_count_) {
210  // calculate jacobian as jacobian(0.5*(u_i+u_j)
211  RangeType u_mean = u_i + u_j;
212  u_mean *= RangeFieldType(0.5);
213  const auto& local_flux = std::get<0>(local_functions_tuple);
214  thread_local std::unique_ptr<MatrixType> jacobian;
215  thread_local std::unique_ptr<MatrixType> Qdense;
216  if (!jacobian) {
217  jacobian = XT::Common::make_unique<MatrixType>();
218  Qdense = XT::Common::make_unique<MatrixType>();
219  }
220  helper<dimDomain>::get_jacobian(direction, local_flux, x_local, u_mean, *jacobian, param_inside_);
221  // get matrix of eigenvectors A and eigenvalues
222  static XT::Common::Configuration eigensolver_options(
223  {"type", "check_for_inf_nan", "check_evs_are_real", "check_evs_are_positive", "check_eigenvectors_are_real"},
224  {EigenSolverType::types()[0].c_str(), "1", "1", "0", "1"});
225  const auto eigen_solver = EigenSolverType(*jacobian);
226  const auto eigenvalues = eigen_solver.eigenvalues(eigensolver_options);
227  StateRangeType eigvals_neg(0.);
228  StateRangeType eigvals_pos(0.);
229  for (size_t ii = 0; ii < eigenvalues.size(); ++ii) {
230  if (eigenvalues[ii].real() < 0.)
231  eigvals_neg[ii] = eigenvalues[ii].real();
232  else
233  eigvals_pos[ii] = eigenvalues[ii].real();
234  }
235  const auto eigenvectors_dense = eigen_solver.real_eigenvectors_as_matrix(eigensolver_options);
236  thread_local SparseMatrixType eigenvectors(dimRange, dimRange, size_t(0));
237  eigenvectors = *eigenvectors_dense;
238  // calculate QR decomposition with column pivoting A = QRP^T
239  FieldVector<size_t, dimRange> permutations;
240  XT::LA::qr_decomposition(*eigenvectors_dense, permutations, *Qdense);
241  FieldVector<size_t, dimRange> inverse_permutations;
242  for (size_t ii = 0; ii < dimRange; ++ii)
243  inverse_permutations[permutations[ii]] = ii;
244  thread_local SparseMatrixType R(dimRange, dimRange, size_t(0));
245  R = *eigenvectors_dense;
246  // we want to compute C_{+,-} = A D_{+,-} A^{-1}, where D_+ is the diagonal matrix containing only positive
247  // eigenvalues and D_- contains only negative eigenvalues. Using the QR decomposition, we get
248  // C = A D A^{-1} = A D (QRP^T)^{-1} = A D P R^{-1} Q^T. For the transpose of C, we get
249  // C^T = Q R^{-T} P^T D A^T. We calculate C^T by first solving (column-wise)
250  // R^T F = P^T D A^T and then computing C^T = Q F;
251  thread_local CscSparseMatrixType F_neg(dimRange, dimRange, size_t(0));
252  thread_local CscSparseMatrixType F_pos(dimRange, dimRange, size_t(0));
253  solve_upper_triangular_transposed(F_neg, R, inverse_permutations, eigvals_neg, eigenvectors);
254  solve_upper_triangular_transposed(F_pos, R, inverse_permutations, eigvals_pos, eigenvectors);
255 
256  jacobian_neg()[direction] = *Qdense;
257  jacobian_pos()[direction].deep_copy(jacobian_neg()[direction]);
258  jacobian_neg()[direction].rightmultiply(F_neg);
259  jacobian_pos()[direction].rightmultiply(F_pos);
260 
261  if (is_linear_) {
262  jacobian = nullptr;
263  Qdense = nullptr;
264  ++local_initialization_counts_[direction];
265  }
266  } // if (local_initialization_counts_[direction] != initialization_count_)
267  } // void calculate_jacobians(...)
