71  template <class SourceType, class IntersectionType, class SpaceType, class VectorType>
72  void apply(const SourceType& source,
73  const IntersectionType& intersection,
74  LocalDiscreteFunction<SpaceType, VectorType>& local_range_entity,
75  LocalDiscreteFunction<SpaceType, VectorType>& local_range_neighbor) const
76  {
77  const auto entity = intersection.inside();
78  const auto neighbor = intersection.outside();
79  const auto local_source_entity = source.local_function(entity);
80  const auto local_source_neighbor = source.local_function(neighbor);
81  const auto local_functions_tuple_entity = numerical_flux_.local_functions(entity);
82  const auto local_functions_tuple_neighbor = numerical_flux_.local_functions(neighbor);
83  ResultType result(0);
84  for (const auto& quad_point : quadrature_) {
85  result += ResultType(numerical_flux_.evaluate(local_functions_tuple_entity,
86  local_functions_tuple_neighbor,
87  *local_source_entity,
88  *local_source_neighbor,
89  intersection,
90  quad_point.position()))
91  * quad_point.weight() * intersection.geometry().integrationElement(quad_point.position());
92  }
93  local_range_entity.vector().add(result * (1.0 / entity.geometry().volume()));
94  local_range_neighbor.vector().add(result * (-1.0 / neighbor.geometry().volume()));
95  }

98  const QuadratureType& quadrature_;
99  const NumericalFluxType numerical_flux_;
