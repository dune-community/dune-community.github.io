154 template <class GL, int p, class R>
155 class DiscontinuousLagrangeSpace
156  : public DgSpaceInterface<internal::DiscontinuousLagrangeSpaceTraits<GL, p, R>, GL::dimension, 1>
157 {
158 public:
159  using Traits = internal::DiscontinuousLagrangeSpaceTraits<GL, p, R>;
160 
161 private:
162  using BaseType = DgSpaceInterface<Traits, GL::dimension, 1>;
163  using ThisType = DiscontinuousLagrangeSpace<GL, p, R>;
164  using D = typename GL::ctype;
165  static const constexpr size_t d = BaseType::dimDomain;
166  using FiniteElementType = LagrangeLocalFiniteElement<EquidistantPointSet, d, D, R>;
167  using DofCommunicationChooserType = typename Traits::DofCommunicationChooserType;
168 
169 public:
170  using typename BaseType::GridLayerType;
171  using typename BaseType::EntityType;
172  using typename BaseType::MapperType;
173  using typename BaseType::BaseFunctionSetType;
174  using DomainType = typename BaseFunctionSetType::DomainType;
175  using DofCommunicatorType = typename Traits::DofCommunicatorType;
176 
177  DiscontinuousLagrangeSpace(GridLayerType grd_lr)
178  : grid_layer_(grd_lr)
179  , communicator_(DofCommunicationChooserType::create(grid_layer_))
180  , backend_(0)
181  , finite_elements_(new std::map<GeometryType, std::shared_ptr<FiniteElementType>>())
182  , lagrange_points_(new std::map<GeometryType, std::vector<DomainType>>())
183  , mapper_(nullptr)
184  {
185  // create finite elements and lagrange points
186  for (auto&& geometry_type : grid_layer_.indexSet().types(0)) {
187  // if (geometry_type == GeometryType(GeometryType::pyramid, 3))
188  // DUNE_THROW(space_error, "Discontinuous Lagrange space does not seem to have working jacobians on pyramid
189  // grids!");
190  auto fe = std::make_shared<FiniteElementType>(geometry_type, p);
191  const auto& lp = fe->localInterpolation().lagrangePoints();
192  std::vector<DomainType> lagrange_points(lp.size());
193  for (unsigned int ii = 0; ii < lp.size(); ++ii)
194  lagrange_points[ii] = lp[ii].point();
195  lagrange_points_->insert(std::make_pair(geometry_type, std::move(lagrange_points)));
196  finite_elements_->insert(std::make_pair(geometry_type, std::move(fe)));
197  }
198  // check
199  // if (d == 3 && finite_elements_->size() != 1)
200  // DUNE_THROW(space_error, "Discontinuous Lagrange space with multiple finite elements in 3d not supported
201  // (yet)!");
202  // create mapper
203  mapper_ = std::make_shared<MapperType>(grid_layer_, finite_elements_);
204  }
205

207  DiscontinuousLagrangeSpace(ThisType&&) = default;
208 
209  ThisType& operator=(const ThisType&) = delete;
210  ThisType& operator=(ThisType&&) = delete;
211

213  {
214  return grid_layer_;
215  }
216 
217  const double& backend() const
218  {
219  return backend_;
220  }
221 
222  const MapperType& mapper() const
223  {
224  return *mapper_;
225  }

227  BaseFunctionSetType base_function_set(const EntityType& entity) const
228  {
229  const auto finite_element_search_result = finite_elements_->find(entity.geometry().type());
230  if (finite_element_search_result == finite_elements_->end())
231  DUNE_THROW(XT::Common::Exceptions::internal_error,
232  "This must not happen, the grid layer did not report all geometry types!"
233  << "\n entity.geometry().type() = "
234  << entity.geometry().type());
235  const auto& finite_element = *finite_element_search_result->second;
236  return BaseFunctionSetType(entity, finite_element);
237  }
238 
239  DofCommunicatorType& dof_communicator() const
240  {
241  DofCommunicationChooserType::prepare(*this, *communicator_);
242  return *communicator_;
243  }
244 
245  std::vector<DomainType> lagrange_points(const EntityType& entity) const
246  {
247  const auto lagrange_points_search_result = lagrange_points_->find(entity.geometry().type());
248  if (lagrange_points_search_result == lagrange_points_->end())
249  DUNE_THROW(XT::Common::Exceptions::internal_error,
250  "This must not happen, the grid layer did not report all geometry types!"
251  << "\n entity.geometry().type() = "
252  << entity.geometry().type());
253  return lagrange_points_search_result->second;
254  }

257  const GridLayerType grid_layer_;
258  mutable std::shared_ptr<DofCommunicatorType> communicator_;
259  const double backend_;
260  std::shared_ptr<std::map<GeometryType, std::shared_ptr<FiniteElementType>>> finite_elements_;
261  std::shared_ptr<std::map<GeometryType, std::vector<DomainType>>> lagrange_points_;
262  std::shared_ptr<MapperType> mapper_;
263 }; // class DiscontinuousLagrangeSpace
