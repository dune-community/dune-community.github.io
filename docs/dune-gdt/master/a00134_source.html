134  explicit RelativePositivityLocalRealizabilityLimiter(const BasisfunctionType& basis_functions,
135  const QuadratureType& quadrature,

141  void apply_local(const EntityType& entity)
142  {
143  const auto entity_index = index_set_->index(entity);
144  auto& local_reconstructed_values = (*reconstructed_values_)[entity_index];
145 
146  // get cell average
147  const RangeType& u_bar = (*source_values_)[entity_index];
148 
149  // vector to store thetas for each local reconstructed value
150  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), 0.);
151 
152  size_t ll = -1;
153  for (const auto& pair : local_reconstructed_values) {
154  ++ll;
155  const auto& u_l = pair.second;
156  for (size_t ii = 0; ii < u_l.size(); ++ii)
157  if (XT::Common::FloatCmp::ne(u_l[ii], u_bar[ii])) {
158  auto theta_ii = u_l[ii] / (u_l[ii] - u_bar[ii]) + epsilon_;
159  if (XT::Common::FloatCmp::le(theta_ii, 1.))
160  thetas[ll] = std::max(thetas[ll], theta_ii);
161  }
162  } // ll
163  for (auto& theta : thetas)
164  theta = std::min(theta, 1.);
165 
166  auto theta_entity_it = std::max_element(thetas.begin(), thetas.end());
167  auto theta_entity = theta_entity_it == thetas.end() ? 0. : *theta_entity_it;
168  if (theta_entity > 0.) {
169  for (auto& pair : local_reconstructed_values) {
170  auto& u = pair.second;
171  auto u_scaled = u;
172  u_scaled *= (1 - theta_entity);
173  auto u_bar_scaled = u_bar;
174  u_bar_scaled *= theta_entity;
175  u = u_scaled + u_bar_scaled;
176  }
177  }
178  } // void apply_local(...)
179 
180 private:
181  using BaseType::source_values_;
182  using BaseType::index_set_;
183  using BaseType::reconstructed_values_;
184  using BaseType::epsilon_;

267  // cell averages includes left and right boundary values as the two last indices in each dimension
268  explicit DgLocalRealizabilityLimiter(const BasisfunctionType& basis_functions,
269  const QuadratureType& quadrature,

276  void apply_local(const EntityType& entity)
277  {
278  const auto entity_index = index_set_->index(entity);
279  auto& local_reconstructed_values = (*reconstructed_values_)[entity_index];
280 
281  // get cell average
282  const RangeType u_bar = (*source_values_)[entity_index];
283 
284  // vector to store thetas for each local reconstructed value
285  std::vector<RangeFieldType> thetas(local_reconstructed_values.size(), -epsilon_);
286 
287  size_t ll = -1;
288  for (const auto& pair : local_reconstructed_values) {
289  ++ll;
290  const auto& u_l = pair.second;
291  FieldVector<RangeFieldType, dimRange / 2> local_thetas(-epsilon_);
292  for (size_t ii = 0; ii < dimRange / 2; ++ii) {
293  const auto& u0 = u_l[2 * ii];
294  const auto& u1 = u_l[2 * ii + 1];
295  const auto& ubar0 = u_bar[2 * ii];
296  const auto& ubar1 = u_bar[2 * ii + 1];
297  const auto& vj = triangulation_[ii];
298  const auto& vjplus1 = triangulation_[ii];
299  FieldVector<RangeFieldType, 3> thetas_ii;
300  thetas_ii[0] = u0 / (ubar0 - u0);
301  thetas_ii[1] = (u0 * vj - u1) / ((ubar1 - u1) - (ubar0 - u0) * vj);
302  thetas_ii[2] = (u0 * vjplus1 - u1) / ((ubar1 - u1) - (ubar0 - u0) * vjplus1);
303  for (size_t kk = 0; kk < 3; ++kk) {
304  if (thetas_ii[kk] < -epsilon_ || thetas_ii[kk] > 1.)
305  thetas_ii[kk] = -epsilon_;
306  local_thetas[ii] = std::max(local_thetas[ii], thetas_ii[kk]);
307  } // kk
308  } // ii
309  thetas[ll] = *std::max_element(local_thetas.begin(), local_thetas.end());
310  } // ll
311  auto theta_entity = *std::max_element(thetas.begin(), thetas.end());
312  theta_entity = std::min(theta_entity + epsilon_, 1.);
313  if (theta_entity > 0.) {
314  std::cout << "Realizability limiter applied, theta = " << theta_entity << std::endl;
315  for (auto& pair : local_reconstructed_values) {
316  auto& u = pair.second;
317  auto u_scaled = u;
318  u_scaled *= (1 - theta_entity);
319  auto u_bar_scaled = u_bar;
320  u_bar_scaled *= theta_entity;
321  u = u_scaled + u_bar_scaled;
322  }
323  }
324  } // void apply_local(...)

328  using BaseType::index_set_;
329  using BaseType::reconstructed_values_;
330  using BaseType::epsilon_;

332 }; // class DgLocalRealizabilityLimiter

514  ++ll;

517  if (XT::Common::FloatCmp::eq(u_l, u_bar))
