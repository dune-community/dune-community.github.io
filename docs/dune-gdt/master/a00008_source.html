51  static void assemble(const TestSpaceType& test_space,
52  const AnsatzSpaceType& ansatz_space,
53  const LocalVolumeTwoFormType& local_volume_two_form,
54  const EntityType& entity,
55  MatrixType& global_matrix)
56  {
57 
58  DUNE_THROW_IF(global_matrix.rows() != test_space.mapper().size(),
59  XT::Common::Exceptions::shapes_do_not_match,
60  "global_matrix.rows() = " << global_matrix.rows() << "\n "
61  << "test_space.mapper().size()"
62  << test_space.mapper().size());
63  DUNE_THROW_IF(global_matrix.cols() != ansatz_space.mapper().size(),
64  XT::Common::Exceptions::shapes_do_not_match,
65  "global_matrix.cols() = " << global_matrix.cols() << "\n "
66  << "ansatz_space.mapper().size()"
67  << ansatz_space.mapper().size());
68  // prepare
69  const size_t rows = test_space.mapper().numDofs(entity);
70  const size_t cols = ansatz_space.mapper().numDofs(entity);
71  DynamicMatrix<FieldType> local_matrix(rows, cols, 0.); // \todo: make mutable member, after SMP refactor
72  // apply local two-form
73  const auto test_base = test_space.base_function_set(entity);
74  const auto ansatz_base = ansatz_space.base_function_set(entity);
75  assert(test_base.size() == rows);
76  assert(ansatz_base.size() == cols);
77  local_volume_two_form.apply2(test_base, ansatz_base, local_matrix);
78  // write local matrix to global
79  const auto global_row_indices =
80  test_space.mapper().globalIndices(entity); // \todo: make mutable member, after SMP refactor
81  const auto global_col_indices =
82  ansatz_space.mapper().globalIndices(entity); // \todo: make mutable member, after SMP refactor
83  assert(global_row_indices.size() == rows);
84  assert(global_col_indices.size() == cols);
85  for (size_t ii = 0; ii < rows; ++ii) {
86  const auto& local_matrix_row = local_matrix[ii];
87  const size_t global_ii = global_row_indices[ii];
88  for (size_t jj = 0; jj < cols; ++jj) {
89  const size_t global_jj = global_col_indices[jj];
90  global_matrix.add_to_entry(global_ii, global_jj, local_matrix_row[jj]);
91  }
92  } // write local matrix to global
93  } // ... assemble(...)
