240 template <size_t rangeDim, class MatrixType, class EigenMatrixType, class AnalyticalFluxType>
241 struct EigenvectorInitializer<1, rangeDim, MatrixType, EigenMatrixType, AnalyticalFluxType>
242 {
243  static void initialize(const AnalyticalFluxType& analytical_flux,
244  const bool is_linear,
245  const bool use_linear_reconstruction,
246  MatrixType& eigenvectors,
247  MatrixType& eigenvectors_inverse)
248  {
249  if (use_linear_reconstruction) {
250  assert(is_linear && "Linear reconstruction is only implemented for linear analytical fluxes!");
251  // calculate matrix of eigenvectors of A, where A is the jacobian of the linear analytical flux, i.e. u_t + A*u_x
252  // = 0. As the analytical flux is linear, the jacobian A is constant, so it is enough to evaluate at 0.
253  ::Eigen::EigenSolver<typename EigenMatrixType::BackendType> eigen_solver(
254  Dune::XT::Common::from_string<EigenMatrixType>(
255  Dune::XT::Common::to_string(analytical_flux.jacobian(typename AnalyticalFluxType::RangeType(0))))
256  .backend());
257  assert(eigen_solver.info() == ::Eigen::Success);
258  const auto eigen_eigenvectors = eigen_solver.eigenvectors();
259 #ifndef NDEBUG
260  for (size_t ii = 0; ii < rangeDim; ++ii)
261  for (size_t jj = 0; jj < rangeDim; ++jj)
262  assert(eigen_eigenvectors(ii, jj).imag() < 1e-15);
263 #endif
264  eigenvectors = Dune::XT::Common::from_string<MatrixType>(
265  Dune::XT::Common::to_string(EigenMatrixType(eigen_eigenvectors.real())));
266  eigenvectors_inverse = Dune::XT::Common::from_string<MatrixType>(
267  Dune::XT::Common::to_string(EigenMatrixType(eigen_eigenvectors.inverse().real())));
268  }
269  }
270 }; // struct EigenvectorInitializer<1, ...>
