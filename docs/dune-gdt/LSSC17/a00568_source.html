120 template <typename SpaceType>
121 template <class CommunicatorType>
122 void GenericParallelHelper<SpaceType>::createIndexSetAndProjectForAMG(CommunicatorType& c)
123 {
124 
125  // ********************************************************************************
126  // In the following, the code will always assume that all DOFs stored in a single
127  // block of the BCRSMatrix are attached to the same entity and can be handled
128  // identically. For that reason, the code often restricts itself to inspecting the
129  // first entry of the blocks in the diverse BlockVectors.
130  // ********************************************************************************
131 
132  const auto& view = space_.grid_layer();
133  const auto vector_size = space_.mapper().size();
134 
135  // Do we need to communicate at all?
136  const bool need_communication = view.comm().size() > 1;
137 
138  // First find out which dofs we share with other processors
139  BoolVector sharedDOF(vector_size, false);
140 
141  if (need_communication) {
142  GDT::SharedDOFDataHandle<SpaceType, BoolVector> data_handle(space_, sharedDOF, false);
143  view.communicate(data_handle, _all_all_interface, Dune::ForwardCommunication);
144  }
145 
146  // Count shared dofs that we own
147  using GlobalIndex = typename CommunicatorType::ParallelIndexSet::GlobalIndex;
148  GlobalIndex count{0};
149 
150  for (size_t i = 0; i < sharedDOF.size(); ++i) {
151  if (owned_for_amg(i) && sharedDOF[i])
152  ++count;
153  }
154 
155  // Communicate per-rank count of owned and shared DOFs to all processes.
156  std::vector<GlobalIndex> counts(view.comm().size());
157  view.comm().allgather(&count, 1, &(counts[0]));
158 
159  // Compute start index start_p = \sum_{i=0}^{i<p} counts_i
160  GlobalIndex start = std::accumulate(counts.begin(), counts.begin() + rank_, GlobalIndex(0));
161 
162  using GlobalIndexVector = CommunicatedVector<GlobalIndex>;
163  GlobalIndexVector scalarIndices(vector_size, std::numeric_limits<GlobalIndex>::max());
164 
165  for (size_t i = 0; i < sharedDOF.size(); ++i) {
166  if (owned_for_amg(i) && sharedDOF[i]) {
167  scalarIndices[i] = start;
168  ++start;
169  }
170  }
171 
172  // Publish global indices for the shared DOFS to other processors.
173  if (need_communication) {
174  MinDataHandle<SpaceType, GlobalIndexVector> data_handle(space_, scalarIndices);
175  view.communicate(data_handle, _interiorBorder_all_interface, Dune::ForwardCommunication);
176  }
177 
178  // Setup the index set
179  c.indexSet().beginResize();
180  for (size_t i = 0; i < scalarIndices.size(); ++i) {
181  Dune::OwnerOverlapCopyAttributeSet::AttributeSet attr;
182  if (scalarIndices[i] != std::numeric_limits<GlobalIndex>::max()) {
183  // global index exist in index set
184  if (owned_for_amg(i)) {
185  // This dof is managed by us.
186  attr = Dune::OwnerOverlapCopyAttributeSet::owner;
187  } else {
188  attr = Dune::OwnerOverlapCopyAttributeSet::copy;
189  }
190  c.indexSet().add(scalarIndices[i], typename CommunicatorType::ParallelIndexSet::LocalIndex(i, attr));
191  }
192  }
193  c.indexSet().endResize();
194 
195  // Compute neighbors using communication
196  std::set<int> neighbors;
197 
198  if (need_communication) {
199  SpaceNeighborDataHandle<SpaceType, int> data_handle(space_, rank_, neighbors);
200  view.communicate(data_handle, _all_all_interface, Dune::ForwardCommunication);
201  }
202 
203  c.remoteIndices().setNeighbours(neighbors);
204  c.remoteIndices().template rebuild<false>();
205 }
