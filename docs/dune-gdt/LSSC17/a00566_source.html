103 private:
104  template <class SourceType, class RangeType>
105  void apply_dg(const SourceType& source, RangeType& range) const
106  {
107  range.vector() *= 0.;
108  // data structures we need
109  // * a map from a global vertex index to global DoF indices
110  // given a vertex, one obtains a set of all global DoF ids, which are associated with this vertex
111  std::map<size_t, std::set<size_t>> global_vertex_id_to_global_DoF_id_map;
112  // * a map from a global DoF index to the global index of its associated vertex
113  std::vector<size_t> global_DoF_id_to_global_vertex_id_map(source.space().mapper().size());
114  // * a set to hold the global id of all boundary vertices
115  std::set<size_t> boundary_vertices;
116 
117  const auto entity_it_end = grid_layer_.template end<0>();
118  // walk the grid to create the maps explained above and to find the boundary vertices
119  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
120  const auto& entity = *entity_it;
121  const size_t num_vertices = entity.subEntities(dimDomain);
122  const auto basis = source.space().base_function_set(entity);
123  if (basis.size() != num_vertices)
124  DUNE_THROW(Dune::XT::Common::Exceptions::internal_error, "basis.size() = " << basis.size());
125 
126  // loop over all vertices of the entitity, to find their associated global DoF indices
127  for (size_t local_vertex_id = 0; local_vertex_id < num_vertices; ++local_vertex_id) {
128  const auto vertex = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(local_vertex_id));
129  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
130  const auto vertex_center = vertex.geometry().center();
131  // find the local basis function which corresponds to this vertex
132  const auto basis_values = basis.evaluate(entity.geometry().local(vertex_center));
133  if (basis_values.size() != num_vertices)
134  DUNE_THROW(Dune::XT::Common::Exceptions::internal_error, "basis_values.size() = " << basis_values.size());
135  size_t ones = 0;
136  size_t zeros = 0;
137  size_t failures = 0;
138  size_t local_DoF_index = 0;
139  for (size_t ii = 0; ii < basis.size(); ++ii) {
140  if (std::abs(basis_values[ii][0] - 1.0) < 1e-14) {
141  local_DoF_index = ii;
142  ++ones;
143  } else if (std::abs(basis_values[ii][0] - 0.0) < 1e-14)
144  ++zeros;
145  else
146  ++failures;
147  }
148  if (ones != 1 || zeros != (basis.size() - 1) || failures > 0) {
149  std::stringstream ss;
150  ss << "ones = " << ones << ", zeros = " << zeros << ", failures = " << failures
151  << ", num_vertices = " << num_vertices << ", entity " << grid_layer_.indexSet().index(entity)
152  << ", vertex " << local_vertex_id << ": [ " << vertex_center << "], ";
153  XT::Common::print(basis_values, "basis_values", ss);
154  DUNE_THROW(Dune::XT::Common::Exceptions::internal_error, ss.str());
155  }
156  // now we know that the local DoF index of this vertex is ii
157  const size_t global_DoF_index = source.space().mapper().mapToGlobal(entity, local_DoF_index);
158  global_DoF_id_to_global_vertex_id_map[global_DoF_index] = global_vertex_id;
159  global_vertex_id_to_global_DoF_id_map[global_vertex_id].insert(global_DoF_index);
160  } // loop over all vertices
161 
162  if (zero_boundary_) {
163  // in order to determine the boundary vertices, we need to
164  // loop over all intersections
165  const auto intersectionEndIt = grid_layer_.iend(entity);
166  for (auto intersectionIt = grid_layer_.ibegin(entity); intersectionIt != intersectionEndIt; ++intersectionIt) {
167  const auto& intersection = *intersectionIt;
168  if (intersection.boundary() && !intersection.neighbor()) {
169  const auto& intersection_geometry = intersection.geometry();
170  for (auto local_intersection_corner_id : Dune::XT::Common::value_range(intersection_geometry.corners())) {
171  const auto global_intersection_corner = intersection_geometry.corner(local_intersection_corner_id);
172  // now, we need to find the entity's vertex this intersection's corner point equals to, so we
173  // loop over all vertices of the entity
174  for (size_t local_vertex_id = 0; local_vertex_id < num_vertices; ++local_vertex_id) {
175  const auto vertex = entity.template subEntity<dimDomain>(boost::numeric_cast<int>(local_vertex_id));
176  const auto global_vertex_id = grid_layer_.indexSet().index(vertex);
177  const auto vertex_center = vertex.geometry().center();
178  if (XT::Common::FloatCmp::eq(global_intersection_corner, vertex_center))
179  boundary_vertices.insert(global_vertex_id);
180  } // loop over all vertices of the entity
181  } // loop over all intersection corners
182  } // if (intersection.boundary() && !intersection.neighbor())
183  } // loop over all intersections
184  } // if(zero_boundary)
185  } // walk the grid for the first time
186 
187  // walk the grid for the second time
188  for (auto entity_it = grid_layer_.template begin<0>(); entity_it != entity_it_end; ++entity_it) {
189  const auto& entity = *entity_it;
190  const auto num_vertices = entity.subEntities(dimDomain);
191  // get the local functions
192  const auto local_source = source.local_discrete_function(entity);
193  const auto& local_source_DoF_vector = local_source->vector();
194 
195  // * loop over all local DoFs
196  for (size_t local_DoF_id = 0; local_DoF_id < num_vertices; ++local_DoF_id) {
197  const size_t global_DoF_index = source.space().mapper().mapToGlobal(entity, local_DoF_id);
198  const size_t global_vertex_id = global_DoF_id_to_global_vertex_id_map[global_DoF_index];
199  // if we are on the domain boundary
200  if (zero_boundary_ && boundary_vertices.count(global_vertex_id)) {
201  // set the dof to zero (we have dirichlet zero)
202  range.vector().set_entry(global_DoF_index, FieldType(0));
203  } else {
204  // do the oswald projection
205  const size_t num_DoFS_per_vertex = global_vertex_id_to_global_DoF_id_map[global_vertex_id].size();
206  // * get the source DoF
207  const FieldType source_DoF_value = local_source_DoF_vector.get(local_DoF_id);
208  // * and add it to all target DoFs
209  for (size_t target_global_DoF_id : global_vertex_id_to_global_DoF_id_map[global_vertex_id])
210  range.vector().add_to_entry(target_global_DoF_id, source_DoF_value / num_DoFS_per_vertex);
211  } // if (boundary_vertices.find(global_vertex_id))
212  } // loop over all local DoFs
213  } // walk the grid for the second time
214  } // ... apply_dg(...)
