84 template <class VectorType>
85 struct ChooseLimiter<SlopeLimiters::minmod, VectorType>
86 {
87  static VectorType
88  limit(const VectorType& slope_left, const VectorType& slope_right, const VectorType& /*centered_slope*/)
89  {
90  VectorType ret;
91  for (size_t ii = 0; ii < slope_left.size(); ++ii) {
92  const auto slope_left_abs = std::abs(slope_left[ii]);
93  const auto slope_right_abs = std::abs(slope_right[ii]);
94  if (slope_left_abs < slope_right_abs && slope_left[ii] * slope_right[ii] > 0)
95  ret[ii] = slope_left[ii];
96  else if (Dune::XT::Common::FloatCmp::ge(slope_left_abs, slope_right_abs) && slope_left[ii] * slope_right[ii] > 0)
97  ret[ii] = slope_right[ii];
98  else
99  ret[ii] = 0.0;
100  }
101  return ret;
102  }
103 };

168  template <class SourceType, class IntersectionType, class SpaceType, class VectorType>
169  void apply(const SourceType& source,
170  const IntersectionType& intersection,
171  LocalDiscreteFunction<SpaceType, VectorType>& local_range_entity,
172  LocalDiscreteFunction<SpaceType, VectorType>& local_range_neighbor) const
173  {
174  const auto entity = intersection.inside();
175  const auto neighbor = intersection.outside();
176  const auto local_source_entity = source.local_function(entity);
177  const auto local_source_neighbor = source.local_function(neighbor);
178  const auto geometry_intersection = intersection.geometry();
179  const auto local_functions_tuple_entity = numerical_flux_.local_functions(entity);
180  const auto local_functions_tuple_neighbor = numerical_flux_.local_functions(neighbor);
181  const Dune::XT::Common::FieldVector<typename LocalDiscreteFunction<SpaceType, VectorType>::DomainFieldType,
182  LocalDiscreteFunction<SpaceType, VectorType>::dimRange>
183  result = numerical_flux_.evaluate(local_functions_tuple_entity,
184  local_functions_tuple_neighbor,
185  *local_source_entity,
186  *local_source_neighbor,
187  intersection,
188  geometry_intersection.local(geometry_intersection.center()));
189  local_range_entity.vector().add(result * (1.0 / entity.geometry().volume()));
190  local_range_neighbor.vector().add(result * (-1.0 / neighbor.geometry().volume()));
191  }
192 
193 private:
194  const NumericalFluxType numerical_flux_;
195 };

282  template <class SourceType, class RangeSpaceType, class VectorType>
283  void apply(const SourceType& source, LocalDiscreteFunction<RangeSpaceType, VectorType>& local_range) const
284  {
285  const auto& entity = local_range.entity();
286  const auto& grid_layer = local_range.space().grid_layer();
287  // get reconstruction vector and mapper
288  auto& reconstruction_vector = local_range.vector();
289  const auto& reconstruction_mapper = local_range.space().mapper();
290  const auto entity_center = entity.geometry().center();
291  // walk over intersections to get values of discrete function on left and right neighbor entity
292  typename SourceType::RangeType u_left, u_right;
293  typename SourceType::RangeType u_entity =
294  source.local_discrete_function(entity)->evaluate(entity.geometry().local(entity_center));
295  const auto i_it_end = grid_layer.iend(entity);
296  for (auto i_it = grid_layer.ibegin(entity); i_it != i_it_end; ++i_it) {
297  const auto& intersection = *i_it;
298  if (intersection.neighbor()) {
299  const auto neighbor = intersection.outside();
300  const auto neighbor_center = neighbor.geometry().center();
301  const bool boundary = intersection.boundary();
302  if ((neighbor_center[0] < entity_center[0] && !boundary) || (neighbor_center[0] > entity_center[0] && boundary))
303  u_left = source.local_discrete_function(neighbor)->evaluate(neighbor.geometry().local(neighbor_center));
304  else
305  u_right = source.local_discrete_function(neighbor)->evaluate(neighbor.geometry().local(neighbor_center));
306  } else {
307  if (intersection.geometry().center()[0] < entity_center[0])
308  u_left = boundary_values_.local_function(entity)->evaluate(intersection.geometryInInside().center());
309  else
310  u_right = boundary_values_.local_function(entity)->evaluate(intersection.geometryInInside().center());
311  }
312  }
313 
314  // diagonalize the system of equations from u_t + A*u_x = 0 to w_t + D*w_x = 0 where D = R^(-1)*A*R, w = R^(-1)*u
315  // and R matrix of eigenvectors of A
316  const XTFieldVectorType w_left(eigenvectors_inverse_ * u_left);
317  const XTFieldVectorType w_right(eigenvectors_inverse_ * u_right);
318  const XTFieldVectorType w_entity(eigenvectors_inverse_ * u_entity);
319 
320  const XTFieldVectorType w_slope_left = w_entity - w_left;
321  const XTFieldVectorType w_slope_right = w_right - w_entity;
322  const XTFieldVectorType w_centered_slope = w_right * RangeFieldType(0.5) - w_left * RangeFieldType(0.5);
323  const XTFieldVectorType w_slope =
324  internal::ChooseLimiter<slope_limiter, XTFieldVectorType>::limit(w_slope_left, w_slope_right, w_centered_slope);
325  const XTFieldVectorType half_w_slope = w_slope * RangeFieldType(0.5);
326  const XTFieldVectorType w_reconstructed_left = w_entity - half_w_slope;
327  const XTFieldVectorType w_reconstructed_right = w_entity + half_w_slope;
328 
329  // convert back to u variable
330  const XTFieldVectorType u_reconstructed_left(eigenvectors_ * w_reconstructed_left);
331  const XTFieldVectorType u_reconstructed_right(eigenvectors_ * w_reconstructed_right);
332 
333  for (size_t factor_index = 0; factor_index < dimRange; ++factor_index) {
334  // set values on dofs, dof with local index 0 for each factor space corresponds to basis function 1 - x, local
335  // index 1 to x
336  reconstruction_vector.set(reconstruction_mapper.mapToLocal(factor_index, entity, 0),
337  u_reconstructed_left[factor_index]);
338  reconstruction_vector.set(reconstruction_mapper.mapToLocal(factor_index, entity, 1),
339  u_reconstructed_right[factor_index]);
340  }
341  } // void apply(...)
342 
343 private:
344  const MatrixType& eigenvectors_;
345  const MatrixType& eigenvectors_inverse_;
346  const BoundaryValueFunctionImp& boundary_values_;
347 };
