162  virtual TimeFieldType solve(const TimeFieldType t_end,
163  const TimeFieldType initial_dt,
164  const size_t num_save_steps,
165  const bool save_solution,
166  const bool output_progress,
167  const bool visualize,
168  const std::string filename_prefix,
169  SolutionType& sol)
170  {
171  TimeFieldType dt = initial_dt;
172  TimeFieldType t = current_time();
173  assert(Dune::XT::Common::FloatCmp::ge(t_end - t, 0.0));
174  size_t time_step_counter = 0;
175 
176  const TimeFieldType save_interval = (t_end - t) / num_save_steps;
177  TimeFieldType next_save_time = t + save_interval > t_end ? t_end : t + save_interval;
178  size_t save_step_counter = 1;
179 
180  // save/visualize initial solution
181  if (save_solution)
182  sol.insert(std::make_pair(t, current_solution()));
183  if (visualize)
184  current_solution().visualize(filename_prefix, Dune::XT::Common::to_string(0));
185 
186  while (Dune::XT::Common::FloatCmp::lt(t, t_end)) {
187  TimeFieldType max_dt = dt;
188  // match saving times and t_end exactly
189  if (Dune::XT::Common::FloatCmp::gt(t + dt, t_end))
190  max_dt = t_end - t;
191  if (Dune::XT::Common::FloatCmp::gt(t + dt, next_save_time) && num_save_steps != size_t(-1))
192  max_dt = std::min(next_save_time - t, max_dt);
193 
194  // do a timestep
195  dt = step(dt, max_dt);
196  t = current_time();
197 
198  // augment time step counter
199  ++time_step_counter;
200 
201  // check if data should be written in this timestep (and write)
202  if (Dune::XT::Common::FloatCmp::ge(t, next_save_time) || num_save_steps == size_t(-1)) {
203  if (save_solution)
204  sol.insert(sol.end(), std::make_pair(t, current_solution()));
205  if (visualize)
206  current_solution().visualize(filename_prefix, Dune::XT::Common::to_string(save_step_counter));
207  if (output_progress)
208  std::cout << "time step " << time_step_counter << " done, time =" << t << ", current dt= " << dt << std::endl;
209  next_save_time += save_interval;
210  ++save_step_counter;
211  }
212  } // while (t < t_end)
213 
214  return dt;
215  } // ... solve(...)
