246  {

248  RealIntersectionIteratorType outside_i_it = real_grid_view_.ibegin(outside_);
249  const RealIntersectionIteratorType outside_i_it_end = real_grid_view_.iend(outside_);
250  // walk over outside intersections and find an intersection on the boundary that differs only in one coordinate
251  for (; outside_i_it != outside_i_it_end; ++outside_i_it) {
252  const BaseType& curr_outside_intersection = *outside_i_it;
253  if (curr_outside_intersection.boundary()) {

263  }
264  }

535 private:
536  // function that is used in the constructor to create the index map and the counter for the PeriodicIndexSet
537  template <int codim>
538  static void create_index_map(const DomainType& lower_left,
539  const DomainType& upper_right,
540  const std::bitset<dimDomain>& periodic_directions,
541  std::vector<IndexMapType>& index_maps,
542  ThisType* this_ptr,
543  const BaseType& real_grid_view,
544  std::vector<IndexType>& entity_counts,
545  std::map<Dune::GeometryType, size_t>& type_counts,
546  std::vector<std::set<IndexType>>& entities_to_skip_vector,
547  std::vector<std::vector<IndexType>>& new_indices_vector)
548  {
549  EntityInlevelSearch<BaseType, codim> entity_search_codim(*this_ptr);
550  IndexMapType index_map;
551  std::map<IndexType, size_t> entity_to_vector_index_map;
552  std::vector<DomainType> periodic_coords_vector;
553  const auto& real_index_set = real_grid_view.indexSet();
554  if (!use_less_memory)
555  new_indices_vector[codim].resize(real_index_set.size(codim));
556  auto& new_indices = new_indices_vector[codim];
557  IndexType num_codim_entities = 0;
558  std::set<IndexType> free_indices;
559  std::set<IndexType> entities_to_skip;
560  for (const auto& entity : entities(*this_ptr, Dune::Codim<codim>())) {
561  // increase codim counter
562  ++num_codim_entities;
563  // increase GeometryType counter
564  const auto geometry_type = entity.type();
565  if (type_counts.count(geometry_type))
566  ++(type_counts.at(geometry_type));
567  else
568  type_counts.insert(std::make_pair(geometry_type, size_t(1)));
569  // check if entity is on a periodic boundary and, if it is, replace index by indices of periodic entity
570  auto periodic_coords = entity.geometry().center();
571  std::size_t num_upper_right_coords = 0;
572  for (std::size_t ii = 0; ii < dimDomain; ++ii) {
573  if (periodic_directions[ii]) {
574  if (XT::Common::FloatCmp::eq(periodic_coords[ii], upper_right[ii])) {
575  ++num_upper_right_coords;
576  periodic_coords[ii] = lower_left[ii];
577  }
578  }
579  }
580  if (num_upper_right_coords > 0) { // find periodic adjacent entity
581  // dont count this entity as it is identified with the single periodic equivalent entity
582  --num_codim_entities;
583  if (type_counts.count(geometry_type))
584  --(type_counts.at(geometry_type));
585  else
586  type_counts.insert(std::make_pair(geometry_type, size_t(-1)));
587  free_indices.insert(real_index_set.index(entity));
588  entities_to_skip.insert(real_index_set.index(entity));
589  periodic_coords_vector.push_back(periodic_coords);
590  entity_to_vector_index_map.insert(
591  std::make_pair(real_index_set.index(entity), periodic_coords_vector.size() - 1));
592  }
593  } // walk entities in a given codimension
594 
595  entity_counts[codim] = num_codim_entities;
596  entities_to_skip_vector[codim] = entities_to_skip;
597 
598  // find periodic entities
599  const auto periodic_entity_ptrs = entity_search_codim(periodic_coords_vector);
600 
601  // assign new indices to the entities
602  IndexType current_new_index = 0;
603 
604  // to save memory, only entities with an index greater than the number of entities in this codim will get a new
605  // index
606  if (use_less_memory) {
607  for (const auto& index : free_indices)
608  if (index >= num_codim_entities)
609  free_indices.erase(index);
610  }
611 
612  // assign new indices to entities that are not replaced by their periodic equivalent entity
613  for (const auto& entity : entities(*this_ptr, Dune::Codim<codim>())) {
614  const auto old_index = real_index_set.index(entity);
615  if (!entity_to_vector_index_map.count(old_index)) {
616  if (!use_less_memory) {
617  new_indices[old_index] = current_new_index;
618  ++current_new_index;
619  } else {
620  if (old_index >= num_codim_entities) {
621  assert(free_indices.begin() != free_indices.end());
622  auto free_index = *free_indices.begin();
623  index_map.insert(std::make_pair(old_index, free_index));
624  free_indices.erase(free_index);
625  }
626  }
627  }
628  }
629 
630  // assign index of periodic equivalent entity to entities that are replaced
631  for (const auto& pair : entity_to_vector_index_map) {
632  assert(periodic_entity_ptrs.size() > pair.second);
633  const auto& periodic_entity_ptr = periodic_entity_ptrs.at(pair.second);
634  if (periodic_entity_ptr == nullptr)
635  DUNE_THROW(Dune::InvalidStateException, "Could not find periodic neighbor entity");
636  auto index_of_periodic_entity = real_index_set.index(*periodic_entity_ptr);
637  if (!use_less_memory)
638  new_indices[pair.first] = new_indices[index_of_periodic_entity];
639  else
640  index_map.insert(std::make_pair(pair.first,
641  index_map.count(index_of_periodic_entity)
642  ? index_map.at(index_of_periodic_entity)
643  : index_of_periodic_entity));
644  }
645  index_maps[codim] = index_map;
646  }

692  const std::bitset<dimDomain>& /*periodic_directions*/,
693  std::vector<IndexMapType>& /*index_maps*/,

695  const BaseType& /*real_grid_view*/,

704 public:
705  PeriodicGridViewImp(const BaseType& real_grid_view, const std::bitset<dimDomain> periodic_directions)
706  : BaseType(real_grid_view)
707  , entity_to_intersection_map_map_(std::make_shared<std::map<IndexType, IntersectionMapType>>())
708  , index_maps_(std::make_shared<std::vector<IndexMapType>>(dimDomain + 1))
709  , periodic_directions_(periodic_directions)
710  , entity_counts_(std::make_shared<std::vector<IndexType>>(dimDomain + 1))
711  , type_counts_(std::make_shared<std::map<Dune::GeometryType, size_t>>())
712  , entities_to_skip_vector_(std::make_shared<std::vector<std::set<IndexType>>>(dimDomain + 1))
713  , new_indices_vector_(std::make_shared<std::vector<std::vector<IndexType>>>(dimDomain + 1))
714  , real_index_set_(BaseType::indexSet())
715  {
716  // find lower left and upper right corner of the grid
717  auto entity_it = BaseType::template begin<0>();
718  nonperiodic_pair_ = std::make_pair(bool(false), EntityType(*entity_it));
719  DomainType lower_left = entity_it->geometry().center();
720  DomainType upper_right = lower_left;
721  for (const auto& entity : Dune::elements(*this)) {
722  if (entity.hasBoundaryIntersections()) {
723  const auto i_it_end = BaseType::iend(entity);
724  for (auto i_it = BaseType::ibegin(entity); i_it != i_it_end; ++i_it) {
725  const RealIntersectionType& intersection = *i_it;
726  const auto intersection_coords = intersection.geometry().center();
727  for (std::size_t ii = 0; ii < dimDomain; ++ii) {
728  if (intersection_coords[ii] > upper_right[ii])
729  upper_right[ii] = intersection_coords[ii];
730  if (intersection_coords[ii] < lower_left[ii])
731  lower_left[ii] = intersection_coords[ii];
732  }
733  }
734  }
735  }
736 
737  /* walk the grid and create a map that maps each entity(index) on the boundary to a map that maps every
738  local intersection index to a std::pair< bool, EntityType >, where the bool component of the pair
739  indicates whether the intersection is on a periodic boundary and the second component is the periodic neighbor
740  entity if the first component is true. If the first component is false, the Entity is not meant to be used. */
741  EntityInlevelSearch<BaseType> entity_search(*this);
742  DomainType periodic_neighbor_coords;
743  IntersectionMapType intersection_neighbor_map;
744  std::vector<DomainType> periodic_neighbor_coords_vector;
745  /* we don't want to do the entitysearch on each entity separately, so we collect the coordinates in a vector to do
746  * the search after the gridwalk. This map maps the entity indices to a map mapping local intersection indices to
747  * the index of the corresponding coordinate in the periodic_neighbor_coords_vector. */
748  std::map<IndexType, std::map<IntersectionIndexType, size_t>> entity_to_intersection_to_vector_index_map;
749  std::map<IntersectionIndexType, size_t> intersection_to_vector_index_map;
750  for (const auto& entity : Dune::elements(*this)) {
751  // count entities per geometry type for the PeriodicIndexSet
752  const auto geometry_type = entity.type();
753  if (type_counts_->count(geometry_type))
754  ++(type_counts_->at(geometry_type));
755  else
756  type_counts_->insert(std::make_pair(geometry_type, size_t(1)));
757  if (entity.hasBoundaryIntersections()) {
758  intersection_neighbor_map.clear();
759  intersection_to_vector_index_map.clear();
760  const auto i_it_end = BaseType::iend(entity);
761  for (auto i_it = BaseType::ibegin(entity); i_it != i_it_end; ++i_it) {
762  const RealIntersectionType& intersection = *i_it;
763  const IntersectionIndexType index_in_inside = intersection.indexInInside();
764  bool is_periodic = false;
765  if (intersection.boundary()) {
766  periodic_neighbor_coords = intersection.geometry().center();
767  size_t num_boundary_coords = 0;
768  for (std::size_t ii = 0; ii < dimDomain; ++ii) {
769  if (periodic_directions_[ii]) {
770  if (XT::Common::FloatCmp::eq(periodic_neighbor_coords[ii], lower_left[ii])) {
771  is_periodic = true;
772  periodic_neighbor_coords[ii] =
773  upper_right[ii] - 1.0 / 100.0 * (entity.geometry().center()[ii] - lower_left[ii]);
774  ++num_boundary_coords;
775  } else if (XT::Common::FloatCmp::eq(periodic_neighbor_coords[ii], upper_right[ii])) {
776  is_periodic = true;
777  periodic_neighbor_coords[ii] =
778  lower_left[ii] + 1.0 / 100.0 * (upper_right[ii] - entity.geometry().center()[ii]);
779  ++num_boundary_coords;
780  }
781  }
782  }
783  if (is_periodic) {
784  assert(num_boundary_coords == 1);
785  periodic_neighbor_coords_vector.push_back(periodic_neighbor_coords);
786  intersection_to_vector_index_map.insert(
787  std::make_pair(index_in_inside, periodic_neighbor_coords_vector.size() - 1));
788  } else {
789  intersection_neighbor_map.insert(
790  std::make_pair(index_in_inside, std::make_pair(is_periodic, EntityType(entity))));
791  }
792  } else {
793  intersection_neighbor_map.insert(
794  std::make_pair(index_in_inside, std::make_pair(bool(false), EntityType(entity))));
795  }
796  }
797  entity_to_intersection_to_vector_index_map.insert(
798  std::make_pair(real_index_set_.index(entity), intersection_to_vector_index_map));
799  entity_to_intersection_map_map_->insert(
800  std::make_pair(real_index_set_.index(entity), intersection_neighbor_map));
801  }
802  } // walk Entities
803 
804  // search the periodic neighbor entities
805  const auto outside_ptrs = entity_search(periodic_neighbor_coords_vector);
806 
807  for (auto& pair : entity_to_intersection_to_vector_index_map) {
808  for (auto& intersection_to_vector : pair.second) {
809  assert(outside_ptrs.size() > intersection_to_vector.second);
810  const auto& outside_ptr = outside_ptrs.at(intersection_to_vector.second);
811  if (outside_ptr == nullptr)
812  DUNE_THROW(Dune::InvalidStateException, "Could not find periodic neighbor entity");
813  assert(entity_to_intersection_map_map_->count(pair.first));
814  entity_to_intersection_map_map_->at(pair.first)
815  .insert(std::make_pair(intersection_to_vector.first, std::make_pair(bool(true), *outside_ptr)));
816  }
817  }
818 
819  /* walk the grid for each codimension from 1 to dimDomain and create a map mapping indices from entitys of that
820  * codimension on a periodic boundary to the index of the corresponding periodic equivalent entity that has the
821  * most coordinates in common with the lower left corner of the grid */
822  (*entity_counts_)[0] = real_grid_view.indexSet().size(0);
823  static_for_loop_for_index_maps<1, dimDomain + 1>()(lower_left,
824  upper_right,
825  periodic_directions_,
826  *index_maps_,
827  this,
828  real_grid_view,
829  *entity_counts_,
830  *type_counts_,
831  *entities_to_skip_vector_,
832  *new_indices_vector_);
833  // create index_set
834  index_set_ =
835  std::make_shared<IndexSet>(real_index_set_, *index_maps_, *entity_counts_, *type_counts_, *new_indices_vector_);
836  } // constructor PeriodicGridViewImp(...)
