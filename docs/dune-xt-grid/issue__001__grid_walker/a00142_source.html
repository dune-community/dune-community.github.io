333  template <class EntityRange>
334  void walk_range(const EntityRange& entity_range)
335  {
336 #ifdef __INTEL_COMPILER
337  const auto it_end = entity_range.end();
338  for (auto it = entity_range.begin(); it != it_end; ++it) {
339  const EntityType& entity = *it;
340 #else
341  for (const EntityType& entity : entity_range) {
342 #endif
343  // apply codim0 functors
344  apply_local(entity);
345 
346  // only walk the intersections, if there are codim1 functors present
347  if (codim1_functors_.size() > 0) {
348  // walk the intersections
349  for (auto&& intersection : intersections(this->grid_view_, entity)) {
350 
351  // apply codim1 functors
352  if (intersection.neighbor()) {
353  const auto neighbor = intersection.outside();
354  apply_local(intersection, entity, neighbor);
355  } else
356  apply_local(intersection, entity, entity);
357 
358  } // walk the intersections
359  } // only walk the intersections, if there are codim1 functors present
360  }
