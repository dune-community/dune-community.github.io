87 template <class G, class I, class D>
88 typename std::enable_if<Dune::Intersection<G, I>::dimension == 2, bool>::type
89 contains(const Dune::Intersection<G, I>& intersection,
90  const Dune::FieldVector<D, 2>& global_point,
91  const D& tolerance = Common::FloatCmp::DefaultEpsilon<D>::value())
92 {
93  const auto& geometry = intersection.geometry();
94  // get the global coordinates of the intersections corners
95  assert(geometry.corners() == 2);
96  const auto corner_0 = geometry.corner(0);
97  const auto corner_1 = geometry.corner(1);
98  // A line is given by $y = a*x + b$. Computing a and b fails for certain intersections (for instance those
99  // parallel to the y axis). So in order to check if the point is on the line between the corners we consider the
100  // vectors pointing from the point to each corner. If those are not orthogonal to the intersections normal, the point
101  // cannot lie on the line between the two corners.
102  const auto normal = intersection.centerUnitOuterNormal();
103  for (auto vector : {corner_0 - global_point, corner_1 - global_point})
104  if (vector * normal > tolerance)
105  return false;
106  // Now that we know the point is on the line, check if it is outside the bounding box of the two corners.
107  if (Common::FloatCmp::lt(global_point[0], std::min(corner_0[0], corner_1[0]), tolerance)
108  || Common::FloatCmp::gt(global_point[0], std::max(corner_0[0], corner_1[0]), tolerance))
109  return false;
110  if (Common::FloatCmp::lt(global_point[1], std::min(corner_0[1], corner_1[1]), tolerance)
111  || Common::FloatCmp::gt(global_point[1], std::max(corner_0[1], corner_1[1]), tolerance))
112  return false;
113  // At this point we cannot reject the assumption that the point lies on the line between the two corners.
114  return true;
115 } // ... contains(...)
