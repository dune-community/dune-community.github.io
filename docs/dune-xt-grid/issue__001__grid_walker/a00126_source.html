38 struct Statistics
39 {
40  size_t numberOfEntities;
41  size_t numberOfIntersections;
42  size_t numberOfInnerIntersections;
43  size_t numberOfBoundaryIntersections;
44  double maxGridWidth;
45  template <class GridViewType>
46  Statistics(const GridViewType& gridView)
47  : numberOfEntities(gridView.size(0))
48  , numberOfIntersections(0)
49  , numberOfInnerIntersections(0)
50  , numberOfBoundaryIntersections(0)
51  , maxGridWidth(0)
52  {
53  for (auto&& entity : elements(gridView)) {
54  for (auto&& intIt : intersections(gridView, entity)) {
55  ++numberOfIntersections;
56  maxGridWidth = std::max(intIt.geometry().volume(), maxGridWidth);
57  // if we are inside the grid
58  numberOfInnerIntersections += (intIt.neighbor() && !intIt.boundary());
59  // if we are on the boundary of the grid
60  numberOfBoundaryIntersections += (!intIt.neighbor() && intIt.boundary());
61  }
62  }
63  }
64 };

126  virtual void apply_local(const EntityType& ent)
127  {
128  const auto& geo = ent.geometry();
129  entity_volume_(geo.volume());
130  entity_width_(entity_diameter(ent));
131  for (auto i : Common::value_range(geo.corners())) {
132  const auto& corner(geo.corner(i));
133  for (size_t k = 0; k < GridType::dimensionworld; ++k)
