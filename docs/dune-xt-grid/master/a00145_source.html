80  static GridProvider<GridType> create(const std::string& filename_prefix)
81  {
82  std::ostream& out = Dune::XT::Common::Logger().devnull();
83 
84  // set up the grid factory
85  GridFactory<GridType> factory;
86 
87  // read the vertices
88  const std::string vertexFileName = filename_prefix + ".vrt";
89  std::ifstream vertexFile(vertexFileName);
90  if (!vertexFile)
91  DUNE_THROW(Dune::IOError, "Could not open " << vertexFileName);
92  std::string line;
93  if (!std::getline(vertexFile, line))
94  DUNE_THROW(Dune::IOError, "File " << vertexFileName << " is too short!");
95  if (!(line == "PROSTAR_VERTEX"))
96  DUNE_THROW(Dune::IOError,
97  "First line of File " << vertexFileName << " (" << line << "is not equal to 'PROSTAR_VERTEX' !");
98  if (!std::getline(vertexFile, line))
99  DUNE_THROW(Dune::IOError, "File " << vertexFileName << " is too short!");
100  size_t numberOfVertices = 0;
101  Dune::FieldVector<double, dimDomain> position;
102  out << "Reading " << vertexFileName << " ... " << std::flush;
103  while (std::getline(vertexFile, line)) {
104  numberOfVertices++;
105  const std::vector<double> items = Dune::XT::Common::tokenize<double>(line, " ");
106  if (items.size() != dimDomain + 1)
107  DUNE_THROW(Dune::IOError, "Error: " << items.size() << " = items.size() != dim + 1 = " << dimDomain + 1 << "!");
108  for (size_t ii = 0; ii < dimDomain; ++ii)
109  position[ii] = items[ii + 1];
110  factory.insertVertex(position);
111  }
112  out << "done: " << numberOfVertices << " vertices read." << std::endl;
113 
114  // read the elements
115  std::string elementFileName = filename_prefix + ".cel";
116  std::ifstream elementFile(elementFileName);
117  if (!elementFile)
118  DUNE_THROW(Dune::IOError, "Could not open " << elementFileName);
119  if (!std::getline(elementFile, line))
120  DUNE_THROW(Dune::IOError, "File " << elementFileName << " is too short!");
121  if (!(line == "PROSTAR_CELL"))
122  DUNE_THROW(Dune::IOError,
123  "First line of File " << elementFileName << " (" << line << "is not equal to 'PROSTAR_CELL' !");
124  if (!std::getline(elementFile, line))
125  DUNE_THROW(Dune::IOError, "File " << elementFileName << " is too short!");
126  size_t numberOfElements = 0;
127  size_t numberOfPrisms = 0;
128  size_t numberOfCubes = 0;
129  size_t numberOfVerticesCube = pow(2, dimDomain);
130  size_t numberOfVerticesPrism = 6;
131  std::vector<unsigned int> cubeVertices(numberOfVerticesCube); // unsigned int required by the grid factory
132  std::vector<unsigned int> prismVertices(numberOfVerticesPrism);
133  std::string firstLine;
134  std::string secondLine;
135  out << "Reading " << elementFileName << " ... " << std::flush;
136  while (std::getline(elementFile, firstLine)) {
137  if (!std::getline(elementFile, secondLine))
138  DUNE_THROW(Dune::IOError,
139  "No vertex data available in file " << elementFileName << " for element " << numberOfElements + 1
140  << "!");
141  numberOfElements++;
142 
143  const auto items1 = Dune::XT::Common::tokenize<int>(firstLine, " ");
144  auto items2 = Dune::XT::Common::tokenize<int>(secondLine, " ");
145 
146  // Erase zeros at the beginning of the line
147  items2.erase(std::remove(items2.begin(), items2.end(), 0), items2.end());
148 
149  if (!(items1[0] == items2[0]) || !(numberOfElements == items2[0]))
150  DUNE_THROW(Dune::IOError, "Elementindices do not correspond!");
151 
152  if (items2.size() == numberOfVerticesCube + 1) { // cube
153  numberOfCubes++;
154  for (size_t k = 0; k < numberOfVerticesCube; k++)
155  cubeVertices[k] = items2[k + 1] - 1;
156  if (dimDomain > 1)
157  std::swap(cubeVertices[2], cubeVertices[3]);
158  if (dimDomain == 3)
159  std::swap(cubeVertices[6], cubeVertices[7]);
160  factory.insertElement(Dune::GeometryType(Dune::GeometryType::cube, dimDomain), cubeVertices);
161  } else if ((items2.size() == numberOfVerticesPrism + 1) && (dimDomain == 3)) { // prism
162  numberOfPrisms++;
163  for (size_t k = 0; k < numberOfVerticesPrism; k++)
164  prismVertices[k] = items2[k + 1] - 1;
165  factory.insertElement(Dune::GeometryType(Dune::GeometryType::prism, dimDomain), prismVertices);
166  } else // neither cube or prism
167  DUNE_THROW(Dune::IOError, "Type of element " << numberOfElements << " is not cube or prism!");
168  } // while loop
169 
170  if (!(numberOfElements == (numberOfCubes + numberOfPrisms)))
171  DUNE_THROW(Dune::IOError,
172  "Number of Elements (" << numberOfElements << ") is not equal to number of cubes (" << numberOfCubes
173  << ") and number of prisms ("
174  << numberOfPrisms
175  << ").");
176 
177  out << "done: " << numberOfElements << " elements read (" << numberOfPrisms << " prisms and " << numberOfCubes
178  << " cubes)." << std::endl;
179 
180  // finish the construction of the grid object
181  out << "Starting createGrid() ... " << std::endl;
182 
183  return GridProvider<GridType>(factory.createGrid());
184  } // ... create(...)
