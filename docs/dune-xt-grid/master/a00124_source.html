132  template <class Entity>
133  double operator()(const Entity& entity) const
134  {
135  double ret(0.0);
136  int numberOfBoundarySegments(0);
137  bool isOnBoundary = false;
138  const auto leafview = grid_.leafGridView();
139  for (auto&& intersection : intersections(leafview, entity)) {
140  if (!intersection.neighbor() && intersection.boundary()) {
141  isOnBoundary = true;
142  numberOfBoundarySegments += 1;
143  ret += double(intersection.boundaryId());
144  }
145  }
146  if (isOnBoundary) {
147  ret /= double(numberOfBoundarySegments);
148  }
149  return ret;
150  }
151  };

162  template <class Entity>
163  double operator()(const Entity& entity) const
164  {
165  typedef typename Entity::Geometry EntityGeometryType;
166 
167  typedef Dune::FieldVector<typename EntityGeometryType::ctype, EntityGeometryType::coorddimension> DomainType;
168 
169  const EntityGeometryType& geometry = entity.geometry();
170 
171  DomainType baryCenter(0.0);
172 
173  for (auto corner : Common::value_range(geometry.corners())) {
174  baryCenter += geometry.corner(corner);
175  }
176  baryCenter /= geometry.corners();
177 
178  double ret(0.0);
179 
180  if (!((baryCenter[0] < 0.0) || (baryCenter[0] > 1.0))) { // only in unit square
181  if (!((baryCenter[1] < 0.0) || (baryCenter[1] > 1.0))) {
182  ret = 1.0;
183  }
184  }
185  return ret;
186  }
187  };
