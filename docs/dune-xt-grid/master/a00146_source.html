170  createCubeGrid(const Dune::FieldVector<ctype, dim>& lowerLeft,
171  const Dune::FieldVector<ctype, dim>& upperRight,
172  const Dune::array<unsigned int, dim>& elements_in,
173  Dune::array<unsigned int, dim> overlap = default_overlap<GridType>(),
174  Dune::MPIHelper::MPICommunicator communicator = Dune::MPIHelper::getCommunicator())
175  {
176  const auto no_periodic_direction = std::bitset<dim>();
177  std::array<int, dim> elements;
178  std::copy(elements_in.begin(), elements_in.end(), elements.begin());
179  auto overlap_check = overlap;
180  overlap_check.fill(overlap[0]);
181  for (auto i : Common::value_range(1, dim))
182  if (overlap[i] != overlap[0])
183  DUNE_THROW(Dune::InvalidStateException, "YaspGrid only supports uniform overlap");
184  return std::make_shared<GridType>(lowerLeft, upperRight, elements, no_periodic_direction, overlap[0], communicator);
185  }
