34  {
35  const auto end = gv.template end<0>();
36  // GridView's size() would return the wrong count for non AllPartition Views/Iterators
37  const auto totalsize = std::distance(gv.template begin<0>(), end);
38  auto it = gv.template begin<0>();
39  entry_points_.reserve(partitions+1);
40  stride_ = totalsize / partitions;
41  overflow_ = totalsize % partitions;
42  for(Size p = 0; p < partitions; ++p)
43  {
