177  virtual void apply_local(const typename BaseType::IntersectionType& intersection,
178  const typename BaseType::EntityType& inside_entity,
179  const typename BaseType::EntityType& /*outside_entity*/)
180  {
181  const PgfCoordWrapper center(inside_entity.geometry().center());
182  const float fac = 0.16 * level_;
183  PgfCoordWrapper a(intersection.geometry().corner(0));
184  PgfCoordWrapper b(intersection.geometry().corner(1));
185  PgfCoordWrapper a_c = center - a;
186  PgfCoordWrapper b_c = center - b;
187  a_c *= fac;
188  b_c *= fac;
189  a += a_c;
190  b += b_c;
191  char buffer[150] = {'\0'};
192  std::snprintf(buffer,
193  150,
194  "\\draw[draw=%s,line width=\\gridlinewidth pt,line cap=round] (%f,%f)--(%f,%f);\n",
195  this->color_.c_str(),
196  a[0],
197  a[1],
198  b[0],
199  b[1]);
200  this->file_ << buffer;
201  this->file_.flush();
202  }
203 
204 private:
205  const int level_;
