216  void visualize_with_boundary(const Common::Configuration& boundary_info_cfg, const std::string filename) const
217  {
218  if (GridType::dimension > 3) // give us a call if you have any idea!
219  DUNE_THROW(NotImplemented, "For grids of dimension > 3!");
220  // boundary info
221  typedef XT::Grid::BoundaryInfoFactory<typename LevelGridViewType::Intersection> BoundaryInfoFactory;
222  auto boundary_info_ptr = BoundaryInfoFactory::create(boundary_info_cfg.get<std::string>("type"), boundary_info_cfg);
223  for (auto lvl : Common::value_range(max_level() + 1)) {
224  auto grid_view = level_view(lvl);
225  // vtk writer
226  Dune::VTKWriter<LevelGridViewType> vtkwriter(grid_view);
227  // codim 0 entity id
228  std::vector<double> entityId = generateEntityVisualization(grid_view);
229  vtkwriter.addCellData(entityId, "entity_id__level_" + Common::to_string(lvl));
230 #if defined(DUNE_GRID_EXPERIMENTAL_GRID_EXTENSIONS) && DUNE_GRID_EXPERIMENTAL_GRID_EXTENSIONS
231  // boundary id
232  std::vector<double> boundaryId = generateBoundaryIdVisualization(grid_view);
233  vtkwriter.addCellData(boundaryId, "boundary_id__level_" + Common::to_string(lvl));
234 #endif
235  // dirichlet values
236  std::vector<double> dirichlet = generateBoundaryVisualization(grid_view, *boundary_info_ptr, "dirichlet");
237  vtkwriter.addCellData(dirichlet, "isDirichletBoundary__level_" + Common::to_string(lvl));
238  // neumann values
239  std::vector<double> neumann = generateBoundaryVisualization(grid_view, *boundary_info_ptr, "neumann");
240  vtkwriter.addCellData(neumann, "isNeumannBoundary__level_" + Common::to_string(lvl));
241  // write

243  }

246 #if defined(DUNE_GRID_EXPERIMENTAL_GRID_EXTENSIONS) && DUNE_GRID_EXPERIMENTAL_GRID_EXTENSIONS
247  std::vector<double> generateBoundaryIdVisualization(const LevelGridViewType& gridView) const
248  {
249  std::vector<double> data(gridView.indexSet().size(0));
250  // walk the grid
251  const auto it_end = gridView.template end<0>();
252  for (auto it = gridView.template begin<0>(); it != it_end; ++it) {
253  const auto& entity = *it;
254  const auto& index = gridView.indexSet().index(entity);
255  data[index] = 0.0;
256  size_t numberOfBoundarySegments = 0;
257  bool isOnBoundary = false;
258  const auto intersectionItEnd = gridView.iend(entity);
259  for (auto intersectionIt = gridView.ibegin(entity); intersectionIt != intersectionItEnd; ++intersectionIt) {
260  if (!intersectionIt->neighbor() && intersectionIt->boundary()) {
261  isOnBoundary = true;
262  numberOfBoundarySegments += 1;
263  data[index] += double(intersectionIt->boundaryId());
264  }
265  }
266  if (isOnBoundary) {
267  data[index] /= double(numberOfBoundarySegments);
268  }
269  } // walk the grid
270  return data;
271  }

279  template <class BoundaryInfoType>
280  std::vector<double> generateBoundaryVisualization(const LevelGridViewType& gridView,
281  const BoundaryInfoType& boundaryInfo,
282  const std::string type) const
283  {
284  constexpr DirichletBoundary dirichlet_type{};
285  constexpr NeumannBoundary neumann_type{};
286  std::vector<double> data(gridView.indexSet().size(0));
287  // walk the grid
288  for (auto&& entity : elements(gridView)) {
289  const auto& index = gridView.indexSet().index(entity);
290  data[index] = 0.0;
291  for (auto intersectionIt = gridView.ibegin(entity); intersectionIt != gridView.iend(entity); ++intersectionIt) {
292  if (type == "dirichlet") {
293  if (boundaryInfo.type(*intersectionIt) == dirichlet_type)
294  data[index] = 1.0;
295  } else if (type == "neumann") {
296  if (boundaryInfo.type(*intersectionIt) == neumann_type)
297  data[index] = 1.0;
298  } else
299  DUNE_THROW(Common::Exceptions::internal_error, "Unknown type '" << type << "'!");
300  }
301  } // walk the grid
302  return data;
303  } // std::vector< double > generateBoundaryVisualization(...) const
