61  static std::unique_ptr<ThisType> create(const Common::Configuration cfg = normalbased_boundaryinfo_default_config())
62  {
63  const Common::Configuration default_cfg = normalbased_boundaryinfo_default_config();
64  // get default
65  const std::string default_type = cfg.get("default", default_cfg.get<std::string>("default"));
66  if (default_type != "dirichlet" && default_type != "neumann")
67  DUNE_THROW(Common::Exceptions::configuration_error, "Wrong default '" << default_type << "' given!");
68  const bool default_to_dirichlet = default_type == "dirichlet";
69  // get tolerance
70  const DomainFieldType tol = cfg.get("compare_tolerance", default_cfg.get<DomainFieldType>("compare_tolerance"));
71  // get dirichlet and neumann
72  std::vector<WorldType> dirichlets = getVectors(cfg, "dirichlet");
73  std::vector<WorldType> neumanns = getVectors(cfg, "neumann");
74  // return
75  return Common::make_unique<ThisType>(default_to_dirichlet, dirichlets, neumanns, tol);
76  } // ... create(...)

78  NormalBasedBoundaryInfo(const bool default_to_dirichlet = true,
79  const std::vector<WorldType> dirichlet_normals = std::vector<WorldType>(),
80  const std::vector<WorldType> neumann_normals = std::vector<WorldType>(),
81  const DomainFieldType tol = 1e-10)
82  : default_to_dirichlet_(default_to_dirichlet)
83  , dirichlet_normals_(dirichlet_normals)
84  , neumann_normals_(neumann_normals)
85  , tol_(tol)
86  {
87  // normalize
88  for (auto& normal : dirichlet_normals_)
89  normal /= normal.two_norm();
90  for (auto& normal : neumann_normals_)
91  normal /= normal.two_norm();
92  // sanity check
93  for (auto& dirichletNormal : dirichlet_normals_) {
94  if (contains(dirichletNormal, neumann_normals_))
95  DUNE_THROW(Common::Exceptions::wrong_input_given,
96  "Given normals are too close for given tolerance '" << tol << "'!");
97  }

100  virtual const BoundaryType& type(const IntersectionType& intersection) const override final
101  {
102  if (!intersection.boundary())
103  return no_boundary_;
104  const WorldType outerNormal = intersection.centerUnitOuterNormal();
105  if (contains(outerNormal, dirichlet_normals_))
106  return dirichlet_boundary_;
107  else if (contains(outerNormal, neumann_normals_))
108  return neumann_boundary_;
109  else
110  return dirichlet_boundary_;
111  } // ... type(...)

114  static std::vector<WorldType> getVectors(const Common::Configuration& config, const std::string key)
115  {
116  std::vector<WorldType> ret;
117  if (config.has_sub(key)) {
118  bool found = true;
119  size_t counter = 0;
120  while (found) {
121  const std::string localKey = key + "." + Dune::XT::Common::to_string(counter);
122  if (config.has_key(localKey))
123  ret.push_back(config.get<WorldType>(localKey, dimWorld));
124  else
125  found = false;
126  ++counter;
127  }
128  } else if (config.has_key(key))
129  ret.push_back(config.get<WorldType>(key, dimWorld));
130  return ret;
131  } // ... getVectors(...)
