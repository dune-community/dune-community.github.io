88  loop_body(const EntityType& entity, const std::size_t& type_index, const IndexType& old_index)
89  {
90  // check if entity is on a periodic boundary
91  auto periodic_coords = entity.geometry().center();
92  std::size_t num_upper_right_coords = 0;
93  for (std::size_t ii = 0; ii < dimDomain; ++ii) {
94  if (periodic_directions_[ii]) {
95  if (XT::Common::FloatCmp::eq(periodic_coords[ii], upper_right_[ii])) {
96  ++num_upper_right_coords;

98  }
99  }
100  }
101

105  ++current_new_index_[type_index];
106  // increase GeometryType counter

116  template <class EntityType, size_t cd = codim>
117  typename std::enable_if<cd == 0, void>::type
118  loop_body(const EntityType& entity, const std::size_t& type_index, const IndexType& entity_index)
119  {
120  if (entity.hasBoundaryIntersections()) {
121  IntersectionMapType intersection_neighbor_map(entity.subEntities(1));
122  const auto i_it_end = real_grid_view_.iend(entity);
123  for (auto i_it = real_grid_view_.ibegin(entity); i_it != i_it_end; ++i_it) {
124  const auto& intersection = *i_it;
125  const int index_in_inside = intersection.indexInInside();
126  if (intersection.boundary()) {
127  bool is_periodic = false;
128  auto periodic_neighbor_coords = intersection.geometry().center();
129  size_t num_boundary_coords = 0;
130  for (std::size_t ii = 0; ii < dimDomain; ++ii) {
131  if (periodic_directions_[ii]) {
132  if (XT::Common::FloatCmp::eq(periodic_neighbor_coords[ii], lower_left_[ii])) {
133  is_periodic = true;
134  periodic_neighbor_coords[ii] =
135  upper_right_[ii] - 1.0 / 100.0 * (entity.geometry().center()[ii] - lower_left_[ii]);
136  ++num_boundary_coords;
137  } else if (XT::Common::FloatCmp::eq(periodic_neighbor_coords[ii], upper_right_[ii])) {
138  is_periodic = true;
139  periodic_neighbor_coords[ii] =
140  lower_left_[ii] + 1.0 / 100.0 * (upper_right_[ii] - entity.geometry().center()[ii]);
141  ++num_boundary_coords;
142  }
143  }
144  }
145  if (is_periodic) {
146  assert(num_boundary_coords == 1);
147  periodic_coords_.push_back(periodic_neighbor_coords);
148  periodic_coords_index_.push_back(std::make_tuple(type_index, entity_index, index_in_inside));
149  } else {
150  intersection_neighbor_map[index_in_inside] = nonperiodic_pair_;
151  }
152  } else {
153  intersection_neighbor_map[index_in_inside] = nonperiodic_pair_;
154  }
155  }
156  entity_to_intersection_map_map_[type_index][entity_index] = intersection_neighbor_map;
157  } // if (entity.hasBoundaryIntersections)
158  } // loop body codim 0

811  auto entity_it = BaseType::template begin<0>();

818  for (auto i_it = BaseType::ibegin(entity); i_it != i_it_end; ++i_it) {

821  upper_right[ii] = std::max(upper_right[ii], intersection_coords[ii]);
