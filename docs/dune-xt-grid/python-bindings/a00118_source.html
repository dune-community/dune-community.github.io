120  static GridProvider<GridType> create(const FieldVector<typename GridType::ctype, GridType::dimension>& lower_left,
121  const FieldVector<typename GridType::ctype, GridType::dimension>& upper_right,
122  const std::array<unsigned int, GridType::dimension>& num_elements,
123  const unsigned int num_refinements,
124  const std::array<unsigned int, GridType::dimension>& overlap_size)
125  {
126  static const int variant = ElementVariant<GridType>::id;
127  static_assert(variant == 1 || variant == 2, "variant has to be 1 or 2!");
128  for (unsigned int dd = 0; dd < GridType::dimension; ++dd) {
129  if (!(lower_left[dd] < upper_right[dd]))
130  DUNE_THROW(Common::Exceptions::wrong_input_given,
131  "lower_left has to be elementwise smaller than upper_right!\n\n"
132  << lower_left[dd]
133  << " vs. "
134  << upper_right[dd]);
135  }
136  std::shared_ptr<GridType> grd_ptr(nullptr);
137  switch (variant) {
138  case 1:
139  grd_ptr = StructuredGridFactory<GridType>::createCubeGrid(lower_left, upper_right, num_elements, overlap_size);
140  break;
141  case 2:
142  default:
143  grd_ptr = StructuredGridFactory<GridType>::createSimplexGrid(lower_left, upper_right, num_elements);
144  break;
145  }
146  grd_ptr->loadBalance();
147 #if HAVE_ALBERTA
148  if (!std::is_same<GridType, AlbertaGrid<GridType::dimension, GridType::dimension>>::value)
149 #endif
150  grd_ptr->preAdapt();
151  grd_ptr->globalRefine(boost::numeric_cast<int>(num_refinements));
152 #if HAVE_ALBERTA
153  if (!std::is_same<GridType, AlbertaGrid<GridType::dimension, GridType::dimension>>::value)
154 #endif
155  grd_ptr->postAdapt();
156  grd_ptr->loadBalance();
157  return GridProvider<GridType>(grd_ptr);
158  } // ... create(...)

177  static GridProvider<GridType> create(const Common::Configuration& cfg = cube_gridprovider_default_config())
178  {
179  static const size_t d = GridType::dimension;
180  auto overlap_size = cube_gridprovider_default_config().template get<std::vector<unsigned int>>("overlap_size");
181  std::array<unsigned int, d> overlap_size_array;
182  if (overlap_size.size() >= d) {
183  overlap_size_array = Common::make_array<unsigned int, d>(overlap_size);
184  } else if (overlap_size.size() >= 1) {
185  overlap_size_array = Common::make_array<unsigned int, d>(overlap_size);
186  } else {
187  DUNE_THROW(Common::Exceptions::wrong_input_given,
188  "overlap_size has to be a single number or a vector with at least " << d << "elements, has only "
189  << overlap_size.size()
190  << " elements!");
191  }
192  auto lower_left = cfg.get(
193  "lower_left",
194  cube_gridprovider_default_config().template get<FieldVector<typename GridType::ctype, GridType::dimension>>(
195  "lower_left"));
196  auto upper_right = cfg.get(
197  "upper_right",
198  cube_gridprovider_default_config().template get<FieldVector<typename GridType::ctype, GridType::dimension>>(
199  "upper_right"));
200  auto num_elements = cfg.get(
201  "num_elements", cube_gridprovider_default_config().template get<std::vector<unsigned int>>("num_elements"));
202  std::array<unsigned int, d> num_elements_array;
203  if (num_elements.size() >= d) {
204  num_elements_array = Common::make_array<unsigned int, d>(num_elements);
205  } else if (num_elements.size() >= 1) {
206  num_elements_array = Common::make_array<unsigned int, d>(num_elements);
207  } else {
208  DUNE_THROW(Common::Exceptions::wrong_input_given,
209  "num_elements has to be a single number or a vector with at least " << d << "elements, has only "
210  << num_elements.size()
211  << " elements!");
212  }
213  auto num_refinements =
214  cfg.get("num_refinements", cube_gridprovider_default_config().template get<unsigned int>("num_refinements"));
215  return create(lower_left, upper_right, num_elements_array, num_refinements, overlap_size_array);
216  } // ... create(...)
