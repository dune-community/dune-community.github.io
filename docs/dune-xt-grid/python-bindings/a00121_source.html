55  Dune::LeafMultipleCodimMultipleGeomTypeMapper<Grid, P0Layout> mapper(grid);
56 
57  std::vector<double> values(mapper.size());

134  template <class Entity>
135  double operator()(const Entity& entity) const
136  {
137  double ret(0.0);
138  int numberOfBoundarySegments(0);
139  bool isOnBoundary = false;
140  const auto leafview = grid_.leafGridView();
141  for (auto&& intersection : intersections(leafview, entity)) {
142  if (!intersection.neighbor() && intersection.boundary()) {
143  isOnBoundary = true;
144  numberOfBoundarySegments += 1;
145  ret += double(intersection.boundaryId());
146  }
147  }
148  if (isOnBoundary) {
149  ret /= double(numberOfBoundarySegments);
150  }
151  return ret;
152  }
153  };

164  template <class Entity>
165  double operator()(const Entity& entity) const
166  {
167  typedef typename Entity::Geometry EntityGeometryType;
168 
169  typedef Dune::FieldVector<typename EntityGeometryType::ctype, EntityGeometryType::coorddimension> DomainType;
170 
171  const EntityGeometryType& geometry = entity.geometry();
172 
173  DomainType baryCenter(0.0);
174 
175  for (auto corner : Common::value_range(geometry.corners())) {
176  baryCenter += geometry.corner(corner);
177  }
178  baryCenter /= geometry.corners();
179 
180  double ret(0.0);
181 
182  if (!((baryCenter[0] < 0.0) || (baryCenter[0] > 1.0))) { // only in unit square
183  if (!((baryCenter[1] < 0.0) || (baryCenter[1] > 1.0))) {
184  ret = 1.0;
185  }
186  }
187  return ret;
188  }
189  };
