70 template <int codim, int worlddim, class GridImp, template <int, int, class> class EntityImp>
71 double entity_diameter(const Dune::Entity<codim, worlddim, GridImp, EntityImp>& entity)
72 {
73  auto max_dist = std::numeric_limits<typename GridImp::ctype>::min();
74  const auto& geometry = entity.geometry();
75  for (auto i : Common::value_range(geometry.corners())) {
76  const auto xi = geometry.corner(i);
77  for (auto j : Common::value_range(i + 1, geometry.corners())) {
78  auto xj = geometry.corner(j);
79  xj -= xi;
80  max_dist = std::max(max_dist, xj.two_norm());
81  }
82  }
83  return max_dist;
84 } // entity_diameter
