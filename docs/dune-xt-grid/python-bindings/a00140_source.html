50 struct CheckInside
51 {
52  template <class GeometryType, class GlobalCoordinateType>
53  static bool check(const GeometryType& geometry, const GlobalCoordinateType& point)
54  {
55  if (codim == point.size()) {
56  return Common::FloatCmp::eq(geometry.center(), point);
57  } else {
58  if (Common::FloatCmp::ne(geometry.global(geometry.local(point)), point))
59  return false;
60  const auto& refElement = reference_element(geometry);
61  return refElement.checkInside(geometry.local(point));
62  }

115  template <class PointContainerType>
116  EntityVectorType operator()(const PointContainerType& points)
117  {
118  const IteratorType begin = gridview_.template begin<codim>();
119  const IteratorType end = gridview_.template end<codim>();
120  EntityVectorType ret(points.size());
121  typename EntityVectorType::size_type idx(0);
122  for (const auto& point : points) {
123  IteratorType it_current = *it_last_;
124  bool it_reset = true;
125  typename EntityVectorType::value_type tmp_ptr(nullptr);
126  for (; it_current != end; ++it_current) {
127  if ((tmp_ptr = check_add(*it_current, point))) {
128  ret[idx++] = std::move(tmp_ptr);
129  tmp_ptr = nullptr;
130  it_reset = false;
131  *it_last_ = it_current;
132  break;
133  }
134  }
135  if (!it_reset)
136  continue;
137  for (it_current = begin; it_current != *it_last_; ++it_current) {
138  if ((tmp_ptr = check_add(*it_current, point))) {
139  ret[idx++] = std::move(tmp_ptr);
140  tmp_ptr = nullptr;
141  it_reset = false;
142  *it_last_ = it_current;
143  break;
144  }
145  }
146  }

156 class EntityHierarchicSearch : public EntitySearchBase<GridViewType>

172  template <class PointContainerType>
173  EntityVectorType operator()(const PointContainerType& points) const
174  {
175  auto level = std::min(gridview_.grid().maxLevel(), start_level_);
176  auto range = elements(gridview_.grid().levelView(level));
177  return process(points, range);
178  }
179 
180 private:
181  template <class QuadpointContainerType, class RangeType>
182  EntityVectorType process(const QuadpointContainerType& quad_points, const RangeType& range) const
183  {
184  EntityVectorType ret;
185 
186  for (const auto& my_ent : range) {
187  const auto my_level = my_ent.level();
188  const auto& geometry = my_ent.geometry();
189  const auto& refElement = reference_element(geometry);
190  for (const auto& point : quad_points) {
191  if (refElement.checkInside(geometry.local(point))) {
192  // if I cannot descend further add this entity even if it's not my view
193  if (gridview_.grid().maxLevel() <= my_level || gridview_.contains(my_ent)) {
194  ret.emplace_back(my_ent);
195  } else {
196  const auto h_end = my_ent.hend(my_level + 1);
197  const auto h_begin = my_ent.hbegin(my_level + 1);
198  const auto h_range = boost::make_iterator_range(h_begin, h_end);
199  const auto kids = process(QuadpointContainerType(1, point), h_range);
200  ret.insert(ret.end(), kids.begin(), kids.end());
201  }
202  }
203  }
204  }
205  return ret;
206  }
207 }; // class EntityHierarchicSearch
