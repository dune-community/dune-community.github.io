213 template <class GridViewImp>
214 class InnerIntersectionsPrimally
215  : public internal::WhichIntersectionBase<GridViewImp, InnerIntersectionsPrimally<GridViewImp>>
216 {
217  typedef WhichIntersection<GridViewImp> BaseType;
218 
219 public:
220  using typename BaseType::GridViewType;
221  using typename BaseType::IntersectionType;
222 
223  virtual bool apply_on(const GridViewType& grid_view, const IntersectionType& intersection) const override final
224  {
225  if (intersection.neighbor() && !intersection.boundary()) {
226  const auto insideEntityPtr = intersection.inside();
227  const auto& insideEntity = *insideEntityPtr;
228  const auto outsideNeighborPtr = intersection.outside();
229  const auto& outsideNeighbor = *outsideNeighborPtr;
230  return grid_view.indexSet().index(insideEntity) < grid_view.indexSet().index(outsideNeighbor);
231  } else
232  return false;
233  }
