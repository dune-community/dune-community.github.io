88 template <class G, class I, class D>
89 typename std::enable_if<Dune::Intersection<G, I>::dimension == 2, bool>::type
90 contains(const Dune::Intersection<G, I>& intersection,
91  const Dune::FieldVector<D, 2>& global_point,
92  const D& tolerance = Common::FloatCmp::DefaultEpsilon<D>::value())
93 {
94  const auto& geometry = intersection.geometry();
95  // get the global coordinates of the intersections corners
96  assert(geometry.corners() == 2);
97  const auto corner_0 = geometry.corner(0);
98  const auto corner_1 = geometry.corner(1);
99  // A line is given by $y = a*x + b$. Computing a and b fails for certain intersections (for instance those
100  // parallel to the y axis). So in order to check if the point is on the line between the corners we consider the
101  // vectors pointing from the point to each corner. If those are not orthogonal to the intersections normal, the point
102  // cannot lie on the line between the two corners.
103  const auto normal = intersection.centerUnitOuterNormal();
104  for (auto vector : {corner_0 - global_point, corner_1 - global_point})
105  if (vector * normal > tolerance)
106  return false;
107  // Now that we know the point is on the line, check if it is outside the bounding box of the two corners.
108  if (Common::FloatCmp::lt(global_point[0], std::min(corner_0[0], corner_1[0]), tolerance)
109  || Common::FloatCmp::gt(global_point[0], std::max(corner_0[0], corner_1[0]), tolerance))
110  return false;
111  if (Common::FloatCmp::lt(global_point[1], std::min(corner_0[1], corner_1[1]), tolerance)
112  || Common::FloatCmp::gt(global_point[1], std::max(corner_0[1], corner_1[1]), tolerance))
113  return false;
114  // At this point we cannot reject the assumption that the point lies on the line between the two corners.
115  return true;
116 } // ... contains(...)

127 template <class G, class I, class D>
128 typename std::enable_if<Dune::Intersection<G, I>::dimension == 3, bool>::type
129 contains(const Dune::Intersection<G, I>& intersection,
130  const Dune::FieldVector<D, 3>& global_point,
131  const D& tolerance = Common::FloatCmp::DefaultEpsilon<D>::value())
132 {
133  const auto& geometry = intersection.geometry();
134  // get the global coordinates of the intersections corners, there should be at least 3 (ignore the fourth if there is
135  // one, 3 points is enough in 3d)
136  assert(geometry.corners() >= 3);
137  std::vector<Dune::FieldVector<D, 3>> points(4, Dune::FieldVector<D, 3>(0.));
138  for (size_t ii = 0; ii < 3; ++ii)
139  points[ii] = geometry.corner(ii);
140  points[3] = global_point;
141  // form a matrix of these points, where the top three entries of each column are given by the three entries of each
142  // point and the bottom row is given by one, i.e.
143  // a_0 b_0 c_0 d_0
144  // a_1 b_1 c_1 d_1
145  // a_2 b_2 c_2 d_2
146  // 1 1 1 1
147  FieldMatrix<D, 4, 4> matrix(1.); // ensures the 1 on the last row
148  for (size_t ii = 0; ii < 3; ++ii) // only set the first three rows
149  for (size_t jj = 0; jj < 4; ++jj)
150  matrix[ii][jj] = points[jj][ii];
151  // the point lies on the plane given by the corners if the determinant of this matrix is zero
152  const D det = matrix.determinant();
153  return std::abs(det) < tolerance;
154 } // ... contains(...)
