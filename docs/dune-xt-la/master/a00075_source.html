199  void apply(const IstlDenseVector<S>& rhs, IstlDenseVector<S>& solution, const Common::Configuration& opts) const
200  {
201  typedef IstlSolverTraits<S, CommunicatorType> Traits;
202  typedef typename Traits::IstlVectorType IstlVectorType;
203  typedef typename Traits::MatrixOperatorType MatrixOperatorType;
204  typedef BiCGSTABSolver<IstlVectorType> BiCgSolverType;
205  InverseOperatorResult solver_result;
206  auto scalar_product = Traits::make_scalarproduct(communicator_.access());
207 
208  try {
209  if (!opts.has_key("type"))
210  DUNE_THROW(Common::Exceptions::configuration_error,
211  "Given options (see below) need to have at least the key 'type' set!\n\n"
212  << opts);
213  const auto type = opts.get<std::string>("type");
214  SolverUtils::check_given(type, types());
215  const Common::Configuration default_opts = options(type);
216  IstlDenseVector<S> writable_rhs = rhs.copy();
217 
218  if (type.substr(0, 13) == "bicgstab.amg.") {
219  solver_result = AmgApplicator<S, CommunicatorType>(matrix_, communicator_.access())
220  .call(writable_rhs, solution, opts, default_opts, type.substr(13));
221  } else if (type == "bicgstab.ilut") {
222  auto matrix_operator = Traits::make_operator(matrix_.backend(), communicator_.access());
223  typedef SeqILUn<typename MatrixType::BackendType, IstlVectorType, IstlVectorType> SequentialPreconditionerType;
224  SequentialPreconditionerType seq_preconditioner(
225  matrix_.backend(),
226  opts.get("preconditioner.iterations", default_opts.get<int>("preconditioner.iterations")),
227  opts.get("preconditioner.relaxation_factor", default_opts.get<S>("preconditioner.relaxation_factor")));
228  auto preconditioner = Traits::make_preconditioner(seq_preconditioner, communicator_.access());
229  BiCgSolverType solver(matrix_operator,
230  scalar_product,
231  preconditioner,
232  opts.get("precision", default_opts.get<R>("precision")),
233  opts.get("max_iter", default_opts.get<int>("max_iter")),
234  verbosity(opts, default_opts));
235  solver.apply(solution.backend(), writable_rhs.backend(), solver_result);
236  } else if (type == "bicgstab.ssor") {
237  auto matrix_operator = Traits::make_operator(matrix_.backend(), communicator_.access());
238  typedef SeqSSOR<typename MatrixType::BackendType, IstlVectorType, IstlVectorType> SequentialPreconditionerType;
239  SequentialPreconditionerType seq_preconditioner(
240  matrix_.backend(),
241  opts.get("preconditioner.iterations", default_opts.get<int>("preconditioner.iterations")),
242  opts.get("preconditioner.relaxation_factor", default_opts.get<S>("preconditioner.relaxation_factor")));
243  auto preconditioner = Traits::make_preconditioner(seq_preconditioner, communicator_.access());
244  BiCgSolverType solver(matrix_operator,
245  scalar_product,
246  preconditioner,
247  opts.get("precision", default_opts.get<S>("precision")),
248  opts.get("max_iter", default_opts.get<int>("max_iter")),
249  verbosity(opts, default_opts));
250  solver.apply(solution.backend(), writable_rhs.backend(), solver_result);
251  } else if (type == "bicgstab") {
252  auto matrix_operator = Traits::make_operator(matrix_.backend(), communicator_.access());
253  constexpr auto cat = decltype(matrix_operator)::category;
254  typedef IdentityPreconditioner<MatrixOperatorType, cat> SequentialPreconditioner;
255  SequentialPreconditioner seq_preconditioner;
256  auto preconditioner = Traits::make_preconditioner(seq_preconditioner, communicator_.access());
257  // define the BiCGStab as the actual solver
258  BiCgSolverType solver(matrix_operator,
259  scalar_product,
260  preconditioner,
261  opts.get("precision", default_opts.get<S>("precision")),
262  opts.get("max_iter", default_opts.get<int>("max_iter")),
263  verbosity(opts, default_opts));
264  solver.apply(solution.backend(), writable_rhs.backend(), solver_result);
265 #if HAVE_UMFPACK
266  } else if (type == "umfpack") {
267  UMFPack<typename MatrixType::BackendType> solver(matrix_.backend(),
268  opts.get("verbose", default_opts.get<int>("verbose")));
269  solver.apply(solution.backend(), writable_rhs.backend(), solver_result);
270 #endif // HAVE_UMFPACK
271 #if HAVE_SUPERLU
272  } else if (type == "superlu") {
273  SuperLU<typename MatrixType::BackendType> solver(matrix_.backend(),
274  opts.get("verbose", default_opts.get<int>("verbose")));
275 
276  solver.apply(solution.backend(), writable_rhs.backend(), solver_result);
277 #endif // HAVE_SUPERLU
278  } else
279  DUNE_THROW(Common::Exceptions::internal_error,
280  "Given type '" << type << "' is not supported, although it was reported by types()!");
281  if (!solver_result.converged)
282  DUNE_THROW(Exceptions::linear_solver_failed_bc_it_did_not_converge,
283  "The dune-istl backend reported 'InverseOperatorResult.converged == false'!\n"
284  << "Those were the given options:\n\n"
285  << opts);
286 
287  // check (use writable_rhs as tmp)
288  const R post_check_solves_system_threshold =
289  opts.get("post_check_solves_system", default_opts.get<R>("post_check_solves_system"));
290  if (post_check_solves_system_threshold > 0) {
291  matrix_.mv(solution, writable_rhs);
292  communicator_.access().copyOwnerToAll(writable_rhs.backend(), writable_rhs.backend());
293  writable_rhs -= rhs;
294  const R sup_norm = writable_rhs.sup_norm();
295  if (sup_norm > post_check_solves_system_threshold || Common::isnan(sup_norm) || Common::isinf(sup_norm))
296  DUNE_THROW(Exceptions::linear_solver_failed_bc_the_solution_does_not_solve_the_system,
297  "The computed solution does not solve the system (although the dune-istl backend "
298  << "reported no error) and you requested checking (see options below)!\n"
299  << "If you want to disable this check, set 'post_check_solves_system = 0' in the options."
300  << "\n\n"
301  << " (A * x - b).sup_norm() = "
302  << sup_norm
303  << "\n\n"
304  << "Those were the given options:\n\n"
305  << opts);
306  }
307  } catch (ISTLError& e) {
308  DUNE_THROW(Common::Exceptions::linear_solver_failed, "The dune-istl backend reported: " << e.what());
309  }
310  } // ... apply(...)
