101  backend_ = std::make_shared<BackendType>(internal::boost_numeric_cast<EIGEN_size_t>(rr),
102  internal::boost_numeric_cast<EIGEN_size_t>(cc));
103  if (rr > 0 && cc > 0) {
104  if (size_t(pattern_in.size()) != rr)
105  DUNE_THROW(Common::Exceptions::shapes_do_not_match,
106  "The size of the pattern (" << pattern_in.size() << ") does not match the number of rows of this ("
107  << rr
108  << ")!");
109  for (size_t row = 0; row < size_t(pattern_in.size()); ++row) {
110  backend_->startVec(internal::boost_numeric_cast<EIGEN_size_t>(row));
111  const auto& columns = pattern_in.inner(row);
112  for (auto& column : columns) {
113 #ifndef NDEBUG
114  if (column >= cc)
115  DUNE_THROW(Common::Exceptions::shapes_do_not_match,

118  << ")!");

142  explicit EigenRowMajorSparseMatrix(const BackendType& mat,
143  const bool prune = false,
144  const typename Common::FloatCmp::DefaultEpsilon<ScalarType>::Type eps =
145  Common::FloatCmp::DefaultEpsilon<ScalarType>::value())
146  {
147  if (prune) {
148  // we do this here instead of using pattern(true), since we can build the triplets along the way which is more
149  // efficient
150  typedef ::Eigen::Triplet<ScalarType> TripletType;
151  const ScalarType zero(0);
152  std::vector<TripletType> triplets;
153  triplets.reserve(mat.nonZeros());
154  for (EIGEN_size_t row = 0; row < mat.outerSize(); ++row) {
155  for (typename BackendType::InnerIterator row_it(mat, row); row_it; ++row_it) {
156  const EIGEN_size_t col = row_it.col();
157  const auto val = mat.coeff(row, col);
158  if (Common::FloatCmp::ne<Common::FloatCmp::Style::absolute>(val, zero, eps))
159  triplets.emplace_back(row, col, val);
160  }
161  }
162  backend_ = std::make_shared<BackendType>(mat.rows(), mat.cols());
163  backend_->setFromTriplets(triplets.begin(), triplets.end());
164  } else
165  backend_ = std::make_shared<BackendType>(mat);
166  } // EigenRowMajorSparseMatrix(...)

394  pattern(const bool prune = false,
395  const ScalarType eps = Common::FloatCmp::DefaultEpsilon<ScalarType>::value()) const override
396  {
397  std::lock_guard<std::mutex> DUNE_UNUSED(lock)(mutex_);
398  SparsityPatternDefault ret(rows());
399  const auto zero = typename Common::FloatCmp::DefaultEpsilon<ScalarType>::Type(0);
400  if (prune) {
401  for (EIGEN_size_t row = 0; row < backend().outerSize(); ++row) {
402  for (typename BackendType::InnerIterator row_it(backend(), row); row_it; ++row_it) {
403  const EIGEN_size_t col = row_it.col();
404  const auto val = backend().coeff(row, col);
405  if (Common::FloatCmp::ne(val, zero, eps))
406  ret.insert(boost::numeric_cast<size_t>(row), boost::numeric_cast<size_t>(col));
407  }
408  }
409  } else {
410  for (EIGEN_size_t row = 0; row < backend().outerSize(); ++row) {
411  for (typename BackendType::InnerIterator row_it(backend(), row); row_it; ++row_it)
412  ret.insert(boost::numeric_cast<size_t>(row), boost::numeric_cast<size_t>(row_it.col()));
413  }
