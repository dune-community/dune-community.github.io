106  template <class T1, class T2>
107  void
108  apply(const EigenBaseVector<T1, S>& rhs, EigenBaseVector<T2, S>& solution, const Common::Configuration& opts) const
109  {
110  if (!opts.has_key("type"))
111  DUNE_THROW(Common::Exceptions::configuration_error,
112  "Given options (see below) need to have at least the key 'type' set!\n\n"
113  << opts);
114  const auto type = opts.get<std::string>("type");
115  SolverUtils::check_given(type, types());
116  const Common::Configuration default_opts = options(type);
117  // check for inf or nan
118  const bool check_for_inf_nan = opts.get("check_for_inf_nan", default_opts.get<bool>("check_for_inf_nan"));
119  if (check_for_inf_nan) {
120  for (size_t ii = 0; ii < matrix_.rows(); ++ii) {
121  for (size_t jj = 0; jj < matrix_.cols(); ++jj) {
122  const S& val = matrix_.backend()(ii, jj);
123  if (Common::isnan(val) || Common::isinf(val)) {
124  std::stringstream msg;
125  msg << "Given matrix contains inf or nan and you requested checking (see options below)!\n"
126  << "If you want to disable this check, set 'check_for_inf_nan = 0' in the options.\n\n"
127  << "Those were the given options:\n\n"
128  << opts;
129  if (rhs.size() <= internal::max_size_to_print)
130  msg << "\nThis was the given matrix:\n\n" << matrix_ << "\n";
131  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements, msg.str());
132  }
133  }
134  }
135  for (size_t ii = 0; ii < rhs.size(); ++ii) {
136  const S& val = rhs[ii];
137  if (Common::isnan(val) || Common::isinf(val)) {
138  std::stringstream msg;
139  msg << "Given rhs contains inf or nan and you requested checking (see options below)!\n"
140  << "If you want to disable this check, set 'check_for_inf_nan = 0' in the options.\n\n"
141  << "Those were the given options:\n\n"
142  << opts;
143  if (rhs.size() <= internal::max_size_to_print)
144  msg << "\nThis was the given right hand side:\n\n" << rhs << "\n";
145  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements, msg.str());
146  }
147  }
148  }
149  // check for symmetry (if solver needs it)
150  if (type == "ldlt" || type == "llt") {
151  const R pre_check_symmetry_threshhold = opts.get("pre_check_symmetry", default_opts.get<R>("pre_check_symmetry"));
152  if (pre_check_symmetry_threshhold > 0) {
153  const MatrixType tmp(matrix_.backend() - matrix_.backend().adjoint());
154  // serialize difference to compute L^\infty error (no copy done here)
155  const R error = std::max(tmp.backend().cwiseAbs().minCoeff(), tmp.backend().cwiseAbs().maxCoeff());
156  if (error > pre_check_symmetry_threshhold) {
157  std::stringstream msg;
158  msg << "Given matrix is not symmetric and you requested checking (see options below)!\n"
159  << "If you want to disable this check, set 'pre_check_symmetry = 0' in the options.\n\n"
160  << " (A - A').sup_norm() = " << error << "\n\n"
161  << "Those were the given options:\n\n"
162  << opts;
163  if (rhs.size() <= internal::max_size_to_print)
164  msg << "\nThis was the given matrix A:\n\n" << matrix_ << "\n";
165  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements, msg.str());
166  }
167  }
168  }
169  // solve
170  if (type == "qr.colpivhouseholder") {
171  solution.backend() = matrix_.backend().colPivHouseholderQr().solve(rhs.backend());
172  } else if (type == "qr.fullpivhouseholder")
173  solution.backend() = matrix_.backend().fullPivHouseholderQr().solve(rhs.backend());
174  else if (type == "qr.householder")
175  solution.backend() = matrix_.backend().householderQr().solve(rhs.backend());
176  else if (type == "lu.fullpiv")
177  solution.backend() = matrix_.backend().fullPivLu().solve(rhs.backend());
178  else if (type == "llt")
179  solution.backend() = matrix_.backend().llt().solve(rhs.backend());
180  else if (type == "ldlt")
181  solution.backend() = matrix_.backend().ldlt().solve(rhs.backend());
182  else if (type == "lu.partialpiv")
183  solution.backend() = matrix_.backend().partialPivLu().solve(rhs.backend());
184  else
185  DUNE_THROW(Common::Exceptions::internal_error,
186  "Given type '" << type << "' is not supported, although it was reported by types()!");
187  // check
188  if (check_for_inf_nan)
189  for (size_t ii = 0; ii < solution.size(); ++ii) {
190  const S& val = solution[ii];
191  if (Common::isnan(val) || Common::isinf(val)) {
192  std::stringstream msg;
193  msg << "The computed solution contains inf or nan and you requested checking (see options "
194  << "below)!\n"
195  << "If you want to disable this check, set 'check_for_inf_nan = 0' in the options.\n\n"
196  << "Those were the given options:\n\n"
197  << opts;
198  if (rhs.size() <= internal::max_size_to_print)
199  msg << "\nThis was the given matrix A:\n\n"
200  << matrix_ << "\nThis was the given right hand side b:\n\n"
201  << rhs << "\nThis is the computed solution:\n\n"
202  << solution << "\n";
203  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements, msg.str());
204  }
205  }
206  const R post_check_solves_system_threshold =
207  opts.get("post_check_solves_system", default_opts.get<R>("post_check_solves_system"));
208  if (post_check_solves_system_threshold > 0) {
209  auto tmp = rhs.copy();
210  tmp.backend() = matrix_.backend() * solution.backend() - rhs.backend();
211  const R sup_norm = tmp.sup_norm();
212  if (sup_norm > post_check_solves_system_threshold || Common::isnan(sup_norm) || Common::isinf(sup_norm)) {
213  std::stringstream msg;
214  msg << "The computed solution does not solve the system (although the eigen backend reported "
215  << "'Success') and you requested checking (see options below)!\n"
216  << "If you want to disable this check, set 'post_check_solves_system = 0' in the options."
217  << "\n\n"
218  << " (A * x - b).sup_norm() = " << tmp.sup_norm() << "\n\n"
219  << "Those were the given options:\n\n"
220  << opts;
221  if (rhs.size() <= internal::max_size_to_print)
222  msg << "\nThis was the given matrix A:\n\n"
223  << matrix_ << "\nThis was the given right hand side b:\n\n"
224  << rhs << "\nThis is the computed solution:\n\n"
225  << solution << "\n";
226  DUNE_THROW(Exceptions::linear_solver_failed_bc_the_solution_does_not_solve_the_system, msg.str());
227  }
228  }
229  } // ... apply(...)

231 private:
232  const MatrixType& matrix_;
233 }; // class Solver

338  apply(const EigenBaseVector<T1, S>& rhs, EigenBaseVector<T2, S>& solution, const Common::Configuration& opts) const
339  {
340  if (!opts.has_key("type"))
341  DUNE_THROW(Common::Exceptions::configuration_error,
342  "Given options (see below) need to have at least the key 'type' set!\n\n"
343  << opts);
344  const auto type = opts.get<std::string>("type");
345  SolverUtils::check_given(type, types());
346  const Common::Configuration default_opts = options(type);
347  // check for inf or nan
348  const bool check_for_inf_nan = opts.get("check_for_inf_nan", default_opts.get<bool>("check_for_inf_nan"));
349  if (check_for_inf_nan) {
350  // iterates over the non-zero entries of matrix_.backend() and checks them
351  typedef typename MatrixType::BackendType::InnerIterator InnerIterator;
352  for (EIGEN_size_t ii = 0; ii < matrix_.backend().outerSize(); ++ii) {
353  for (InnerIterator it(matrix_.backend(), ii); it; ++it) {
354  if (Common::isnan(std::real(it.value())) || Common::isnan(std::imag(it.value()))
355  || Common::isinf(std::abs(it.value())))
356  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements,
357  "Given matrix contains inf or nan and you requested checking (see options below)!\n"
358  << "If you want to disable this check, set 'check_for_inf_nan = 0' in the options.\n\n"
359  << "Those were the given options:\n\n"
360  << opts);
361  }
362  }
363  for (size_t ii = 0; ii < rhs.size(); ++ii) {
364  const S& val = rhs[ii];
365  if (Common::isnan(val) || Common::isinf(val))
366  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements,
367  "Given rhs contains inf or nan and you requested checking (see options below)!\n"
368  << "If you want to disable this check, set 'check_for_inf_nan = 0' in the options.\n\n"
369  << "Those were the given options:\n\n"
370  << opts);
371  }
372  }
373  // check for symmetry (if solver needs it)
374  if (type.substr(0, 3) == "cg." || type == "ldlt.simplicial" || type == "llt.simplicial") {
375  const R pre_check_symmetry_threshhold = opts.get("pre_check_symmetry", default_opts.get<R>("pre_check_symmetry"));
376  if (pre_check_symmetry_threshhold > 0) {
377  ColMajorBackendType colmajor_copy(matrix_.backend());
378  colmajor_copy -= matrix_.backend().adjoint();
379  // iterates over non-zero entries as above
380  typedef typename ColMajorBackendType::InnerIterator InnerIterator;
381  for (EIGEN_size_t ii = 0; ii < colmajor_copy.outerSize(); ++ii) {
382  for (InnerIterator it(colmajor_copy, ii); it; ++it) {
383  if (std::max(std::abs(std::real(it.value())), std::abs(std::imag(it.value())))
384  > pre_check_symmetry_threshhold)
385  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements,
386  "Given matrix is not symmetric/hermitian and you requested checking (see options below)!\n"
387  << "If you want to disable this check, set 'pre_check_symmetry = 0' in the options.\n\n"
388  << "Those were the given options:\n\n"
389  << opts);
390  }
391  }
392  }
393  }
394  ::Eigen::ComputationInfo info;
395  if (type == "cg.diagonal.lower") {
396  typedef ::Eigen::ConjugateGradient<typename MatrixType::BackendType,
397  ::Eigen::Lower,
398  ::Eigen::DiagonalPreconditioner<S>>
399  SolverType;
400  SolverType solver(matrix_.backend());
401  solver.setMaxIterations(opts.get("max_iter", default_opts.get<int>("max_iter")));
402  solver.setTolerance(opts.get("precision", default_opts.get<R>("precision")));
403  solution.backend() = solver.solve(rhs.backend());
404  info = solver.info();
405  } else if (type == "cg.diagonal.upper") {
406  typedef ::Eigen::ConjugateGradient<typename MatrixType::BackendType,
407  ::Eigen::Upper,
408  ::Eigen::DiagonalPreconditioner<S>>
409  SolverType;
410  SolverType solver(matrix_.backend());
411  solver.setMaxIterations(opts.get("max_iter", default_opts.get<int>("max_iter")));
412  solver.setTolerance(opts.get("precision", default_opts.get<R>("precision")));
413  solution.backend() = solver.solve(rhs.backend());
414  info = solver.info();
415  } else if (type == "cg.identity.lower") {
416  typedef ::Eigen::ConjugateGradient<typename MatrixType::BackendType,
417  ::Eigen::Lower,
418  ::Eigen::IdentityPreconditioner>
419  SolverType;
420  SolverType solver(matrix_.backend());
421  solver.setMaxIterations(opts.get("max_iter", default_opts.get<int>("max_iter")));
422  solver.setTolerance(opts.get("precision", default_opts.get<R>("precision")));
423  solution.backend() = solver.solve(rhs.backend());
424  info = solver.info();
425  } else if (type == "cg.identity.upper") {
426  typedef ::Eigen::ConjugateGradient<typename MatrixType::BackendType,
427  ::Eigen::Lower,
428  ::Eigen::IdentityPreconditioner>
429  SolverType;
430  SolverType solver(matrix_.backend());
431  solver.setMaxIterations(opts.get("max_iter", default_opts.get<int>("max_iter")));
432  solver.setTolerance(opts.get("precision", default_opts.get<R>("precision")));
433  solution.backend() = solver.solve(rhs.backend());
434  info = solver.info();
