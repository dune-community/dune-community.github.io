645 class CommonSparseMatrix : public MatrixInterface<internal::CommonSparseMatrixTraits<ScalarImp>, ScalarImp>
646 {
647  typedef CommonSparseMatrix<ScalarImp> ThisType;
648  typedef MatrixInterface<internal::CommonSparseMatrixTraits<ScalarImp>, ScalarImp> MatrixInterfaceType;
649 
650 public:
651  typedef internal::CommonSparseMatrixTraits<ScalarImp> Traits;
652  typedef typename Traits::EntriesVectorType EntriesVectorType;
653  typedef typename Traits::ScalarType ScalarType;
654  typedef typename Traits::RealType RealType;
655  typedef std::vector<size_t> IndexVectorType;
656 
660  CommonSparseMatrix(const size_t rr, const size_t cc, const SparsityPatternDefault& patt)
661  : num_rows_(rr)
662  , num_cols_(cc)
663  , entries_(std::make_shared<EntriesVectorType>())
664  , row_pointers_(std::make_shared<IndexVectorType>(num_rows_ + 1))
665  , column_indices_(std::make_shared<IndexVectorType>())
666  {
667  if (num_rows_ > 0 && num_cols_ > 0) {
668  if (size_t(patt.size()) != num_rows_)
669  DUNE_THROW(XT::Common::Exceptions::shapes_do_not_match,
670  "The size of the pattern (" << patt.size() << ") does not match the number of rows of this ("
671  << num_rows_
672  << ")!");
673  for (size_t row = 0; row < num_rows_; ++row) {
674  const auto& columns = patt.inner(row);
675  const auto num_nonzero_entries_in_row = columns.size();
676  row_pointers_->operator[](row + 1) = row_pointers_->operator[](row) + num_nonzero_entries_in_row;
677  for (size_t col = 0; col < num_nonzero_entries_in_row; ++col) {
678 #ifndef NDEBUG
679  if (col >= num_cols_)
680  DUNE_THROW(XT::Common::Exceptions::shapes_do_not_match,
681  "The size of row " << row << " of the pattern does not match the number of columns of this ("
682  << num_cols_
683  << ")!");
684 #endif // NDEBUG
685  column_indices_->push_back(columns[col]);
686  }
687  entries_->resize(column_indices_->size());
688  }
689  }
690  }
691 
692  CommonSparseMatrix(const size_t rr = 0, const size_t cc = 0, const ScalarType& value = ScalarType(0))
693  : num_rows_(rr)
694  , num_cols_(cc)
695  , entries_(std::make_shared<EntriesVectorType>(num_rows_ * num_cols_, value))
696  , row_pointers_(std::make_shared<IndexVectorType>(num_rows_ + 1))
697  , column_indices_(std::make_shared<IndexVectorType>())
698  {
699  IndexVectorType row_column_indices(num_cols_);
700  for (size_t col = 0; col < num_cols_; ++col) {
701  row_column_indices[col] = col;
702  }
703  for (size_t row = 0; row < num_rows_; ++row) {
704  row_pointers_->operator[](row + 1) = (row + 1) * num_cols_;
705  column_indices_->insert(column_indices_->end(), row_column_indices.begin(), row_column_indices.end());
706  }
707  }

725  explicit CommonSparseMatrix(
726  const OtherMatrixType& mat,
727  const typename std::enable_if<Common::MatrixAbstraction<OtherMatrixType>::is_matrix, bool>::type prune = false,
728  const typename Common::FloatCmp::DefaultEpsilon<ScalarType>::Type eps =
729  Common::FloatCmp::DefaultEpsilon<ScalarType>::value())
730  : num_rows_(Common::MatrixAbstraction<OtherMatrixType>::rows(mat))
731  , num_cols_(Common::MatrixAbstraction<OtherMatrixType>::cols(mat))
732  , entries_(std::make_shared<EntriesVectorType>())
733  , row_pointers_(std::make_shared<IndexVectorType>(num_rows_ + 1))
734  , column_indices_(std::make_shared<IndexVectorType>())
735  {
736  for (size_t rr = 0; rr < num_rows_; ++rr) {
737  size_t num_nonzero_entries_in_row = 0;
738  for (size_t cc = 0; cc < num_cols_; ++cc) {
739  const auto& value = Common::MatrixAbstraction<OtherMatrixType>::get_entry(mat, rr, cc);
740  if (!prune || XT::Common::FloatCmp::ne(value, ScalarType(0), eps)) {
741  ++num_nonzero_entries_in_row;
742  entries_->push_back(value);
743  column_indices_->push_back(cc);
744  }
745  }
746  row_pointers_->operator[](rr + 1) = row_pointers_->operator[](rr) + num_nonzero_entries_in_row;
747  }

917  size_t get_entry_index(const size_t rr, const size_t cc, const bool throw_if_not_in_pattern = true) const
918  {
919  const auto& row_offset = row_pointers_->operator[](rr);
920  const auto& next_row_offset = row_pointers_->operator[](rr + 1);
921  const auto column_indices_it = column_indices_->begin() + row_offset;
922  const auto column_indices_it_end = column_indices_->begin() + next_row_offset;
923  const auto entry_it = std::find(column_indices_it, column_indices_it_end, cc);
924  if (entry_it != column_indices_it_end)
925  return row_offset + std::distance(column_indices_it, entry_it);
926  if (throw_if_not_in_pattern)
927  DUNE_THROW(Common::Exceptions::index_out_of_range, "Entry is not in the sparsity pattern!");
928  return size_t(-1);
929  }
