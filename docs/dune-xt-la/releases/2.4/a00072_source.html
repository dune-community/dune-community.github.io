150  virtual size_t apply(const MatrixType& _systemMatrix,
151  const VectorType& _rhsVector,
152  VectorType& solutionVector,
153  const Dune::ParameterTree description = defaultSettings()) const
154  {
155  const size_t maxIter = description.get<size_t>("maxIter");
156  const ScalarType precision = description.get<ScalarType>("precision");
157  // init system matrix and right hand side
158  MatrixType& systemMatrix = const_cast<MatrixType&>(_systemMatrix);
159  VectorType& rhsVector = const_cast<VectorType&>(_rhsVector);
160  dCSRmat A;
161  A.row = systemMatrix.rows();
162  A.col = systemMatrix.cols();
163  A.nnz = systemMatrix.backend().nonZeros();
164  A.IA = systemMatrix.backend().outerIndexPtr();
165  A.JA = systemMatrix.backend().innerIndexPtr();
166  A.val = systemMatrix.backend().valuePtr();
167  dvector f, x;
168  f.row = rhsVector.size();
169  f.val = rhsVector.backend().data();
170  x.row = rhsVector.backend().rows();
171  x.val = solutionVector.backend().data();
172  // init parameters
173  input_param inparam = initInputParams(maxIter, precision, description);
174  itsolver_param itparam = initItsolverParams(maxIter, precision, description);
175  AMG_param amgparam = initAMGParams(1, precision, description); // the 1 is on purpose!
176  ILU_param iluparam = initIluParams(maxIter, precision, description);
177  Schwarz_param swzparam = initSchwarzParams(maxIter, precision, description);
178  // call fasp (this is taken from the fasp example test)
179  int status = -1;
180  // Preconditioned Krylov methods
181  if (inparam.solver_type >= 1 && inparam.solver_type <= 20) {
182  // Using no preconditioner for Krylov iterative methods
183  if (inparam.precond_type == PREC_NULL) {
184  status = fasp_solver_dcsr_krylov(&A, &f, &x, &itparam);
185  }
186  // Using diag(A) as preconditioner for Krylov iterative methods
187  else if (inparam.precond_type == PREC_DIAG) {
188  status = fasp_solver_dcsr_krylov_diag(&A, &f, &x, &itparam);
189  }
190  // Using AMG as preconditioner for Krylov iterative methods
191  else if (inparam.precond_type == PREC_AMG || inparam.precond_type == PREC_FMG) {
192  if (inparam.print_level > PRINT_NONE)
193  fasp_param_amg_print(&amgparam);
194  status = fasp_solver_dcsr_krylov_amg(&A, &f, &x, &itparam, &amgparam);
195  }
196  // Using ILU as preconditioner for Krylov iterative methods Q: Need to change!
197  else if (inparam.precond_type == PREC_ILU) {
198  if (inparam.print_level > PRINT_NONE)
199  fasp_param_ilu_print(&iluparam);
200  status = fasp_solver_dcsr_krylov_ilu(&A, &f, &x, &itparam, &iluparam);
201  }
202  // Using Schwarz as preconditioner for Krylov iterative methods
203  else if (inparam.precond_type == PREC_SCHWARZ) {
204  if (inparam.print_level > PRINT_NONE)
205  fasp_param_schwarz_print(&swzparam);
206  status = fasp_solver_dcsr_krylov_schwarz(&A, &f, &x, &itparam, &swzparam);
207  } else {
208  printf("### ERROR: Wrong preconditioner type %d!!!\n", inparam.precond_type);
209  status = ERROR_SOLVER_PRECTYPE;
210  }
211  }
212  // AMG as the iterative solver
213  else if (inparam.solver_type == SOLVER_AMG) {
214  if (inparam.print_level > PRINT_NONE)
215  fasp_param_amg_print(&amgparam);
216  fasp_solver_amg(&A, &f, &x, &amgparam);
217  }
218  // Full AMG as the iterative solver
219  else if (inparam.solver_type == SOLVER_FMG) {
220  if (inparam.print_level > PRINT_NONE)
221  fasp_param_amg_print(&amgparam);
222  fasp_solver_famg(&A, &f, &x, &amgparam);
223  } else {
224  DUNE_THROW(Dune::RangeError, "### ERROR: Wrong solver type: " << inparam.solver_type << "!");
225  status = ERROR_SOLVER_TYPE;
226  }
227  if (status > 0)
228  return 0;
229  else
230  return 3;
231  } // ... apply(...)
