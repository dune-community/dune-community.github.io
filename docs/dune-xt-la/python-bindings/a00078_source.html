162  virtual SparsityPatternDefault pattern(const bool prune = false,
163  const typename Common::FloatCmp::DefaultEpsilon<ScalarType>::Type eps =
164  Common::FloatCmp::DefaultEpsilon<ScalarType>::value()) const
165  {
166  SparsityPatternDefault ret(rows());
167  const ScalarType zero(0);
168  if (prune) {
169  for (size_t ii = 0; ii < rows(); ++ii)
170  for (size_t jj = 0; jj < cols(); ++jj)
171  if (Common::FloatCmp::ne<Common::FloatCmp::Style::absolute>(get_entry(ii, jj), zero, eps))
172  ret.insert(ii, jj);
173  } else {
174  for (size_t ii = 0; ii < rows(); ++ii)
175  for (size_t jj = 0; jj < cols(); ++jj)
176  ret.insert(ii, jj);
177  }
178  ret.sort();
179  return ret;
180  } // ... pattern(...)

191  virtual derived_type pruned(const typename Common::FloatCmp::DefaultEpsilon<ScalarType>::Type eps =

207 template <class T, class S>
208 std::ostream& operator<<(std::ostream& out, const MatrixInterface<T, S>& matrix)
209 {
210  out << "[";
211  const size_t rows = matrix.rows();
212  const size_t cols = matrix.cols();
213  if (rows > 0 && cols > 0) {
214  for (size_t ii = 0; ii < rows; ++ii) {
215  if (ii > 0)
216  out << "\n ";
217  out << "[" << matrix.get_entry(ii, 0);
218  for (size_t jj = 1; jj < cols; ++jj)
219  out << " " << matrix.get_entry(ii, jj);
220  out << "]";
221  if (rows > 1 && ii < (rows - 1))
222  out << ",";
223  }
224  out << "]";
225  } else
226  out << "[ ]]";
227  return out;
