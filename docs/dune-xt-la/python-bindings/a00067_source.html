72  void apply(const CommonDenseVector<S>& rhs, CommonDenseVector<S>& solution, const Common::Configuration& opts) const
73  {
74  if (!opts.has_key("type"))
75  DUNE_THROW(Common::Exceptions::configuration_error,
76  "Given options (see below) need to have at least the key 'type' set!\n\n"
77  << opts);
78  const auto type = opts.get<std::string>("type");
79  SolverUtils::check_given(type, types());
80  const Common::Configuration default_opts = options(type);
81  // solve
82  try {
83  matrix_.backend().solve(solution.backend(), rhs.backend());
84  } catch (FMatrixError&) {
85  DUNE_THROW(Exceptions::linear_solver_failed_bc_data_did_not_fulfill_requirements,
86  "The dune-common backend reported 'FMatrixError'!\n"
87  << "Those were the given options:\n\n"
88  << opts);
89  }
90  // check
91  const R post_check_solves_system_threshold =
92  opts.get("post_check_solves_system", default_opts.get<R>("post_check_solves_system"));
93  if (post_check_solves_system_threshold > 0) {
94  auto tmp = rhs.copy();
95  matrix_.mv(solution, tmp);
96  tmp -= rhs;
97  const R sup_norm = tmp.sup_norm();
98  if (sup_norm > post_check_solves_system_threshold || Common::isnan(sup_norm) || Common::isinf(sup_norm))
99  DUNE_THROW(Exceptions::linear_solver_failed_bc_the_solution_does_not_solve_the_system,
100  "The computed solution does not solve the system (although the dune-common backend "
101  << "reported no error) and you requested checking (see options below)! "
102  << "If you want to disable this check, set 'post_check_solves_system = 0' in the options."
103  << "\n\n"
104  << " (A * x - b).sup_norm() = "
105  << tmp.sup_norm()
106  << "\n\n"
107  << "Those were the given options:\n\n"
108  << opts);
109  }
110  } // ... apply(...)
