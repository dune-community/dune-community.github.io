131  template <class stl_container_type>
132  MinMaxAvg(const stl_container_type& elements)
133  {
134  static_assert((boost::is_same<ElementType, typename stl_container_type::value_type>::value),
135  "cannot assign mismatching types");
136  acc_ = std::for_each(elements.begin(), elements.end(), acc_);
137  }
138 
139  std::size_t count() const
140  {
141  return boost::accumulators::count(acc_);
142  }
143  ElementType sum() const
144  {
145  return boost::accumulators::sum(acc_);
146  }
147  ElementType min() const
148  {
149  return boost::accumulators::min(acc_);
150  }
151  ElementType max() const
152  {
153  return boost::accumulators::max(acc_);
154  }
155  ElementType average() const
156  {
157  // for integer ElementType this just truncates from floating-point
158  return ElementType(boost::accumulators::mean(acc_));
159  }

164  }

166  void output(std::ostream& stream)

168  stream << boost::format("min: %e\tmax: %e\tavg: %e\n") % min() % max() % average();

173  boost::accumulators::tag::mean, boost::accumulators::tag::count,
174  boost::accumulators::tag::sum> StatsType;
175  boost::accumulators::accumulator_set<ElementType, StatsType> acc_;
