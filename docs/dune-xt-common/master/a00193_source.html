287  // this class run test on generated types in thos round and go to next*/
288  template <class UIterator = typename begin<UTypes>::type, class VIterator = typename begin<VTypes>::type>
289  struct Generate
290  {
291  // generate <<next>> target type
292  typedef typename Next<UIterator, VIterator>::type next_type;
293 
294  static void Run()
295  {
296  // increment recursion counter
297  ++total_recursions;
298 
299  // test on the generated types of this round of recursion
300  TestFunc::template run<typename deref<UIterator>::type, typename deref<VIterator>::type>();
301 
302  // go to the next round of recursion
303  next_type::Run();
304  }
305 
306  template <class... Args>
307  static void Run(Args&&... args)
308  {
309  // increment recursion counter
310  ++total_recursions;
311 
312  // test on the generated types of this round of recursion
313  TestFunc::template run<typename deref<UIterator>::type, typename deref<VIterator>::type>(
314  std::forward<Args>(args)...);
315 
316  // go to the next round of recursion
