67 {
68  const std::string str_out = boost::algorithm::trim_copy(str_in);

155 template <class V>
156 static inline typename std::enable_if<is_complex<V>::value, V>::type
157 from_string(std::string ss, const size_t /*size*/ = 0, const size_t /*cols*/ = 0)
158 {
159  boost::algorithm::trim(ss);
160  if (ss.size() < 1)
161  DUNE_THROW(Exceptions::conversion_error, "Error converting " << ss << " (too short)");
162  using namespace std;
163  typedef typename V::value_type T;
164  T re(0), im(0);
165  const auto sign_pos = ss.find("+", 1) != string::npos ? ss.find("+", 1) : ss.find("-", 1);
166  re = from_string<T>(ss.substr(0, sign_pos));
167  if (sign_pos != string::npos) {
168  ss = ss.substr(sign_pos);
169  const auto im_pos = ss.find("i");
170  if (im_pos == string::npos)
171  DUNE_THROW(Exceptions::conversion_error, "Error converting " << ss << " no imaginary unit");
172  im = from_string<T>(ss.substr(0, im_pos));
173  }
174  return V(re, im);
175 }

177 template <class VectorType>
178 static inline typename std::enable_if<is_vector<VectorType>::value, VectorType>::type
179 from_string(std::string ss, const size_t size, const size_t DXTC_DEBUG_ONLY(cols) = 0)
180 {
181  auto vector_str = ss;
182  typedef typename VectorAbstraction<VectorType>::S S;
183  assert(cols == 0);
184  // check if this is a vector
185  if (vector_str.substr(0, 1) == "[" && vector_str.substr(vector_str.size() - 1, 1) == "]") {
186  vector_str = vector_str.substr(1, vector_str.size() - 2);
187  // we treat this as a vector and split along ' '
188  const auto tokens = tokenize<std::string>(vector_str, " ", boost::algorithm::token_compress_on);
189  if (size > 0 && tokens.size() < size)
190  DUNE_THROW(Exceptions::conversion_error,
191  "Vector expression (see below) has only " << tokens.size() << " elements but " << size
192  << " elements were requested!"
193  << "\n"
194  << "'["
195  << vector_str
196  << "]'");
197  const size_t automatic_size = (size > 0) ? std::min(tokens.size(), size) : tokens.size();
198  const size_t actual_size =
199  VectorAbstraction<VectorType>::has_static_size ? VectorAbstraction<VectorType>::static_size : automatic_size;
200  if (actual_size > automatic_size)
201  DUNE_THROW(Exceptions::conversion_error,
202  "Vector expression (see below) has only " << automatic_size << " elements but " << actual_size
203  << " elements are required for this VectorType ("
204  << Typename<VectorType>::value()
205  << ")!"
206  << "\n"
207  << "'["
208  << vector_str
209  << "]'");
210  VectorType ret = VectorAbstraction<VectorType>::create(actual_size);
211  for (size_t ii = 0; ii < actual_size; ++ii)
212  ret[ii] = from_string<S>(trim_copy_safely(tokens[ii]));
213  return ret;
214  } else {
215  // we treat this as a scalar
216  const auto val = from_string<S>(trim_copy_safely(vector_str));
217  const size_t automatic_size = (size == 0 ? 1 : size);
218  const size_t actual_size =
219  VectorAbstraction<VectorType>::has_static_size ? VectorAbstraction<VectorType>::static_size : automatic_size;
220  if (actual_size > automatic_size && automatic_size != 1)
221  DUNE_THROW(Exceptions::conversion_error,
222  "Vector expression (see below) has only " << automatic_size << " elements but " << actual_size
223  << " elements are required for this VectorType ("
224  << Typename<VectorType>::value()
225  << ")!"
226  << "\n"
227  << "'["
228  << vector_str
229  << "]'");
230  VectorType ret = VectorAbstraction<VectorType>::create(actual_size);
231  for (size_t ii = 0; ii < std::min(actual_size, ret.size()); ++ii)
232  ret[ii] = val;
233  return ret;
234  }
235 } // ... from_string(...)

237 template <class MatrixType>
238 static inline typename std::enable_if<is_matrix<MatrixType>::value, MatrixType>::type
239 from_string(std::string matrix_str, const size_t rows, const size_t cols)
240 {
241  typedef typename MatrixAbstraction<MatrixType>::S S;
242  // check if this is a matrix
243  if (matrix_str.substr(0, 1) == "[" && matrix_str.substr(matrix_str.size() - 1, 1) == "]") {
244  matrix_str = matrix_str.substr(1, matrix_str.size() - 2);
245  // we treat this as a matrix and split along ';' to obtain the rows
246  const auto row_tokens = tokenize<std::string>(matrix_str, ";", boost::algorithm::token_compress_on);
247  if (rows > 0 && row_tokens.size() < rows)
248  DUNE_THROW(Exceptions::conversion_error,
249  "Matrix expression (see below) has only " << row_tokens.size() << " rows but " << rows
250  << " rows were requested!"
251  << "\n"
252  << "'["
253  << matrix_str
254  << "]'");
255  const size_t automatic_rows = (rows > 0) ? std::min(row_tokens.size(), rows) : row_tokens.size();
256  const size_t actual_rows =
257  MatrixAbstraction<MatrixType>::has_static_size ? MatrixAbstraction<MatrixType>::static_rows : automatic_rows;
258  if (actual_rows > automatic_rows)
259  DUNE_THROW(Exceptions::conversion_error,
260  "Matrix expression (see below) has only " << automatic_rows << " rows but " << actual_rows
261  << " rows are required for this MatrixType ("
262  << Typename<MatrixType>::value()
263  << ")!"
264  << "\n"
265  << "'["
266  << matrix_str
267  << "]'");
268  // compute the number of columns the matrix will have
269  size_t min_cols = std::numeric_limits<size_t>::max();
270  for (size_t rr = 0; rr < actual_rows; ++rr) {
271  const auto row_token = boost::algorithm::trim_copy(row_tokens[rr]);
272  // we treat this as a vector, so we split along ' '
273  const auto column_tokens = tokenize<std::string>(row_token, " ", boost::algorithm::token_compress_on);
274  min_cols = std::min(min_cols, column_tokens.size());
275  }
276  if (cols > 0 && min_cols < cols)
277  DUNE_THROW(Exceptions::conversion_error,
278  "Matrix expression (see below) has only " << min_cols << " columns but " << cols
279  << " columns were requested!"
280  << "\n"
281  << "'["
282  << matrix_str
283  << "]'");
284  const auto automatic_cols = (cols > 0) ? std::min(min_cols, cols) : min_cols;
285  const size_t actual_cols =
286  MatrixAbstraction<MatrixType>::has_static_size ? MatrixAbstraction<MatrixType>::static_cols : automatic_cols;
287  if (actual_cols > automatic_cols)
288  DUNE_THROW(Exceptions::conversion_error,
289  "Matrix expression (see below) has only " << automatic_cols << " cols but " << actual_cols
290  << " cols are required for this MatrixType ("
291  << Typename<MatrixType>::value()
292  << ")!"
293  << "\n"
294  << "'["
295  << matrix_str
296  << "]'");
297  MatrixType ret = MatrixAbstraction<MatrixType>::create(actual_rows, actual_cols);
298  // now we do the same again and build the actual matrix
299  for (size_t rr = 0; rr < actual_rows; ++rr) {
300  const std::string row_token = boost::algorithm::trim_copy(row_tokens[rr]);
301  const auto column_tokens = tokenize<std::string>(row_token, " ", boost::algorithm::token_compress_on);
302  for (size_t cc = 0; cc < actual_cols; ++cc)
303  MatrixAbstraction<MatrixType>::set_entry(ret, rr, cc, from_string<S>(trim_copy_safely(column_tokens[cc])));
304  }
305  return ret;
306  } else {
307  // we treat this as a scalar
308  const S val = from_string<S>(trim_copy_safely(matrix_str));
309  const size_t automatic_rows = (rows == 0 ? 1 : rows);
310  const size_t actual_rows =
311  MatrixAbstraction<MatrixType>::has_static_size ? MatrixAbstraction<MatrixType>::static_rows : automatic_rows;
312  if (actual_rows > automatic_rows)
313  DUNE_THROW(Exceptions::conversion_error,
314  "Matrix expression (see below) has only " << automatic_rows << " rows but " << actual_rows
315  << " rows are required for this MatrixType ("
316  << Typename<MatrixType>::value()
317  << ")!"
318  << "\n"
319  << "'["
320  << matrix_str
321  << "]'");
322  const size_t automatic_cols = (cols == 0 ? 1 : cols);
323  const size_t actual_cols =
324  MatrixAbstraction<MatrixType>::has_static_size ? MatrixAbstraction<MatrixType>::static_cols : automatic_cols;
325  if (actual_cols > automatic_cols)
326  DUNE_THROW(Exceptions::conversion_error,
327  "Matrix expression (see below) has only " << automatic_cols << " cols but " << actual_cols
328  << " cols are required for this MatrixType ("
329  << Typename<MatrixType>::value()
330  << ")!"
331  << "\n"
332  << "'["
333  << matrix_str
334  << "]'");
335  MatrixType ret = MatrixAbstraction<MatrixType>::create(actual_rows, actual_cols);
336  for (size_t rr = 0; rr < std::min(actual_rows, MatrixAbstraction<MatrixType>::rows(ret)); ++rr)
337  for (size_t cc = 0; cc < std::min(actual_cols, MatrixAbstraction<MatrixType>::cols(ret)); ++cc)
338  MatrixAbstraction<MatrixType>::set_entry(ret, rr, cc, val);
339  return ret;
340  }
