120 template <class T, size_t N>
121 std::array<T, N> make_array(const std::vector<T>& v)
122 {
123  if (v.size() == 1) {
124  return make_array<T, N>(v[0]);
125  } else if (v.size() < N) {
126  std::array<T, N> error_msg;
127  DUNE_THROW(Exceptions::shapes_do_not_match,
128  "Trying to create a " << get_typename(error_msg) << " from a " << get_typename(v) << " of size "
129  << v.size());
130  } else {
131  std::array<T, N> ret;
132  for (size_t nn = 0; nn < N; ++nn) {
133  ret[nn] = v[nn];
134  }
135  return ret;
136  }
137 }
