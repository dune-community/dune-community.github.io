124 template <class T, class sequence = std::vector<T>>
125 typename std::enable_if<!std::is_enum<T>::value, sequence>::type value_range(const T start, const T end,
126  const T increment = Epsilon<T>::value)
127 {
128  // sadly, no overloaded version of std::abs is available for
129  // unsigned long long, so we compute the absolute value of increment
130  // ourselves
131  const auto incrementAbs = abs(increment);
132  assert(incrementAbs > 0);
133  sequence ret(typename sequence::size_type(((end > start) ? end - start : start - end) / incrementAbs), start);
134  typename sequence::size_type i = 0;
135  std::generate(std::begin(ret), std::end(ret), [&]() { return T(start + (increment * i++)); });
136  return ret;
137 }
