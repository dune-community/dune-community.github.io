130  const boost::algorithm::token_compress_mode_type mode)
131 {
132  std::vector<std::string> strings;
133  boost::algorithm::split(strings, msg, boost::algorithm::is_any_of(separators), mode);
134  std::vector<T> ret(strings.size());
135  size_t i = 0;
136  // special case for empty strings to avoid non-default init
137  std::generate(
138  std::begin(ret), std::end(ret), [&]() { return strings[i++].empty() ? T() : from_string<T>(strings[i - 1]); });
139  return ret;
140 } // ... tokenize(...)
