83 template <class Matrix>
84 void matrix_to_gnuplot_stream(const Matrix& matrix, std::ostream& stream)
85 {
86  unsigned long nz = 0;
87 
88  const auto cols = matrix.cols();
89  for (auto row : value_range(matrix.rows())) {
90  for (auto col : value_range(cols)) {
91  if (matrix.find(row, col))
92  stream << row << "\t" << col << "\t" << matrix(row, col) << "\n";
93  }
94  nz += matrix.numNonZeros(row);
95  stream << "#non zeros in row " << row << " " << matrix.numNonZeros(row) << " (of " << matrix.cols() << " cols)\n";
96  }
97  stream << "#total non zeros " << nz << " of " << matrix.rows() * matrix.cols() << " entries\n";
98 } // matrix_to_gnuplot_stream
