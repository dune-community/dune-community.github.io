312  T get_valid_value(const std::string& key, T def,
313  const ValidatorInterface<typename internal::Typer<T>::type, Validator>& validator,
314  const size_t size, const size_t cols) const
315  {
316  std::string valstring = BaseType::get(key, to_string(def));
317  try {
318  T val = from_string<T>(valstring, size, cols);
319  if (validator(val))
