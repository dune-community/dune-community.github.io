72  static std::vector<RangeType> read_values_from_file(const std::string& filename,
73  const RangeFieldType& min,
74  const RangeFieldType& max,
75  const RangeType& unit_range)
76 
77  {
78  if (!(max > min))
79  DUNE_THROW(Dune::RangeError, "max (is " << max << ") has to be larger than min (is " << min << ")!");
80  const RangeFieldType scale = (max - min) / (internal::model1_max_value - internal::model1_min_value);
81  const RangeFieldType shift = min - scale * internal::model1_min_value;
82  // read all the data from the file
83  std::ifstream datafile(filename);
84  if (datafile.is_open()) {
85  static const size_t entriesPerDim = model1_x_elements * model1_y_elements * model1_z_elements;
86  // create storage (there should be exactly 6000 values in the file, but we onyl read the first 2000)
87  std::vector<RangeType> data(entriesPerDim, unit_range);
88  double tmp = 0;
89  size_t counter = 0;
90  while (datafile >> tmp && counter < entriesPerDim)
91  data[counter++] *= (tmp * scale) + shift;
92  datafile.close();
93  if (counter != entriesPerDim)
94  DUNE_THROW(Dune::IOError,
95  "wrong number of entries in '" << filename << "' (are " << counter << ", should be " << entriesPerDim
96  << ")!");
97  return data;
98  } else
99  DUNE_THROW(Exceptions::spe10_data_file_missing, "could not open '" << filename << "'!");
100  } // ... read_values_from_file(...)
