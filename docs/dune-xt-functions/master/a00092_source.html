33 struct Ellipsoid

36  DomainType center;

39  bool contains(DomainType point) const
40  {

42  double sum = 0;

44  const auto pii = shifted[ii];

48  }

170  RandomEllipsoidsFunction(const Common::FieldVector<DomainFieldType, dimDomain>& lowerLeft,
171  const Common::FieldVector<DomainFieldType, dimDomain>& upperRight,
172  const Common::Configuration& ellipsoid_cfg,
173  const std::string nm = static_id())
174  : lowerLeft_(lowerLeft)
175  , upperRight_(upperRight)
176  , name_(nm)
177  , ellipsoid_cfg_(ellipsoid_cfg)
178  {
179  typedef unsigned long UL;
180  const UL level_0_count = ellipsoid_cfg.get("ellipsoids.count", 10);
181  const UL max_depth = ellipsoid_cfg.get("ellipsoids.recursion_depth", 1);
182  const UL children = ellipsoid_cfg.get<UL>("ellipsoids.children", 3u); //, ValidateLess<UL>(0));
183  const UL total_count = level_0_count + level_0_count * std::pow(children, max_depth + 1);
184  ellipsoids_.resize(level_0_count);
185  ellipsoids_.reserve(total_count);
186  const auto seed = DomainFieldType(ellipsoid_cfg.get("ellipsoids.seed", 0));
187  const auto min_radius = ellipsoid_cfg.get("ellipsoids.min_radius", 0.01);
188  const auto max_radius = ellipsoid_cfg.get("ellipsoids.max_radius", 0.02);
189  const auto child_displacement = ellipsoid_cfg.get("ellipsoids.max_child_displacement", max_radius);
190  typedef Common::DefaultRNG<DomainFieldType> RNG;
191  RNG center_rng(0, 1, seed);
192  RNG radii_rng(min_radius, max_radius, seed);
193  RNG dist_rng(min_radius, child_displacement, seed);
194  RNG sign_rng(-1, 1, seed);
195 
196  const auto parent_range = Common::value_range(0ul, level_0_count);
197  for (auto ii : parent_range) {
198  std::generate(
199  ellipsoids_[ii].center.begin(), ellipsoids_[ii].center.end(), [&center_rng]() { return center_rng(); });
200  std::generate(ellipsoids_[ii].radii.begin(), ellipsoids_[ii].radii.end(), [&radii_rng]() { return radii_rng(); });
201  }
202 
203  std::function<void(UL, const EllipsoidType&)> recurse_add = [&](UL current_level, const EllipsoidType& parent) {
204  if (current_level > max_depth)
205  return;
206  for (const auto unused_counter : Common::value_range(children)) {
207  EllipsoidType child = parent;
208  const double scale = std::pow(ellipsoid_cfg.get("ellipsoids.recursion_scale", 0.5), current_level);
209  const auto displace = [&](DomainFieldType& coord) {
210  const auto disp = dist_rng() * signum(sign_rng());
211  coord += disp;
212  DXTC_LOG_DEBUG_0 << disp << ";";
213  };
214  std::for_each(child.center.begin(), child.center.end(), displace);
215  std::generate(child.radii.begin(), child.radii.end(), [&radii_rng, scale]() { return radii_rng() * scale; });
216  ellipsoids_.push_back(child);
217  recurse_add(current_level + 1, child);
218  }
219  };
220 
221  for (auto ii : parent_range) {
222  recurse_add(0, ellipsoids_[ii]);
223  }
224  DXTC_LOG_DEBUG_0 << "generated " << ellipsoids_.size() << " of " << total_count << "\n";
225  to_file(*Common::make_ofstream("ellipsoids.txt"));
226  }

258  bounding_box(const EntityType& entity) const
259  {
260  typename EllipsoidType::DomainType ll, ur;
261  typedef Common::MinMaxAvg<DomainFieldType> MinMaxAvgType;
262  std::array<MinMaxAvgType, dimDomain> coord_limits;
263  const auto& geo = entity.geometry();
264  for (auto i : Common::value_range(geo.corners())) {
265  const auto& corner(geo.corner(i));
266  for (size_t k = 0; k < dimDomain; ++k)
267  coord_limits[k](corner[k]);
268  }
269  for (auto ii : Common::value_range(dimDomain)) {
270  ll[ii] = coord_limits[ii].min();
271  ur[ii] = coord_limits[ii].max();
272 
273  return std::make_pair(std::move(ll), std::move(ur));
274  }

278  virtual std::unique_ptr<LocalfunctionType> local_function(const EntityType& entity) const override
279  {
280  const auto local_value = DomainFieldType(ellipsoid_cfg_.get("ellipsoids.local_value", 1.));
281  // decide on the subdomain the center of the entity belongs to
282  const auto max_radius = ellipsoid_cfg_.get("ellipsoids.max_radius", 0.04);
283  typename EllipsoidType::DomainType ll, ur;
284  std::tie(ll, ur) = bounding_box(entity);
285  std::for_each(ll.begin(), ll.end(), [&max_radius](DomainFieldType& lc) { lc -= 1.1 * max_radius; });
286  std::for_each(ur.begin(), ur.end(), [&max_radius](DomainFieldType& uc) { uc += 1.1 * max_radius; });
287  std::vector<EllipsoidType> local_ellipsoids;
288  for (const auto& ellipsoid : ellipsoids_) {
289  bool inside = true;
290  for (auto ii : Common::value_range(dimDomain)) {
291  inside = inside && FloatCmp::ge(ellipsoid.center[ii], ll[ii]);
292  inside = inside && FloatCmp::le(ellipsoid.center[ii], ur[ii]);
293  }
294  if (inside) {
295  local_ellipsoids.push_back(ellipsoid);
296  }
297  }
298  std::vector<EllipsoidType> tmp = ellipsoids_;
299  return std::unique_ptr<Localfunction>(new Localfunction(entity, local_value, std::move(tmp)));
300  } // ... local_function(...)
