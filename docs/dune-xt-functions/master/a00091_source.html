79  // 3 is the maximum space dimension

81  permIntervalls_[dim] = std::min(unsigned(std::floor(x[dim] / deltas_[dim])), num_elements[dim] - 1);

100  double val;
101  if (!file) { // file couldn't be opened
102  return;
