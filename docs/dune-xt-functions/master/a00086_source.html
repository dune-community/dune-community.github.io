119  static std::unique_ptr<ThisType> create(const Common::Configuration config = default_config(),
120  const std::string sub_name = static_id())
121  {
122  const Common::Configuration cfg = config.has_sub(sub_name) ? config.sub(sub_name) : config;
123  const Common::Configuration def_cfg = default_config();
124  std::vector<std::tuple<DomainType, DomainType, RangeFieldType>> values;
125  DomainType tmp_lower;
126  DomainType tmp_upper;
127  size_t cc = 0;
128  while (cfg.has_sub(Common::to_string(cc))) {
129  const Common::Configuration local_cfg = cfg.sub(Common::to_string(cc));
130  if (local_cfg.has_key("domain") && local_cfg.has_key("value")) {
131  auto domains = local_cfg.get<FieldMatrix<DomainFieldType, d, 2>>("domain");
132  for (size_t dd = 0; dd < d; ++dd) {
133  tmp_lower[dd] = domains[dd][0];
134  tmp_upper[dd] = domains[dd][1];
135  }
136  auto val = local_cfg.get<RangeFieldType>("value");
137  values.emplace_back(tmp_lower, tmp_upper, val);
138  } else
139  break;
140  ++cc;
141  }
