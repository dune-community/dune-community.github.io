79  static std::unique_ptr<ThisType> create(const Common::Configuration config = default_config(),
80  const std::string sub_name = static_id())
81  {
82  // get correct config
83  const Common::Configuration cfg = config.has_sub(sub_name) ? config.sub(sub_name) : config;
84  const Common::Configuration default_cfg = default_config();
85  // get expression
86  ExpressionStringVectorType expression_as_vectors;
87  // try to get expression as FieldVector (if dimRangeCols == 1) or as FieldMatrix (else)
88  try {
89  get_expression_helper(cfg, expression_as_vectors, internal::ChooseVariant<dimRangeCols>());
90  } catch (Common::Exceptions::conversion_error) {
91  // if dimRangeCols == 1 and we could not get expression as FieldVector, get it as FieldMatrix with one col
92  if (dimRangeCols == 1) { // the 2 in ChooseVariant is here on purpose, anything > 1 will suffice
93  get_expression_helper(cfg, expression_as_vectors, internal::ChooseVariant<2>());
94  } else { // if dimRangeCols > 1 do the same again (to throw exception without catching it)
95  get_expression_helper(cfg, expression_as_vectors, internal::ChooseVariant<dimRangeCols>());
96  }
97  }
98  // get gradient
99  GradientStringVectorType gradient_as_vectors;
100  if (cfg.has_key("gradient")) {
101  if (cfg.has_key("gradient.0"))
102  assert((cfg.get<std::string>("gradient") == cfg.get<std::string>("gradient.0"))
103  && "gradient and gradient.0 differ but should be synonymous!");
104  get_gradient(cfg, gradient_as_vectors, "gradient");
105  } else if (cfg.has_key("gradient.0")) {
106  get_gradient(cfg, gradient_as_vectors, "gradient.0");
107  }
108  // create
109  return Common::make_unique<ThisType>(cfg.get("variable", default_cfg.get<std::string>("variable")),
110  expression_as_vectors,
111  cfg.get("order", default_cfg.get<size_t>("order")),

218  virtual void evaluate(const DomainType& xx, RangeType& ret) const override
219  {
220  evaluate_helper(xx, ret, internal::ChooseVariant<dimRangeCols>());
221 #ifndef NDEBUG
222 #ifndef DUNE_XT_FUNCTIONS_EXPRESSION_DISABLE_CHECKS
223  bool failure = false;
224  std::string error_type;
225  for (size_t rr = 0; rr < dimRange; ++rr) {
226  *tmp_row_ = ret[rr];
227  for (size_t cc = 0; cc < dimRangeCols; ++cc) {
228  if (isnan(tmp_row_->operator[](cc))) {
229  failure = true;
230  error_type = "NaN";
231  } else if (isinf(tmp_row_->operator[](cc))) {
232  failure = true;
233  error_type = "inf";
234  } else if (std::abs(tmp_row_->operator[](cc)) > (0.9 * std::numeric_limits<double>::max())) {
235  failure = true;
236  error_type = "an unlikely value";
237  }
238  if (failure)
239  DUNE_THROW(Common::Exceptions::internal_error,
240  "evaluating this function yielded "
241  << error_type
242  << "!\n"
243  << "The variable of this function is: "
244  << function_->variable()
245  << "\n"
246  << "The expression of this functional is: "
247  << function_->expression().at(rr * dimRangeCols + cc)
248  << "\n"
249  << "You tried to evaluate it with: xx = "
250  << xx
251  << "\n"
252  << "The result was: "
253  << tmp_row_->operator[](cc)
254  << "\n\n"
255  << "You can disable this check by defining DUNE_XT_FUNCTIONS_EXPRESSION_DISABLE_CHECKS\n");

272 private:
273  // fill the rows of the dimRange x dimRangeCols matrix (aka vector< vector< string > > expression) in a vector of
274  // length dimRange*dimRangeCols, e.g. [3 4; 1 2] becomes [3 4 1 2], in order to create function_
275  void build_function(const std::string variable, const ExpressionStringVectorType& expressions)
276  {
277  assert(expressions.size() >= dimRange);
278  std::vector<std::string> reordered_expressions;
279  for (size_t rr = 0; rr < dimRange; ++rr) {
280  assert(expressions[rr].size() >= dimRangeCols);
281  for (size_t cc = 0; cc < dimRangeCols; ++cc) {
282  reordered_expressions.emplace_back(expressions[rr][cc]);
283  }
284  }
285  function_ = std::make_shared<MathExpressionFunctionType>(variable, reordered_expressions);
286  } // ... build_function(...)

288  void build_gradients(const std::string variable, const GradientStringVectorType& gradient_expressions)
289  {
290  assert(gradient_expressions.size() == 0 || gradient_expressions.size() >= dimRangeCols);
291  if (gradient_expressions.size() > 0) {
292  for (size_t cc = 0; cc < dimRangeCols; ++cc) {
293  gradients_.emplace_back(std::vector<std::shared_ptr<const MathExpressionGradientType>>());
294  assert(gradient_expressions[cc].size() >= dimRange);
295  for (size_t rr = 0; rr < dimRange; ++rr) {
296  const auto& gradient_expression = gradient_expressions[cc][rr];
297  assert(gradient_expression.size() >= dimDomain);

366  static void get_gradient(const Common::Configuration& cfg, GradientStringVectorType& gradient_as_vectors,
367  const std::string first_gradient_key)
368  {
369  // create vector of gradient keys
370  std::vector<std::string> gradient_keys(1, first_gradient_key);
371  for (size_t cc = 1; cc < dimRangeCols; ++cc)
372  gradient_keys.emplace_back("gradient." + Common::to_string(cc));
373  // get gradient as FieldMatrix for every key
374  for (std::string key : gradient_keys) {
375  ExpressionStringVectorType gradient_as_vectors_component;
376  typedef typename Dune::FieldMatrix<std::string, dimRange, dimDomain> JacobianMatrixType;
377  const JacobianMatrixType gradient_as_matrix = cfg.get<JacobianMatrixType>(key);
378  // convert FieldMatrix to ExpressionStringVectorType
379  for (size_t rr = 0; rr < dimRange; ++rr) {
380  std::vector<std::string> gradient_expression;
381  for (size_t ii = 0; ii < dimDomain; ++ii)
382  gradient_expression.emplace_back(gradient_as_matrix[rr][ii]);
383  gradient_as_vectors_component.emplace_back(gradient_expression);
384  }
385  gradient_as_vectors.emplace_back(gradient_as_vectors_component);
386  }
387  } // ... get_gradient(...)
