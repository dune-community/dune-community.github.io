134  virtual void evaluate(const DomainType& xx, RangeType& ret) const override
135  {
136  ret = value_;
137  for (size_t dd = 0; dd < dimDomain; ++dd) {
138  const auto& left = lower_left_[dd];
139  const auto& right = upper_right_[dd];
140  const auto& point = xx[dd];
141  const auto& delta = boundary_layer_[dd];
142  if (point < left - delta) {
143  // outside
144  ret[0] = 0.0;
145  break;
146  } else if (point < left + delta) {
147  // left boundary layer
148  ret[0] *= phi_left((point - (left + delta)) / (2.0 * delta));
149  } else if (point < right - delta) {
150  // inside
151  // do nothing (keep value)
152  } else if (point < right + delta) {
153  // right boundary layer
154  ret[0] *= phi_right((point - (right - delta)) / (2.0 * delta));
155  } else {
156  // outside
157  ret[0] = 0.0;
158  break;
159  }
160  }
161  } // ... evaluate(...)
