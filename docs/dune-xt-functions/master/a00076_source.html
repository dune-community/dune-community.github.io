69  virtual void evaluate(const DomainType& xx, RangeType& ret) const override
70  {
71  // evaluate inner function
72  const auto inner_value = inner_function_.local_function(entity_)->evaluate(xx);
73  // find entity on outer grid the value of inner function belongs to
74  const auto entity_ptrs = (*entity_search_)(std::vector<typename OuterType::DomainType>(1, inner_value));
75  const auto& entity_ptr = entity_ptrs[0];

78  "Could not find entity, maybe inner function does not map to the domain of outer function");

80  // evaluate outer function
81  outer_function_.local_function(outer_entity)->evaluate(outer_entity.geometry().local(inner_value), ret);
82  }
