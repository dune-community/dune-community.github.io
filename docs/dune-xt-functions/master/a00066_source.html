158  CheckerboardFunction(const Common::FieldVector<DomainFieldType, dimDomain>& lowerLeft,
159  const Common::FieldVector<DomainFieldType, dimDomain>& upperRight,
160  const Common::FieldVector<size_t, dimDomain>& numElements, const std::vector<RangeType>& values,
161  const std::string nm = static_id())

163  , upperRight_(new Common::FieldVector<DomainFieldType, dimDomain>(upperRight))

165  , values_(new std::vector<RangeType>(values))

168  // checks
169  size_t totalSubdomains = 1;
170  for (size_t dd = 0; dd < dimDomain; ++dd) {
171  const auto& ll = (*lowerLeft_)[dd];
172  const auto& ur = (*upperRight_)[dd];
173  const auto& ne = (*numElements_)[dd];
174  if (!(ll < ur))
175  DUNE_THROW(Dune::RangeError, "lowerLeft has to be elementwise smaller than upperRight!");
176  totalSubdomains *= ne;
177  }
178  if (values_->size() < totalSubdomains)

199  virtual std::unique_ptr<LocalfunctionType> local_function(const EntityType& entity) const override
200  {
201  // decide on the subdomain the center of the entity belongs to
202  const auto center = entity.geometry().center();
203  std::vector<size_t> whichPartition(dimDomain, 0);
204  const auto& ll = *lowerLeft_;
205  const auto& ur = *upperRight_;
206  const auto& ne = *numElements_;
207  for (size_t dd = 0; dd < dimDomain; ++dd) {
208  // for points that are on upperRight_[d], this selects one partition too much
209  // so we need to cap this
210  whichPartition[dd] =
211  std::min(size_t(std::floor(ne[dd] * ((center[dd] - ll[dd]) / (ur[dd] - ll[dd])))), ne[dd] - 1);
212  }
213  size_t subdomain = 0;
214  if (dimDomain == 1)
215  subdomain = whichPartition[0];
216  else if (dimDomain == 2)
217  subdomain = whichPartition[0] + whichPartition[1] * ne[0];
218  else
219  subdomain = whichPartition[0] + whichPartition[1] * ne[0] + whichPartition[2] * ne[1] * ne[0];
220  // return the component that belongs to the subdomain
221  return std::unique_ptr<Localfunction>(new Localfunction(entity, (*values_)[subdomain]));
222  } // ... local_function(...)
